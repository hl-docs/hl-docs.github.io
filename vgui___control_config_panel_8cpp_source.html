226void ControlConfigPanel::AddCVarFromInputStream(InputStream* is)
227{
228 if(is==null)
229 {
230 return;
231 }
232
233 DataInputStream dis(is);
234
235 bool success;
236
237 while(1)
238 {
239 char buf[256],cvar[128],desc[128];
240 dis.readLine(buf,256,success);
241 if(!success)
242 {
243 break;
244 }
245 if(sscanf(buf,"\"%[^\"]\" \"%[^\"]\"",cvar,desc)==2)
246 {
247 AddCVar(cvar,desc);
248 }
249 }
250}
