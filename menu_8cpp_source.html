104static inline const char* ParseEscapeToken( const char* token )
105{
106 if ( *token != '\\' )
107 return token;
108
109 token++;
110
111 switch ( *token )
112 {
113 case '\0':
114 return token;
115
116 case 'w':
117 menu_r = 255;
118 menu_g = 255;
119 menu_b = 255;
120 break;
121
122 case 'd':
123 menu_r = 100;
124 menu_g = 100;
125 menu_b = 100;
126 break;
127
128 case 'y':
129 menu_r = 255;
130 menu_g = 210;
131 menu_b = 64;
132 break;
133
134 case 'r':
135 menu_r = 210;
136 menu_g = 24;
137 menu_b = 0;
138 break;
139
140 case 'R':
141 menu_x = ScreenWidth/2;
142 menu_ralign = TRUE;
143 break;
144 }
145
146 return ++token;
147}

158int CHudMenu::Draw( float flTime )
159{
160 // check for if menu is set to disappear
161 if ( m_flShutoffTime > 0 )
162 {
163 if ( m_flShutoffTime <= gHUD.m_flTime )
164 { // times up, shutoff
165 m_fMenuDisplayed = 0;
166 m_iFlags &= ~HUD_ACTIVE;
167 return 1;
168 }
169 }
170
171 // don't draw the menu if the scoreboard is being shown
172 if ( gViewPort && gViewPort->IsScoreBoardVisible() )
173 return 1;
174
175 // count the number of newlines in the menu string
176 int nlc = 0;
177 int i;
178 for ( i = 0; i < MAX_MENU_STRING && g_szMenuString[i] != '\0'; i++ )
179 {
180 if ( g_szMenuString[i] == '\n' )
181 nlc++;
182 }
183
184 // calculate vertical center position
185 int y = (ScreenHeight/2) - ((nlc/2)*12) - 40;
186
187 menu_r = 255;
188 menu_g = 255;
189 menu_b = 255;
190 menu_x = 20;
191 menu_ralign = FALSE;
192
193 const char* sptr = g_szMenuString;
194
195 while ( *sptr != '\0' )
196 {
197 if ( *sptr == '\\' )
198 {
199 sptr = ParseEscapeToken( sptr );
200 }
201 else if ( *sptr == '\n' )
202 {
203 menu_ralign = FALSE;
204 menu_x = 20;
205 y += (12);
206
207 sptr++;
208 }
209 else
210 {
211 char menubuf[ 80 ];
212 const char *ptr = sptr;
213 while ( *sptr != '\0' && *sptr != '\n' && *sptr != '\\')
214 {
215 sptr++;
216 }
217 strncpy( menubuf, ptr, min( ( sptr - ptr), (int)sizeof( menubuf ) ));
218 menubuf[ min( ( sptr - ptr), (int)(sizeof( menubuf )-1) ) ] = '\0';
219
220 if ( menu_ralign )
221 {
222 menu_x = gHUD.DrawHudStringReverse( menu_x, y, 0, menubuf, menu_r, menu_g, menu_b );
223 }
224 else
225 {
226 menu_x = gHUD.DrawHudString( menu_x, y, 320, menubuf, menu_r, menu_g, menu_b );
227 }
228 }
229 }
230
231 return 1;
232}

272int CHudMenu::MsgFunc_ShowMenu( const char *pszName, int iSize, void *pbuf )
273{
274 char *temp = NULL;
275
276 BEGIN_READ( pbuf, iSize );
277
278 m_bitsValidSlots = READ_SHORT();
279 int DisplayTime = READ_CHAR();
280 int NeedMore = READ_BYTE();
281
282 if ( DisplayTime > 0 )
283 m_flShutoffTime = DisplayTime + gHUD.m_flTime;
284 else
285 m_flShutoffTime = -1;
286
287 if ( m_bitsValidSlots )
288 {
289 if ( !m_fWaitingForMore )
290 {
291 strncpy( g_szPrelocalisedMenuString, READ_STRING(), MAX_MENU_STRING );
292 }
293 else
294 {
295 strncat( g_szPrelocalisedMenuString, READ_STRING(), MAX_MENU_STRING - strlen(g_szPrelocalisedMenuString) );
296 }
297 g_szPrelocalisedMenuString[MAX_MENU_STRING-1] = 0;
298
299 if ( !NeedMore )
300 {
301 strcpy( g_szMenuString, gHUD.m_TextMessage.BufferedLocaliseTextString( g_szPrelocalisedMenuString ) );
302
303 if ( KB_ConvertString( g_szMenuString, &temp ) )
304 {
305 strcpy( g_szMenuString, temp );
306 free( temp );
307 }
308 }
309
310 m_fMenuDisplayed = 1;
311 m_iFlags |= HUD_ACTIVE;
312 }
313 else
314 {
315 m_fMenuDisplayed = 0;
316 m_iFlags &= ~HUD_ACTIVE;
317 }
318
319 m_fWaitingForMore = NeedMore;
320
321 return 1;
322}
