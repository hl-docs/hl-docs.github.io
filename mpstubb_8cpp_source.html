181void CBaseMonster::Look(int iDistance)
182{
183 int iSighted = 0; // Variable to track sighted entities
184
185 // Clear visibility information from the last frame
186 ClearConditions(bits_COND_SEE_HATE | bits_COND_SEE_DISLIKE | bits_COND_SEE_ENEMY | bits_COND_SEE_FEAR | bits_COND_SEE_NEMESIS | bits_COND_SEE_CLIENT);
187
188 m_pLink = NULL; // Initialize the link to the first visible entity
189
190 CBaseEntity *pSightEnt = NULL; // The current visible entity being processed
191
192 CBaseEntity *pList[100]; // Array to hold entities in the sight box
193
194 Vector delta = Vector(iDistance, iDistance, iDistance); // Create a vector for the sight box dimensions
195
196 // Find only monsters/clients in the box, NOT limited to PVS
197 int count = UTIL_EntitiesInBox(pList, 100, pev->origin - delta, pev->origin + delta, FL_CLIENT | FL_MONSTER);
198 for (int i = 0; i < count; i++)
199 {
200 pSightEnt = pList[i]; // Get the current entity
201 if (pSightEnt != this && pSightEnt->pev->health > 0) // Ensure it's not the monster itself and is alive
202 {
203 // Check if the entity is relevant and visible
204 if (IRelationship(pSightEnt) != R_NO && FInViewCone(pSightEnt) && !FBitSet(pSightEnt->pev->flags, FL_NOTARGET) && FVisible(pSightEnt))
205 {
206 if (pSightEnt->IsPlayer())
207 {
208 // If we see a client, remember that (mostly for scripted AI)
209 iSighted |= bits_COND_SEE_CLIENT; // Set the client sight condition
210 }
211
212 pSightEnt->m_pLink = m_pLink; // Link the visible entity
213 m_pLink = pSightEnt; // Update the head of the link list
214
215 if (pSightEnt == m_hEnemy)
216 {
217 // If the entity is the current enemy, store that now
218 iSighted |= bits_COND_SEE_ENEMY; // Set the enemy sight condition
219 }
220
221 // Update conditions based on the relationship with the sighted entity
222 switch (IRelationship(pSightEnt))
223 {
224 case R_NM:
225 iSighted |= bits_COND_SEE_NEMESIS; // Set nemesis condition
226 break;
227 case R_HT:
228 iSighted |= bits_COND_SEE_HATE; // Set hate condition
229 break;
230 case R_DL:
231 iSighted |= bits_COND_SEE_DISLIKE; // Set dislike condition
232 break;
233 case R_FR:
234 iSighted |= bits_COND_SEE_FEAR; // Set fear condition
235 break;
236 case R_AL:
237 break; // No action for ally
238 default:
239 ALERT(at_aiconsole, "%s can't assess %s\n", STRING(pev->classname), STRING(pSightEnt->pev->classname)); // Log an error for unassessable entities
240 break;
241 }
242 }
243 }
244 }
245
246 SetConditions(iSighted); // Update the monster's conditions based on sighted entities
247}

257CBaseEntity *CBaseMonster::BestVisibleEnemy(void)
258{
259 CBaseEntity *pReturn = NULL; // Pointer to the best visible enemy
260 CBaseEntity *pNextEnt; // Pointer to the next entity in the list
261 int iNearest = 8192; // Initialize nearest distance to a large value
262 int iDist; // Variable to hold the distance to the next entity
263 int iBestRelationship = R_NO; // Variable to hold the best relationship value
264
265 pNextEnt = m_pLink; // Start with the head of the linked list
266
267 while (pNextEnt != NULL) // Iterate through the linked list
268 {
269 if (pNextEnt->IsAlive()) // Check if the entity is alive
270 {
271 if (IRelationship(pNextEnt) > iBestRelationship) // Check if this entity has a better relationship
272 {
273 // Update the best visible enemy
274 iBestRelationship = IRelationship(pNextEnt);
275 iNearest = (pNextEnt->pev->origin - pev->origin).Length(); // Calculate distance
276 pReturn = pNextEnt; // Set the current entity as the best visible enemy
277 }
278 else if (IRelationship(pNextEnt) == iBestRelationship) // Check if the relationship is the same
279 {
280 // Check distance to determine the best visible enemy
281 iDist = (pNextEnt->pev->origin - pev->origin).Length();
282 if (iDist <= iNearest) // If this entity is closer
283 {
284 iNearest = iDist; // Update nearest distance
285 iBestRelationship = IRelationship(pNextEnt); // Update best relationship
286 pReturn = pNextEnt; // Set the current entity as the best visible enemy
287 }
288 }
289 }
290
291 pNextEnt = pNextEnt->m_pLink; // Move to the next entity in the list
292 }
293
294 return pReturn; // Return the best visible enemy
295}
