63void CBasePlayer::Observer_FindNextPlayer(bool bReverse)
64{
65 int iStart;
66 if (m_hObserverTarget)
67 iStart = ENTINDEX(m_hObserverTarget->edict()); // Start from the current target
68 else
69 iStart = ENTINDEX(edict()); // Start from the current player
70
71 int iCurrent = iStart;
72 m_hObserverTarget = NULL; // Reset observer target
73 int iDir = bReverse ? -1 : 1; // Determine search direction
74
75 do
76 {
77 iCurrent += iDir; // Move to the next player index
78
79 // Loop through the clients
80 if (iCurrent > gpGlobals->maxClients)
81 iCurrent = 1; // Wrap around to the first player
82 if (iCurrent < 1)
83 iCurrent = gpGlobals->maxClients; // Wrap around to the last player
84
85 CBaseEntity *pEnt = UTIL_PlayerByIndex(iCurrent); // Get the player entity
86 if (!pEnt)
87 continue; // Skip if entity is invalid
88 if (pEnt == this)
89 continue; // Skip if it's the observer themselves
90 // Don't observe other observers or players who haven't picked a class yet
91 if (((CBasePlayer*)pEnt)->IsObserver() || (pEnt->pev->effects & EF_NODRAW))
92 continue;
93
94 // Additional checks can be added here
95
96 m_hObserverTarget = pEnt; // Set the new observer target
97 break;
98
99 } while (iCurrent != iStart); // Continue until we loop back to the start
100
101 // Did we find a target?
102 if (m_hObserverTarget)
103 {
104 // Move to the target
105 UTIL_SetOrigin(pev, m_hObserverTarget->pev->origin); // Set observer's position to target's position
106
107 // Store the target in pev for the physics DLL
108 if (pev->iuser1 != OBS_ROAMING)
109 pev->iuser2 = ENTINDEX(m_hObserverTarget->edict()); // Store target index
110 }
111}

120void CBasePlayer::Observer_HandleButtons()
121{
122 // Slow down mouse clicks
123 if (m_flNextObserverInput > gpGlobals->time)
124 return; // Exit if input is not ready
125
126 // Jump changes observer modes
127 if (m_afButtonPressed & IN_JUMP)
128 {
129 if (pev->iuser1 == OBS_CHASE_LOCKED)
130 Observer_SetMode(OBS_CHASE_FREE);
131 else if (pev->iuser1 == OBS_CHASE_FREE)
132 Observer_SetMode(OBS_IN_EYE);
133 else if (pev->iuser1 == OBS_IN_EYE)
134 Observer_SetMode(OBS_ROAMING);
135 else if (pev->iuser1 == OBS_ROAMING)
136 Observer_SetMode(OBS_MAP_FREE);
137 else if (pev->iuser1 == OBS_MAP_FREE)
138 Observer_SetMode(OBS_MAP_CHASE);
139 else
140 Observer_SetMode(OBS_CHASE_FREE); // Default mode
141
142 m_flNextObserverInput = gpGlobals->time + 0.2; // Set input cooldown
143 }
144
145 // Attack moves to the next player
146 if (m_afButtonPressed & IN_ATTACK)
147 {
148 Observer_FindNextPlayer(false); // Find the next player in the forward direction
149 m_flNextObserverInput = gpGlobals->time + 0.2; // Set input cooldown
150 }
151
152 // Attack2 moves to the previous player
153 if (m_afButtonPressed & IN_ATTACK2)
154 {
155 Observer_FindNextPlayer(true); // Find the previous player
156 m_flNextObserverInput = gpGlobals->time + 0.2; // Set input cooldown
157 }
158}

166void CBasePlayer::Observer_CheckTarget()
167{
168 if (pev->iuser1 == OBS_ROAMING)
169 return; // No need to check if in roaming mode
170
171 // Try to find a target if there is none
172 if (m_hObserverTarget == NULL)
173 {
174 Observer_FindNextPlayer(false); // Find the next player
175
176 if (m_hObserverTarget == NULL)
177 {
178 // No target found at all
179 int lastMode = pev->iuser1; // Store the last mode
180 Observer_SetMode(OBS_ROAMING); // Switch to roaming mode
181 m_iObserverLastMode = lastMode; // Don't overwrite user's last mode
182 return; // Still have no target, exit
183 }
184 }
185
186 CBasePlayer* target = (CBasePlayer*)(UTIL_PlayerByIndex(ENTINDEX(m_hObserverTarget->edict()))); // Get the target player
187
188 if (!target)
189 {
190 Observer_FindNextPlayer(false); // Find the next player if target is invalid
191 return;
192 }
193
194 // Check if the target is dead
195 if (target->pev->deadflag == DEAD_DEAD)
196 {
197 if ((target->m_fDeadTime + 2.0f) < gpGlobals->time)
198 {
199 // 3 seconds after death, change target
200 Observer_FindNextPlayer(false);
201 return;
202 }
203 }
204}

212void CBasePlayer::Observer_CheckProperties()
213{
214 // Check if in eye mode and if there is a valid target
215 if (pev->iuser1 == OBS_IN_EYE && m_hObserverTarget != NULL)
216 {
217 CBasePlayer* target = (CBasePlayer*)(UTIL_PlayerByIndex(ENTINDEX(m_hObserverTarget->edict()))); // Get the target player
218
219 if (!target)
220 return; // Exit if target is invalid
221
222 int weapon = (target->m_pActiveItem != NULL) ? target->m_pActiveItem->m_iId : 0; // Get the target's weapon ID
223 // Use FOV of tracked client
224 if (m_iFOV != target->m_iFOV || m_iObserverWeapon != weapon)
225 {
226 m_iFOV = target->m_iFOV; // Update observer FOV
227 m_iClientFOV = m_iFOV; // Update client FOV
228 // Write FOV before weapon data, so zoomed crosshair is set correctly
229 MESSAGE_BEGIN(MSG_ONE, gmsgSetFOV, NULL, pev);
230 WRITE_BYTE(m_iFOV); // Send updated FOV
231 MESSAGE_END();
232
233 m_iObserverWeapon = weapon; // Update observer weapon
234 // Send weapon update
235 MESSAGE_BEGIN(MSG_ONE, gmsgCurWeapon, NULL, pev);
236 WRITE_BYTE(1); // 1 = current weapon, not on target
237 WRITE_BYTE(m_iObserverWeapon); // Send weapon ID
238 WRITE_BYTE(0); // Clip
239 MESSAGE_END();
240 }
241 }
242 else
243 {
244 m_iFOV = 90; // Reset FOV to default
245
246 if (m_iObserverWeapon != 0)
247 {
248 m_iObserverWeapon = 0; // Reset observer weapon
249
250 MESSAGE_BEGIN(MSG_ONE, gmsgCurWeapon, NULL, pev);
251 WRITE_BYTE(1); // 1 = current weapon
252 WRITE_BYTE(m_iObserverWeapon); // Send weapon ID
253 WRITE_BYTE(0); // Clip
254 MESSAGE_END();
255 }
256 }

267void CBasePlayer::Observer_SetMode(int iMode)
268{
269 // Abort if changing to the current mode
270 if (iMode == pev->iuser1)
271 return;
272
273 // Validate mode
274 if (iMode < OBS_CHASE_LOCKED || iMode > OBS_MAP_CHASE)
275 iMode = OBS_IN_EYE; // Default to eye mode
276
277 // Verify observer target again
278 if (m_hObserverTarget != NULL)
279 {
280 CBaseEntity *pEnt = m_hObserverTarget;
281
282 if ((pEnt == this) || (pEnt == NULL))
283 m_hObserverTarget = NULL; // Clear target if invalid
284 else if (((CBasePlayer*)pEnt)->IsObserver() || (pEnt->pev->effects & EF_NODRAW))
285 m_hObserverTarget = NULL; // Clear target if it's an observer or not drawn
286 }
287
288 // Set spectator mode
289 pev->iuser1 = iMode;
290
291 // If not roaming, we need a valid target to track
292 if ((iMode != OBS_ROAMING) && (m_hObserverTarget == NULL))
293 {
294 Observer_FindNextPlayer(false); // Find the next player
295
296 // If no valid target found, switch to roaming
297 if (m_hObserverTarget == NULL)
298 {
299 ClientPrint(pev, HUD_PRINTCENTER, "#Spec_NoTarget"); // Notify no target
300 pev->iuser1 = OBS_ROAMING; // Switch to roaming mode
301 }
302 }
303
304 // Set target if not roaming
305 if (pev->iuser1 == OBS_ROAMING)
306 {
307 pev->iuser2 = 0; // Clear target index
308 }
309 else
310 pev->iuser2 = ENTINDEX(m_hObserverTarget->edict()); // Store target index
311
312 pev->iuser3 = 0; // Clear second target from death cam
313
314 // Print spectator mode on client screen
315 char modemsg[16];
316 sprintf(modemsg, "#Spec_Mode%i", pev->iuser1);
317 ClientPrint(pev, HUD_PRINTCENTER, modemsg); // Notify current mode
318
319 m_iObserverLastMode = iMode; // Store last mode
320}
