217void CPython::PrimaryAttack()
218{
219 // Don't fire underwater
220 if (m_pPlayer->pev->waterlevel == 3)
221 {
222 PlayEmptySound(); // Play empty sound
223 m_flNextPrimaryAttack = 0.15; // Set next primary attack time
224 return;
225 }
226
227 if (m_iClip <= 0)
228 {
229 if (!m_fFireOnEmpty)
230 Reload(); // Reload if not firing on empty
231 else
232 {
233 EMIT_SOUND(ENT(m_pPlayer->pev), CHAN_WEAPON, "weapons/357_cock1.wav", 0.8, ATTN_NORM); // Play cock sound
234 m_flNextPrimaryAttack = 0.15; // Set next primary attack time
235 }
236
237 return;
238 }
239
240 m_pPlayer->m_iWeaponVolume = LOUD_GUN_VOLUME; // Set weapon volume
241 m_pPlayer->m_iWeaponFlash = BRIGHT_GUN_FLASH; // Set weapon flash
242
243 m_iClip--; // Decrease ammo in clip
244
245 m_pPlayer->pev->effects = (int)(m_pPlayer->pev->effects) | EF_MUZZLEFLASH; // Set muzzle flash effect
246
247 // Player "shoot" animation
248 m_pPlayer->SetAnimation(PLAYER_ATTACK1);
249
250 UTIL_MakeVectors(m_pPlayer->pev->v_angle + m_pPlayer->pev->punchangle); // Calculate firing direction
251
252 Vector vecSrc = m_pPlayer->GetGunPosition(); // Get gun position
253 Vector vecAiming = m_pPlayer->GetAutoaimVector(AUTOAIM_10DEGREES); // Get aiming vector
254
255 Vector vecDir;
256 vecDir = m_pPlayer->FireBulletsPlayer(1, vecSrc, vecAiming, VECTOR_CONE_1DEGREES, 8192, BULLET_PLAYER_357, 0, 0, m_pPlayer->pev, m_pPlayer->random_seed); // Fire bullets
257
258 int flags;
259#if defined(CLIENT_WEAPONS)
260 flags = FEV_NOTHOST; // Set flags for client weapons
261#else
262 flags = 0; // No flags for server
263#endif
264
265 PLAYBACK_EVENT_FULL(flags, m_pPlayer->edict(), m_usFirePython, 0.0, (float *)&g_vecZero, (float *)&g_vecZero, vecDir.x, vecDir.y, 0, 0, 0, 0); // Play firing event
266
267 if (!m_iClip && m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0)
268 // HEV suit - indicate out of ammo condition
269 m_pPlayer->SetSuitUpdate("!HEV_AMO0", FALSE, 0); // Update suit
270
271 m_flNextPrimaryAttack = 0.75; // Set next primary attack time
272 m_flTimeWeaponIdle = UTIL_SharedRandomFloat(m_pPlayer->random_seed, 10, 15); // Set idle time
273}

309void CPython::WeaponIdle(void)
310{
311 ResetEmptySound(); // Reset empty sound
312
313 m_pPlayer->GetAutoaimVector(AUTOAIM_10DEGREES); // Get autoaim vector
314
315 if (m_flTimeWeaponIdle > UTIL_WeaponTimeBase())
316 return; // Exit if still in idle time
317
318 int iAnim;
319 float flRand = UTIL_SharedRandomFloat(m_pPlayer->random_seed, 0, 1); // Get random value
320 if (flRand <= 0.5)
321 {
322 iAnim = PYTHON_IDLE1; // Select idle animation
323 m_flTimeWeaponIdle = (70.0 / 30.0); // Set idle time
324 }
325 else if (flRand <= 0.7)
326 {
327 iAnim = PYTHON_IDLE2; // Select idle animation
328 m_flTimeWeaponIdle = (60.0 / 30.0); // Set idle time
329 }
330 else if (flRand <= 0.9)
331 {
332 iAnim = PYTHON_IDLE3; // Select idle animation
333 m_flTimeWeaponIdle = (88.0 / 30.0); // Set idle time
334 }
335 else
336 {
337 iAnim = PYTHON_FIDGET; // Select fidget animation
338 m_flTimeWeaponIdle = (170.0 / 30.0); // Set idle time
339 }
340
341 int bUseScope = FALSE;
342#ifdef CLIENT_DLL
343 bUseScope = bIsMultiplayer(); // Check if in multiplayer
344#else
345 bUseScope = g_pGameRules->IsMultiplayer(); // Check if in multiplayer
346#endif
347
348 SendWeaponAnim(iAnim, UseDecrement() ? 1 : 0, bUseScope); // Play idle animation
349}
