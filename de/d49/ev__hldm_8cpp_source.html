107float EV_HLDM_PlayTextureSound( int idx, pmtrace_t *ptr, float *vecSrc, float *vecEnd, int iBulletType )
108{
109 // hit the world, try to play sound based on texture material type
110 char chTextureType = CHAR_TEX_CONCRETE;
111 float fvol;
112 float fvolbar;
113 char *rgsz[4];
114 int cnt;
115 float fattn = ATTN_NORM;
116 int entity;
117 char *pTextureName;
118 char texname[ 64 ];
119 char szbuffer[ 64 ];
120
121 entity = gEngfuncs.pEventAPI->EV_IndexFromTrace( ptr );
122
123 // FIXME check if playtexture sounds movevar is set
124 //
125
126 chTextureType = 0;
127
128 // Player
129 if ( entity >= 1 && entity <= gEngfuncs.GetMaxClients() )
130 {
131 // hit body
132 chTextureType = CHAR_TEX_FLESH;
133 }
134 else if ( entity == 0 )
135 {
136 // get texture from entity or world (world is ent(0))
137 pTextureName = (char *)gEngfuncs.pEventAPI->EV_TraceTexture( ptr->ent, vecSrc, vecEnd );
138
139 if ( pTextureName )
140 {
141 strcpy( texname, pTextureName );
142 pTextureName = texname;
143
144 // strip leading '-0' or '+0~' or '{' or '!'
145 if (*pTextureName == '-' || *pTextureName == '+')
146 {
147 pTextureName += 2;
148 }
149
150 if (*pTextureName == '{' || *pTextureName == '!' || *pTextureName == '~' || *pTextureName == ' ')
151 {
152 pTextureName++;
153 }
154
155 // '}}'
156 strcpy( szbuffer, pTextureName );
157 szbuffer[ CBTEXTURENAMEMAX - 1 ] = 0;
158
159 // get texture type
160 chTextureType = PM_FindTextureType( szbuffer );
161 }
162 }
163
164 switch (chTextureType)
165 {
166 default:
167 case CHAR_TEX_CONCRETE: fvol = 0.9; fvolbar = 0.6;
168 rgsz[0] = "player/pl_step1.wav";
169 rgsz[1] = "player/pl_step2.wav";
170 cnt = 2;
171 break;
172 case CHAR_TEX_METAL: fvol = 0.9; fvolbar = 0.3;
173 rgsz[0] = "player/pl_metal1.wav";
174 rgsz[1] = "player/pl_metal2.wav";
175 cnt = 2;
176 break;
177 case CHAR_TEX_DIRT: fvol = 0.9; fvolbar = 0.1;
178 rgsz[0] = "player/pl_dirt1.wav";
179 rgsz[1] = "player/pl_dirt2.wav";
180 rgsz[2] = "player/pl_dirt3.wav";
181 cnt = 3;
182 break;
183 case CHAR_TEX_VENT: fvol = 0.5; fvolbar = 0.3;
184 rgsz[0] = "player/pl_duct1.wav";
185 rgsz[1] = "player/pl_duct1.wav";
186 cnt = 2;
187 break;
188 case CHAR_TEX_GRATE: fvol = 0.9; fvolbar = 0.5;
189 rgsz[0] = "player/pl_grate1.wav";
190 rgsz[1] = "player/pl_grate4.wav";
191 cnt = 2;
192 break;
193 case CHAR_TEX_TILE: fvol = 0.8; fvolbar = 0.2;
194 rgsz[0] = "player/pl_tile1.wav";
195 rgsz[1] = "player/pl_tile3.wav";
196 rgsz[2] = "player/pl_tile2.wav";
197 rgsz[3] = "player/pl_tile4.wav";
198 cnt = 4;
199 break;
200 case CHAR_TEX_SLOSH: fvol = 0.9; fvolbar = 0.0;
201 rgsz[0] = "player/pl_slosh1.wav";
202 rgsz[1] = "player/pl_slosh3.wav";
203 rgsz[2] = "player/pl_slosh2.wav";
204 rgsz[3] = "player/pl_slosh4.wav";
205 cnt = 4;
206 break;
207 case CHAR_TEX_WOOD: fvol = 0.9; fvolbar = 0.2;
208 rgsz[0] = "debris/wood1.wav";
209 rgsz[1] = "debris/wood2.wav";
210 rgsz[2] = "debris/wood3.wav";
211 cnt = 3;
212 break;
213 case CHAR_TEX_GLASS:
214 case CHAR_TEX_COMPUTER:

218 rgsz[2] = "debris/glass3.wav";
219 cnt = 3;

222 if (iBulletType == BULLET_PLAYER_CROWBAR)
223 return 0.0; // crowbar already makes this sound
224 fvol = 1.0; fvolbar = 0.2;
225 rgsz[0] = "weapons/bullet_hit1.wav";
226 rgsz[1] = "weapons/bullet_hit2.wav";
227 fattn = 1.0;
228 cnt = 2;
229 break;
230 }
231
232 // play material hit sound
233 gEngfuncs.pEventAPI->EV_PlaySound( 0, ptr->endpos, CHAN_STATIC, rgsz[gEngfuncs.pfnRandomLong(0,cnt-1)], fvol, fattn, 0, 96 + gEngfuncs.pfnRandomLong(0,0xf) );

357int EV_HLDM_CheckTracer( int idx, float *vecSrc, float *end, float *forward, float *right, int iBulletType, int iTracerFreq, int *tracerCount )
358{
359 int tracer = 0;
360 int i;
361 qboolean player = idx >= 1 && idx <= gEngfuncs.GetMaxClients() ? true : false;
362
363 if ( iTracerFreq != 0 && ( (*tracerCount)++ % iTracerFreq) == 0 )
364 {
365 vec3_t vecTracerSrc;
366
367 if ( player )
368 {
369 vec3_t offset( 0, 0, -4 );
370
371 // adjust tracer position for player
372 for ( i = 0; i < 3; i++ )
373 {
374 vecTracerSrc[ i ] = vecSrc[ i ] + offset[ i ] + right[ i ] * 2 + forward[ i ] * 16;
375 }
376 }
377 else
378 {
379 VectorCopy( vecSrc, vecTracerSrc );
380 }
381
382 if ( iTracerFreq != 1 ) // guns that always trace also always decal
383 tracer = 1;
384
385 switch( iBulletType )
386 {
387 case BULLET_PLAYER_MP5:
388 case BULLET_MONSTER_MP5:
389 case BULLET_MONSTER_9MM:
390 case BULLET_MONSTER_12MM:
391 default:
392 EV_CreateTracer( vecTracerSrc, end );
393 break;
394 }
395 }
396
397 return tracer;
398}

422void EV_HLDM_FireBullets( int idx, float *forward, float *right, float *up, int cShots, float *vecSrc, float *vecDirShooting, float flDistance, int iBulletType, int iTracerFreq, int *tracerCount, float flSpreadX, float flSpreadY )
423{
424 int i;
425 pmtrace_t tr;
426 int iShot;
427 int tracer;
428
429 for ( iShot = 1; iShot <= cShots; iShot++ )
430 {
431 vec3_t vecDir, vecEnd;
432
433 float x, y, z;
434 //We randomize for the Shotgun.
435 if ( iBulletType == BULLET_PLAYER_BUCKSHOT )
436 {
437 do {
438 x = gEngfuncs.pfnRandomFloat(-0.5,0.5) + gEngfuncs.pfnRandomFloat(-0.5,0.5);
439 y = gEngfuncs.pfnRandomFloat(-0.5,0.5) + gEngfuncs.pfnRandomFloat(-0.5,0.5);
440 z = x*x+y*y;
441 } while (z > 1);
442
443 for ( i = 0 ; i < 3; i++ )
444 {
445 vecDir[i] = vecDirShooting[i] + x * flSpreadX * right[ i ] + y * flSpreadY * up [ i ];
446 vecEnd[i] = vecSrc[ i ] + flDistance * vecDir[ i ];
447 }
448 }//But other guns already have their spread randomized in the synched spread.
449 else
450 {
451
452 for ( i = 0 ; i < 3; i++ )
453 {
454 vecDir[i] = vecDirShooting[i] + flSpreadX * right[ i ] + flSpreadY * up [ i ];
455 vecEnd[i] = vecSrc[ i ] + flDistance * vecDir[ i ];
456 }
457 }
458
459 gEngfuncs.pEventAPI->EV_SetUpPlayerPrediction( false, true );
460
461 // Store off the old count
462 gEngfuncs.pEventAPI->EV_PushPMStates();
463
464 // Now add in all of the players.
465 gEngfuncs.pEventAPI->EV_SetSolidPlayers ( idx - 1 );
466
467 gEngfuncs.pEventAPI->EV_SetTraceHull( 2 );
468 gEngfuncs.pEventAPI->EV_PlayerTrace( vecSrc, vecEnd, PM_STUDIO_BOX, -1, &tr );
469
470 tracer = EV_HLDM_CheckTracer( idx, vecSrc, tr.endpos, forward, right, iBulletType, iTracerFreq, tracerCount );
471
472 // do damage, paint decals
473 if ( tr.fraction != 1.0 )
474 {
475 switch(iBulletType)
476 {
477 default:
478 case BULLET_PLAYER_9MM:
479
480 EV_HLDM_PlayTextureSound( idx, &tr, vecSrc, vecEnd, iBulletType );
481 EV_HLDM_DecalGunshot( &tr, iBulletType );
482
483 break;
484 case BULLET_PLAYER_MP5:
485
486 if ( !tracer )
487 {
488 EV_HLDM_PlayTextureSound( idx, &tr, vecSrc, vecEnd, iBulletType );
489 EV_HLDM_DecalGunshot( &tr, iBulletType );
490 }
491 break;
492 case BULLET_PLAYER_BUCKSHOT:
493
494 EV_HLDM_DecalGunshot( &tr, iBulletType );
495
496 break;
497 case BULLET_PLAYER_357:
498
499 EV_HLDM_PlayTextureSound( idx, &tr, vecSrc, vecEnd, iBulletType );
500 EV_HLDM_DecalGunshot( &tr, iBulletType );
501
502 break;
503
504 }
505 }
506
507 gEngfuncs.pEventAPI->EV_PopPMStates();
508 }
