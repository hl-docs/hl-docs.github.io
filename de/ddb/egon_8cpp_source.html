245void CEgon::Attack( void )
246{
247 // don't fire underwater
248 if ( m_pPlayer->pev->waterlevel == 3 )
249 {
250
251 if ( m_fireState != FIRE_OFF || m_pBeam )
252 {
253 EndAttack();
254 }
255 else
256 {
257 PlayEmptySound( );
258 }
259 return;
260 }
261
262 UTIL_MakeVectors( m_pPlayer->pev->v_angle + m_pPlayer->pev->punchangle );
263 Vector vecAiming = gpGlobals->v_forward;
264 Vector vecSrc = m_pPlayer->GetGunPosition( );
265
266 int flags;
267#if defined( CLIENT_WEAPONS )
268 flags = FEV_NOTHOST;
269#else
270 flags = 0;
271#endif
272
273 switch( m_fireState )
274 {
275 case FIRE_OFF:
276 {
277 if ( !HasAmmo() )
278 {
279 m_flNextPrimaryAttack = m_flNextSecondaryAttack = UTIL_WeaponTimeBase() + 0.25;
280 PlayEmptySound( );
281 return;
282 }
283
284 m_flAmmoUseTime = gpGlobals->time;// start using ammo ASAP.
285
286 PLAYBACK_EVENT_FULL( flags, m_pPlayer->edict(), m_usEgonFire, 0.0, (float *)&g_vecZero, (float *)&g_vecZero, 0.0, 0.0, m_fireState, m_fireMode, 1, 0 );
287
288 m_shakeTime = 0;
289
290 m_pPlayer->m_iWeaponVolume = EGON_PRIMARY_VOLUME;
291 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 0.1;
292 pev->fuser1 = UTIL_WeaponTimeBase() + 2;
293
294 pev->dmgtime = gpGlobals->time + GetPulseInterval();
295 m_fireState = FIRE_CHARGE;
296 }
297 break;
298
299 case FIRE_CHARGE:
300 {
301 Fire( vecSrc, vecAiming );
302 m_pPlayer->m_iWeaponVolume = EGON_PRIMARY_VOLUME;
303
304 if ( pev->fuser1 <= UTIL_WeaponTimeBase() )
305 {
306 PLAYBACK_EVENT_FULL( flags, m_pPlayer->edict(), m_usEgonFire, 0, (float *)&g_vecZero, (float *)&g_vecZero, 0.0, 0.0, m_fireState, m_fireMode, 0, 0 );
307 pev->fuser1 = 1000;
308 }
309
310 if ( !HasAmmo() )
311 {
312 EndAttack();
313 m_flNextPrimaryAttack = m_flNextSecondaryAttack = UTIL_WeaponTimeBase() + 1.0;
314 }
315
316 }
317 break;
318 }
319}

342void CEgon::Fire( const Vector &vecOrigSrc, const Vector &vecDir )
343{
344 Vector vecDest = vecOrigSrc + vecDir * 2048;
345 edict_t *pentIgnore;
346 TraceResult tr;
347
348 pentIgnore = m_pPlayer->edict();
349 Vector tmpSrc = vecOrigSrc + gpGlobals->v_up * -8 + gpGlobals->v_right * 3;
350
351 // ALERT( at_console, "." );
352
353 UTIL_TraceLine( vecOrigSrc, vecDest, dont_ignore_monsters, pentIgnore, &tr );
354
355 if (tr.fAllSolid)
356 return;
357
358#ifndef CLIENT_DLL
359 CBaseEntity *pEntity = CBaseEntity::Instance(tr.pHit);
360
361 if (pEntity == NULL)
362 return;
363
364 if ( g_pGameRules->IsMultiplayer() )
365 {
366 if ( m_pSprite && pEntity->pev->takedamage )
367 {
368 m_pSprite->pev->effects &= ~EF_NODRAW;
369 }
370 else if ( m_pSprite )
371 {
372 m_pSprite->pev->effects |= EF_NODRAW;
373 }
374 }
375
376
377#endif
378
379 float timedist;
380
381 switch ( m_fireMode )
382 {
383 case FIRE_NARROW:
384#ifndef CLIENT_DLL
385 if ( pev->dmgtime < gpGlobals->time )
386 {
387 // Narrow mode only does damage to the entity it hits
388 ClearMultiDamage();
389 if (pEntity->pev->takedamage)
390 {
391 pEntity->TraceAttack( m_pPlayer->pev, gSkillData.plrDmgEgonNarrow, vecDir, &tr, DMG_ENERGYBEAM );
392 }
393 ApplyMultiDamage(m_pPlayer->pev, m_pPlayer->pev);
394
395 if ( g_pGameRules->IsMultiplayer() )
396 {
397 // multiplayer uses 1 ammo every 1/10th second
398 if ( gpGlobals->time >= m_flAmmoUseTime )
399 {
400 UseAmmo( 1 );
401 m_flAmmoUseTime = gpGlobals->time + 0.1;
402 }
403 }
404 else
405 {
406 // single player, use 3 ammo/second
407 if ( gpGlobals->time >= m_flAmmoUseTime )
408 {
409 UseAmmo( 1 );
410 m_flAmmoUseTime = gpGlobals->time + 0.166;
411 }
412 }
413
414 pev->dmgtime = gpGlobals->time + GetPulseInterval();
415 }
416#endif
417 timedist = ( pev->dmgtime - gpGlobals->time ) / GetPulseInterval();
418 break;
419
420 case FIRE_WIDE:
421#ifndef CLIENT_DLL
422 if ( pev->dmgtime < gpGlobals->time )
423 {
424 // wide mode does damage to the ent, and radius damage
425 ClearMultiDamage();
426 if (pEntity->pev->takedamage)
427 {
428 pEntity->TraceAttack( m_pPlayer->pev, gSkillData.plrDmgEgonWide, vecDir, &tr, DMG_ENERGYBEAM | DMG_ALWAYSGIB);
429 }
430 ApplyMultiDamage(m_pPlayer->pev, m_pPlayer->pev);
431
432 if ( g_pGameRules->IsMultiplayer() )
433 {
434 // radius damage a little more potent in multiplayer.
435 ::RadiusDamage( tr.vecEndPos, pev, m_pPlayer->pev, gSkillData.plrDmgEgonWide/4, 128, CLASS_NONE, DMG_ENERGYBEAM | DMG_BLAST | DMG_ALWAYSGIB );
436 }
437
438 if ( !m_pPlayer->IsAlive() )
439 return;
440
441 if ( g_pGameRules->IsMultiplayer() )
442 {
443 //multiplayer uses 5 ammo/second
444 if ( gpGlobals->time >= m_flAmmoUseTime )
445 {
446 UseAmmo( 1 );
447 m_flAmmoUseTime = gpGlobals->time + 0.2;
448 }
449 }
450 else
451 {
452 // Wide mode uses 10 charges per second in single player
453 if ( gpGlobals->time >= m_flAmmoUseTime )
454 {
455 UseAmmo( 1 );
456 m_flAmmoUseTime = gpGlobals->time + 0.1;
457 }
458 }
459
460 pev->dmgtime = gpGlobals->time + GetDischargeInterval();
461 if ( m_shakeTime < gpGlobals->time )
462 {
463 UTIL_ScreenShake( tr.vecEndPos, 5.0, 150.0, 0.75, 250.0 );
464 m_shakeTime = gpGlobals->time + 1.5;
465 }
466 }
467#endif
468 timedist = ( pev->dmgtime - gpGlobals->time ) / GetDischargeInterval();
469 break;
470 }
471
472 if ( timedist < 0 )
473 timedist = 0;
474 else if ( timedist > 1 )
475 timedist = 1;
476 timedist = 1-timedist;
477
478 UpdateEffect( tmpSrc, tr.vecEndPos, timedist );
479}

491void CEgon::UpdateEffect( const Vector &startPoint, const Vector &endPoint, float timeBlend )
492{
493#ifndef CLIENT_DLL
494 if ( !m_pBeam )
495 {
496 CreateEffect();
497 }
498
499 m_pBeam->SetStartPos( endPoint );
500 m_pBeam->SetBrightness( 255 - (timeBlend*180) );
501 m_pBeam->SetWidth( 40 - (timeBlend*20) );
502
503 if ( m_fireMode == FIRE_WIDE )
504 m_pBeam->SetColor( 30 + (25*timeBlend), 30 + (30*timeBlend), 64 + 80*fabs(sin(gpGlobals->time*10)) );
505 else
506 m_pBeam->SetColor( 60 + (25*timeBlend), 120 + (30*timeBlend), 64 + 80*fabs(sin(gpGlobals->time*10)) );
507
508
509 UTIL_SetOrigin( m_pSprite->pev, endPoint );
510 m_pSprite->pev->frame += 8 * gpGlobals->frametime;
511 if ( m_pSprite->pev->frame > m_pSprite->Frames() )
512 m_pSprite->pev->frame = 0;
513
514 m_pNoise->SetStartPos( endPoint );
515
516#endif
517
518}

612void CEgon::WeaponIdle( void )
613{
614 ResetEmptySound( );
615
616 if ( m_flTimeWeaponIdle > gpGlobals->time )
617 return;
618
619 if ( m_fireState != FIRE_OFF )
620 EndAttack();
621
622 int iAnim;
623
624 float flRand = RANDOM_FLOAT(0,1);
625
626 if ( flRand <= 0.5 )
627 {
628 iAnim = EGON_IDLE1;
629 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat( m_pPlayer->random_seed, 10, 15 );
630 }
631 else
632 {
633 iAnim = EGON_FIDGET1;
634 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 3;
635 }
636
637 SendWeaponAnim( iAnim );
638 m_deployed = TRUE;
639}
