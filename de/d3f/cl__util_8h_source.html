220inline char *safe_strcpy( char *dst, const char *src, int len_dst)
221{
222 if( len_dst <= 0 )
223 {
224 return NULL; // this is bad
225 }
226
227 strncpy(dst,src,len_dst);
228 dst[ len_dst - 1 ] = '\0';
229
230 return dst;
231}

245inline int safe_sprintf( char *dst, int len_dst, const char *format, ...)
246{
247 if( len_dst <= 0 )
248 {
249 return -1; // this is bad
250 }
251
252 va_list v;
253
254 va_start(v, format);
255
256 _vsnprintf(dst,len_dst,format,v);
257
258 va_end(v);
259
260 dst[ len_dst - 1 ] = '\0';
261
262 return 0;
263}
