148void CGenericMonster::Spawn()
149{
150 Precache();
151
152 SET_MODEL(ENT(pev), STRING(pev->model));
153
154 // Set the size based on the model type
155 if (FStrEq(STRING(pev->model), "models/player.mdl") || FStrEq(STRING(pev->model), "models/holo.mdl"))
156 UTIL_SetSize(pev, VEC_HULL_MIN, VEC_HULL_MAX);
157 else
158 UTIL_SetSize(pev, VEC_HUMAN_HULL_MIN, VEC_HUMAN_HULL_MAX);
159
160 pev->solid = SOLID_SLIDEBOX; // Set solid type for collision
161 pev->movetype = MOVETYPE_STEP; // Set movement type
162 m_bloodColor = BLOOD_COLOR_RED; // Set blood color
163 pev->health = 8; // Set initial health
164 m_flFieldOfView = 0.5; // Set field of view width
165 m_MonsterState = MONSTERSTATE_NONE; // Initialize monster state
166
167 MonsterInit(); // Initialize the monster
168
169 // Check for special spawn flags
170 if (pev->spawnflags & SF_GENERICMONSTER_NOTSOLID)
171 {
172 pev->solid = SOLID_NOT; // Set to not solid if flag is set
173 pev->takedamage = DAMAGE_NO; // Disable damage
174 }
175}
