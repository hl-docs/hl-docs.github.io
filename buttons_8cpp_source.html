274BOOL CMultiSource::IsTriggered(CBaseEntity *)
275{
276 // Is everything triggered?
277 int i = 0;
278
279 // Still initializing?
280 if (pev->spawnflags & SF_MULTI_INIT)
281 return 0;
282
283 while (i < m_iTotal)
284 {
285 if (m_rgTriggered[i] == 0)
286 break;
287 i++;
288 }
289
290 if (i == m_iTotal)
291 {
292 if (!m_globalstate || gGlobalState.EntityGetState(m_globalstate) == GLOBAL_ON)
293 return 1;
294 }
295
296 return 0;
297}

305void CMultiSource::Register(void)
306{
307 edict_t *pentTarget = NULL;
308
309 m_iTotal = 0;
310 memset(m_rgEntities, 0, MS_MAX_TARGETS * sizeof(EHANDLE));
311
312 SetThink(&CMultiSource::SUB_DoNothing);
313
314 // search for all entities which target this multisource (pev->targetname)
315
316 pentTarget = FIND_ENTITY_BY_STRING(NULL, "target", STRING(pev->targetname));
317
318 while (!FNullEnt(pentTarget) && (m_iTotal < MS_MAX_TARGETS))
319 {
320 CBaseEntity *pTarget = CBaseEntity::Instance(pentTarget);
321 if (pTarget)
322 m_rgEntities[m_iTotal++] = pTarget;
323
324 pentTarget = FIND_ENTITY_BY_STRING(pentTarget, "target", STRING(pev->targetname));
325 }
326
327 pentTarget = FIND_ENTITY_BY_STRING(NULL, "classname", "multi_manager");
328 while (!FNullEnt(pentTarget) && (m_iTotal < MS_MAX_TARGETS))
329 {
330 CBaseEntity *pTarget = CBaseEntity::Instance(pentTarget);
331 if (pTarget && pTarget->HasTarget(pev->targetname))
332 m_rgEntities[m_iTotal++] = pTarget;
333
334 pentTarget = FIND_ENTITY_BY_STRING(pentTarget, "classname", "multi_manager");
335 }
336
337 pev->spawnflags &= ~SF_MULTI_INIT;
338}

478{
479 BUTTON_CODE code = ButtonResponseToTouch();
480

484 SetTouch( NULL );
485

498 }
499 else // code == BUTTON_ACTIVATE
500 ButtonActivate( );
501
502 return 0;
503}

530void CBaseButton::Spawn( )
531{
532 char *pszSound;
533
534 //----------------------------------------------------
535 //determine sounds for buttons
536 //a sound of 0 should not make a sound
537 //----------------------------------------------------
538 pszSound = ButtonSound( m_sounds );
539 PRECACHE_SOUND(pszSound);
540 pev->noise = ALLOC_STRING(pszSound);
541
542 Precache();
543
544 if ( FBitSet ( pev->spawnflags, SF_BUTTON_SPARK_IF_OFF ) )// this button should spark in OFF state
545 {
546 SetThink ( &CBaseButton::ButtonSpark );
547 pev->nextthink = gpGlobals->time + 0.5;// no hurry, make sure everything else spawns
548 }
549
550 SetMovedir(pev);
551
552 pev->movetype = MOVETYPE_PUSH;
553 pev->solid = SOLID_BSP;
554 SET_MODEL(ENT(pev), STRING(pev->model));
555
556 if (pev->speed == 0)
557 pev->speed = 40;
558
559 if (pev->health > 0)
560 {
561 pev->takedamage = DAMAGE_YES;
562 }
563
564 if (m_flWait == 0)
565 m_flWait = 1;
566 if (m_flLip == 0)
567 m_flLip = 4;
568
569 m_toggle_state = TS_AT_BOTTOM;
570 m_vecPosition1 = pev->origin;
571 // Subtract 2 from size because the engine expands bboxes by 1 in all directions making the size too big
572 m_vecPosition2 = m_vecPosition1 + (pev->movedir * (fabs( pev->movedir.x * (pev->size.x-2) ) + fabs( pev->movedir.y * (pev->size.y-2) ) + fabs( pev->movedir.z * (pev->size.z-2) ) - m_flLip));
573
574
575 // Is this a non-moving button?
576 if ( ((m_vecPosition2 - m_vecPosition1).Length() < 1) || (pev->spawnflags & SF_BUTTON_DONTMOVE) )
577 m_vecPosition2 = m_vecPosition1;
578
579 m_fStayPushed = (m_flWait == -1 ? TRUE : FALSE);
580 m_fRotating = FALSE;
581
582 // if the button is flagged for USE button activation only, take away it's touch function and add a use function
583
584 if ( FBitSet ( pev->spawnflags, SF_BUTTON_TOUCH_ONLY ) ) // touchable button
585 {
586 SetTouch( &CBaseButton::ButtonTouch );
587 }
588 else
589 {
590 SetTouch ( NULL );
591 SetUse ( &CBaseButton::ButtonUse );
592 }
593}

758void CBaseButton:: ButtonTouch( CBaseEntity *pOther )
759{
760 // Ignore touches by anything but players
761 if (!FClassnameIs(pOther->pev, "player"))
762 return;
763
764 m_hActivator = pOther;
765
766 BUTTON_CODE code = ButtonResponseToTouch();
767
768 if ( code == BUTTON_NOTHING )
769 return;
770
771 if (!UTIL_IsMasterTriggered(m_sMaster, pOther))
772 {
773 // play button locked sound
774 PlayLockSounds(pev, &m_ls, TRUE, TRUE);
775 return;
776 }
777
778 // Temporarily disable the touch function, until movement is finished.
779 SetTouch( NULL );
780
781 if ( code == BUTTON_RETURN )
782 {
783 EMIT_SOUND(ENT(pev), CHAN_VOICE, (char*)STRING(pev->noise), 1, ATTN_NORM);
784 SUB_UseTargets( m_hActivator, USE_TOGGLE, 0 );
785 ButtonReturn();
786 }
787 else // code == BUTTON_ACTIVATE
788 ButtonActivate( );
789}

797void CBaseButton::ButtonActivate( )
798{
799 EMIT_SOUND(ENT(pev), CHAN_VOICE, (char*)STRING(pev->noise), 1, ATTN_NORM);
800
801 if (!UTIL_IsMasterTriggered(m_sMaster, m_hActivator))
802 {
803 // button is locked, play locked sound
804 PlayLockSounds(pev, &m_ls, TRUE, TRUE);
805 return;
806 }
807 else
808 {
809 // button is unlocked, play unlocked sound
810 PlayLockSounds(pev, &m_ls, FALSE, TRUE);
811 }
812
813 ASSERT(m_toggle_state == TS_AT_BOTTOM);
814 m_toggle_state = TS_GOING_UP;
