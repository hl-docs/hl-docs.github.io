146void CHandGrenade::Holster( int skiplocal /* = 0 */ )
147{
148 m_pPlayer->m_flNextAttack = UTIL_WeaponTimeBase() + 0.5;
149
150 if ( m_pPlayer->m_rgAmmo[ m_iPrimaryAmmoType ] )
151 {
152 SendWeaponAnim( HANDGRENADE_HOLSTER );
153 }
154 else
155 {
156 // no more grenades!
157 m_pPlayer->pev->weapons &= ~(1<<WEAPON_HANDGRENADE);
158 SetThink( &CHandGrenade::DestroyItem );
159 pev->nextthink = gpGlobals->time + 0.1;
160 }
161
162 EMIT_SOUND(ENT(m_pPlayer->pev), CHAN_WEAPON, "common/null.wav", 1.0, ATTN_NORM);
163}

191void CHandGrenade::WeaponIdle( void )
192{
193 if ( m_flReleaseThrow == 0 && m_flStartThrow )
194 m_flReleaseThrow = gpGlobals->time;
195
196 if ( m_flTimeWeaponIdle > UTIL_WeaponTimeBase() )
197 return;
198
199 if ( m_flStartThrow )
200 {
201 Vector angThrow = m_pPlayer->pev->v_angle + m_pPlayer->pev->punchangle;
202
203 if ( angThrow.x < 0 )
204 angThrow.x = -10 + angThrow.x * ( ( 90 - 10 ) / 90.0 );
205 else
206 angThrow.x = -10 + angThrow.x * ( ( 90 + 10 ) / 90.0 );
207
208 float flVel = ( 90 - angThrow.x ) * 4;
209 if ( flVel > 500 )
210 flVel = 500;
211
212 UTIL_MakeVectors( angThrow );
213
214 Vector vecSrc = m_pPlayer->pev->origin + m_pPlayer->pev->view_ofs + gpGlobals->v_forward * 16;
215
216 Vector vecThrow = gpGlobals->v_forward * flVel + m_pPlayer->pev->velocity;
217
218 // alway explode 3 seconds after the pin was pulled
219 float time = m_flStartThrow - gpGlobals->time + 3.0;
220 if (time < 0)
221 time = 0;
222
223 CGrenade::ShootTimed( m_pPlayer->pev, vecSrc, vecThrow, time );
224
225 if ( flVel < 500 )
226 {
227 SendWeaponAnim( HANDGRENADE_THROW1 );
228 }
229 else if ( flVel < 1000 )
230 {
231 SendWeaponAnim( HANDGRENADE_THROW2 );
232 }
233 else
234 {
235 SendWeaponAnim( HANDGRENADE_THROW3 );
236 }
237
238 // player "shoot" animation
239 m_pPlayer->SetAnimation( PLAYER_ATTACK1 );
240
241 m_flReleaseThrow = 0;
242 m_flStartThrow = 0;
243 m_flNextPrimaryAttack = GetNextAttackDelay(0.5);
244 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 0.5;
245
246 m_pPlayer->m_rgAmmo[ m_iPrimaryAmmoType ]--;
247
248 if ( !m_pPlayer->m_rgAmmo[ m_iPrimaryAmmoType ] )
249 {
250 // just threw last grenade
251 // set attack times in the future, and weapon idle in the future so we can see the whole throw
252 // animation, weapon idle will automatically retire the weapon for us.
253 m_flTimeWeaponIdle = m_flNextSecondaryAttack = m_flNextPrimaryAttack = GetNextAttackDelay(0.5);// ensure that the animation can finish playing
254 }
255 return;
256 }
257 else if ( m_flReleaseThrow > 0 )
258 {
259 // we've finished the throw, restart.
260 m_flStartThrow = 0;
261
262 if ( m_pPlayer->m_rgAmmo[ m_iPrimaryAmmoType ] )
263 {
264 SendWeaponAnim( HANDGRENADE_DRAW );
265 }
266 else
267 {
268 RetireWeapon();
269 return;
270 }
271
272 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat( m_pPlayer->random_seed, 10, 15 );
273 m_flReleaseThrow = -1;
274 return;
275 }
276
277 if ( m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] )
278 {
279 int iAnim;
280 float flRand = UTIL_SharedRandomFloat( m_pPlayer->random_seed, 0, 1 );
281 if (flRand <= 0.75)
282 {
283 iAnim = HANDGRENADE_IDLE;
284 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat( m_pPlayer->random_seed, 10, 15 );// how long till we do this again.
285 }
286 else
287 {
288 iAnim = HANDGRENADE_FIDGET;
289 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 75.0 / 30.0;
290 }
291
292 SendWeaponAnim( iAnim );
293 }
294}
