170void CBarnacle::BarnacleThink(void)
171{
172 CBaseEntity *pTouchEnt;
173 CBaseMonster *pVictim;
174 float flLength;
175
176 pev->nextthink = gpGlobals->time + 0.1; // Set the next think time
177
178 if (m_hEnemy != NULL) // Check if the barnacle has prey
179 {
180 if (!m_hEnemy->IsAlive()) // Check if the prey is alive
181 {
182 // Reset barnacle if prey is dead
183 m_fLiftingPrey = FALSE; // Not lifting prey anymore
184 m_hEnemy = NULL; // Clear enemy reference
185 return;
186 }
187
188 if (m_fLiftingPrey) // Check if the barnacle is lifting prey
189 {
190 if (m_hEnemy != NULL && m_hEnemy->pev->deadflag != DEAD_NO) // Check if prey is dead
191 {
192 m_hEnemy = NULL; // Clear enemy reference
193 m_fLiftingPrey = FALSE; // Not lifting prey anymore
194 return;
195 }
196
197 // Still pulling prey
198 Vector vecNewEnemyOrigin = m_hEnemy->pev->origin; // Get the current position of the prey
199 vecNewEnemyOrigin.x = pev->origin.x; // Align x position with barnacle
200 vecNewEnemyOrigin.y = pev->origin.y; // Align y position with barnacle
201
202 // Guess where the neck is
203 vecNewEnemyOrigin.x -= 6 * cos(m_hEnemy->pev->angles.y * M_PI / 180.0);
204 vecNewEnemyOrigin.y -= 6 * sin(m_hEnemy->pev->angles.y * M_PI / 180.0);
205
206 m_flAltitude -= BARNACLE_PULL_SPEED; // Adjust altitude
207 vecNewEnemyOrigin.z += BARNACLE_PULL_SPEED; // Raise prey
208
209 if (fabs(pev->origin.z - (vecNewEnemyOrigin.z + m_hEnemy->pev->view_ofs.z - 8)) < BARNACLE_BODY_HEIGHT)
210 {
211 // Prey has just been lifted into position
212 m_fLiftingPrey = FALSE; // Stop lifting prey
213
214 EMIT_SOUND(ENT(pev), CHAN_WEAPON, "barnacle/bcl_bite3.wav", 1, ATTN_NORM); // Play bite sound
215
216 pVictim = m_hEnemy->MyMonsterPointer(); // Get the victim pointer
217
218 m_flKillVictimTime = gpGlobals->time + 10; // Set time to kill victim
219
220 if (pVictim)
221 {
222 pVictim->BarnacleVictimBitten(pev); // Notify victim of being bitten
223 SetActivity(ACT_EAT); // Set activity to eat
224 }
225 }
226
227 UTIL_SetOrigin(m_hEnemy->pev, vecNewEnemyOrigin); // Update the position of the prey
228 }
229 else
230 {
231 // Prey is lifted fully into feeding position
232 pVictim = m_hEnemy->MyMonsterPointer(); // Get the victim pointer
233
234 if (m_flKillVictimTime != -1 && gpGlobals->time > m_flKillVictimTime)
235 {
236 // Kill the prey
237 if (pVictim)
238 {
239 pVictim->TakeDamage(pev, pev, pVictim->pev->health, DMG_SLASH | DMG_ALWAYSGIB); // Inflict damage
240 m_cGibs = 3; // Set gib count
241 }
242
243 return; // Exit function
244 }
245
246 // Bite prey occasionally
247 if (pVictim && (RANDOM_LONG(0, 49) == 0))
248 {
249 switch (RANDOM_LONG(0, 2))
250 {
251 case 0: EMIT_SOUND(ENT(pev), CHAN_WEAPON, "barnacle/bcl_chew1.wav", 1, ATTN_NORM); break;
252 case 1: EMIT_SOUND(ENT(pev), CHAN_WEAPON, "barnacle/bcl_chew2.wav", 1, ATTN_NORM); break;
253 case 2: EMIT_SOUND(ENT(pev), CHAN_WEAPON, "barnacle/bcl_chew3.wav", 1, ATTN_NORM); break;
254 }
255
256 pVictim->BarnacleVictimBitten(pev); // Notify victim of being bitten
257 }
258 }
259 }
260 else
261 {
262 // Barnacle has no prey, idle and check for tongue contact
263 if (FNullEnt(FIND_CLIENT_IN_PVS(edict())))
264 pev->nextthink = gpGlobals->time + RANDOM_FLOAT(1, 1.5); // Stagger think time
265
266 if (m_fSequenceFinished)
267 {
268 SetActivity(ACT_IDLE); // Set activity to idle
269 m_flTongueAdj = -100; // Reset tongue adjustment
270 }
271
272 if (m_cGibs && RANDOM_LONG(0, 99) == 1)
273 {
274 // Cough up a gib
275 CGib::SpawnRandomGibs(pev, 1, 1); // Spawn gib
276 m_cGibs--; // Decrease gib count
277
278 switch (RANDOM_LONG(0, 2))
279 {
280 case 0: EMIT_SOUND(ENT(pev), CHAN_WEAPON, "barnacle/bcl_chew1.wav", 1, ATTN_NORM); break;
281 case 1: EMIT_SOUND(ENT(pev), CHAN_WEAPON, "barnacle/bcl_chew2.wav", 1, ATTN_NORM); break;
282 case 2: EMIT_SOUND(ENT(pev), CHAN_WEAPON, "barnacle/bcl_chew3.wav", 1, ATTN_NORM); break;
283 }
284 }
285
286 pTouchEnt = TongueTouchEnt(&flLength); // Check for entities touching the tongue
287
288 if (pTouchEnt != NULL && m_fTongueExtended)
289 {
290 // Tongue is extended and touching an entity
291 if (pTouchEnt->FBecomeProne())
292 {
293 EMIT_SOUND(ENT(pev), CHAN_WEAPON, "barnacle/bcl_alert2.wav", 1, ATTN_NORM); // Play alert sound
294
295 SetSequenceByName("attack1"); // Set attack animation
296 m_flTongueAdj = -20; // Adjust tongue position
297
298 m_hEnemy = pTouchEnt; // Set the enemy to the touched entity
299
300 pTouchEnt->pev->movetype = MOVETYPE_FLY; // Set movement type
301 pTouchEnt->pev->velocity = g_vecZero; // Stop movement
302 pTouchEnt->pev->basevelocity = g_vecZero; // Reset base velocity
303 pTouchEnt->pev->origin.x = pev->origin.x; // Align x position
304 pTouchEnt->pev->origin.y = pev->origin.y; // Align y position
305
306 m_fLiftingPrey = TRUE; // Indicate that we should lift prey
307 m_flKillVictimTime = -1; // Reset kill time while lifting
308
309 m_flAltitude = (pev->origin.z - pTouchEnt->EyePosition().z); // Set altitude
310 }
311 }
312 else
313 {
314 // Adjust tongue position if necessary
315 if (m_flAltitude < flLength)
316 {
317 // Lower tongue slowly if too high
318 m_flAltitude += BARNACLE_PULL_SPEED;
319 m_fTongueExtended = FALSE; // Tongue is not extended
320 }
321 else
322 {
323 m_flAltitude = flLength; // Set altitude to length
324 m_fTongueExtended = TRUE; // Tongue is extended
325 }
326 }
327 }
328
329 SetBoneController(0, -(m_flAltitude + m_flTongueAdj)); // Update bone controller
330 StudioFrameAdvance(0.1); // Advance the animation frame
331}

342void CBarnacle::Killed(entvars_t *pevAttacker, int iGib)
343{
344 CBaseMonster *pVictim;
345
346 pev->solid = SOLID_NOT; // Set solid type to not solid
347 pev->takedamage = DAMAGE_NO; // Disable damage
348
349 if (m_hEnemy != NULL) // Check if there is an enemy
350 {
351 pVictim = m_hEnemy->MyMonsterPointer(); // Get the victim pointer
352
353 if (pVictim)
354 {
355 pVictim->BarnacleVictimReleased(); // Notify victim of release
356 }
357 }
358
359 switch (RANDOM_LONG(0, 1))
360 {
361 case 0: EMIT_SOUND(ENT(pev), CHAN_WEAPON, "barnacle/bcl_die1.wav", 1, ATTN_NORM); break;
362 case 1: EMIT_SOUND(ENT(pev), CHAN_WEAPON, "barnacle/bcl_die3.wav", 1, ATTN_NORM); break;
363 }
364
365 SetActivity(ACT_DIESIMPLE); // Set death activity
366 SetBoneController(0, 0); // Reset bone controller
367
368 StudioFrameAdvance(0.1); // Advance the animation frame
369
370 pev->nextthink = gpGlobals->time + 0.1; // Set the next think time
371 SetThink(&CBarnacle::WaitTillDead); // Set think function to wait until dead
372}

426CBaseEntity *CBarnacle::TongueTouchEnt(float *pflLength)
427{
428 TraceResult tr;
429 float length;
430
431 // Trace to hit architecture and see if the tongue needs to change position
432 UTIL_TraceLine(pev->origin, pev->origin - Vector(0, 0, 2048), ignore_monsters, ENT(pev), &tr);
433 length = fabs(pev->origin.z - tr.vecEndPos.z); // Calculate length of the trace
434 if (pflLength)
435 {
436 *pflLength = length; // Store the length if provided
437 }
438
439 Vector delta = Vector(BARNACLE_CHECK_SPACING, BARNACLE_CHECK_SPACING, 0);
440 Vector mins = pev->origin - delta; // Calculate minimum bounds
441 Vector maxs = pev->origin + delta; // Calculate maximum bounds
442 maxs.z = pev->origin.z; // Set z position
443 mins.z -= length; // Adjust minimum z position
444
445 CBaseEntity *pList[10];
446 int count = UTIL_EntitiesInBox(pList, 10, mins, maxs, (FL_CLIENT | FL_MONSTER)); // Get entities in the box
447 if (count)
448 {
449 for (int i = 0; i < count; i++)
450 {
451 // Only clients and monsters
452 if (pList[i] != this && IRelationship(pList[i]) > R_NO && pList[i]->pev->deadflag == DEAD_NO)
453 {
454 // This entity is one of our enemies. Barnacle tries to eat it.
455 return pList[i]; // Return the entity
456 }
457 }
458 }
459
460 return NULL; // No entity found
461}
