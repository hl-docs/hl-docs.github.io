102void CTripmineGrenade::Spawn( void )
103{
104 Precache( );
105 // motor
106 pev->movetype = MOVETYPE_FLY;
107 pev->solid = SOLID_NOT;
108
109 SET_MODEL(ENT(pev), "models/v_tripmine.mdl");
110 pev->frame = 0;
111 pev->body = 3;
112 pev->sequence = TRIPMINE_WORLD;
113 ResetSequenceInfo( );
114 pev->framerate = 0;
115
116 UTIL_SetSize(pev, Vector( -8, -8, -8), Vector(8, 8, 8));
117 UTIL_SetOrigin( pev, pev->origin );
118
119 if (pev->spawnflags & 1)
120 {
121 // power up quickly
122 m_flPowerUp = gpGlobals->time + 1.0;
123 }
124 else
125 {
126 // power up in 2.5 seconds
127 m_flPowerUp = gpGlobals->time + 2.5;
128 }
129
130 SetThink( &CTripmineGrenade::PowerupThink );
131 pev->nextthink = gpGlobals->time + 0.2;
132
133 pev->takedamage = DAMAGE_YES;
134 pev->dmg = gSkillData.plrDmgTripmine;
135 pev->health = 1; // don't let die normally
136
137 if (pev->owner != NULL)
138 {
139 // play deploy sound
140 EMIT_SOUND( ENT(pev), CHAN_VOICE, "weapons/mine_deploy.wav", 1.0, ATTN_NORM );
141 EMIT_SOUND( ENT(pev), CHAN_BODY, "weapons/mine_charge.wav", 0.2, ATTN_NORM ); // chargeup
142
143 m_pRealOwner = pev->owner;// see CTripmineGrenade for why.
144 }
145
146 UTIL_MakeAimVectors( pev->angles );
147
148 m_vecDir = gpGlobals->v_forward;
149 m_vecEnd = pev->origin + m_vecDir * 2048;
150}

192void CTripmineGrenade::PowerupThink( void )
193{
194 TraceResult tr;
195
196 if (m_hOwner == NULL)
197 {
198 // find an owner
199 edict_t *oldowner = pev->owner;
200 pev->owner = NULL;
201 UTIL_TraceLine( pev->origin + m_vecDir * 8, pev->origin - m_vecDir * 32, dont_ignore_monsters, ENT( pev ), &tr );
202 if (tr.fStartSolid || (oldowner && tr.pHit == oldowner))
203 {
204 pev->owner = oldowner;
205 m_flPowerUp += 0.1;
206 pev->nextthink = gpGlobals->time + 0.1;
207 return;
208 }
209 if (tr.flFraction < 1.0)
210 {
211 pev->owner = tr.pHit;
212 m_hOwner = CBaseEntity::Instance( pev->owner );
213 m_posOwner = m_hOwner->pev->origin;
214 m_angleOwner = m_hOwner->pev->angles;
215 }
216 else
217 {
218 STOP_SOUND( ENT(pev), CHAN_VOICE, "weapons/mine_deploy.wav" );
219 STOP_SOUND( ENT(pev), CHAN_BODY, "weapons/mine_charge.wav" );
220 SetThink(&CTripmineGrenade::SUB_Remove );
221 pev->nextthink = gpGlobals->time + 0.1;
222 ALERT( at_console, "WARNING:Tripmine at %.0f, %.0f, %.0f removed\n", pev->origin.x, pev->origin.y, pev->origin.z );
223 KillBeam();
224 return;
225 }
226 }
227 else if (m_posOwner != m_hOwner->pev->origin || m_angleOwner != m_hOwner->pev->angles)
228 {
229 // disable
230 STOP_SOUND( ENT(pev), CHAN_VOICE, "weapons/mine_deploy.wav" );
231 STOP_SOUND( ENT(pev), CHAN_BODY, "weapons/mine_charge.wav" );
232 CBaseEntity *pMine = Create( "weapon_tripmine", pev->origin + m_vecDir * 24, pev->angles );
233 pMine->pev->spawnflags |= SF_NORESPAWN;
234
235 SetThink( &CTripmineGrenade::SUB_Remove );
236 KillBeam();
237 pev->nextthink = gpGlobals->time + 0.1;
238 return;
239 }
240 // ALERT( at_console, "%d %.0f %.0f %0.f\n", pev->owner, m_pOwner->pev->origin.x, m_pOwner->pev->origin.y, m_pOwner->pev->origin.z );
241
242 if (gpGlobals->time > m_flPowerUp)
243 {
244 // make solid
245 pev->solid = SOLID_BBOX;
246 UTIL_SetOrigin( pev, pev->origin );
247
248 MakeBeam( );
249
250 // play enabled sound
251 EMIT_SOUND_DYN( ENT(pev), CHAN_VOICE, "weapons/mine_activate.wav", 0.5, ATTN_NORM, 1.0, 75 );
252 }
253 pev->nextthink = gpGlobals->time + 0.1;
254}

308void CTripmineGrenade::BeamBreakThink( void )
309{
310 BOOL bBlowup = 0;
311
312 TraceResult tr;
313
314 // HACKHACK Set simple box using this really nice global!
315 gpGlobals->trace_flags = FTRACE_SIMPLEBOX;
316 UTIL_TraceLine( pev->origin, m_vecEnd, dont_ignore_monsters, ENT( pev ), &tr );
317
318 // ALERT( at_console, "%f : %f\n", tr.flFraction, m_flBeamLength );
319
320 // respawn detect.
321 if ( !m_pBeam )
322 {
323 MakeBeam( );
324 if ( tr.pHit )
325 m_hOwner = CBaseEntity::Instance( tr.pHit ); // reset owner too
326 }
327
328 if (fabs( m_flBeamLength - tr.flFraction ) > 0.001)
329 {
330 bBlowup = 1;
331 }
332 else
333 {
334 if (m_hOwner == NULL)
335 bBlowup = 1;
336 else if (m_posOwner != m_hOwner->pev->origin)
337 bBlowup = 1;
338 else if (m_angleOwner != m_hOwner->pev->angles)
339 bBlowup = 1;
340 }
341
342 if (bBlowup)
343 {
344 // a bit of a hack, but all CGrenade code passes pev->owner along to make sure the proper player gets credit for the kill
345 // so we have to restore pev->owner from pRealOwner, because an entity's tracelines don't strike it's pev->owner which meant
346 // that a player couldn't trigger his own tripmine. Now that the mine is exploding, it's safe the restore the owner so the
347 // CGrenade code knows who the explosive really belongs to.
348 pev->owner = m_pRealOwner;
349 pev->health = 0;
350 Killed( VARS( pev->owner ), GIB_NORMAL );
351 return;
352 }
353
354 pev->nextthink = gpGlobals->time + 0.1;
355}

543void CTripmine::PrimaryAttack( void )
544{
545 if (m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0)
546 return;
547
548 UTIL_MakeVectors( m_pPlayer->pev->v_angle + m_pPlayer->pev->punchangle );
549 Vector vecSrc = m_pPlayer->GetGunPosition( );
550 Vector vecAiming = gpGlobals->v_forward;
551
552 TraceResult tr;
553
554 UTIL_TraceLine( vecSrc, vecSrc + vecAiming * 128, dont_ignore_monsters, ENT( m_pPlayer->pev ), &tr );
555
556 int flags;
557#ifdef CLIENT_WEAPONS
558 flags = FEV_NOTHOST;
559#else
560 flags = 0;
561#endif
562
563 PLAYBACK_EVENT_FULL( flags, m_pPlayer->edict(), m_usTripFire, 0.0, (float *)&g_vecZero, (float *)&g_vecZero, 0.0, 0.0, 0, 0, 0, 0 );
564
565 if (tr.flFraction < 1.0)
566 {
567 CBaseEntity *pEntity = CBaseEntity::Instance( tr.pHit );
568 if ( pEntity && !(pEntity->pev->flags & FL_CONVEYOR) )
569 {
570 Vector angles = UTIL_VecToAngles( tr.vecPlaneNormal );
571
572 CBaseEntity *pEnt = CBaseEntity::Create( "monster_tripmine", tr.vecEndPos + tr.vecPlaneNormal * 8, angles, m_pPlayer->edict() );
573
574 m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType]--;
575
576 // player "shoot" animation
577 m_pPlayer->SetAnimation( PLAYER_ATTACK1 );
578
579 if ( m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0 )
580 {
581 // no more mines!
582 RetireWeapon();
583 return;
584 }
585 }
586 else
587 {
588 // ALERT( at_console, "no deploy\n" );
589 }
590 }
591 else
592 {
593
594 }
595
596 m_flNextPrimaryAttack = GetNextAttackDelay(0.3);
597 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat( m_pPlayer->random_seed, 10, 15 );
598}

606void CTripmine::WeaponIdle( void )
607{
608 if ( m_flTimeWeaponIdle > UTIL_WeaponTimeBase() )
609 return;
610
611 if ( m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] > 0 )
612 {
613 SendWeaponAnim( TRIPMINE_DRAW );
614 }
615 else
616 {
617 RetireWeapon();
618 return;
619 }
620
621 int iAnim;
622 float flRand = UTIL_SharedRandomFloat( m_pPlayer->random_seed, 0, 1 );
623 if (flRand <= 0.25)
624 {
625 iAnim = TRIPMINE_IDLE1;
626 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 90.0 / 30.0;
627 }
628 else if (flRand <= 0.75)
629 {
630 iAnim = TRIPMINE_IDLE2;
631 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 60.0 / 30.0;
632 }
633 else
634 {
635 iAnim = TRIPMINE_FIDGET;
636 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 100.0 / 30.0;
637 }
638
639 SendWeaponAnim( iAnim );
