385void CMultiManager::KeyValue(KeyValueData *pkvd)
386{
387 // Handle the "wait" key to set the wait time
388 if (FStrEq(pkvd->szKeyName, "wait"))
389 {
390 m_flWait = atof(pkvd->szValue);
391 pkvd->fHandled = TRUE;
392 }
393 else // Add this field to the target list
394 {
395 // This assumes that additional fields are target names and their values are delay values.
396 if (m_cTargets < MAX_MULTI_TARGETS)
397 {
398 char tmp[128];
399
400 UTIL_StripToken(pkvd->szKeyName, tmp);
401 m_iTargetName[m_cTargets] = ALLOC_STRING(tmp);
402 m_flTargetDelay[m_cTargets] = atof(pkvd->szValue);
403 m_cTargets++;
404 pkvd->fHandled = TRUE;
405 }

414void CMultiManager::Spawn(void)
415{
416 pev->solid = SOLID_NOT;
417 SetUse(&CMultiManager::ManagerUse);
418 SetThink(&CMultiManager::ManagerThink);
419
420 // Sort targets using a simple bubble sort
421 int swapped = 1;
422
423 while (swapped)
424 {
425 swapped = 0;
426 for (int i = 1; i < m_cTargets; i++)
427 {
428 if (m_flTargetDelay[i] < m_flTargetDelay[i - 1])
429 {
430 // Swap out of order elements
431 int name = m_iTargetName[i];
432 float delay = m_flTargetDelay[i];
433 m_iTargetName[i] = m_iTargetName[i - 1];
434 m_flTargetDelay[i] = m_flTargetDelay[i - 1];
435 m_iTargetName[i - 1] = name;
436 m_flTargetDelay[i - 1] = delay;
437 swapped = 1;
438 }
439 }
440 }

466void CMultiManager::ManagerThink(void)
467{
468 float time;
469
470 time = gpGlobals->time - m_startTime;
471 while (m_index < m_cTargets && m_flTargetDelay[m_index] <= time)
472 {
473 FireTargets(STRING(m_iTargetName[m_index]), m_hActivator, this, USE_TOGGLE, 0);
474 m_index++;
475 }
476
477 if (m_index >= m_cTargets) // Have we fired all targets?
478 {
479 SetThink(NULL);
480 if (IsClone())
481 {
482 UTIL_Remove(this);
483 return;
484 }
485 SetUse(&CMultiManager::ManagerUse); // Allow manager re-use
486 }
487 else
488 pev->nextthink = m_startTime + m_flTargetDelay[m_index];
489}

499CMultiManager *CMultiManager::Clone(void)
500{
501 CMultiManager *pMulti = GetClassPtr((CMultiManager *)NULL);
502
503 edict_t *pEdict = pMulti->pev->pContainingEntity;
504 memcpy(pMulti->pev, pev, sizeof(*pev));
505 pMulti->pev->pContainingEntity = pEdict;
506
507 pMulti->pev->spawnflags |= SF_MULTIMAN_CLONE;
508 pMulti->m_cTargets = m_cTargets;
509 memcpy(pMulti->m_iTargetName, m_iTargetName, sizeof(m_iTargetName));
510 memcpy(pMulti->m_flTargetDelay, m_flTargetDelay, sizeof(m_flTargetDelay));
511
512 return pMulti;
513}

526void CMultiManager::ManagerUse(CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
527{
528 // In multiplayer games, clone the MM and execute in the clone (like a thread)
529 // to allow multiple players to trigger the same multi-manager
530 if (ShouldClone())
531 {
532 CMultiManager *pClone = Clone();
533 pClone->ManagerUse(pActivator, pCaller, useType, value);
534 return;
535 }
536
537 m_hActivator = pActivator;
538 m_index = 0;
539 m_startTime = gpGlobals->time;
540
541 SetUse(NULL); // Disable use until all targets have fired
542
543 SetThink(&CMultiManager::ManagerThink);
544 pev->nextthink = gpGlobals->time;
545}

1137void CBaseTrigger::HurtTouch(CBaseEntity *pOther)
1138{

1140
1141 if (!pOther->pev->takedamage)
1142 return;
1143
1144 if ((pev->spawnflags & SF_TRIGGER_HURT_CLIENTONLYTOUCH) && !pOther->IsPlayer())
1145 {
1146 // This trigger is only allowed to touch clients, and this isn't a client.
1147 return;
1148 }
1149
1150 if ((pev->spawnflags & SF_TRIGGER_HURT_NO_CLIENTS) && pOther->IsPlayer())
1151 return;
1152
1153 // HACKHACK -- In multiplayer, players touch this based on packet receipt.
1154 // So the players who send packets later aren't always hurt. Keep track of
1155 // how much time has passed and whether or not you've touched that player
1156 if (g_pGameRules->IsMultiplayer())
1157 {
1158 if (pev->dmgtime > gpGlobals->time)
1159 {
1160 if (gpGlobals->time != pev->pain_finished)
1161 { // Too early to hurt again, and not same frame with a different entity
1162 if (pOther->IsPlayer())
1163 {
1164 int playerMask = 1 << (pOther->entindex() - 1);
1165
1166 // If I've already touched this player (this time), then bail out
1167 if (pev->impulse & playerMask)
1168 return;
1169
1170 // Mark this player as touched
1171 // BUGBUG - There can be only 32 players!
1172 pev->impulse |= playerMask;
1173 }
1174 else
1175 {
1176 return;
1177 }
1178 }
1179 }
1180 else
1181 {
1182 // New clock, "un-touch" all players
1183 pev->impulse = 0;
1184 if (pOther->IsPlayer())
1185 {
1186 int playerMask = 1 << (pOther->entindex() - 1);
1187
1188 // Mark this player as touched
1189 // BUGBUG - There can be only 32 players!
1190 pev->impulse |= playerMask;
1191 }
1192 }
1193 }
1194 else // Original code -- single player
1195 {
1196 if (pev->dmgtime > gpGlobals->time && gpGlobals->time != pev->pain_finished)
1197 { // Too early to hurt again, and not same frame with a different entity
1198 return;
1199 }
1200 }
1201
1202 // If this is time-based damage (poison, radiation), override the pev->dmg with a
1203 // default for the given damage type. Monsters only take time-based damage
1204 // while touching the trigger. Player continues taking damage for a while after
1205 // leaving the trigger
1206 fldmg = pev->dmg * 0.5; // 0.5 seconds worth of damage, pev->dmg is damage/second
1207
1208 if (fldmg < 0)
1209 pOther->TakeHealth(-fldmg, m_bitsDamageInflict);
1210 else
1211 pOther->TakeDamage(pev, pev, fldmg, m_bitsDamageInflict);
1212
1213 // Store pain time so we can get all of the other entities on this frame
1214 pev->pain_finished = gpGlobals->time;
1215
1216 // Apply damage every half second
1217 pev->dmgtime = gpGlobals->time + 0.5; // Half second delay until this trigger can hurt toucher again
1218
1219 if (pev->target)
1220 {
1221 // Trigger has a target it wants to fire.
1222 if (pev->spawnflags & SF_TRIGGER_HURT_CLIENTONLYFIRE)
1223 {
1224 // If the toucher isn't a client, don't fire the target!
1225 if (!pOther->IsPlayer())
1226 {
1227 return;
1228 }
1229 }
1230
1231 SUB_UseTargets(pOther, USE_TOGGLE, 0);
1232 if (pev->spawnflags & SF_TRIGGER_HURT_TARGETONCE)
1233 pev->target = 0;
1234 }
