52void CShower::Spawn( void )
53{
54 pev->velocity = RANDOM_FLOAT( 200, 300 ) * pev->angles;
55 pev->velocity.x += RANDOM_FLOAT(-100.f,100.f);
56 pev->velocity.y += RANDOM_FLOAT(-100.f,100.f);
57 if ( pev->velocity.z >= 0 )
58 pev->velocity.z += 200;
59 else
60 pev->velocity.z -= 200;
61 pev->movetype = MOVETYPE_BOUNCE;
62 pev->gravity = 0.5;
63 pev->nextthink = gpGlobals->time + 0.1;
64 pev->solid = SOLID_NOT;

66 UTIL_SetSize(pev, g_vecZero, g_vecZero );
67 pev->effects |= EF_NODRAW;
68 pev->speed = RANDOM_FLOAT( 0.5, 1.5 );
69
70 pev->angles = g_vecZero;
71}

103void CShower::Touch( CBaseEntity *pOther )
104{
105 if ( pev->flags & FL_ONGROUND )
106 pev->velocity = pev->velocity * 0.1;
107 else
108 pev->velocity = pev->velocity * 0.6;
109
110 if ( (pev->velocity.x*pev->velocity.x+pev->velocity.y*pev->velocity.y) < 10.0 )
111 pev->speed = 0;

209void CEnvExplosion::Use(CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
210{
211 TraceResult tr;
212
213 pev->model = iStringNull; // Set model to invisible
214 pev->solid = SOLID_NOT; // Set solid type to intangible
215
216 Vector vecSpot = pev->origin + Vector(0, 0, 8); // Set trace start position
217
218 UTIL_TraceLine(vecSpot, vecSpot + Vector(0, 0, -40), ignore_monsters, ENT(pev), &tr); // Trace downwards
219
220 // Pull out of the wall a bit
221 if (tr.flFraction != 1.0)
222 {
223 pev->origin = tr.vecEndPos + (tr.vecPlaneNormal * (m_iMagnitude - 24) * 0.6); // Adjust position
224 }
225
226 // Draw decal if not suppressed
227 if (!(pev->spawnflags & SF_ENVEXPLOSION_NODECAL))
228 {
229 if (RANDOM_FLOAT(0, 1) < 0.5)
230 {
231 UTIL_DecalTrace(&tr, DECAL_SCORCH1); // Draw scorch decal
232 }
233 else
234 {
235 UTIL_DecalTrace(&tr, DECAL_SCORCH2); // Draw another scorch decal
236 }
237 }
238
239 // Draw fireball effect
240 if (!(pev->spawnflags & SF_ENVEXPLOSION_NOFIREBALL))
241 {
242 MESSAGE_BEGIN(MSG_PAS, SVC_TEMPENTITY, pev->origin);
243 WRITE_BYTE(TE_EXPLOSION);
244 WRITE_COORD(pev->origin.x);
245 WRITE_COORD(pev->origin.y);
246 WRITE_COORD(pev->origin.z);
247 WRITE_SHORT(g_sModelIndexFireball);
248 WRITE_BYTE((BYTE)m_spriteScale); // Scale for fireball
249 WRITE_BYTE(15); // Framerate
250 WRITE_BYTE(TE_EXPLFLAG_NONE);
251 MESSAGE_END();
252 }
253 else
254 {
255 MESSAGE_BEGIN(MSG_PAS, SVC_TEMPENTITY, pev->origin);
256 WRITE_BYTE(TE_EXPLOSION);
257 WRITE_COORD(pev->origin.x);
258 WRITE_COORD(pev->origin.y);
259 WRITE_COORD(pev->origin.z);
260 WRITE_SHORT(g_sModelIndexFireball);
261 WRITE_BYTE(0); // No sprite
262 WRITE_BYTE(15); // Framerate
263 WRITE_BYTE(TE_EXPLFLAG_NONE);
264 MESSAGE_END();
265 }
266
267 // Apply damage if not suppressed
268 if (!(pev->spawnflags & SF_ENVEXPLOSION_NODAMAGE))
269 {
270 RadiusDamage(pev, pev, m_iMagnitude, CLASS_NONE, DMG_BLAST); // Apply blast damage
271 }
272
273 SetThink(&CEnvExplosion::Smoke); // Set the next think function to Smoke
274 pev->nextthink = gpGlobals->time + 0.3; // Schedule next think
275
276 // Draw sparks if not suppressed
277 if (!(pev->spawnflags & SF_ENVEXPLOSION_NOSPARKS))
278 {
279 int sparkCount = RANDOM_LONG(0, 3); // Random number of sparks
280
281 for (int i = 0; i < sparkCount; i++)
282 {
283 Create("spark_shower", pev->origin, tr.vecPlaneNormal, NULL); // Create sparks
284 }
285 }
286}

328void ExplosionCreate(const Vector &center, const Vector &angles, edict_t *pOwner, int magnitude, BOOL doDamage)
329{
330 KeyValueData kvd;
331 char buf[128];
332
333 CBaseEntity *pExplosion = CBaseEntity::Create("env_explosion", center, angles, pOwner); // Create explosion entity
334 sprintf(buf, "%3d", magnitude); // Format magnitude
335 kvd.szKeyName = "iMagnitude"; // Set key name
336 kvd.szValue = buf; // Set key value
337 pExplosion->KeyValue(&kvd); // Set the magnitude
338

341
342 pExplosion->Spawn(); // Spawn the explosion entity
343 pExplosion->Use(NULL, NULL, USE_TOGGLE, 0); // Activate the explosion
