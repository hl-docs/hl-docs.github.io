133 virtual void setText(int textBufferLen, const char* text)
134 {
135 _dualImage->GetImage(0)->setText(text);
136
137 // calculate the text size
138 Font *font = _dualImage->GetImage(0)->getFont();
139 _gap = 0;
140 for (const char *ch = text; *ch != 0; ch++)
141 {
142 int a, b, c;
143 font->getCharABCwide(*ch, a, b, c);
144 _gap += (a + b + c);
145 }
146
147 _gap += XRES(5);
148 }
149
150 virtual void setText(const char* text)
151 {
152 // strip any non-alnum characters from the end
153 char buf[512];
154 strcpy(buf, text);
155
156 int len = strlen(buf);
157 while (len && isspace(buf[--len]))
158 {
159 buf[len] = 0;
160 }
161
162 CLabelHeader::setText(0, buf);
163 }
