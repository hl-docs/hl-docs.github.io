237void UTIL_StringToVector( float * pVector, const char *pString )
238{
239 char *pstr, *pfront, tempString[128];
240 int j;
241
242 strcpy( tempString, pString );
243 pstr = pfront = tempString;
244
245 for ( j = 0; j < 3; j++ )
246 {
247 pVector[j] = atof( pfront );
248
249 while ( *pstr && *pstr != ' ' )
250 pstr++;
251 if (!*pstr)
252 break;
253 pstr++;
254 pfront = pstr;
255 }
256
257 if (j < 2)
258 {
259 for (j = j+1;j < 3; j++)
260 pVector[j] = 0;
261 }

272int UTIL_FindEntityInMap(char * name, float * origin, float * angle)
273{
274 int n,found = 0;
275 char keyname[256];
276 char token[1024];
277
278 cl_entity_t * pEnt = gEngfuncs.GetEntityByIndex( 0 ); // get world model
279
280 if ( !pEnt ) return 0;
281
282 if ( !pEnt->model ) return 0;
283
284 char * data = pEnt->model->entities;
285
286 while (data)
287 {
288 data = gEngfuncs.COM_ParseFile(data, token);
289
290 if ( (token[0] == '}') || (token[0]==0) )
291 break;
292
293 if (!data)
294 {
295 gEngfuncs.Con_DPrintf("UTIL_FindEntityInMap: EOF without closing brace\n");
296 return 0;
297 }
298
299 if (token[0] != '{')
300 {
301 gEngfuncs.Con_DPrintf("UTIL_FindEntityInMap: expected {\n");
302 return 0;
303 }
304
305 // we parse the first { now parse entities properties
306
307 while ( 1 )
308 {
309 // parse key
310 data = gEngfuncs.COM_ParseFile(data, token);
311 if (token[0] == '}')
312 break; // finish parsing this entity
313
314 if (!data)
315 {
316 gEngfuncs.Con_DPrintf("UTIL_FindEntityInMap: EOF without closing brace\n");
317 return 0;
318 };
319
320 strcpy (keyname, token);
321
322 // another hack to fix keynames with trailing spaces
323 n = strlen(keyname);
324 while (n && keyname[n-1] == ' ')
325 {
326 keyname[n-1] = 0;
327 n--;
328 }
329
330 // parse value
331 data = gEngfuncs.COM_ParseFile(data, token);
332 if (!data)
333 {
334 gEngfuncs.Con_DPrintf("UTIL_FindEntityInMap: EOF without closing brace\n");
335 return 0;
336 };
337
338 if (token[0] == '}')
339 {
340 gEngfuncs.Con_DPrintf("UTIL_FindEntityInMap: closing brace without data");
341 return 0;
342 }
343
344 if (!strcmp(keyname,"classname"))
345 {
346 if (!strcmp(token, name ))
347 {
348 found = 1; // thats our entity
349 }
350 };
351
352 if( !strcmp( keyname, "angle" ) )
353 {
354 float y = atof( token );
355
356 if (y >= 0)
357 {
358 angle[0] = 0.0f;
359 angle[1] = y;
360 }
361 else if ((int)y == -1)
362 {
363 angle[0] = -90.0f;
364 angle[1] = 0.0f;;
365 }
366 else
367 {
368 angle[0] = 90.0f;
369 angle[1] = 0.0f;
370 }
371
372 angle[2] = 0.0f;
373 }
374
375 if( !strcmp( keyname, "angles" ) )
376 {
377 UTIL_StringToVector(angle, token);
378 }
379
380 if (!strcmp(keyname,"origin"))
381 {
382 UTIL_StringToVector(origin, token);
383
384 };
385
386 } // while (1)
387
388 if (found)
389 return 1;
390
391 }
392
393 return 0; // we search all entities, but didn't found the correct
394
395}

530bool CHudSpectator::GetDirectorCamera(vec3_t &position, vec3_t &angle)
531{
532 float now = gHUD.m_flTime;
533 float fov = 90.0f;
534
535 if ( m_ChaseEntity )
536 {
537 cl_entity_t * ent = gEngfuncs.GetEntityByIndex( m_ChaseEntity );
538
539 if ( ent )
540 {
541 vec3_t vt = ent->curstate.origin;
542
543 if ( m_ChaseEntity <= gEngfuncs.GetMaxClients() )
544 {
545 if ( ent->curstate.solid == SOLID_NOT )
546 {
547 vt[2]+= -8 ; // PM_DEAD_VIEWHEIGHT
548 }
549 else if (ent->curstate.usehull == 1 )
550 {
551 vt[2]+= 12; // VEC_DUCK_VIEW;
552 }
553 else
554 {
555 vt[2]+= 28; // DEFAULT_VIEWHEIGHT
556 }
557 }
558
559 vt = vt - position;
560 VectorAngles(vt, angle);
561 angle[0] = -angle[0];
562 return true;
563 }
564 else
565 {
566 return false;
567 }
568 }
569
570 if ( !m_IsInterpolating )
571 return false;
572
573 if ( m_WayPoint < 0 || m_WayPoint >= (m_NumWayPoints-1) )
574 return false;
575
576 cameraWayPoint_t * wp1 = &m_CamPath[m_WayPoint];
577 cameraWayPoint_t * wp2 = &m_CamPath[m_WayPoint+1];
578
579 if ( now < wp1->time )
580 return false;
581
582 while ( now > wp2->time )
583 {
584 // go to next waypoint, if possible
585 m_WayPoint++;
586
587 if ( m_WayPoint >= (m_NumWayPoints-1) )
588 {
589 m_IsInterpolating = false;
590 return false; // there is no following waypoint
591 }
592
593 wp1 = wp2;
594 wp2 = &m_CamPath[m_WayPoint+1];
595
596 if ( m_WayPoint > 0 )
597 {
598 // we have a predecessor
599
600 if ( m_WayPoint < (m_NumWayPoints-1) )
601 {
602 // we have also a successor
603 SetWayInterpolation( &m_CamPath[m_WayPoint-1], wp1, wp2, &m_CamPath[m_WayPoint+2] );
604 }
605 else
606 {
607 SetWayInterpolation( &m_CamPath[m_WayPoint-1], wp1, wp2, NULL );
608 }
609 }
610 else if ( m_WayPoint < (m_NumWayPoints-1) )
611 {
612 // we only have a successor
613 SetWayInterpolation( NULL, wp1, wp2, &m_CamPath[m_WayPoint+2] );
614
615 }
616 else
617 {
618 // we have only two waypoints
619 SetWayInterpolation( NULL, wp1, wp2, NULL );
620 }
621 }
622
623 if ( wp2->time <= wp1->time )
624 return false;
625
626 float fraction = (now - wp1->time) / (wp2->time - wp1->time);
627
628 if ( fraction < 0.0f )
629 fraction = 0.0f;
630 else if ( fraction > 1.0f )
631 fraction = 1.0f;
632
633 m_WayInterpolation.Interpolate( fraction, position, angle, &fov );
634
635 // gEngfuncs.Con_Printf("Interpolate time: %.2f, fraction %.2f, point : %.2f,%.2f,%.2f\n", now, fraction, position[0], position[1], position[2] );
636
637 SetCameraView( position, angle, fov );
638
639 return true;
640
641}

734
735
736 // loop through all the players and draw additional infos to their sprites on the map
737 for (int i = 0; i < MAX_PLAYERS; i++)
738 {

761 DrawConsoleString( m_vPlayerPos[i][0]-lx,m_vPlayerPos[i][1], string);

763 }

777void CHudSpectator::DirectorMessage( int iSize, void *pbuf )
778{
779 float f1,f2;
780 char * string;
781 vec3_t v1,v2;
782 int i1,i2,i3;
783
784 BEGIN_READ( pbuf, iSize );
785
786 int cmd = READ_BYTE();
787
788 switch ( cmd ) // director command byte
789 {
790 case DRC_CMD_START :
791 // now we have to do some things clientside, since the proxy doesn't know our mod
792 g_iPlayerClass = 0;
793 g_iTeamNumber = 0;
794
795 // fake a InitHUD & ResetHUD message
796 gHUD.MsgFunc_InitHUD(NULL,0, NULL);
797 gHUD.MsgFunc_ResetHUD(NULL, 0, NULL);
798
799 break;
800
801 case DRC_CMD_EVENT : // old director style message
802 m_lastPrimaryObject = READ_WORD();
803 m_lastSecondaryObject = READ_WORD();
804 m_iObserverFlags = READ_LONG();
805
806 if ( m_autoDirector->value )
807 {
808 if ( (g_iUser2 != m_lastPrimaryObject) || (g_iUser3 != m_lastSecondaryObject) )
809 V_ResetChaseCam();
810
811 g_iUser2 = m_lastPrimaryObject;
812 g_iUser3 = m_lastSecondaryObject;
813 m_IsInterpolating = false;
814 m_ChaseEntity = 0;
815 }
816
817 // gEngfuncs.Con_Printf("Director Camera: %i %i\n", firstObject, secondObject);
818 break;
819 case DRC_CMD_MODE :
820 if ( m_autoDirector->value )
821 {
822 SetModes( READ_BYTE(), -1 );
823 }
824 break;
825
826
827 case DRC_CMD_CAMERA :
828 v1[0] = READ_COORD(); // position
829 v1[1] = READ_COORD();
830 v1[2] = READ_COORD(); // vJumpOrigin
831
832 v2[0] = READ_COORD(); // view angle
833 v2[1] = READ_COORD(); // vJumpAngles
834 v2[2] = READ_COORD();
835 f1 = READ_BYTE(); // fov
836 i1 = READ_WORD(); // target
837
838 if ( m_autoDirector->value )
839 {
840 SetModes( OBS_ROAMING, -1 );
841 SetCameraView(v1, v2, f1);
842 m_ChaseEntity = i1;
843 }
844 break;
845
846 case DRC_CMD_MESSAGE:
847 {
848 client_textmessage_t * msg = &m_HUDMessages[m_lastHudMessage];
849
850 msg->effect = READ_BYTE(); // effect
851
852 UnpackRGB( (int&)msg->r1, (int&)msg->g1, (int&)msg->b1, READ_LONG() ); // color
853 msg->r2 = msg->r1;
854 msg->g2 = msg->g1;
855 msg->b2 = msg->b1;
856 msg->a2 = msg->a1 = 0xFF; // not transparent
857
858 msg->x = READ_FLOAT(); // x pos
859 msg->y = READ_FLOAT(); // y pos
860
861 msg->fadein = READ_FLOAT(); // fadein
862 msg->fadeout = READ_FLOAT(); // fadeout
863 msg->holdtime = READ_FLOAT(); // holdtime
864 msg->fxtime = READ_FLOAT(); // fxtime;
865
866 strncpy( m_HUDMessageText[m_lastHudMessage], READ_STRING(), 128 );
867 m_HUDMessageText[m_lastHudMessage][127]=0; // text
868
869 msg->pMessage = m_HUDMessageText[m_lastHudMessage];
870 msg->pName = "HUD_MESSAGE";
871
872 gHUD.m_Message.MessageAdd( msg );
873
874 m_lastHudMessage++;
875 m_lastHudMessage %= MAX_SPEC_HUD_MESSAGES;
876
877 }
878
879 break;
880
881 case DRC_CMD_SOUND :
882 string = READ_STRING();
883 f1 = READ_FLOAT();
