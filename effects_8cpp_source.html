78{
79 Precache( );
80 SET_MODEL( ENT(pev), STRING(pev->model) ); // Set size
81
82 pev->solid = SOLID_NOT; // Remove model & collisions
83 pev->renderamt = 0; // The engine won't draw this model if this is set to 0 and blending is on
84 pev->rendermode = kRenderTransTexture;
85 int speed = pev->speed > 0 ? pev->speed : -pev->speed;
86
87 // HACKHACK!!! - Speed in rendercolor
88 pev->rendercolor.x = speed >> 8;
89 pev->rendercolor.y = speed & 255;
90 pev->rendercolor.z = (pev->speed < 0) ? 1 : 0;
91
92
93 if ( !(pev->spawnflags & SF_BUBBLES_STARTOFF) )
94 {
95 SetThink( &CBubbling::FizzThink );
96 pev->nextthink = gpGlobals->time + 2.0;
97 m_state = 1;
98 }
99 else
100 m_state = 0;
101}

408void CBeam::RelinkBeam(void)
409{
410 const Vector& startPos = GetStartPos(), & endPos = GetEndPos();
411
412 pev->mins.x = min(startPos.x, endPos.x);
413 pev->mins.y = min(startPos.y, endPos.y);
414 pev->mins.z = min(startPos.z, endPos.z);
415 pev->maxs.x = max(startPos.x, endPos.x);
416 pev->maxs.y = max(startPos.y, endPos.y);
417 pev->maxs.z = max(startPos.z, endPos.z);
418 pev->mins = pev->mins - pev->origin;
419 pev->maxs = pev->maxs - pev-> second;
420
421 UTIL_SetSize(pev, pev->mins, pev->maxs);
422 UTIL_SetOrigin(pev, pev->origin);
423}

453CBaseEntity* CBeam::RandomTargetname(const char* szName)
454{
455 int total = 0;
456
457 CBaseEntity* pEntity = NULL;
458 CBaseEntity* pNewEntity = NULL;
459 while ((pNewEntity = UTIL_FindEntityByTargetname(pNewEntity, szName)) != NULL)
460 {
461 total++;
462 if (RANDOM_LONG(0, total - 1) < 1)
463 pEntity = pNewEntity;
464 }
465 return pEntity;

809void CLightning::StrikeThink(void)
810{
811 if (m_life != 0)
812 {
813 if (pev->spawnflags & SF_BEAM_RANDOM)
814 pev->nextthink = gpGlobals->time + m_life + RANDOM_FLOAT(0, m_restrike);
815 else
816 pev->nextthink = gpGlobals->time + m_life + m_restrike;
817 }
818 m_active = 1;
819
820 if (FStringNull(m_iszEndEntity))
821 {
822 if (FStringNull(m_iszStartEntity))
823 {
824 RandomArea();
825 }
826 else
827 {
828 CBaseEntity* pStart = RandomTargetname(STRING(m_iszStartEntity));
829 if (pStart != NULL)
830 RandomPoint(pStart->pev->origin);
831 else
832 ALERT(at_console, "env_beam: unknown entity \"%s\"\n", STRING(m_iszStartEntity));
833 }
834 return;
835 }
836
837 CBaseEntity* pStart = RandomTargetname(STRING(m_iszStartEntity));
838 CBaseEntity* pEnd = RandomTargetname(STRING(m_iszEndEntity));
839
840 if (pStart != NULL && pEnd != NULL)
841 {
842 if (IsPointEntity(pStart) || IsPointEntity(pEnd))
843 {
844 if (pev->spawnflags & SF_BEAM_RING)
845 {
846 // don't work
847 return;
848 }
849 }
850
851 MESSAGE_BEGIN(MSG_BROADCAST, SVC_TEMPENTITY);
852 if (IsPointEntity(pStart) || IsPointEntity(pEnd))
853 {
854 if (!IsPointEntity(pEnd)) // One point entity must be in pEnd
855 {
856 CBaseEntity* pTemp;
857 pTemp = pStart;
858 pStart = pEnd;
859 pEnd = pTemp;
860 }
861 if (!IsPointEntity(pStart)) // One sided
862 {
863 WRITE_BYTE(TE_BEAMENTPOINT);
864 WRITE_SHORT(pStart->entindex());
865 WRITE_COORD(pEnd->pev->origin.x);
866 WRITE_COORD(pEnd->pev->origin.y);
867 WRITE_COORD(pEnd->pev->origin.z);
868 }
869 else
870 {
871 WRITE_BYTE(TE_BEAMPOINTS);
872 WRITE_COORD(pStart->pev->origin.x);
873 WRITE_COORD(pStart->pev->origin.y);
874 WRITE_COORD(pStart->pev->origin.z);
875 WRITE_COORD(pEnd->pev->origin.x);
876 WRITE_COORD(pEnd->pev->origin.y);
877 WRITE_COORD(pEnd->pev->origin.z);
878 }
879 }
880 else
881 {
882 if (pev->spawnflags & SF_BEAM_RING)
883 WRITE_BYTE(TE_BEAMRING);
884 else
885 WRITE_BYTE(TE_BEAMENTS);
886 WRITE_SHORT(pStart->entindex());
887 WRITE_SHORT(pEnd->entindex());
888 }
889
890 WRITE_SHORT(m_spriteTexture);
891 WRITE_BYTE(m_frameStart); // framestart
892 WRITE_BYTE((int)pev->framerate); // framerate
893 WRITE_BYTE((int)(m_life * 10.0)); // life
894 WRITE_BYTE(m_boltWidth); // width
895 WRITE_BYTE(m_noiseAmplitude); // noise
896 WRITE_BYTE((int)pev->rendercolor.x); // r, g, b
897 WRITE_BYTE((int)pev->rendercolor.y); // r, g, b
898 WRITE_BYTE((int)pev->rendercolor.z); // r, g, b
899 WRITE_BYTE(pev->renderamt); // brightness
900 WRITE_BYTE(m_speed); // speed
901 MESSAGE_END();
902 DoSparks(pStart->pev->origin, pEnd->pev->origin);
903 if (pev->dmg > 0)
904 {
905 TraceResult tr;
906 UTIL_TraceLine(pStart->pev->origin, pEnd->pev->origin, dont_ignore_monsters, NULL, &tr);
907 BeamDamageInstant(&tr, pev->dmg);
908 }
909 }

919void CBeam::BeamDamage(TraceResult* ptr)
920{
921 RelinkBeam();
922 if (ptr->flFraction != 1.0 && ptr->pHit != NULL)
923 {
924 CBaseEntity* pHit = CBaseEntity::Instance(ptr->pHit);
925 if (pHit)
926 {
927 ClearMultiDamage();
928 pHit->TraceAttack(pev, pev->dmg * (gpGlobals->time - pev->dmgtime), (ptr->vecEndPos - pev->origin).Normalize(), ptr, DMG_ENERGYBEAM);
929 ApplyMultiDamage(pev, pev);
930 if (pev->spawnflags & SF_BEAM_DECALS)
931 {
932 if (pHit->IsBSPModel())
933 UTIL_DecalTrace(ptr, DECAL_BIGSHOT1 + RANDOM_LONG(0, 4));
934 }
935 }
936 }
937 pev->dmgtime = gpGlobals->time;
938}

1007void CLightning::RandomArea(void)
1008{
1009 int iLoops = 0;
1010
1011 for (iLoops = 0; iLoops < 10; iLoops++)
1012 {
1013 Vector vecSrc = pev->origin;
1014
1015 Vector vecDir1 = Vector(RANDOM_FLOAT(-1.0, 1.0), RANDOM_FLOAT(-1.0, 1.0), RANDOM_FLOAT(-1.0, 1.0));
1016 vecDir1 = vecDir1.Normalize();
1017 TraceResult tr1;
1018 UTIL_TraceLine(vecSrc, vecSrc + vecDir1 * m_radius, ignore_monsters, ENT(pev), &tr1);
1019
1020 if (tr1.flFraction == 1.0)
1021 continue;
1022
1023 Vector vecDir2;
1024 do {
1025 vecDir2 = Vector(RANDOM_FLOAT(-1.0, 1.0), RANDOM_FLOAT(-1.0, 1.0), RANDOM_FLOAT(-1.0, 1.0));
1026 } while (DotProduct(vecDir1, vecDir2) > 0);
1027 vecDir2 = vecDir2.Normalize();
1028 TraceResult tr2;
1029 UTIL_TraceLine(vecSrc, vecSrc + vecDir2 * m_radius, ignore_monsters, ENT(pev), &tr2);
1030
1031 if (tr2.flFraction == 1.0)
1032 continue;
1033
1034 if ((tr1.vecEndPos - tr2.vecEndPos).Length() < m_radius * 0.1)
1035 continue;
1036
1037 UTIL_TraceLine(tr1.vecEndPos, tr2.vecEndPos, ignore_monsters, ENT(pev), &tr2);
1038
1039 if (tr2.flFraction != 1.0)
1040 continue;
1041
1042 Zap(tr1.vecEndPos, tr2.vecEndPos);
1043
1044 break;
1045 }
1046}
