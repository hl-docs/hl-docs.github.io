107int ScrollTextUp(void)
108{
109 ConsolePrint(g_szLineBuffer[0]); // move the first line into the console buffer
110 g_szLineBuffer[MAX_LINES][0] = 0;
111 memmove(g_szLineBuffer[0], g_szLineBuffer[1], sizeof(g_szLineBuffer) - sizeof(g_szLineBuffer[0])); // overwrite the first line
112 memmove(&g_pflNameColors[0], &g_pflNameColors[1], sizeof(g_pflNameColors) - sizeof(g_pflNameColors[0]));
113 memmove(&g_iNameLengths[0], &g_iNameLengths[1], sizeof(g_iNameLengths) - sizeof(g_iNameLengths[0]));
114 g_szLineBuffer[MAX_LINES - 1][0] = 0;
115
116 if (g_szLineBuffer[0][0] == ' ') // also scroll up following lines
117 {
118 g_szLineBuffer[0][0] = 2;
119 return 1 + ScrollTextUp();
120 }
121
122 return 1;
123}

134int CHudSayText::Draw(float flTime)
135{
136 int y = Y_START;
137
138 if ((gViewPort && gViewPort->AllowedToPrintText() == FALSE) || !m_HUD_saytext->value)
139 return 1;
140
141 // make sure the scrolltime is within reasonable bounds, to guard against the clock being reset
142 flScrollTime = min(flScrollTime, flTime + m_HUD_saytext_time->value);
143
144 if (flScrollTime <= flTime)
145 {
146 if (*g_szLineBuffer[0])
147 {
148 flScrollTime = flTime + m_HUD_saytext_time->value;
149 // push the console up
150 ScrollTextUp();
151 }
152 else
153 { // buffer is empty, just disable drawing of this section
154 m_iFlags &= ~HUD_ACTIVE;
155 }
156 }
157
158 for (int i = 0; i < MAX_LINES; i++)
159 {
160 if (*g_szLineBuffer[i])
161 {
162 if (*g_szLineBuffer[i] == 2 && g_pflNameColors[i])
163 {
164 // it's a saytext string
165 char *buf = static_cast<char *>(_alloca(strlen(g_szLineBuffer[i])));
166 if (buf)
167 {
168 // draw the first x characters in the player color
169 strncpy(buf, g_szLineBuffer[i], min(g_iNameLengths[i], MAX_PLAYER_NAME_LENGTH + 32));
170 buf[min(g_iNameLengths[i], MAX_PLAYER_NAME_LENGTH + 31)] = 0;
171 gEngfuncs.pfnDrawSetTextColor(g_pflNameColors[i][0], g_pflNameColors[i][1], g_pflNameColors[i][2]);
172 int x = DrawConsoleString(LINE_START, y, buf + 1); // don't draw the control code at the start
173 strncpy(buf, g_szLineBuffer[i] + g_iNameLengths[i], strlen(g_szLineBuffer[i]));
174 buf[strlen(g_szLineBuffer[i] + g_iNameLengths[i]) - 1] = '\0';
175 // color is reset after each string draw
176 DrawConsoleString(x, y, buf);
177 }
178 else
179 {
180 assert("Not able to alloca chat buffer!\n");
181 }
182 }
183 else
184 {
185 // normal draw
186 DrawConsoleString(LINE_START, y, g_szLineBuffer[i]);
187 }
188 }
189
190 y += line_height;
191 }
192
193 return 1;
194}

226void CHudSayText::SayTextPrint(const char *pszBuf, int iBufSize, int clientIndex)
227{
228 if (gViewPort && gViewPort->AllowedToPrintText() == FALSE)
229 {
230 // Print it straight to the console
231 ConsolePrint(pszBuf);
232 return;
233 }
234
235 int i;
236 // find an empty string slot
237 for (i = 0; i < MAX_LINES; i++)
238 {
239 if (!*g_szLineBuffer[i])
240 break;
241 }
242 if (i == MAX_LINES)
243 {
244 // force scroll buffer up
245 ScrollTextUp();
246 i = MAX_LINES - 1;
247 }
248
249 g_iNameLengths[i] = 0;
250 g_pflNameColors[i] = NULL;
251
252 // if it's a say message, search for the players name in the string
253 if (*pszBuf == 2 && clientIndex > 0)
254 {
255 gEngfuncs.pfnGetPlayerInfo(clientIndex, &g_PlayerInfoList[clientIndex]);
256 const char *pName = g_PlayerInfoList[clientIndex].name;
257
258 if (pName)
259 {
260 const char *nameInString = strstr(pszBuf, pName);
261
262 if (nameInString)
263 {
264 g_iNameLengths[i] = strlen(pName) + (nameInString - pszBuf);
265 g_pflNameColors[i] = GetClientColor(clientIndex);
266 }
267 }
268 }
269
270 strncpy(g_szLineBuffer[i], pszBuf, max(iBufSize, MAX_CHARS_PER_LINE));
271
272 // make sure the text fits in one line
273 EnsureTextFitsInOneLineAndWrapIfHaveTo(i);
274
275 // Set scroll time
276 if (i == 0)
277 {
278 flScrollTime = gHUD.m_flTime + m_HUD_saytext_time->value;
279 }
280
281 m_iFlags |= HUD_ACTIVE;
282 PlaySound("misc/talk.wav", 1);
283
284 Y_START = ScreenHeight - 60 - (line_height * (MAX_LINES + 2));
285}

295void CHudSayText::EnsureTextFitsInOneLineAndWrapIfHaveTo(int line)
296{
297 int line_width = 0;
298 GetConsoleStringSize(g_szLineBuffer[line], &line_width, &line_height);
299
300 if ((line_width + LINE_START) > MAX_LINE_WIDTH)
301 { // string is too long to fit on line
302 // scan the string until we find what word is too long, and wrap the end of the sentence after the word
303 int length = LINE_START;
304 int tmp_len = 0;
305 char *last_break = NULL;
306 for (char *x = g_szLineBuffer[line]; *x != 0; x++)
307 {
308 // check for a color change, if so skip past it
309 if (x[0] == '/' && x[1] == '(')
310 {
311 x += 2;
312 // skip forward until past mode specifier
313 while (*x != 0 && *x != ')')
314 x++;
315
316 if (*x != 0)
317 x++;
318
319 if (*x == 0)
320 break;
321 }
322
323 char buf[2];
324 buf[1] = 0;
325
326 if (*x == ' ' && x != g_szLineBuffer[line]) // store each line break, except for the very first character
327 last_break = x;
328
329 buf[0] = *x; // get the length of the current character
330 GetConsoleStringSize(buf, &tmp_len, &line_height);
331 length += tmp_len;
332
333 if (length > MAX_LINE_WIDTH)
334 { // needs to be broken up
335 if (!last_break)
336 last_break = x - 1;
337
338 x = last_break;
339
340 // find an empty string slot
341 int j;
342 do
343 {
344 for (j = 0; j < MAX_LINES; j++)
345 {
346 if (!*g_szLineBuffer[j])
347 break;
348 }
349 if (j == MAX_LINES)
350 {
351 // need to make more room to display text, scroll stuff up then fix the pointers
352 int linesmoved = ScrollTextUp();
353 line -= linesmoved;
354 last_break = last_break - (sizeof(g_szLineBuffer[0]) * linesmoved);
355 }
356 }
357 while (j == MAX_LINES);
358
359 // copy remaining string into next buffer, making sure it starts with a space character
360 if ((char)*last_break == (char)' ')
361 {
362 int linelen = strlen(g_szLineBuffer[j]);
363 int remaininglen = strlen(last_break);
364
365 if ((linelen - remaininglen) <= MAX_CHARS_PER_LINE)
366 strcat(g_szLineBuffer[j], last_break);
367 }
368 else
369 {
370 if ((strlen(g_szLineBuffer[j]) - strlen(last_break) - 2) < MAX_CHARS_PER_LINE)
371 {
372 strcat(g_szLineBuffer[j], " ");
373 strcat(g_szLineBuffer[j], last_break);
374 }
375 }
376
377 *last_break = 0; // cut off the last string
378
379 EnsureTextFitsInOneLineAndWrapIfHaveTo(j);
380 break;
381 }
382 }
383 }
