119int CHudDeathNotice::Draw(float flTime)
120{
121 int x, y, r, g, b;
122
123 for (int i = 0; i < MAX_DEATHNOTICES; i++)
124 {
125 if (rgDeathNoticeList[i].iId == 0)
126 break;
127
128 if (rgDeathNoticeList[i].flDisplayTime < flTime)
129 {
130 memmove(&rgDeathNoticeList[i], &rgDeathNoticeList[i+1], sizeof(DeathNoticeItem) * (MAX_DEATHNOTICES - i));
131 i--;
132 continue;
133 }
134
135 rgDeathNoticeList[i].flDisplayTime = min(rgDeathNoticeList[i].flDisplayTime, gHUD.m_flTime + DEATHNOTICE_DISPLAY_TIME);
136
137 if (gViewPort && gViewPort->AllowedToPrintText())
138 {
139 y = DEATHNOTICE_TOP + 2 + (20 * i);
140
141 int id = (rgDeathNoticeList[i].iId == -1) ? m_HUD_d_skull : rgDeathNoticeList[i].iId;
142 x = ScreenWidth - ConsoleStringLen(rgDeathNoticeList[i].szVictim) -
143 (gHUD.GetSpriteRect(id).right - gHUD.GetSpriteRect(id).left);
144
145 if (!rgDeathNoticeList[i].iSuicide)
146 {
147 x -= (5 + ConsoleStringLen(rgDeathNoticeList[i].szKiller));
148 if (rgDeathNoticeList[i].KillerColor)
149 gEngfuncs.pfnDrawSetTextColor(
150 rgDeathNoticeList[i].KillerColor[0],
151 rgDeathNoticeList[i].KillerColor[1],
152 rgDeathNoticeList[i].KillerColor[2]
153 );
154 x = 5 + DrawConsoleString(x, y, rgDeathNoticeList[i].szKiller);
155 }
156
157 r = 255; g = 80; b = 0;
158 if (rgDeathNoticeList[i].iTeamKill)
159 {
160 r = 10; g = 240; b = 10;
161 }
162
163 SPR_Set(gHUD.GetSprite(id), r, g, b);
164 SPR_DrawAdditive(0, x, y, &gHUD.GetSpriteRect(id));
165
166 x += (gHUD.GetSpriteRect(id).right - gHUD.GetSpriteRect(id).left);
167
168 if (rgDeathNoticeList[i].iNonPlayerKill == FALSE)
169 {
170 if (rgDeathNoticeList[i].VictimColor)
171 gEngfuncs.pfnDrawSetTextColor(
172 rgDeathNoticeList[i].VictimColor[0],
173 rgDeathNoticeList[i].VictimColor[1],
174 rgDeathNoticeList[i].VictimColor[2]
175 );
176 x = DrawConsoleString(x, y, rgDeathNoticeList[i].szVictim);
177 }
178 }
179 }
180
181 return 1;
182}

194int CHudDeathNotice::MsgFunc_DeathMsg(const char *pszName, int iSize, void *pbuf)
195{
196 m_iFlags |= HUD_ACTIVE;
197
198 BEGIN_READ(pbuf, iSize);
199
200 int killer = READ_BYTE();
201 int victim = READ_BYTE();
202
203 char killedwith[32];
204 strcpy(killedwith, "d_");
205 strncat(killedwith, READ_STRING(), 32);
206
207 if (gViewPort)
208 gViewPort->DeathMsg(killer, victim);
209
210 gHUD.m_Spectator.DeathMessage(victim);
211
212 int i;
213 for (i = 0; i < MAX_DEATHNOTICES; i++)
214 {
215 if (rgDeathNoticeList[i].iId == 0)
216 break;
217 }
218
219 if (i == MAX_DEATHNOTICES)
220 {
221 memmove(rgDeathNoticeList, rgDeathNoticeList + 1, sizeof(DeathNoticeItem) * MAX_DEATHNOTICES);
222 i = MAX_DEATHNOTICES - 1;
223 }
224
225 if (gViewPort)
226 gViewPort->GetAllPlayersInfo();
227
228 // Killer
229 char *killer_name = g_PlayerInfoList[killer].name;
230 if (!killer_name)
231 {
232 killer_name = "";
233 rgDeathNoticeList[i].szKiller[0] = 0;
234 }
235 else
236 {
237 rgDeathNoticeList[i].KillerColor = GetClientColor(killer);
238 strncpy(rgDeathNoticeList[i].szKiller, killer_name, MAX_PLAYER_NAME_LENGTH);
239 rgDeathNoticeList[i].szKiller[MAX_PLAYER_NAME_LENGTH - 1] = 0;
240 }
241
242 // Victim
243 char *victim_name = ((char)victim != -1) ? g_PlayerInfoList[victim].name : NULL;
244 if (!victim_name)
245 {
246 victim_name = "";
247 rgDeathNoticeList[i].szVictim[0] = 0;
248 }
249 else
250 {
251 rgDeathNoticeList[i].VictimColor = GetClientColor(victim);
252 strncpy(rgDeathNoticeList[i].szVictim, victim_name, MAX_PLAYER_NAME_LENGTH);
253 rgDeathNoticeList[i].szVictim[MAX_PLAYER_NAME_LENGTH - 1] = 0;
254 }
255
256 if ((char)victim == -1)
257 {
258 rgDeathNoticeList[i].iNonPlayerKill = TRUE;
259 strcpy(rgDeathNoticeList[i].szVictim, killedwith + 2);
260 }
261 else
262 {
263 if (killer == victim || killer == 0)
264 rgDeathNoticeList[i].iSuicide = TRUE;
265
266 if (!strcmp(killedwith, "d_teammate"))
267 rgDeathNoticeList[i].iTeamKill = TRUE;
268 }
269
270 int spr = gHUD.GetSpriteIndex(killedwith);
271 rgDeathNoticeList[i].iId = spr;
272
273 DEATHNOTICE_DISPLAY_TIME = CVAR_GET_FLOAT("hud_deathnotice_time");
274 rgDeathNoticeList[i].flDisplayTime = gHUD.m_flTime + DEATHNOTICE_DISPLAY_TIME;
275
276 // Console output
277 if (rgDeathNoticeList[i].iNonPlayerKill)
278 {
279 ConsolePrint(rgDeathNoticeList[i].szKiller);
280 ConsolePrint(" killed a ");
281 ConsolePrint(rgDeathNoticeList[i].szVictim);
282 }
283 else
284 {
285 if (rgDeathNoticeList[i].iSuicide)
286 {
287 ConsolePrint(rgDeathNoticeList[i].szVictim);
288 ConsolePrint(!strcmp(killedwith, "d_world") ? " died" : " killed self");
289 }
290 else if (rgDeathNoticeList[i].iTeamKill)
291 {
292 ConsolePrint(rgDeathNoticeList[i].szKiller);
293 ConsolePrint(" killed his teammate ");
294 ConsolePrint(rgDeathNoticeList[i].szVictim);
295 }
296 else
297 {
298 ConsolePrint(rgDeathNoticeList[i].szKiller);
299 ConsolePrint(" killed ");
300 ConsolePrint(rgDeathNoticeList[i].szVictim);
301 }
302
303 if (killedwith && *killedwith && (*killedwith > 13) && strcmp(killedwith, "d_world") && !rgDeathNoticeList[i].iTeamKill)
304 {
305 ConsolePrint(" with ");
306 if (!strcmp(killedwith + 2, "egon")) strcpy(killedwith, "d_gluon gun");
307 if (!strcmp(killedwith + 2, "gauss")) strcpy(killedwith, "d_tau cannon");
308 ConsolePrint(killedwith + 2);
309 }
310 }
311
312 ConsolePrint("\n");
313
314 return 1;
315}
