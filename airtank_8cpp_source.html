162void CAirtank::TankTouch(CBaseEntity *pOther)
163{
164 if (!pOther->IsPlayer())
165 return;
166
167 if (!m_state)
168 {
169 // "no oxygen" sound
170 EMIT_SOUND(ENT(pev), CHAN_BODY, "player/pl_swim2.wav", 1.0, ATTN_NORM);
171 return;
172 }
173
174 // give player 12 more seconds of air
175 pOther->pev->air_finished = gpGlobals->time + 12;
176
177 // suit recharge sound
178 EMIT_SOUND(ENT(pev), CHAN_VOICE, "doors/aliendoor3.wav", 1.0, ATTN_NORM);
179
180 // recharge airtank in 30 seconds
181 pev->nextthink = gpGlobals->time + 30;
182 m_state = 0;
183 SUB_UseTargets(this, USE_TOGGLE, 1);
184}
