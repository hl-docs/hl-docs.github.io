123void CApache::Spawn(void)
124{
125 Precache(); // Preload resources
126
127 // Set movement and solid type
128 pev->movetype = MOVETYPE_FLY; // Set movement type to fly
129 pev->solid = SOLID_BBOX; // Set solid type to bounding box
130
131 SET_MODEL(ENT(pev), "models/apache.mdl"); // Set the model for the Apache
132 UTIL_SetSize(pev, Vector(-32, -32, -64), Vector(32, 32, 0)); // Set the size of the entity
133 UTIL_SetOrigin(pev, pev->origin); // Set the origin of the entity
134
135 pev->flags |= FL_MONSTER; // Set the monster flag
136 pev->takedamage = DAMAGE_AIM; // Set damage type to aim
137 pev->health = gSkillData.apacheHealth; // Set health based on skill data
138
139 m_flFieldOfView = -0.707; // Set field of view to 270 degrees
140
141 pev->sequence = 0; // Initialize sequence
142 ResetSequenceInfo(); // Reset sequence information
143 pev->frame = RANDOM_LONG(0, 0xFF); // Set a random frame
144
145 InitBoneControllers(); // Initialize bone controllers
146
147 // Check spawn flags to determine behavior
148 if (pev->spawnflags & SF_WAITFORTRIGGER)
149 {
150 SetUse(&CApache::StartupUse); // Set use function for waiting for trigger
151 }
152 else
153 {
154 SetThink(&CApache::HuntThink); // Set think function for hunting behavior
155 SetTouch(&CApache::FlyTouch); // Set touch function for flying behavior
156 pev->nextthink = gpGlobals->time + 1.0; // Schedule the next think
157 }
158
159 m_iRockets = 10; // Initialize the number of rockets
160}

509void CApache :: HuntThink( void )
510{
511 StudioFrameAdvance( );
512 pev->nextthink = gpGlobals->time + 0.1;
513
514 ShowDamage( );
515
516 if ( m_pGoalEnt == NULL && !FStringNull(pev->target) )// this monster has a target
517 {
518 m_pGoalEnt = UTIL_FindEntityByTargetname( NULL, STRING( pev->target ) );
519 if (m_pGoalEnt)
520 {
521 m_posDesired = m_pGoalEnt->pev->origin;
522 UTIL_MakeAimVectors( m_pGoalEnt->pev->angles );
523 m_vecGoal = gpGlobals->v_forward;
524 }
525 }
526
527 // if (m_hEnemy == NULL)
528 {
529 Look( 4092 );
530 m_hEnemy = BestVisibleEnemy( );
531 }
532
533 // generic speed up
534 if (m_flGoalSpeed < 800)
535 m_flGoalSpeed += 5;
536
537 if (m_hEnemy != NULL)
538 {
539 // ALERT( at_console, "%s\n", STRING( m_hEnemy->pev->classname ) );
540 if (FVisible( m_hEnemy ))
541 {
542 if (m_flLastSeen < gpGlobals->time - 5)
543 m_flPrevSeen = gpGlobals->time;
544 m_flLastSeen = gpGlobals->time;
545 m_posTarget = m_hEnemy->Center( );
546 }
547 else
548 {
549 m_hEnemy = NULL;
550 }
551 }
552
553 m_vecTarget = (m_posTarget - pev->origin).Normalize();
554
555 float flLength = (pev->origin - m_posDesired).Length();
556
557 if (m_pGoalEnt)
558 {
559 // ALERT( at_console, "%.0f\n", flLength );
560
561 if (flLength < 128)
562 {
563 m_pGoalEnt = UTIL_FindEntityByTargetname( NULL, STRING( m_pGoalEnt->pev->target ) );
564 if (m_pGoalEnt)
565 {
566 m_posDesired = m_pGoalEnt->pev->origin;
567 UTIL_MakeAimVectors( m_pGoalEnt->pev->angles );
568 m_vecGoal = gpGlobals->v_forward;
569 flLength = (pev->origin - m_posDesired).Length();
570 }
571 }
572 }
573 else
574 {
575 m_posDesired = pev->origin;
576 }
577
578 if (flLength > 250) // 500
579 {
580 // float flLength2 = (m_posTarget - pev->origin).Length() * (1.5 - DotProduct((m_posTarget - pev->origin).Normalize(), pev->velocity.Normalize() ));
581 // if (flLength2 < flLength)
582 if (m_flLastSeen + 90 > gpGlobals->time && DotProduct( (m_posTarget - pev->origin).Normalize(), (m_posDesired - pev->origin).Normalize( )) > 0.25)
583 {
584 m_vecDesired = (m_posTarget - pev->origin).Normalize( );
585 }
586 else
587 {
588 m_vecDesired = (m_posDesired - pev->origin).Normalize( );
589 }
590 }
591 else
592 {
593 m_vecDesired = m_vecGoal;
594 }
595
596 Flight( );
597
598 // ALERT( at_console, "%.0f %.0f %.0f\n", gpGlobals->time, m_flLastSeen, m_flPrevSeen );
599 if ((m_flLastSeen + 1 > gpGlobals->time) && (m_flPrevSeen + 2 < gpGlobals->time))
600 {
601 if (FireGun( ))
602 {
603 // slow down if we're fireing
604 if (m_flGoalSpeed > 400)
605 m_flGoalSpeed = 400;
606 }
607
608 // don't fire rockets and gun on easy mode
609 if (g_iSkillLevel == SKILL_EASY)
610 m_flNextRocket = gpGlobals->time + 10.0;
611 }
612
613 UTIL_MakeAimVectors( pev->angles );
614 Vector vecEst = (gpGlobals->v_forward * 800 + pev->velocity).Normalize( );
615 // ALERT( at_console, "%d %d %d %4.2f\n", pev->angles.x < 0, DotProduct( pev->velocity, gpGlobals->v_forward ) > -100, m_flNextRocket < gpGlobals->time, DotProduct( m_vecTarget, vecEst ) );
616
617 if ((m_iRockets % 2) == 1)
618 {
619 FireRocket( );
620 m_flNextRocket = gpGlobals->time + 0.5;
621 if (m_iRockets <= 0)
622 {
623 m_flNextRocket = gpGlobals->time + 10;
624 m_iRockets = 10;
625 }
626 }
627 else if (pev->angles.x < 0 && DotProduct( pev->velocity, gpGlobals->v_forward ) > -100 && m_flNextRocket < gpGlobals->time)
628 {
629 if (m_flLastSeen + 60 > gpGlobals->time)
630 {
631 if (m_hEnemy != NULL)
632 {
633 // make sure it's a good shot
634 if (DotProduct( m_vecTarget, vecEst) > .965)
635 {
636 TraceResult tr;
637
638 UTIL_TraceLine( pev->origin, pev->origin + vecEst * 4096, ignore_monsters, edict(), &tr );
639 if ((tr.vecEndPos - m_posTarget).Length() < 512)
640 FireRocket( );
641 }
642 }
643 else
644 {
645 TraceResult tr;
646
647 UTIL_TraceLine( pev->origin, pev->origin + vecEst * 4096, dont_ignore_monsters, edict(), &tr );
648 // just fire when close
649 if ((tr.vecEndPos - m_posTarget).Length() < 512)
650 FireRocket( );
651 }
652 }
653 }

662void CApache :: Flight( void )
663{
664 // tilt model 5 degrees
665 Vector vecAdj = Vector( 5.0, 0, 0 );
666
667 // estimate where I'll be facing in one seconds
668 UTIL_MakeAimVectors( pev->angles + pev->avelocity * 2 + vecAdj);
669 // Vector vecEst1 = pev->origin + pev->velocity + gpGlobals->v_up * m_flForce - Vector( 0, 0, 384 );
670 // float flSide = DotProduct( m_posDesired - vecEst1, gpGlobals->v_right );
671
672 float flSide = DotProduct( m_vecDesired, gpGlobals->v_right );
673
674 if (flSide < 0)
675 {
676 if (pev->avelocity.y < 60)
677 {
678 pev->avelocity.y += 8; // 9 * (3.0/2.0);
679 }
680 }
681 else
682 {
683 if (pev->avelocity.y > -60)
684 {
685 pev->avelocity.y -= 8; // 9 * (3.0/2.0);
686 }
687 }
688 pev->avelocity.y *= 0.98;
689
690 // estimate where I'll be in two seconds
691 UTIL_MakeAimVectors( pev->angles + pev->avelocity * 1 + vecAdj);
692 Vector vecEst = pev->origin + pev->velocity * 2.0 + gpGlobals->v_up * m_flForce * 20 - Vector( 0, 0, 384 * 2 );
693
694 // add immediate force
695 UTIL_MakeAimVectors( pev->angles + vecAdj);
696 pev->velocity.x += gpGlobals->v_up.x * m_flForce;
697 pev->velocity.y += gpGlobals->v_up.y * m_flForce;
698 pev->velocity.z += gpGlobals->v_up.z * m_flForce;
699 // add gravity
700 pev->velocity.z -= 38.4; // 32ft/sec
701
702
703 float flSpeed = pev->velocity.Length();
704 float flDir = DotProduct( Vector( gpGlobals->v_forward.x, gpGlobals->v_forward.y, 0 ), Vector( pev->velocity.x, pev->velocity.y, 0 ) );
705 if (flDir < 0)
706 flSpeed = -flSpeed;
707
708 float flDist = DotProduct( m_posDesired - vecEst, gpGlobals->v_forward );
709
710 // float flSlip = DotProduct( pev->velocity, gpGlobals->v_right );
711 float flSlip = -DotProduct( m_posDesired - vecEst, gpGlobals->v_right );
712
713 // fly sideways
714 if (flSlip > 0)
715 {
716 if (pev->angles.z > -30 && pev->avelocity.z > -15)
717 pev->avelocity.z -= 4;
718 else
719 pev->avelocity.z += 2;
720 }
721 else
722 {
723

869BOOL CApache :: FireGun( )
870{

872
873 Vector posGun, angGun;
874 GetAttachment( 1, posGun, angGun );
875
876 Vector vecTarget = (m_posTarget - posGun).Normalize( );
877
878 Vector vecOut;
879
880 vecOut.x = DotProduct( gpGlobals->v_forward, vecTarget );
881 vecOut.y = -DotProduct( gpGlobals->v_right, vecTarget );
882 vecOut.z = DotProduct( gpGlobals->v_up, vecTarget );
883
884 Vector angles = UTIL_VecToAngles (vecOut);
885
886 angles.x = -angles.x;
887 if (angles.y > 180)
888 angles.y = angles.y - 360;
889 if (angles.y < -180)
890 angles.y = angles.y + 360;
891 if (angles.x > 180)
892 angles.x = angles.x - 360;
893 if (angles.x < -180)
894 angles.x = angles.x + 360;
895
896 if (angles.x > m_angGun.x)
897 m_angGun.x = min( angles.x, m_angGun.x + 12 );
898 if (angles.x < m_angGun.x)
899 m_angGun.x = max( angles.x, m_angGun.x - 12 );
900 if (angles.y > m_angGun.y)
901 m_angGun.y = min( angles.y, m_angGun.y + 12 );
902 if (angles.y < m_angGun.y)
903 m_angGun.y = max( angles.y, m_angGun.y - 12 );
904
905 m_angGun.y = SetBoneController( 0, m_angGun.y );
906 m_angGun.x = SetBoneController( 1, m_angGun.x );
907
908 Vector posBarrel, angBarrel;
909 GetAttachment( 0, posBarrel, angBarrel );
910 Vector vecGun = (posBarrel - posGun).Normalize( );
911
912 if (DotProduct( vecGun, vecTarget ) > 0.98)
913 {
914#if 1

922 if (!m_pBeam)

933 flNext = gpGlobals->time + 0.5;
