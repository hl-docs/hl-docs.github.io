240// any sounds that may interest them. (smells, too!)

247void CBaseMonster :: Listen ( void )
248{
249 int iSound;
250 int iMySounds;
251 float hearingSensitivity;
252 CSound *pCurrentSound;
253
254 m_iAudibleList = SOUNDLIST_EMPTY;
255 ClearConditions(bits_COND_HEAR_SOUND | bits_COND_SMELL | bits_COND_SMELL_FOOD);
256 m_afSoundTypes = 0;
257
258 iMySounds = ISoundMask();
259
260 if ( m_pSchedule )
261 {
263 // Make sure your schedule AND personal sound masks agree!
264 iMySounds &= m_pSchedule->iSoundMask;
265 }
266
267 iSound = CSoundEnt::ActiveList();
268
269 // Clear these here?
270 ClearConditions( bits_COND_HEAR_SOUND | bits_COND_SMELL_FOOD | bits_COND_SMELL );
271 hearingSensitivity = HearingSensitivity( );
272
273 while ( iSound != SOUNDLIST_EMPTY )
274 {
275 pCurrentSound = CSoundEnt::SoundPointerForIndex( iSound );
276
277 if ( pCurrentSound &&
278 ( pCurrentSound->m_iType & iMySounds ) &&
279 ( pCurrentSound->m_vecOrigin - EarPosition() ).Length() <= pCurrentSound->m_iVolume * hearingSensitivity )
280
281 //if ( ( g_pSoundEnt->m_SoundPool[ iSound ].m_iType & iMySounds ) && ( g_pSoundEnt->m_SoundPool[ iSound ].m_vecOrigin - EarPosition()).Length () <= g_pSoundEnt->m_SoundPool[ iSound ].m_iVolume * hearingSensitivity )
282 {
283 // the monster cares about this sound, and it's close enough to hear.
284 pCurrentSound->m_iNextAudible = m_iAudibleList;
285
286 if ( pCurrentSound->FIsSound() )
287 {
288 // this is an audible sound.
289 SetConditions( bits_COND_HEAR_SOUND );
290 }
291 else
292 {
293 // if not a sound, must be a smell - determine if it's just a scent, or if it's a food scent
294// if ( g_pSoundEnt->m_SoundPool[ iSound ].m_iType & ( bits_SOUND_MEAT | bits_SOUND_CARCASS ) )
295 if ( pCurrentSound->m_iType & ( bits_SOUND_MEAT | bits_SOUND_CARCASS ) )
296 {
297 // the detected scent is a food item, so set both conditions.
298 // !!!BUGBUG - maybe a virtual function to determine whether or not the scent is food?
299 SetConditions( bits_COND_SMELL_FOOD );
300 SetConditions( bits_COND_SMELL );
301 }
302 else
303 {
304 // just a normal scent.
305 SetConditions( bits_COND_SMELL );
306 }
307 }
308
309// m_afSoundTypes |= g_pSoundEnt->m_SoundPool[ iSound ].m_iType;
310 m_afSoundTypes |= pCurrentSound->m_iType;
311
312 m_iAudibleList = iSound;
313 }
314
315// iSound = g_pSoundEnt->m_SoundPool[ iSound ].m_iNext;
316 iSound = pCurrentSound->m_iNext;
317 }

356void CBaseMonster::Look(int iDistance)
357{
358 int iSighted = 0;
359
360 // Clear previous visibility information
361 ClearConditions(bits_COND_SEE_HATE | bits_COND_SEE_DISLIKE | bits_COND_SEE_ENEMY | bits_COND_SEE_FEAR | bits_COND_SEE_NEMESIS | bits_COND_SEE_CLIENT);
362
363 m_pLink = NULL; // Initialize the link to the first visible entity
364
365 CBaseEntity *pSightEnt = NULL; // Current visible entity
366
367 // If the monster is not a prisoner, check for visible entities
368 if (!FBitSet(pev->spawnflags, SF_MONSTER_PRISONER))
369 {
370 CBaseEntity *pList[100]; // Array to hold visible entities
371
372 Vector delta = Vector(iDistance, iDistance, iDistance); // Create a delta vector for the search
373
374 // Find entities in the box around the monster's position
375 int count = UTIL_EntitiesInBox(pList, 100, pev->origin - delta, pev->origin + delta, FL_CLIENT | FL_MONSTER);
376 for (int i = 0; i < count; i++)
377 {
378 pSightEnt = pList[i];
379
380 // Check if the entity is valid and not the monster itself
381 if (pSightEnt != this && !FBitSet(pSightEnt->pev->spawnflags, SF_MONSTER_PRISONER) && pSightEnt->pev->health > 0)
382 {
383 // Check visibility conditions
384 if (IRelationship(pSightEnt) != R_NO && FInViewCone(pSightEnt) && !FBitSet(pSightEnt->pev->flags, FL_NOTARGET) && FVisible(pSightEnt))
385 {
386 if (pSightEnt->IsPlayer())
387 {
388 if (pev->spawnflags & SF_MONSTER_WAIT_TILL_SEEN)
389 {
390 CBaseMonster *pClient = pSightEnt->MyMonsterPointer();
391 // Check if the player is facing the monster
392 if (pSightEnt && !pClient->FInViewCone(this))
393 {
394 continue; // Skip if not in view
395 }
396 else
397 {
398 // Player sees the monster, update state
399 pev->spawnflags &= ~SF_MONSTER_WAIT_TILL_SEEN;
400 }
401 }
402 // Mark that the monster has seen a client
403 iSighted |= bits_COND_SEE_CLIENT;
404 }
405
406 // Link the visible entity
407 pSightEnt->m_pLink = m_pLink;
408 m_pLink = pSightEnt;
409
410 if (pSightEnt == m_hEnemy)
411 {
412 // If the visible entity is the enemy, update conditions
413 iSighted |= bits_COND_SEE_ENEMY;
414 }
415
416 // Update conditions based on the relationship with the entity
417 switch (IRelationship(pSightEnt))
418 {
419 case R_NM:
420 iSighted |= bits_COND_SEE_NEMESIS;
421 break;
422 case R_HT:
423 iSighted |= bits_COND_SEE_HATE;
424 break;
425 case R_DL:
426 iSighted |= bits_COND_SEE_DISLIKE;
427 break;
428 case R_FR:
429 iSighted |= bits_COND_SEE_FEAR;
430 break;
431 case R_AL:
432 break;
433 default:
434 ALERT(at_aiconsole, "%s can't assess %s\n", STRING(pev->classname), STRING(pSightEnt->pev->classname));
435 break;
436 }
437 }
438 }
439 }
440 }
441
442 // Set the sighted conditions
443 SetConditions(iSighted);
444}

465CSound* CBaseMonster::PBestSound(void)
466{
467 int iThisSound;
468 int iBestSound = -1;
469 float flBestDist = 8192; // Initial distance for the best sound
470 float flDist;
471 CSound *pSound;
472
473 iThisSound = m_iAudibleList;
474
475 if (iThisSound == SOUNDLIST_EMPTY)
476 {
477 ALERT(at_aiconsole, "ERROR! monster %s has no audible sounds!\n", STRING(pev->classname));
478 return NULL;
479 }
480
481 while (iThisSound != SOUNDLIST_EMPTY)
482 {
483 pSound = CSoundEnt::SoundPointerForIndex(iThisSound);
484
485 if (pSound && pSound->FIsSound())
486 {
487 flDist = (pSound->m_vecOrigin - EarPosition()).Length();
488
489 if (flDist < flBestDist)
490 {
491 iBestSound = iThisSound;
492 flBestDist = flDist;
493 }
494 }
495
496 iThisSound = pSound->m_iNextAudible;
497 }
498 if (iBestSound >= 0)
499 {
500 pSound = CSoundEnt::SoundPointerForIndex(iBestSound);
501 return pSound;
502 }
503 return NULL;
504}

557// Monster Think - calls out to core AI functions and handles this
558// monster's specific animation events
559//=========================================================
567void CBaseMonster :: MonsterThink ( void )
568{
569 pev->nextthink = gpGlobals->time + 0.1;// keep monster thinking.
570
571
572 RunAI();
573
574 float flInterval = StudioFrameAdvance( ); // animate
575// start or end a fidget
576// This needs a better home -- switching animations over time should be encapsulated on a per-activity basis
577// perhaps MaintainActivity() or a ShiftAnimationOverTime() or something.
578 if ( m_MonsterState != MONSTERSTATE_SCRIPT && m_MonsterState != MONSTERSTATE_DEAD && m_Activity == ACT_IDLE && m_fSequenceFinished )
579 {
580 int iSequence;
581
582 if ( m_fSequenceLoops )
583 {
584 // animation does loop, which means we're playing subtle idle. Might need to
585 // fidget.
586 iSequence = LookupActivity ( m_Activity );
587 }
588 else
589 {
590 // animation that just ended doesn't loop! That means we just finished a fidget
591 // and should return to our heaviest weighted idle (the subtle one)
592 iSequence = LookupActivityHeaviest ( m_Activity );
593 }
594 if ( iSequence != ACTIVITY_NOT_AVAILABLE )
595 {
596 pev->sequence = iSequence; // Set to new anim (if it's there)
597 ResetSequenceInfo( );
598 }
599 }
600
601 DispatchAnimEvents( flInterval );
602
603 if ( !MovementIsComplete() )
604 {
605 Move( flInterval );
606 }
607#if _DEBUG
608 else
609 {
610 if ( !TaskIsRunning() && !TaskIsComplete() )
611 ALERT( at_error, "Schedule stalled!!\n" );
612 }
613#endif
614}

717//=========================================================
718// FRefreshRoute - after calculating a path to the monster's
719// target, this function copies as many waypoints as possible
720// from that path to the monster's Route array
721//=========================================================
730BOOL CBaseMonster :: FRefreshRoute ( void )
731{
732 CBaseEntity *pPathCorner;
733 int i;
734 BOOL returnCode;
735
736 RouteNew();
737
738 returnCode = FALSE;
739
740 switch( m_movementGoal )
741 {
742 case MOVEGOAL_PATHCORNER:
743 {
744 // monster is on a path_corner loop
745 pPathCorner = m_pGoalEnt;
746 i = 0;
747
748 while ( pPathCorner && i < ROUTE_SIZE )
749 {
750 m_Route[ i ].iType = bits_MF_TO_PATHCORNER;
751 m_Route[ i ].vecLocation = pPathCorner->pev->origin;
752
753 pPathCorner = pPathCorner->GetNextTarget();
754
755 // Last path_corner in list?
756 if ( !pPathCorner )
757 m_Route[i].iType |= bits_MF_IS_GOAL;
758
759 i++;
760 }
761 }
762 returnCode = TRUE;
763 break;
764
765 case MOVEGOAL_ENEMY:
766 returnCode = BuildRoute( m_vecEnemyLKP, bits_MF_TO_ENEMY, m_hEnemy );
767 break;
768
769 case MOVEGOAL_LOCATION:
770 returnCode = BuildRoute( m_vecMoveGoal, bits_MF_TO_LOCATION, NULL );
771 break;
772
773 case MOVEGOAL_TARGETENT:
774 if (m_hTargetEnt != NULL)
775 {
776 returnCode = BuildRoute( m_hTargetEnt->pev->origin, bits_MF_TO_TARGETENT, m_hTargetEnt );
777 }
778 break;
779
780 case MOVEGOAL_NODE:
781 returnCode = FGetNodeRoute( m_vecMoveGoal );
782// if ( returnCode )
783// RouteSimplify( NULL );
784 break;
785 }
786
787 return returnCode;
788}

1846BOOL CBaseMonster::FTriangulate(const Vector &vecStart, const Vector &vecEnd, float flDist, CBaseEntity *pTargetEnt, Vector *pApex)
1847{
1848 Vector vecDir;
1849 Vector vecForward;
1850 Vector vecLeft; // The spot we'll try to triangulate to on the left
1851 Vector vecRight; // The spot we'll try to triangulate to on the right
1852 Vector vecTop; // The spot we'll try to triangulate to on the top
1853 Vector vecBottom; // The spot we'll try to triangulate to on the bottom
1854 Vector vecFarSide; // The spot that we'll move to after hitting the triangulated point
1855 int i;
1856 float sizeX, sizeZ;
1857
1858 // Ensure hull width is at least 24
1859 sizeX = pev->size.x;
1860 if (sizeX < 24.0)
1861 sizeX = 24.0;
1862 else if (sizeX > 48.0)
1863 sizeX = 48.0;
1864 sizeZ = pev->size.z;
1865
1866 vecForward = (vecEnd - vecStart).Normalize(); // Normalize the direction to the goal
1867
1868 Vector vecDirUp(0, 0, 1);
1869 vecDir = CrossProduct(vecForward, vecDirUp); // Calculate the left/right direction
1870
1871 // Start checking right about where the object is, picking two equidistant starting points
1872 vecLeft = pev->origin + (vecForward * (flDist + sizeX)) - vecDir * (sizeX * 3);
1873 vecRight = pev->origin + (vecForward * (flDist + sizeX)) + vecDir * (sizeX * 3);
1874 if (pev->movetype == MOVETYPE_FLY)
1875 {
1876 vecTop = pev->origin + (vecForward * flDist) + (vecDirUp * sizeZ * 3);
1877 vecBottom = pev->origin + (vecForward * flDist) - (vecDirUp * sizeZ * 3);
1878 }
1879
1880 vecFarSide = m_Route[m_iRouteIndex].vecLocation; // Store the far side location
1881
1882 vecDir = vecDir * sizeX * 2; // Scale the direction for triangulation
1883 if (pev->movetype == MOVETYPE_FLY)
1884 vecDirUp = vecDirUp * sizeZ * 2; // Scale the upward direction for flying
1885
1886 for (i = 0; i < 8; i++)
1887 {
1888 if (CheckLocalMove(pev->origin, vecRight, pTargetEnt, NULL) == LOCALMOVE_VALID)
1889 {
1890 if (CheckLocalMove(vecRight, vecFarSide, pTargetEnt, NULL) == LOCALMOVE_VALID)
1891 {
1892 if (pApex)
1893 {
1894 *pApex = vecRight; // Set the apex point
1895 }
1896 return TRUE; // Valid path found
1897 }
1898 }
1899 if (CheckLocalMove(pev->origin, vecLeft, pTargetEnt, NULL) == LOCALMOVE_VALID)
1900 {
1901 if (CheckLocalMove(vecLeft, vecFarSide, pTargetEnt, NULL) == LOCALMOVE_VALID)
1902 {
1903 if (pApex)
1904 {
1905 *pApex = vecLeft; // Set the apex point
1906 }
1907 return TRUE; // Valid path found
1908 }
1909 }
1910
1911 if (pev->movetype == MOVETYPE_FLY)
1912 {
1913 if (CheckLocalMove(pev->origin, vecTop, pTargetEnt, NULL) == LOCALMOVE_VALID)
1914 {
1915 if (CheckLocalMove(vecTop, vecFarSide, pTargetEnt, NULL) == LOCALMOVE_VALID)
1916 {
1917 if (pApex)
1918 {
1919 *pApex = vecTop; // Set the apex point
1920 }
1921 return TRUE; // Valid path found
1922 }
1923 }
1924 if (CheckLocalMove(pev->origin, vecBottom, pTargetEnt, NULL) == LOCALMOVE_VALID)
1925 {
1926 if (CheckLocalMove(vecBottom, vecFarSide, pTargetEnt, NULL) == LOCALMOVE_VALID)
1927 {
1928 if (pApex)
1929 {
1930 *pApex = vecBottom; // Set the apex point
1931 }
1932 return TRUE; // Valid path found
1933 }
1934 }
1935 }
1936
1937 // Adjust the left/right/top/bottom points for the next iteration
1938 vecRight = vecRight + vecDir;
1939 vecLeft = vecLeft - vecDir;
1940 if (pev->movetype == MOVETYPE_FLY)
1941 {
1942 vecTop = vecTop + vecDirUp;
1943 vecBottom = vecBottom - vecDirUp;
1944 }
1945 }
1946
1947 return FALSE; // No valid path found
1948}

1963void CBaseMonster::Move(float flInterval)
1964{

1968 Vector vecDir;
1969 Vector vecApex;
1970 CBaseEntity *pTargetEnt;
1971
1972 // Don't move if no valid route
1973 if (FRouteClear())
1974 {
1975 // Refresh the route if there's no movement goal
1976 if (m_movementGoal == MOVEGOAL_NONE || !FRefreshRoute())
1977 {
1978 ALERT(at_aiconsole, "Tried to move with no route!\n");
1979 TaskFail();
1980 return;
1981 }
1982 }
1983
1984 if (m_flMoveWaitFinished > gpGlobals->time)
1985 return; // Wait if the move is still in progress
1986
1987 // Set the target entity for movement checks
1988 pTargetEnt = NULL;
1989
1990 // Calculate direction and distance to the waypoint
1991 vecDir = (m_Route[m_iRouteIndex].vecLocation - pev->origin).Normalize();
1992 flWaypointDist = (m_Route[m_iRouteIndex].vecLocation - pev->origin).Length2D();
1993
1994 MakeIdealYaw(m_Route[m_iRouteIndex].vecLocation); // Set the ideal yaw towards the waypoint
1995 ChangeYaw(pev->yaw_speed); // Change the yaw speed
1996
1997 // Determine the distance to check for movement
1998 flCheckDist = (flWaypointDist < DIST_TO_CHECK) ? flWaypointDist : DIST_TO_CHECK;
1999
2000 // Set the target entity based on the route type
2001 if ((m_Route[m_iRouteIndex].iType & (~bits_MF_NOT_TO_MASK)) == bits_MF_TO_ENEMY)
2002 {
2003 pTargetEnt = m_hEnemy; // Set target entity to enemy
2004 }
2005 else if ((m_Route[m_iRouteIndex].iType & ~bits_MF_NOT_TO_MASK) == bits_MF_TO_TARGETENT)
2006 {
2007 pTargetEnt = m_hTargetEnt; // Set target entity to target
2008 }
2009
2010 // Check for local movement
2011 flDist = 0;
2012 if (CheckLocalMove(pev->origin, pev->origin + vecDir * flCheckDist, pTargetEnt, &flDist) != LOCALMOVE_VALID)
2013 {
2014 CBaseEntity *pBlocker;
2015
2016 // Can't move, stop
2017 Stop();
2018 pBlocker = CBaseEntity::Instance(gpGlobals->trace_ent); // Get the blocking entity
2019 if (pBlocker)
2020 {
2021 DispatchBlocked(edict(), pBlocker->edict()); // Notify the monster is blocked
2022 }
2023
2024 // Handle waiting for a moving blocker
2025 if (pBlocker && m_moveWaitTime > 0 && pBlocker->IsMoving() && !pBlocker->IsPlayer() && (gpGlobals->time - m_flMoveWaitFinished) > 3.0)
2026 {
2027 if (flDist < m_flGroundSpeed)
2028 {
2029 // Wait for a second if there's not enough room to move
2030 m_flMoveWaitFinished = gpGlobals->time + m_moveWaitTime;
2031 return;
2032 }
2033 }
2034 else
2035 {
2036 // Try to triangulate around the obstacle
2037 if (FTriangulate(pev->origin, m_Route[m_iRouteIndex].vecLocation, flDist, pTargetEnt, &vecApex))
2038 {
2039 InsertWaypoint(vecApex, bits_MF_TO_DETOUR); // Insert a detour waypoint
2040 RouteSimplify(pTargetEnt); // Simplify the route
2041 }
2042 else
2043 {
2044 Stop(); // Stop movement
2045 if (m_moveWaitTime > 0 && !(m_afMemory & bits_MEMORY_MOVE_FAILED))
2046 {
2047 FRefreshRoute(); // Refresh the route
2048 if (FRouteClear())
2049 {
2050 TaskFail(); // Fail the task if the route is clear
2051 }
2052 else
2053 {
2054 // Don't get stuck
2055 if ((gpGlobals->time - m_flMoveWaitFinished) < 0.2)
2056 Remember(bits_MEMORY_MOVE_FAILED); // Remember the failure
2057
2058 m_flMoveWaitFinished = gpGlobals->time + 0.1; // Set wait time
2059 }
2060 }
2061 else
2062 {
2063 TaskFail(); // Fail the task
2064 ALERT(at_aiconsole, "%s Failed to move (%d)!\n", STRING(pev->classname), HasMemory(bits_MEMORY_MOVE_FAILED));
2065 }
2066 return;
2067 }
2068 }
2069 }
2070
2071 // Advance to the next target if close enough
2072 if (ShouldAdvanceRoute(flWaypointDist))
2073 {
2074 AdvanceRoute(flWaypointDist); // Advance the route index
2075 }
2076
2077 // Check if waiting for a door
2078 if (m_flMoveWaitFinished > gpGlobals->time)
2079 {
2080 Stop(); // Stop movement
2081 return;
2082 }
2083
2084 // Adjust the movement interval based on the distance
2085 if (flCheckDist < m_flGroundSpeed * flInterval)
2086 {
2087 flInterval = flCheckDist / m_flGroundSpeed; // Adjust the interval
2088 }
2089 MoveExecute(pTargetEnt, vecDir, flInterval); // Execute the movement
2090
2091 if (MovementIsComplete())
2092 {
2093 Stop(); // Stop movement if complete
2094 RouteClear(); // Clear the route
2095 }
2096}
