358BOOL CFuncTank :: OnControls( entvars_t *pevTest )
359{
360 // Only allow control if the player is within the tank's control
361 // radius.
362 if ( !(pev->spawnflags & SF_TANK_CANCONTROL) )
363 return FALSE;
364
365 // Calculate the distance between the player and the tank.
366 Vector offset = pevTest->origin - pev->origin;
367
368 // If the player is within the control radius, allow control.
369 if ( (m_vecControllerUsePos - pevTest->origin).Length() < 30 )
370 return TRUE;
371
372 return FALSE;
373}

444// Called each frame by the player's ItemPostFrame
452void CFuncTank :: ControllerPostFrame( void )
453{
454 ASSERT(m_pController != NULL);
455
456 if ( gpGlobals->time < m_flNextAttack )
457 return;
458
459 // Check if the player is attempting to fire the tank
460 if ( m_pController->pev->button & IN_ATTACK )
461 {
462 // Calculate the forward vector of the tank
463 Vector vecForward;
464 UTIL_MakeVectorsPrivate( pev->angles, vecForward, NULL, NULL );
465
466 // Make sure we fire the tank at the right rate
467 m_fireLast = gpGlobals->time - (1/m_fireRate) - 0.01; // to make sure the gun doesn't fire too many bullets
468
469 // Fire the tank
470 Fire( BarrelPosition(), vecForward, m_pController->pev );
471
472 // HACKHACK -- make some noise (that the AI can hear)
473 if ( m_pController && m_pController->IsPlayer() )
474 ((CBasePlayer *)m_pController)->m_iWeaponVolume = LOUD_GUN_VOLUME;
475
476 // Set the time for the next attack
477 m_flNextAttack = gpGlobals->time + (1/m_fireRate);
478 }
479}

504void CFuncTank :: Use( CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value )
505{
506 if ( pev->spawnflags & SF_TANK_CANCONTROL )
507 { // player controlled turret
508
509 if ( pActivator->Classify() != CLASS_PLAYER )
510 return;
511
512 if ( value == 2 && useType == USE_SET )
513 {
514 ControllerPostFrame();
515 }
516 else if ( !m_pController && useType != USE_OFF )
517 {
518 ((CBasePlayer*)pActivator)->m_pTank = this;
519 StartControl( (CBasePlayer*)pActivator );
520 }
521 else
522 {
523 StopControl();
524 }
525 }
526 else
527 {
528 if ( !ShouldToggle( useType, IsActive() ) )
529 return;
530
531 if ( IsActive() )
532 TankDeactivate();
533 else
534 TankActivate();
535 }
536}

610void CFuncTank::TrackTarget( void )
611{
612 TraceResult tr;
613 edict_t *pPlayer = FIND_CLIENT_IN_PVS( edict() );
614 BOOL updateTime = FALSE, lineOfSight;
615 Vector angles, direction, targetPosition, barrelEnd;
616 edict_t *pTarget;
617
618 // Get a position to aim for
619 if (m_pController)
620 {
621 // Tanks attempt to mirror the player's angles
622 angles = m_pController->pev->v_angle;
623 angles[0] = 0 - angles[0];
624 pev->nextthink = pev->ltime + 0.05;
625 }
626 else
627 {
628 if ( IsActive() )
629 pev->nextthink = pev->ltime + 0.1;
630 else
631 return;
632
633 if ( FNullEnt( pPlayer ) )
634 {
635 if ( IsActive() )
636 pev->nextthink = pev->ltime + 2; // Wait 2 secs
637 return;
638 }
639 pTarget = FindTarget( pPlayer );
640 if ( !pTarget )
641 return;
642
643 // Calculate angle needed to aim at target
644 barrelEnd = BarrelPosition();
645 targetPosition = pTarget->v.origin + pTarget->v.view_ofs;
646 float range = (targetPosition - barrelEnd).Length();
647
648 if ( !InRange( range ) )
649 return;
650
651 UTIL_TraceLine( barrelEnd, targetPosition, dont_ignore_monsters, edict(), &tr );
652
653 lineOfSight = FALSE;
654 // No line of sight, don't track
655 if ( tr.flFraction == 1.0 || tr.pHit == pTarget )
656 {
657 lineOfSight = TRUE;
658
659 CBaseEntity *pInstance = CBaseEntity::Instance(pTarget);
660 if ( InRange( range ) && pInstance && pInstance->IsAlive() )
661 {
662 updateTime = TRUE;
663 m_sightOrigin = UpdateTargetPosition( pInstance );
664 }
665 }
666
667 // Track sight origin
668
669// !!! I'm not sure what i changed
670 direction = m_sightOrigin - pev->origin;
671// direction = m_sightOrigin - barrelEnd;
672 angles = UTIL_VecToAngles( direction );
673
674 // Calculate the additional rotation to point the end of the barrel at the target (not the gun's center)
675 AdjustAnglesForBarrel( angles, direction.Length() );
676 }
677
678 angles.x = -angles.x;
679
680 // Force the angles to be relative to the center position
681 angles.y = m_yawCenter + UTIL_AngleDistance( angles.y, m_yawCenter );
682 angles.x = m_pitchCenter + UTIL_AngleDistance( angles.x, m_pitchCenter );
683
684 // Limit against range in y
685 if ( angles.y > m_yawCenter + m_yawRange )
686 {
687 angles.y = m_yawCenter + m_yawRange;
688 updateTime = FALSE; // Don't update if you saw the player, but out of range
689 }
690 else if ( angles.y < (m_yawCenter - m_yawRange) )
691 {
692 angles.y = (m_yawCenter - m_yawRange);
693 updateTime = FALSE; // Don't update if you saw the player, but out of range
694 }
695
696 if ( updateTime )
697 m_lastSightTime = gpGlobals->time;
698
699 // Move toward target at rate or less
700 float distY = UTIL_AngleDistance( angles.y, pev->angles.y );
701 pev->avelocity.y = distY * 10;
702 if ( pev->avelocity.y > m_yawRate )
703 pev->avelocity.y = m_yawRate;
704 else if ( pev->avelocity.y < -m_yawRate )
705 pev->avelocity.y = -m_yawRate;
706
707 // Limit against range in x
708 if ( angles.x > m_pitchCenter + m_pitchRange )
709 angles.x = m_pitchCenter + m_pitchRange;
710 else if ( angles.x < m_pitchCenter - m_pitchRange )
711 angles.x = m_pitchCenter - m_pitchRange;
712
713 // Move toward target at rate or less
714 float distX = UTIL_AngleDistance( angles.x, pev->angles.x );
715 pev->avelocity.x = distX * 10;
716
717 if ( pev->avelocity.x > m_pitchRate )
718 pev->avelocity.x = m_pitchRate;
719 else if ( pev->avelocity.x < -m_pitchRate )
720 pev->avelocity.x = -m_pitchRate;
721
722 if ( m_pController )
723 return;
724
725 if ( CanFire() && ( (fabs(distX) < m_pitchTolerance && fabs(distY) < m_yawTolerance) || (pev->spawnflags & SF_TANK_LINEOFSIGHT) ) )
726 {
727 BOOL fire = FALSE;
728 Vector forward;
729 UTIL_MakeVectorsPrivate( pev->angles, forward, NULL, NULL );
730
731 if ( pev->spawnflags & SF_TANK_LINEOFSIGHT )
732 {
733 float length = direction.Length();
734 UTIL_TraceLine( barrelEnd, barrelEnd + forward * length, dont_ignore_monsters, edict(), &tr );
735 if ( tr.pHit == pTarget )
736 fire = TRUE;
737 }
738 else
739 fire = TRUE;
740
741 if ( fire )
742 {
743 Fire( BarrelPosition(), forward, pev );
744 }
745 else
746 m_fireLast = 0;
747 }
748 else
749 m_fireLast = 0;
750}

void CFuncTank::AdjustAnglesForBarrel( Vector &angles, float distance )
761{
762 float r2, d2;
763
764
765 if ( m_barrelPos.y != 0 || m_barrelPos.z != 0 )
766 {
767 distance -= m_barrelPos.z;
768 d2 = distance * distance;
769 if ( m_barrelPos.y )
770 {
771 r2 = m_barrelPos.y * m_barrelPos.y;
772 angles.y += (180.0 / M_PI) * atan2( m_barrelPos.y, sqrt( d2 - r2 ) );
773 }
774 if ( m_barrelPos.z )
775 {
776 r2 = m_barrelPos.z * m_barrelPos.z;
777 angles.x += (180.0 / M_PI) * atan2( -m_barrelPos.z, sqrt( d2 - r2 ) );
778 }
779 }
780}

793void CFuncTank::Fire( const Vector &barrelEnd, const Vector &forward, entvars_t *pevAttacker )
794{
795 // If the tank has fired before, animate the smoke and flash sprites
796 // and use the targets
797 if ( m_fireLast != 0 )
798 {
799 if ( m_iszSpriteSmoke )
800 {
801 // Create a sprite for the smoke effect
802 CSprite *pSprite = CSprite::SpriteCreate( STRING(m_iszSpriteSmoke), barrelEnd, TRUE );
803 pSprite->AnimateAndDie( RANDOM_FLOAT( 15.0, 20.0 ) );
804 pSprite->SetTransparency( kRenderTransAlpha, pev->rendercolor.x, pev->rendercolor.y, pev->rendercolor.z, 255, kRenderFxNone );
805 pSprite->pev->velocity.z = RANDOM_FLOAT(40, 80);
806 pSprite->SetScale( m_spriteScale );
807 }
808 if ( m_iszSpriteFlash )
809 {
810 // Create a sprite for the flash effect
811 CSprite *pSprite = CSprite::SpriteCreate( STRING(m_iszSpriteFlash), barrelEnd, TRUE );
812 pSprite->AnimateAndDie( 60 );
813 pSprite->SetTransparency( kRenderTransAdd, 255, 255, 255, 255, kRenderFxNoDissipation );
814 pSprite->SetScale( m_spriteScale );
815
816 // Hack Hack, make it stick around for at least 100 ms.
817 pSprite->pev->nextthink += 0.1;
818 }
819 SUB_UseTargets( this, USE_TOGGLE, 0 );
820 }

839void CFuncTank::TankTrace( const Vector &vecStart, const Vector &vecForward, const Vector &vecSpread, TraceResult &tr )
840{
841 // get circular gaussian spread
842 float x, y, z;
843 do {
844 x = RANDOM_FLOAT(-0.5,0.5) + RANDOM_FLOAT(-0.5,0.5);
845 y = RANDOM_FLOAT(-0.5,0.5) + RANDOM_FLOAT(-0.5,0.5);
846 z = x*x+y*y;
847 } while (z > 1);
848 Vector vecDir = vecForward +
849 x * vecSpread.x * gpGlobals->v_right +
850 y * vecSpread.y * gpGlobals->v_up;
851 Vector vecEnd;
852
853 vecEnd = vecStart + vecDir * 4096;
854 UTIL_TraceLine( vecStart, vecEnd, dont_ignore_monsters, edict(), &tr );
855}
