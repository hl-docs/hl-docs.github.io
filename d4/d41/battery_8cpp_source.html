123int CHudBattery::Draw(float flTime)
124{
125 if (gHUD.m_iHideHUDDisplay & HIDEHUD_HEALTH)
126 return 1;
127
128 int r, g, b, x, y, a;
129 wrect_t rc = *m_prc2;
130
131#if defined(_TFC)
132 float fScale = (m_iBatMax > 0) ? 1.0f / m_iBatMax : 0.0f;
133 rc.top += m_iHeight * ((float)(m_iBatMax - min(m_iBatMax, m_iBat)) * fScale);
134#else
135 rc.top += m_iHeight * ((float)(100 - min(100, m_iBat)) * 0.01f);
136#endif
137
138 UnpackRGB(r, g, b, RGB_YELLOWISH);
139
140 if (!(gHUD.m_iWeaponBits & (1 << WEAPON_SUIT)))
141 return 1;
142
143 // Handle fading effect
144 if (m_fFade)
145 {
146 if (m_fFade > FADE_TIME)
147 m_fFade = FADE_TIME;
148
149 m_fFade -= (gHUD.m_flTimeDelta * 20);
150 a = (m_fFade <= 0) ? 128 : MIN_ALPHA + (m_fFade / FADE_TIME) * 128;
151
152 if (m_fFade <= 0)
153 m_fFade = 0;
154 }
155 else
156 a = MIN_ALPHA;
157
158 ScaleColors(r, g, b, a);
159
160 int iOffset = (m_prc1->bottom - m_prc1->top) / 6;
161 y = ScreenHeight - gHUD.m_iFontHeight - gHUD.m_iFontHeight / 2;
162 x = ScreenWidth / 5;
163
164 // Load sprite handles if not already loaded
165 if (!m_hSprite1)
166 m_hSprite1 = gHUD.GetSprite(gHUD.GetSpriteIndex("suit_empty"));
167 if (!m_hSprite2)
168 m_hSprite2 = gHUD.GetSprite(gHUD.GetSpriteIndex("suit_full"));
169
170 // Draw empty battery background
171 SPR_Set(m_hSprite1, r, g, b);
172 SPR_DrawAdditive(0, x, y - iOffset, m_prc1);
173
174 // Draw battery fill
175 if (rc.bottom > rc.top)
176 {
177 SPR_Set(m_hSprite2, r, g, b);
178 SPR_DrawAdditive(0, x, y - iOffset + (rc.top - m_prc2->top), &rc);
179 }
180
181 // Draw battery numeric value
182 x += (m_prc1->right - m_prc1->left);
183 x = gHUD.DrawHudNumber(x, y, DHN_3DIGITS | DHN_DRAWZERO, m_iBat, r, g, b);
184
185 return 1;
186}
