155int DispatchSpawn( edict_t *pent )
156{
157 CBaseEntity *pEntity = (CBaseEntity *)GET_PRIVATE(pent);
158
159 if (pEntity)
160 {
161 // Initialize the entity's bounding box
162 pEntity->pev->absmin = pEntity->pev->origin - Vector(1,1,1);
163 pEntity->pev->absmax = pEntity->pev->origin + Vector(1,1,1);
164
165 pEntity->Spawn();
166
167 // Re-fetch the entity pointer in case it was deleted during spawn
168 pEntity = (CBaseEntity *)GET_PRIVATE(pent);
169
170 if ( pEntity )
171 {
172 if ( g_pGameRules && !g_pGameRules->IsAllowedToSpawn( pEntity ) )
173 return -1; // Entity should be deleted
174 if ( pEntity->pev->flags & FL_KILLME )
175 return -1;
176 }
177
178 // Handle global entity state
179 if ( pEntity && pEntity->pev->globalname )
180 {
181 const globalentity_t *pGlobal = gGlobalState.EntityFromTable( pEntity->pev->globalname );
182 if ( pGlobal )
183 {
184 if ( pGlobal->state == GLOBAL_DEAD )
185 return -1; // Entity is dead, delete it
186 else if ( !FStrEq( STRING(gpGlobals->mapname), pGlobal->levelName ) )
187 pEntity->MakeDormant(); // Entity hasn't moved to this level yet
188 }
189 else
190 {
191 // Spawned entities default to 'On'
192 gGlobalState.EntityAdd( pEntity->pev->globalname, gpGlobals->mapname, GLOBAL_ON );
193 }
194 }
195 }
196
197 return 0;
198}

310void DispatchSave( edict_t *pent, SAVERESTOREDATA *pSaveData )
311{
312 CBaseEntity *pEntity = (CBaseEntity *)GET_PRIVATE(pent);
313
314 if ( pEntity && pSaveData )
315 {
316 ENTITYTABLE *pTable = &pSaveData->pTable[ pSaveData->currentIndex ];
317
318 if ( pTable->pent != pent )
319 ALERT( at_error, "ENTITY TABLE OR INDEX IS WRONG!!!!\n" );
320
321 if ( pEntity->ObjectCaps() & FCAP_DONT_SAVE )
322 return;
323
324 // Adjust nextthink time for push entities
325 if ( pEntity->pev->movetype == MOVETYPE_PUSH )
326 {
327 float delta = pEntity->pev->nextthink - pEntity->pev->ltime;
328 pEntity->pev->ltime = gpGlobals->time;
329 pEntity->pev->nextthink = pEntity->pev->ltime + delta;
330 }
331
332 pTable->location = pSaveData->size; // Remember entity position for file I/O
333 pTable->classname = pEntity->pev->classname; // Remember entity class for respawn
334
335 CSave saveHelper( pSaveData );
336 pEntity->Save( saveHelper );
337
338 pTable->size = pSaveData->size - pTable->location; // Size of entity block is data size written to block
339 }

377int DispatchRestore( edict_t *pent, SAVERESTOREDATA *pSaveData, int globalEntity )
378{
379 CBaseEntity *pEntity = (CBaseEntity *)GET_PRIVATE(pent);
380
381 if ( pEntity && pSaveData )
382 {
383 entvars_t tmpVars;
384 Vector oldOffset;
385
386 CRestore restoreHelper( pSaveData );
387 if ( globalEntity )
388 {
389 CRestore tmpRestore( pSaveData );
390 tmpRestore.PrecacheMode( 0 );
391 tmpRestore.ReadEntVars( "ENTVARS", &tmpVars );
392
393 // Reset the save pointers for restoring
394 pSaveData->size = pSaveData->pTable[pSaveData->currentIndex].location;
395 pSaveData->pCurrentData = pSaveData->pBaseData + pSaveData->size;
396
397 const globalentity_t *pGlobal = gGlobalState.EntityFromTable( tmpVars.globalname );
398
399 // Check if the global entity is from the correct level
400 if ( !FStrEq( pSaveData->szCurrentMapName, pGlobal->levelName ) )
401 return 0;
402
403 // Compute the new global offset
404 oldOffset = pSaveData->vecLandmarkOffset;
405 CBaseEntity *pNewEntity = FindGlobalEntity( tmpVars.classname, tmpVars.globalname );
406 if ( pNewEntity )
407 {
408 restoreHelper.SetGlobalMode( 1 );
409 pSaveData->vecLandmarkOffset = (pSaveData->vecLandmarkOffset - pNewEntity->pev->mins) + tmpVars.mins;
410 pEntity = pNewEntity;
411 pent = ENT( pEntity->pev );
412 gGlobalState.EntityUpdate( pEntity->pev->globalname, gpGlobals->mapname );
413 }
414 else
415 {
416 return 0;
417 }
418 }
419
420 if ( pEntity->ObjectCaps() & FCAP_MUST_SPAWN )
421 {
422 pEntity->Restore( restoreHelper );
423 pEntity->Spawn();
424 }
425 else
426 {
427 pEntity->Restore( restoreHelper );
428 pEntity->Precache( );
429 }
430
431 // Re-fetch the entity pointer
432 pEntity = (CBaseEntity *)GET_PRIVATE(pent);
433
434 if ( globalEntity )
435 {
436 pSaveData->vecLandmarkOffset = oldOffset;
437 if ( pEntity )
438 {
439 UTIL_SetOrigin( pEntity->pev, pEntity->pev->origin );
440 pEntity->OverrideReset();
441 }
442 }
443 else if ( pEntity && pEntity->pev->globalname )
444 {
445 const globalentity_t *pGlobal = gGlobalState.EntityFromTable( pEntity->pev->globalname );
446 if ( pGlobal )
447 {
448 if ( pGlobal->state == GLOBAL_DEAD )
449 return -1;
450 else if ( !FStrEq( STRING(gpGlobals->mapname), pGlobal->levelName ) )
451 {
452 pEntity->MakeDormant();
453 }
454 }
455 else
456 {
457 ALERT( at_error, "Global Entity %s (%s) not in table!!!\n", STRING(pEntity->pev->globalname), STRING(pEntity->pev->classname) );
458 gGlobalState.EntityAdd( pEntity->pev->globalname, gpGlobals->mapname, GLOBAL_ON );
459 }
460 }
461 }
462 return 0;
463}

649int CBaseEntity :: TakeDamage( entvars_t* pevInflictor, entvars_t* pevAttacker, float flDamage, int bitsDamageType )
650{
651 Vector vecTemp;
652
653 if (!pev->takedamage)
654 return 0;
655
656 // UNDONE: some entity types may be immune or resistant to some bitsDamageType
657
658 // if Attacker == Inflictor, the attack was a melee or other instant-hit attack.
659 // (that is, no actual entity projectile was involved in the attack so use the shooter's origin).
660 if ( pevAttacker == pevInflictor )
661 {
662 vecTemp = pevInflictor->origin - ( VecBModelOrigin(pev) );
663 }
664 else
665 // an actual missile was involved.
666 {
667 vecTemp = pevInflictor->origin - ( VecBModelOrigin(pev) );
668 }
669
670// this global is still used for glass and other non-monster killables, along with decals.
671 g_vecAttackDir = vecTemp.Normalize();
672
673// save damage based on the target's armor level
674
675// figure momentum add (don't let hurt brushes or other triggers move player)
676 if ((!FNullEnt(pevInflictor)) && (pev->movetype == MOVETYPE_WALK || pev->movetype == MOVETYPE_STEP) && (pevAttacker->solid != SOLID_TRIGGER) )
677 {
678 Vector vecDir = pev->origin - (pevInflictor->absmin + pevInflictor->absmax) * 0.5;
679 vecDir = vecDir.Normalize();
680
681 float flForce = flDamage * ((32 * 32 * 72.0) / (pev->size.x * pev->size.y * pev->size.z)) * 5;
682
683 if (flForce > 1000.0)
684 flForce = 1000.0;
685 pev->velocity = pev->velocity + vecDir * flForce;
686 }
687
688// do the damage
689 pev->health -= flDamage;
690 if (pev->health <= 0)
691 {
692 Killed( pevAttacker, GIB_NORMAL );
693 return 0;
694 }
695
696 return 1;
697}

797void SetObjectCollisionBox( entvars_t *pev )
798{
799 if ( (pev->solid == SOLID_BSP) &&
800 (pev->angles.x || pev->angles.y|| pev->angles.z) )
801 { // expand for rotation
802 float max, v;
803 int i;
804
805 max = 0;
806 for (i=0 ; i<3 ; i++)
807 {
808 v = fabs( ((float *)pev->mins)[i]);
809 if (v > max)
810 max = v;
811 v = fabs( ((float *)pev->maxs)[i]);
812 if (v > max)
813 max = v;
814 }
815 for (i=0 ; i<3 ; i++)
816 {
817 ((float *)pev->absmin)[i] = ((float *)pev->origin)[i] - max;
818 ((float *)pev->absmax)[i] = ((float *)pev->origin)[i] + max;
819 }
820 }
821 else
822 {
823 pev->absmin = pev->origin + pev->mins;
824 pev->absmax = pev->origin + pev->maxs;
825 }
826
827 pev->absmin.x -= 1;
828 pev->absmin.y -= 1;
829 pev->absmin.z -= 1;
830 pev->absmax.x += 1;
831 pev->absmax.y += 1;
832 pev->absmax.z += 1;
833}
