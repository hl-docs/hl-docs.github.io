8// This header defines the interface convention used in the valve engine.
9// To make an interface and expose it:
10// 1. Derive from IBaseInterface.
11// 2. The interface must be ALL pure virtuals, and have no data members.
12// 3. Define a name for it.
13// 4. In its implementation file, use EXPOSE_INTERFACE or EXPOSE_SINGLE_INTERFACE.
14
15// Versioning
16// There are two versioning cases that are handled by this:
17// 1. You add functions to the end of an interface, so it is binary compatible with the previous interface. In this case,
18// you need two EXPOSE_INTERFACEs: one to expose your class as the old interface and one to expose it as the new interface.
19// 2. You update an interface so it's not compatible anymore (but you still want to be able to expose the old interface
20// for legacy code). In this case, you need to make a new version name for your new interface, and make a wrapper interface and
21// expose it for the old interface.
22

60// Use this to expose an interface that can have multiple instances.
61// e.g.:
62// EXPOSE_INTERFACE( CInterfaceImp, IInterface, "MyInterface001" )
63// This will expose a class called CInterfaceImp that implements IInterface (a pure class)
64// clients can receive a pointer to this class by calling CreateInterface( "MyInterface001" )
65//
66// In practice, the shared header file defines the interface (IInterface) and version name ("MyInterface001")
67// so that each component can use these names/vtables to communicate
68//
69// A single class can support multiple interfaces through multiple inheritance
70//
71#define EXPOSE_INTERFACE_FN(functionName, interfaceName, versionName) \
72 static InterfaceReg __g_Create##className##_reg(functionName, versionName);
73
74#define EXPOSE_INTERFACE(className, interfaceName, versionName) \
75 static IBaseInterface* __Create##className##_interface() {return (interfaceName *)new className;}\
76 static InterfaceReg __g_Create##className##_reg(__Create##className##_interface, versionName );
77
78// Use this to expose a singleton interface with a global variable you've created.
79#define EXPOSE_SINGLE_INTERFACE_GLOBALVAR(className, interfaceName, versionName, globalVarName) \
80 static IBaseInterface* __Create##className##interfaceName##_interface() {return (interfaceName *)&globalVarName;}\
81 static InterfaceReg __g_Create##className##interfaceName##_reg(__Create##className##interfaceName##_interface, versionName);
82
83// Use this to expose a singleton interface. This creates the global variable for you automatically.
84#define EXPOSE_SINGLE_INTERFACE(className, interfaceName, versionName) \
85 static className __g_##className##_singleton;\
86 EXPOSE_SINGLE_INTERFACE_GLOBALVAR(className, interfaceName, versionName, __g_##className##_singleton)
