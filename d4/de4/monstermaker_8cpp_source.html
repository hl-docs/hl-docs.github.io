99void CMonsterMaker::KeyValue(KeyValueData *pkvd)
100{

102 {

116 else
117 CBaseMonster::KeyValue(pkvd); // Call the base class method for unhandled keys
118}

127void CMonsterMaker::Spawn()
128{
129 pev->solid = SOLID_NOT; // Set solid type to not solid
130
131 m_cLiveChildren = 0; // Initialize the count of live children
132 Precache(); // Preload resources
133
134 if (!FStringNull(pev->targetname)) // Check if the monster maker has a target name
135 {
136 if (pev->spawnflags & SF_MONSTERMAKER_CYCLIC)
137 {
138 SetUse(&CMonsterMaker::CyclicUse); // Set cyclic use function
139 }
140 else
141 {
142 SetUse(&CMonsterMaker::ToggleUse); // Set toggle use function
143 }
144
145 if (FBitSet(pev->spawnflags, SF_MONSTERMAKER_START_ON))
146 { // Start making monsters as soon as the monster maker spawns
147 m_fActive = TRUE;
148 SetThink(&CMonsterMaker::MakerThink); // Set the think function to create monsters
149 }
150 else
151 { // Wait to be activated
152 m_fActive = FALSE;
153 SetThink(&CMonsterMaker::SUB_DoNothing); // Do nothing until activated
154 }
155 }
156 else
157 { // No target name, just start
158 pev->nextthink = gpGlobals->time + m_flDelay; // Set the next think time
159 m_fActive = TRUE;
160 SetThink(&CMonsterMaker::MakerThink); // Set the think function to create monsters
161 }
162
163 m_fFadeChildren = (m_cNumMonsters != 1); // Determine if children should fade based on monster count
164 m_flGround = 0; // Initialize ground level
165}

187void CMonsterMaker::MakeMonster(void)
188{
189 edict_t *pent; // Pointer to the new monster entity
190 entvars_t *pevCreate; // Pointer to the new monster's entity variables
191
192 if (m_iMaxLiveChildren > 0 && m_cLiveChildren >= m_iMaxLiveChildren)
193 { // Check if the maximum number of live children has been reached
194 return; // Exit if too many live monsters
195 }
196
197 if (!m_flGround)
198 { // Set altitude if not already set
199 TraceResult tr; // Trace result for ground detection
200 UTIL_TraceLine(pev->origin, pev->origin - Vector(0, 0, 2048), ignore_monsters, ENT(pev), &tr); // Trace downwards
201 m_flGround = tr.vecEndPos.z; // Set ground level
202 }
203
204 Vector mins = pev->origin - Vector(34, 34, 0); // Calculate minimum bounds for spawn area
205 Vector maxs = pev->origin + Vector(34, 34, 0); // Calculate maximum bounds for spawn area
206 maxs.z = pev->origin.z; // Set maximum Z to current height
207 mins.z = m_flGround; // Set minimum Z to ground level
208
209 CBaseEntity *pList[2]; // Array to hold entities in the spawn area
210 int count = UTIL_EntitiesInBox(pList, 2, mins, maxs, FL_CLIENT | FL_MONSTER); // Check for entities in the area
211 if (count)
212 { // Don't build a stack of monsters
213 return; // Exit if there are entities in the way
214 }
215
216 pent = CREATE_NAMED_ENTITY(m_iszMonsterClassname); // Create the new monster entity
217
218 if (FNullEnt(pent))
219 {
220 ALERT(at_console, "NULL Ent in MonsterMaker!\n"); // Log an error if the entity is null
221 return; // Exit if the entity creation failed
222 }
223
224 // If I have a target, fire!
225 if (!FStringNull(pev->target))
226 {
227 // Delay already overloaded for this entity, so can't call SUB_UseTargets()
228 FireTargets(STRING(pev->target), this, this, USE_TOGGLE, 0); // Fire the target if set
229 }
230
231 pevCreate = VARS(pent); // Get the entity variables for the new monster
232 pevCreate->origin = pev->origin; // Set the origin to the monster maker's origin
233 pevCreate->angles = pev->angles; // Set the angles to the monster maker's angles
234 SetBits(pevCreate->spawnflags, SF_MONSTER_FALL_TO_GROUND); // Set the spawn flag to fall to ground
235
236 // Children hit monsterclip brushes
237 if (pev->spawnflags & SF_MONSTERMAKER_MONSTERCLIP)
238 SetBits(pevCreate->spawnflags, SF_MONSTER_HITMONSTERCLIP); // Set the monsterclip flag
239
240 DispatchSpawn(ENT(pevCreate)); // Spawn the new monster
241 pevCreate->owner = edict(); // Set the owner of the new monster
242
243 if (!FStringNull(pev->netname))
244 { // If I have a netname (overloaded), give the child monster that name as a targetname
245 pevCreate->targetname = pev->netname; // Set the target name for the new monster
246 }
247
248 m_cLiveChildren++; // Increment the count of live children
249 m_cNumMonsters--; // Decrement the count of monsters left to create
250
251 if (m_cNumMonsters == 0)
252 { // Disable this forever if no more monsters to create
253 SetThink(NULL); // Clear the think function
254 SetUse(NULL); // Clear the use function
255 }
256}

286void CMonsterMaker::ToggleUse(CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
287{
288 if (!ShouldToggle(useType, m_fActive))
289 return; // Exit if the toggle condition is not met
290
291 if (m_fActive)
292 {
293 m_fActive = FALSE; // Set to inactive
294 SetThink(NULL); // Clear the think function
295 }
296 else
297 {
298 m_fActive = TRUE; // Set to active
299 SetThink(&CMonsterMaker::MakerThink); // Set the think function to create monsters
300 }
301
302 pev->nextthink = gpGlobals->time; // Update the next think time
303}
