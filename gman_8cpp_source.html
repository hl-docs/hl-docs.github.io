217void CGMan::RunTask(Task_t *pTask)
218{
219 switch (pTask->iTask)
220 {
221 case TASK_WAIT:
222 // Look at who I'm talking to
223 if (m_flTalkTime > gpGlobals->time && m_hTalkTarget != NULL)
224 {
225 float yaw = VecToYaw(m_hTalkTarget->pev->origin - pev->origin) - pev->angles.y;
226
227 if (yaw > 180) yaw -= 360; // Normalize yaw
228 if (yaw < -180) yaw += 360; // Normalize yaw
229
230 // Turn towards vector
231 SetBoneController(0, yaw);
232 }
233 // Look at player, but only if playing a "safe" idle animation
234 else if (m_hPlayer != NULL && pev->sequence == 0)
235 {
236 float yaw = VecToYaw(m_hPlayer->pev->origin - pev->origin) - pev->angles.y;
237
238 if (yaw > 180) yaw -= 360; // Normalize yaw
239 if (yaw < -180) yaw += 360; // Normalize yaw
240
241 // Turn towards vector
242 SetBoneController(0, yaw);
243 }
244 else
245 {
246 SetBoneController(0, 0); // Reset bone controller
247 }
248 CBaseMonster::RunTask(pTask); // Call base class method
249 break;
250 default:
251 SetBoneController(0, 0); // Reset bone controller
252 CBaseMonster::RunTask(pTask); // Call base class method
253 break;
254 }
255}

269int CGMan::TakeDamage(entvars_t* pevInflictor, entvars_t* pevAttacker, float flDamage, int bitsDamageType)
