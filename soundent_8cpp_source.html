172void CSoundEnt :: FreeSound ( int iSound, int iPrevious )
173{
174 if ( !pSoundEnt )
175 {
176 // no sound ent!
177 return;
178 }
179
180 if ( iPrevious != SOUNDLIST_EMPTY )
181 {
182 // iSound is not the head of the active list, so
183 // must fix the index for the Previous sound
184// pSoundEnt->m_SoundPool[ iPrevious ].m_iNext = m_SoundPool[ iSound ].m_iNext;
185 pSoundEnt->m_SoundPool[ iPrevious ].m_iNext = pSoundEnt->m_SoundPool[ iSound ].m_iNext;
186 }
187 else
188 {
189 // the sound we're freeing IS the head of the active list.
190 pSoundEnt->m_iActiveSound = pSoundEnt->m_SoundPool [ iSound ].m_iNext;
191 }
192
193 // make iSound the head of the Free list.
194 pSoundEnt->m_SoundPool[ iSound ].m_iNext = pSoundEnt->m_iFreeSound;
195 pSoundEnt->m_iFreeSound = iSound;
196}

206int CSoundEnt :: IAllocSound( void )
207{
208 int iNewSound;
209
210 if ( m_iFreeSound == SOUNDLIST_EMPTY )
211 {
212 // no free sound!
213 ALERT ( at_console, "Free Sound List is full!\n" );
214 return SOUNDLIST_EMPTY;
215 }
216
217 // there is at least one sound available, so move it to the
218 // Active sound list, and return its SoundPool index.
219
220 iNewSound = m_iFreeSound;// copy the index of the next free sound
221
222 m_iFreeSound = m_SoundPool[ m_iFreeSound ].m_iNext;// move the index down into the free list.
223
224 m_SoundPool[ iNewSound ].m_iNext = m_iActiveSound;// point the new sound at the top of the active list.
225
226 m_iActiveSound = iNewSound;// now make the new sound the top of the active list. You're done.
227
228 return iNewSound;
229}

271void CSoundEnt :: Initialize ( void )
272{
273 int i;
274 int iSound;
275
276 m_cLastActiveSounds;
277 m_iFreeSound = 0;
278 m_iActiveSound = SOUNDLIST_EMPTY;
279
280 for ( i = 0 ; i < MAX_WORLD_SOUNDS ; i++ )
281 {// clear all sounds, and link them into the free sound list.
282 m_SoundPool[ i ].Clear();
283 m_SoundPool[ i ].m_iNext = i + 1;
284 }
285
286 m_SoundPool[ i - 1 ].m_iNext = SOUNDLIST_EMPTY;// terminate the list here.
287
288
289 // now reserve enough sounds for each client
290 for ( i = 0 ; i < gpGlobals->maxClients ; i++ )
291 {
292 iSound = pSoundEnt->IAllocSound();
293
294 if ( iSound == SOUNDLIST_EMPTY )
295 {
296 ALERT ( at_console, "Could not AllocSound() for Client Reserve! (DLL)\n" );
297 return;
298 }
299
300 pSoundEnt->m_SoundPool[ iSound ].m_flExpireTime = SOUND_NEVER_EXPIRE;
301 }
302
303 if ( CVAR_GET_FLOAT("displaysoundlist") == 1 )
304 {
305 m_fShowReport = TRUE;
306 }
307 else
308 {
309 m_fShowReport = FALSE;
310 }
311}

319int CSoundEnt :: ISoundsInList ( int iListType )
320{
321 int i;
322 int iThisSound;
323
324 if ( iListType == SOUNDLISTTYPE_FREE )
325 {
326 iThisSound = m_iFreeSound;
327 }
328 else if ( iListType == SOUNDLISTTYPE_ACTIVE )
329 {
330 iThisSound = m_iActiveSound;
331 }
332 else
333 {
334 ALERT ( at_console, "Unknown Sound List Type!\n" );
335 }
336
337 if ( iThisSound == SOUNDLIST_EMPTY )
338 {
339 return 0;
340 }
341
342 i = 0;
343
344 while ( iThisSound != SOUNDLIST_EMPTY )
345 {
346 i++;
347
348 iThisSound = m_SoundPool[ iThisSound ].m_iNext;
349 }
350
351 return i;
352}
