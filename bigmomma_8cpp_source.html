521void CBigMomma :: HandleAnimEvent( MonsterEvent_t *pEvent )
522{
523 switch( pEvent->event )
524 {
525 case BIG_AE_MELEE_ATTACKBR:
526 case BIG_AE_MELEE_ATTACKBL:
527 case BIG_AE_MELEE_ATTACK1:
528 {
529 Vector forward, right;
530
531 UTIL_MakeVectorsPrivate( pev->angles, forward, right, NULL );
532
533 Vector center = pev->origin + forward * 128;
534 Vector mins = center - Vector( 64, 64, 0 );
535 Vector maxs = center + Vector( 64, 64, 64 );
536
537 CBaseEntity *pList[8];
538 int count = UTIL_EntitiesInBox( pList, 8, mins, maxs, FL_MONSTER|FL_CLIENT );
539 CBaseEntity *pHurt = NULL;
540
541 for ( int i = 0; i < count && !pHurt; i++ )
542 {
543 if ( pList[i] != this )
544 {
545 if ( pList[i]->pev->owner != edict() )
546 pHurt = pList[i];
547 }
548 }
549
550 if ( pHurt )
551 {
552 pHurt->TakeDamage( pev, pev, gSkillData.bigmommaDmgSlash, DMG_CRUSH | DMG_SLASH );
553 pHurt->pev->punchangle.x = 15;
554 switch( pEvent->event )
555 {
556 case BIG_AE_MELEE_ATTACKBR:
557 pHurt->pev->velocity = pHurt->pev->velocity + (forward * 150) + Vector(0,0,250) - (right * 200);
558 break;
559
560 case BIG_AE_MELEE_ATTACKBL:
561 pHurt->pev->velocity = pHurt->pev->velocity + (forward * 150) + Vector(0,0,250) + (right * 200);
562 break;
563
564 case BIG_AE_MELEE_ATTACK1:
565 pHurt->pev->velocity = pHurt->pev->velocity + (forward * 220) + Vector(0,0,200);
566 break;
567 }
568
569 pHurt->pev->flags &= ~FL_ONGROUND;
570 EMIT_SOUND_DYN( edict(), CHAN_WEAPON, RANDOM_SOUND_ARRAY(pAttackHitSounds), 1.0, ATTN_NORM, 0, 100 + RANDOM_LONG(-5,5) );
571 }
572 }
573 break;
574
575 case BIG_AE_SCREAM:
576 EMIT_SOUND_ARRAY_DYN( CHAN_VOICE, pAlertSounds );
577 break;
578
579 case BIG_AE_PAIN_SOUND:
580 EMIT_SOUND_ARRAY_DYN( CHAN_VOICE, pPainSounds );
581 break;
582
583 case BIG_AE_ATTACK_SOUND:
584 EMIT_SOUND_ARRAY_DYN( CHAN_WEAPON, pAttackSounds );
585 break;
586
587 case BIG_AE_BIRTH_SOUND:
588 EMIT_SOUND_ARRAY_DYN( CHAN_BODY, pBirthSounds );
589 break;
590
591 case BIG_AE_SACK:
592 if ( RANDOM_LONG(0,100) < 30 )
593 EMIT_SOUND_ARRAY_DYN( CHAN_BODY, pSackSounds );
594 break;
595
596 case BIG_AE_DEATHSOUND:
597 EMIT_SOUND_ARRAY_DYN( CHAN_VOICE, pDeathSounds );
598 break;
599
600 case BIG_AE_STEP1: // Footstep left
601 case BIG_AE_STEP3: // Footstep back left
602 EMIT_SOUND_ARRAY_DYN( CHAN_ITEM, pFootSounds );
603 break;
604
605 case BIG_AE_STEP4: // Footstep back right
606 case BIG_AE_STEP2: // Footstep right
607 EMIT_SOUND_ARRAY_DYN( CHAN_BODY, pFootSounds );
608 break;
609
610 case BIG_AE_MORTAR_ATTACK1:
611 LaunchMortar();
612 break;
613
614 case BIG_AE_LAY_CRAB:
615 LayHeadcrab();
616 break;
617
618 case BIG_AE_JUMP_FORWARD:
619 ClearBits( pev->flags, FL_ONGROUND );
620
621 UTIL_SetOrigin (pev, pev->origin + Vector ( 0 , 0 , 1) );// take him off ground so engine doesn't instantly reset onground
622 UTIL_MakeVectors ( pev->angles );
623
624 pev->velocity = (gpGlobals->v_forward * 200) + gpGlobals->v_up * 500;
625 break;
626
627 case BIG_AE_EARLY_TARGET:
628 {
629 CBaseEntity *pTarget = m_hTargetEnt;
630 if ( pTarget && pTarget->pev->message )
631 FireTargets( STRING(pTarget->pev->message), this, this, USE_TOGGLE, 0 );
632 Remember( bits_MEMORY_FIRED_NODE );
633 }
634 break;
635
636 default:
637 CBaseMonster::HandleAnimEvent( pEvent );
638 break;
639 }
640}

864{

866

1305Vector VecCheckSplatToss(entvars_t *pev, const Vector &vecSpot1, Vector vecSpot2, float maxHeight)
1306{
1307 TraceResult tr;
1308 Vector vecMidPoint; // halfway point between Spot1 and Spot2
1309 Vector vecApex; // highest point
1310 Vector vecScale;
1311 Vector vecGrenadeVel;
1312 Vector vecTemp;
1313 float flGravity = g_psv_gravity->value;
1314
1315 // calculate the midpoint and apex of the 'triangle'
1316 vecMidPoint = vecSpot1 + (vecSpot2 - vecSpot1) * 0.5;
1317 UTIL_TraceLine(vecMidPoint, vecMidPoint + Vector(0, 0, maxHeight), ignore_monsters, ENT(pev), &tr);
1318 vecApex = tr.vecEndPos;
1319
1320 UTIL_TraceLine(vecSpot1, vecApex, dont_ignore_monsters, ENT(pev), &tr);
1321 if (tr.flFraction != 1.0)
1322 {
1323 // fail!
1324 return g_vecZero;
1325 }
1326
1327 // Don't worry about actually hitting the target, this won't hurt us!
1328
1329 // How high should the grenade travel (subtract 15 so the grenade doesn't hit the ceiling)?
1330 float height = (vecApex.z - vecSpot1.z) - 15;
1331 // How fast does the grenade need to travel to reach that height given gravity?
1332 float speed = sqrt(2 * flGravity * height);
1333
1334 // How much time does it take to get there?
1335 float time = speed / flGravity;
1336 vecGrenadeVel = (vecSpot2 - vecSpot1);
1337 vecGrenadeVel.z = 0;
1338 float distance = vecGrenadeVel.Length();
1339
1340 // Travel half the distance to the target in that time (apex is at the midpoint)
1341 vecGrenadeVel = vecGrenadeVel * (0.5 / time);
1342 // Speed to offset gravity at the desired height
1343 vecGrenadeVel.z = speed;
1344
1345 return vecGrenadeVel;
1346}
