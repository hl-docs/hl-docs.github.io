214void PM_InitTextureTypes()
215{
216 char buffer[512];
217 int i, j;
218 byte *pMemFile;
219 int fileSize, filePos;
220 static qboolean bTextureTypeInit = false;
221
222 if ( bTextureTypeInit )
223 return;
224
225 memset(&(grgszTextureName[0][0]), 0, CTEXTURESMAX * CBTEXTURENAMEMAX);
226 memset(grgchTextureType, 0, CTEXTURESMAX);
227
228 gcTextures = 0;
229 memset(buffer, 0, 512);
230
231 fileSize = pmove->COM_FileSize( "sound/materials.txt" );
232 pMemFile = pmove->COM_LoadFile( "sound/materials.txt", 5, NULL );
233 if ( !pMemFile )
234 return;
235
236 filePos = 0;
237 // for each line in the file...
238 while ( pmove->memfgets( pMemFile, fileSize, &filePos, buffer, 511 ) != NULL && (gcTextures < CTEXTURESMAX) )
239 {
240 // skip whitespace
241 i = 0;
242 while(buffer[i] && isspace(buffer[i]))
243 i++;
244
245 if (!buffer[i])
246 continue;
247
248 // skip comment lines
249 if (buffer[i] == '/' || !isalpha(buffer[i]))
250 continue;
251
252 // get texture type
253 grgchTextureType[gcTextures] = toupper(buffer[i++]);
254
255 // skip whitespace
256 while(buffer[i] && isspace(buffer[i]))
257 i++;
258
259 if (!buffer[i])
260 continue;
261
262 // get sentence name
263 j = i;
264 while (buffer[j] && !isspace(buffer[j]))
265 j++;
266
267 if (!buffer[j])
268 continue;
269
270 // null-terminate name and save in sentences array
271 j = min (j, CBTEXTURENAMEMAX-1+i);
272 buffer[j] = 0;
273 strcpy(&(grgszTextureName[gcTextures++][0]), &(buffer[i]));
274 }
275
276 // Must use engine to free since we are in a .dll
277 pmove->COM_FreeFile ( pMemFile );
278
279 PM_SortTextures();
280
281 bTextureTypeInit = true;
282}

293char PM_FindTextureType( char *name )
294{
295 int left, right, pivot;
296 int val;
297
298 assert( pm_shared_initialized );
299
300 left = 0;
301 right = gcTextures - 1;
302
303 while ( left <= right )
304 {
305 pivot = ( left + right ) / 2;
306
307 val = strnicmp( name, grgszTextureName[ pivot ], CBTEXTURENAMEMAX-1 );
308 if ( val == 0 )
309 {
310 return grgchTextureType[ pivot ];
311 }
312 else if ( val > 0 )
313 {
314 left = pivot + 1;
315 }
316 else if ( val < 0 )
317 {
318 right = pivot - 1;
319 }
320 }
321
322 return CHAR_TEX_CONCRETE;
323}

560void PM_UpdateStepSound( void )
561{
562 int fWalking;
563 float fvol;
564 vec3_t knee;
565 vec3_t feet;
566 vec3_t center;
567 float height;
568 float speed;
569 float velrun;
570 float velwalk;
571 float flduck;
572 int fLadder;
573 int step;
574
575 if ( pmove->flTimeStepSound > 0 )
576 return;
577
578 if ( pmove->flags & FL_FROZEN )
579 return;
580
581 PM_CatagorizeTextureType();
582
583 speed = Length( pmove->velocity );
584
585 // determine if we are on a ladder
586 fLadder = ( pmove->movetype == MOVETYPE_FLY );// IsOnLadder();
587
588 // UNDONE: need defined numbers for run, walk, crouch, crouch run velocities!!!!
589 if ( ( pmove->flags & FL_DUCKING) || fLadder )
590 {
591 velwalk = 60; // These constants should be based on cl_movespeedkey * cl_forwardspeed somehow
592 velrun = 80; // UNDONE: Move walking to server
593 flduck = 100;
594 }
595 else
596 {
597 velwalk = 120;
598 velrun = 210;
599 flduck = 0;
600 }
601
602 // If we're on a ladder or on the ground, and we're moving fast enough,
603 // play step sound. Also, if pmove->flTimeStepSound is zero, get the new
604 // sound right away - we just started moving in new level.
605 if ( (fLadder || ( pmove->onground != -1 ) ) &&
606 ( Length( pmove->velocity ) > 0.0 ) &&
607 ( speed >= velwalk || !pmove->flTimeStepSound ) )
608 {
609 fWalking = speed < velrun;
610
611 VectorCopy( pmove->origin, center );
612 VectorCopy( pmove->origin, knee );
613 VectorCopy( pmove->origin, feet );
614
615 height = pmove->player_maxs[ pmove->usehull ][ 2 ] - pmove->player_mins[ pmove->usehull ][ 2 ];
616
617 knee[2] = pmove->origin[2] - 0.3 * height;
618 feet[2] = pmove->origin[2] - 0.5 * height;
619
620 // find out what we're stepping in or on...
621 if (fLadder)
622 {
623 step = STEP_LADDER;
624 fvol = 0.35;
625 pmove->flTimeStepSound = 350;
626 }
627 else if ( pmove->PM_PointContents ( knee, NULL ) == CONTENTS_WATER )
628 {
629 step = STEP_WADE;
630 fvol = 0.65;
631 pmove->flTimeStepSound = 600;
632 }
633 else if ( pmove->PM_PointContents ( feet, NULL ) == CONTENTS_WATER )
634 {
635 step = STEP_SLOSH;
636 fvol = fWalking ? 0.2 : 0.5;
637 pmove->flTimeStepSound = fWalking ? 400 : 300;
638 }
639 else
640 {
641 // find texture under player, if different from current texture,
642 // get material type
643 step = PM_MapTextureTypeStepType( pmove->chtexturetype );
644
645 switch ( pmove->chtexturetype )
646 {
647 default:
648 case CHAR_TEX_CONCRETE:
649 fvol = fWalking ? 0.2 : 0.5;
650 pmove->flTimeStepSound = fWalking ? 400 : 300;
651 break;
652
653 case CHAR_TEX_METAL:
654 fvol = fWalking ? 0.2 : 0.5;
655 pmove->flTimeStepSound = fWalking ? 400 : 300;
656 break;
657
658 case CHAR_TEX_DIRT:
659 fvol = fWalking ? 0.25 : 0.55;
660 pmove->flTimeStepSound = fWalking ? 400 : 300;
661 break;
662
663 case CHAR_TEX_VENT:
664 fvol = fWalking ? 0.4 : 0.7;
665 pmove->flTimeStepSound = fWalking ? 400 : 300;
666 break;
667
668 case CHAR_TEX_GRATE:
669 fvol = fWalking ? 0.2 : 0.5;
670 pmove->flTimeStepSound = fWalking ? 400 : 300;
671 break;
672
673 case CHAR_TEX_TILE:
674 fvol = fWalking ? 0.2 : 0.5;
675 pmove->flTimeStepSound = fWalking ? 400 : 300;
676 break;
677
678 case CHAR_TEX_SLOSH:
679 fvol = fWalking ? 0.2 : 0.5;
680 pmove->flTimeStepSound = fWalking ? 400 : 300;
681 break;
682 }
683 }
684
685 pmove->flTimeStepSound += flduck; // slower step time if ducking
686
687 // play the sound
688 // 35% volume if ducking
689 if ( pmove->flags & FL_DUCKING )
690 {
691 fvol *= 0.35;
692 }
693
694 PM_PlayStepSound( step, fvol );
695 }

790int PM_ClipVelocity (vec3_t in, vec3_t normal, vec3_t out, float overbounce)
791{
792 float backoff;
793 float change;
794 float angle;
795 int i, blocked;
796
797 angle = normal[ 2 ];
798
799 blocked = 0x00; // Assume unblocked.
800 if (angle > 0) // If the plane that is blocking us has a positive z component, then assume it's a floor.
801 blocked |= 0x01; //
802 if (!angle) // If the plane has no Z, it is vertical (wall/step)
803 blocked |= 0x02; //
804
805 // Determine how far along plane to slide based on incoming direction.
806 // Scale by overbounce factor.
807 backoff = DotProduct (in, normal) * overbounce;
808
809 for (i=0 ; i<3 ; i++)
810 {
811 change = normal[i]*backoff;
812 out[i] = in[i] - change;
813 // If out velocity is too small, zero it out.
814 if (out[i] > -STOP_EPSILON && out[i] < STOP_EPSILON)
815 out[i] = 0;
816 }
817
818 // Return blocking flags.
819 return blocked;
820}

885int PM_FlyMove (void)
886{
887 int bumpcount, numbumps;
888 vec3_t dir;
889 float d;
890 int numplanes;
891 vec3_t planes[MAX_CLIP_PLANES];
892 vec3_t primal_velocity, original_velocity;
893 vec3_t new_velocity;
894 int i, j;
895 pmtrace_t trace;
896 vec3_t end;
897 float time_left, allFraction;
898 int blocked;
899
900 numbumps = 4; // Bump up to four times
901
902 blocked = 0; // Assume not blocked
903 numplanes = 0; // and not sliding along any planes
904 VectorCopy (pmove->velocity, original_velocity); // Store original velocity
905 VectorCopy (pmove->velocity, primal_velocity);
906
907 allFraction = 0;
908 time_left = pmove->frametime; // Total time for this movement operation.
909
910 for (bumpcount=0 ; bumpcount<numbumps ; bumpcount++)
911 {
912 if (!pmove->velocity[0] && !pmove->velocity[1] && !pmove->velocity[2])
913 break;
914
915 // Assume we can move all the way from the current origin to the
916 // end point.
917 for (i=0 ; i<3 ; i++)
918 end[i] = pmove->origin[i] + time_left * pmove->velocity[i];
919
920 // See if we can make it from origin to end point.
921 trace = pmove->PM_PlayerTrace (pmove->origin, end, PM_NORMAL, -1 );
922
923 allFraction += trace.fraction;
924 // If we started in a solid object, or we were in solid space
925 // the whole way, zero out our velocity and return that we
926 // are blocked by floor and wall.
927 if (trace.allsolid)
928 { // entity is trapped in another solid
929 VectorCopy (vec3_origin, pmove->velocity);
930 //Con_DPrintf("Trapped 4\n");
931 return 4;
932 }
933
934 // If we moved some portion of the total distance, then
935 // copy the end position into the pmove->origin and
936 // zero the plane counter.
937 if (trace.fraction > 0)
938 { // actually covered some distance
939 VectorCopy (trace.endpos, pmove->origin);
940 VectorCopy (pmove->velocity, original_velocity);
941 numplanes = 0;
942 }
943
944 // If we covered the entire distance, we are done
945 // and can return.
946 if (trace.fraction == 1)
947 break; // moved the entire distance
948
949 //if (!trace.ent)
950 // Sys_Error ("PM_PlayerTrace: !trace.ent");
951
952 // Save entity that blocked us (since fraction was < 1.0)
953 // for contact
954 // Add it if it's not already in the list!!!
955 PM_AddToTouched(trace, pmove->velocity);
956
957 // If the plane we hit has a high z component in the normal, then
958 // it's probably a floor
959 if (trace.plane.normal[2] > 0.7)
960 {
961 blocked |= 1; // floor
962 }
963 // If the plane has a zero z component in the normal, then it's a
964 // step or wall
965 if (!trace.plane.normal[2])
966 {
967 blocked |= 2; // step / wall
968 //Con_DPrintf("Blocked by %i\n", trace.ent);
969 }
970
971 // Reduce amount of pmove->frametime left by total time left * fraction
972 // that we covered.
973 time_left -= time_left * trace.fraction;
974
975 // Did we run out of planes to clip against?
976 if (numplanes >= MAX_CLIP_PLANES)
977 { // this shouldn't really happen
978 // Stop our movement if so.
979 VectorCopy (vec3_origin, pmove->velocity);
980 //Con_DPrintf("Too many planes 4\n");
981
982 break;
983 }
984
985 // Set up next clipping plane
986 VectorCopy (trace.plane.normal, planes[numplanes]);
987 numplanes++;
988//
989
990// modify original_velocity so it parallels all of the clip planes
991//
992 if ( pmove->movetype == MOVETYPE_WALK &&
993 ((pmove->onground == -1) || (pmove->friction != 1)) ) // relfect player velocity
994 {
995 for ( i = 0; i < numplanes; i++ )
996 {
997 if ( planes[i][2] > 0.7 )
998 {// floor or slope
999 PM_ClipVelocity( original_velocity, planes[i], new_velocity, 1 );
1000 VectorCopy( new_velocity, original_velocity );
1001 }
1002 else
1003 PM_ClipVelocity( original_velocity, planes[i], new_velocity, 1.0 + pmove->movevars->bounce * (1-pmove->friction) );
1004 }
1005
1006 VectorCopy( new_velocity, pmove->velocity );
1007 VectorCopy( new_velocity, original_velocity );
1008 }
1009 else
1010 {
1011 for (i=0 ; i<numplanes ; i++)
1012 {
1013 PM_ClipVelocity (
1014 original_velocity,
1015 planes[i],
1016 pmove->velocity,
1017 1);
1018 for (j=0 ; j<numplanes ; j++)
1019 if (j != i)
1020 {
1021 // Are we now moving against this plane?
1022 if (DotProduct (pmove->velocity, planes[j]) < 0)
1023 break; // not ok
1024 }
1025 if (j == numplanes) // Didn't have to clip, so we're ok
1026 break;
1027 }
1028
1029 // Did we go all the way through plane set
1030 if (i != numplanes)
1031 { // go along this plane
1032 // pmove->velocity is set in clipping call, no need to set again.
1033 ;
1034 }
1035 else
1036 { // go along the crease
1037 if (numplanes != 2)
1038 {
1039 //Con_Printf ("clip velocity, numplanes == %i\n",numplanes);
1040 VectorCopy (vec3_origin, pmove->velocity);
1041 //Con_DPrintf("Trapped 4\n");
1042
1043 break;
1044 }
1045 CrossProduct (planes[0], planes[1], dir);
1046 d = DotProduct (dir, pmove->velocity);
1047 VectorScale (dir, d, pmove->velocity );
1048 }
1049
1050 //
1051 // if original velocity is against the original velocity, stop dead
1052 // to avoid tiny occilations in sloping corners
1053 //
1054 if (DotProduct (pmove->velocity, primal_velocity) <= 0)
1055 {
1056 //Con_DPrintf("Back\n");
1057 VectorCopy (vec3_origin, pmove->velocity);
1058 break;
1059 }
1060 }
1061 }
1062
1063 if ( allFraction == 0 )
1064 {
1065 VectorCopy (vec3_origin, pmove->velocity);
1066 //Con_DPrintf( "Don't stick\n" );
1067 }
1068
1069 return blocked;
1070}
