482void CScientist::StartTask(Task_t* pTask)
483{
484 switch (pTask->iTask)
485 {
486 case TASK_SAY_HEAL:
487 // if ( FOkToSpeak() )
488 Talk(2);
489 m_hTalkTarget = m_hTargetEnt;
490 PlaySentence("SC_HEAL", 2, VOL_NORM, ATTN_IDLE);
491
492 TaskComplete();
493 break;
494
495 case TASK_SCREAM:
496 Scream();
497 TaskComplete();
498 break;
499
500 case TASK_RANDOM_SCREAM:
501 if (RANDOM_FLOAT(0, 1) < pTask->flData)
502 Scream();
503 TaskComplete();
504 break;
505
506 case TASK_SAY_FEAR:
507 if (FOkToSpeak())
508 {
509 Talk(2);
510 m_hTalkTarget = m_hEnemy;
511 if (m_hEnemy->IsPlayer())
512 PlaySentence("SC_PLFEAR", 5, VOL_NORM, ATTN_NORM);
513 else
514 PlaySentence("SC_FEAR", 5, VOL_NORM, ATTN_NORM);
515 }
516 TaskComplete();
517 break;
518
519 case TASK_HEAL:
520 m_IdealActivity = ACT_MELEE_ATTACK1;
521 break;
522
523 case TASK_RUN_PATH_SCARED:
524 m_movementActivity = ACT_RUN_SCARED;
525 break;
526
527 case TASK_MOVE_TO_TARGET_RANGE_SCARED:
528 {
529 if ((m_hTargetEnt->pev->origin - pev->origin).Length() < 1)
530 TaskComplete();
531 else
532 {
533 m_vecMoveGoal = m_hTargetEnt->pev->origin;
534 if (!MoveToTarget(ACT_WALK_SCARED, 0.5))
535 TaskFail();
536 }
537 }
538 break;
539
540 default:
541 CTalkMonster::StartTask(pTask);
542 break;
543 }
544}

554void CScientist::RunTask(Task_t* pTask)
555{
556 switch (pTask->iTask)
557 {
558 case TASK_RUN_PATH_SCARED:
559 if (MovementIsComplete())
560 TaskComplete();
561 if (RANDOM_LONG(0, 31) < 8)
562 Scream();
563 break;
564
565 case TASK_MOVE_TO_TARGET_RANGE_SCARED:
566 {
567 if (RANDOM_LONG(0, 63) < 8)
568 Scream();
569
570 if (m_hEnemy == NULL)
571 {
572 TaskFail();
573 }
574 else
575 {
576 float distance;
577
578 distance = (m_vecMoveGoal - pev->origin).Length2D();
579 // Re-evaluate when you think you're finished, or the target has moved too far
580 if ((distance < pTask->flData) || (m_vecMoveGoal - m_hTargetEnt->pev->origin).Length() > pTask->flData * 0.5)
581 {
582 m_vecMoveGoal = m_hTargetEnt->pev->origin;
583 distance = (m_vecMoveGoal - pev->origin).Length2D();
584 FRefreshRoute();
585 }
586
587 // Set the appropriate activity based on an overlapping range
588 // overlap the range to prevent oscillation
589 if (distance < pTask->flData)
590 {
591 TaskComplete();
592 RouteClear(); // Stop moving
593 }
594 else if (distance < 190 && m_movementActivity != ACT_WALK_SCARED)
595 m_movementActivity = ACT_WALK_SCARED;
596 else if (distance >= 270 && m_movementActivity != ACT_RUN_SCARED)
597 m_movementActivity = ACT_RUN_SCARED;
598 }
599 }
600 break;
601
602 case TASK_HEAL:
603 if (m_fSequenceFinished)
604 {
605 TaskComplete();
606 }
607 else
608 {
609 if (TargetDistance() > 90)
610 TaskComplete();
611 pev->ideal_yaw = UTIL_VecToYaw(m_hTargetEnt->pev->origin - pev->origin);
612 ChangeYaw(pev->yaw_speed);
613 }
614 break;
615 default:
616 CTalkMonster::RunTask(pTask);
617 break;
618 }

707void CScientist::Spawn(void)
708{
709 Precache();
710
711 SET_MODEL(ENT(pev), "models/scientist.mdl");
712 UTIL_SetSize(pev, VEC_HUMAN_HULL_MIN, VEC_HUMAN_HULL_MAX);
713
714 pev->solid = SOLID_SLIDEBOX;
715 pev->movetype = MOVETYPE_STEP;
716 m_bloodColor = BLOOD_COLOR_RED;
717 pev->health = gSkillData.scientistHealth;
718 pev->view_ofs = Vector(0, 0, 50); // position of the eyes relative to monster's origin.
719 m_flFieldOfView = VIEW_FIELD_WIDE; // NOTE: we need a wide field of view so scientists will notice player and say hello
720 m_MonsterState = MONSTERSTATE_NONE;
721
722 // m_flDistTooFar = 256.0;
723
724 m_afCapability = bits_CAP_HEAR | bits_CAP_TURN_HEAD | bits_CAP_OPEN_DOORS | bits_CAP_AUTO_DOORS | bits_CAP_USE;
725
726 // White hands
727 pev->skin = 0;
728
729 if (pev->body == -1)
730 { // -1 chooses a random head
731 pev->body = RANDOM_LONG(0, NUM_SCIENTIST_HEADS - 1); // pick a head, any head
732 }
733
734 // Luther is black, make his hands black
735 if (pev->body == HEAD_LUTHER)
736 pev->skin = 1;
737
738 MonsterInit();
739 SetUse(&CScientist::FollowerUse);
740}
