137void FindHullIntersection( const Vector &vecSrc, TraceResult &tr, float *mins, float *maxs, edict_t *pEntity )
138{
139 int i, j, k;
140 float distance;
141 float *minmaxs[2] = {mins, maxs};
142 TraceResult tmpTrace;
143 Vector vecHullEnd = tr.vecEndPos;
144 Vector vecEnd;
145
146 distance = 1e6f;
147
148 vecHullEnd = vecSrc + ((vecHullEnd - vecSrc)*2);
149 UTIL_TraceLine( vecSrc, vecHullEnd, dont_ignore_monsters, pEntity, &tmpTrace );
150 if ( tmpTrace.flFraction < 1.0 )
151 {
152 tr = tmpTrace;
153 return;
154 }
155
156 for ( i = 0; i < 2; i++ )
157 {
158 for ( j = 0; j < 2; j++ )
159 {
160 for ( k = 0; k < 2; k++ )
161 {
162 vecEnd.x = vecHullEnd.x + minmaxs[i][0];
163 vecEnd.y = vecHullEnd.y + minmaxs[j][1];
164 vecEnd.z = vecHullEnd.z + minmaxs[k][2];
165
166 UTIL_TraceLine( vecSrc, vecEnd, dont_ignore_monsters, pEntity, &tmpTrace );
167 if ( tmpTrace.flFraction < 1.0 )
168 {
169 float thisDistance = (tmpTrace.vecEndPos - vecSrc).Length();
170 if ( thisDistance < distance )
171 {
172 tr = tmpTrace;
173 distance = thisDistance;
174 }
175 }
176 }
177 }
178 }

228int CCrowbar::Swing( int fFirst )
229{
230 int fDidHit = FALSE;
231
232 TraceResult tr;
233
234 UTIL_MakeVectors (m_pPlayer->pev->v_angle);
235 Vector vecSrc = m_pPlayer->GetGunPosition( );
236 Vector vecEnd = vecSrc + gpGlobals->v_forward * 32;
237
238 UTIL_TraceLine( vecSrc, vecEnd, dont_ignore_monsters, ENT( m_pPlayer->pev ), &tr );
239
240#ifndef CLIENT_DLL
241 if ( tr.flFraction >= 1.0 )
242 {
243 UTIL_TraceHull( vecSrc, vecEnd, dont_ignore_monsters, head_hull, ENT( m_pPlayer->pev ), &tr );
244 if ( tr.flFraction < 1.0 )
245 {
246 // Calculate the point of intersection of the line (or hull) and the object we hit
247 // This is and approximation of the "best" intersection
248 CBaseEntity *pHit = CBaseEntity::Instance( tr.pHit );
249 if ( !pHit || pHit->IsBSPModel() )
250 FindHullIntersection( vecSrc, tr, VEC_DUCK_HULL_MIN, VEC_DUCK_HULL_MAX, m_pPlayer->edict() );
251 vecEnd = tr.vecEndPos; // This is the point on the actual surface (the hull could have hit space)
252 }
253 }
254#endif
255
256 PLAYBACK_EVENT_FULL( FEV_NOTHOST, m_pPlayer->edict(), m_usCrowbar,
257 0.0, (float *)&g_vecZero, (float *)&g_vecZero, 0, 0, 0,
258 0.0, 0, 0.0 );
259
260
261 if ( tr.flFraction >= 1.0 )
262 {
263 if (fFirst)
264 {
265 // miss
266 m_flNextPrimaryAttack = GetNextAttackDelay(0.5);
267
268 // player "shoot" animation
269 m_pPlayer->SetAnimation( PLAYER_ATTACK1 );
270 }
271 }
272 else
273 {
274 switch( ((m_iSwing++) % 2) + 1 )
275 {
276 case 0:
277 SendWeaponAnim( CROWBAR_ATTACK1HIT ); break;
278 case 1:
279 SendWeaponAnim( CROWBAR_ATTACK2HIT ); break;
280 case 2:
281 SendWeaponAnim( CROWBAR_ATTACK3HIT ); break;
282 }
283
284 // player "shoot" animation
285 m_pPlayer->SetAnimation( PLAYER_ATTACK1 );
286
287#ifndef CLIENT_DLL
288
289 // hit
290 fDidHit = TRUE;
291 CBaseEntity *pEntity = CBaseEntity::Instance(tr.pHit);
292
293 ClearMultiDamage( );
294
295 if ( (m_flNextPrimaryAttack + 1 < UTIL_WeaponTimeBase() ) || g_pGameRules->IsMultiplayer() )
296 {
297 // first swing does full damage
298 pEntity->TraceAttack(m_pPlayer->pev, gSkillData.plrDmgCrowbar, gpGlobals->v_forward, &tr, DMG_CLUB );
299 }
300 else
301 {
302 // subsequent swings do half
303 pEntity->TraceAttack(m_pPlayer->pev, gSkillData.plrDmgCrowbar / 2, gpGlobals->v_forward, &tr, DMG_CLUB );
304 }
305 ApplyMultiDamage( m_pPlayer->pev, m_pPlayer->pev );
306
307 // play thwack, smack, or dong sound
308 float flVol = 1.0;
309 int fHitWorld = TRUE;
310
311 if (pEntity)
312 {
313 if ( pEntity->Classify() != CLASS_NONE && pEntity->Classify() != CLASS_MACHINE )
314 {
315 // play thwack or smack sound
316 switch( RANDOM_LONG(0,2) )
317 {
318 case 0:
319 EMIT_SOUND(ENT(m_pPlayer->pev), CHAN_ITEM, "weapons/cbar_hitbod1.wav", 1, ATTN_NORM); break;
320 case 1:
321 EMIT_SOUND(ENT(m_pPlayer->pev), CHAN_ITEM, "weapons/cbar_hitbod2.wav", 1, ATTN_NORM); break;
322 case 2:
323 EMIT_SOUND(ENT(m_pPlayer->pev), CHAN_ITEM, "weapons/cbar_hitbod3.wav", 1, ATTN_NORM); break;
324 }
325 m_pPlayer->m_iWeaponVolume = CROWBAR_BODYHIT_VOLUME;
326 if ( !pEntity->IsAlive() )
327 return TRUE;
328 else
329 flVol = 0.1;
330
331 fHitWorld = FALSE;
332 }
333 }
334
335 // play texture hit sound
336 // UNDONE: Calculate the correct point of intersection when we hit with the hull instead of the line
337
338 if (fHitWorld)
339 {
340 float fvolbar = TEXTURETYPE_PlaySound(&tr, vecSrc, vecSrc + (vecEnd-vecSrc)*2, BULLET_PLAYER_CROWBAR);
341
342 if ( g_pGameRules->IsMultiplayer() )
343 {
344 // override the volume here, cause we don't play texture sounds in multiplayer,
345 // and fvolbar is going to be 0 from the above call.
346
347 fvolbar = 1;
348 }
349
350 // also play crowbar strike
351 switch( RANDOM_LONG(0,1) )
352 {
353 case 0:
354 EMIT_SOUND_DYN(ENT(m_pPlayer->pev), CHAN_ITEM, "weapons/cbar_hit1.wav", fvolbar, ATTN_NORM, 0, 98 + RANDOM_LONG(0,3));
355 break;
356 case 1:
357 EMIT_SOUND_DYN(ENT(m_pPlayer->pev), CHAN_ITEM, "weapons/cbar_hit2.wav", fvolbar, ATTN_NORM, 0, 98 + RANDOM_LONG(0,3));
358 break;
359 }
360
361 // delay the decal a bit
362 m_trHit = tr;
363 }
364
365 m_pPlayer->m_iWeaponVolume = flVol * CROWBAR_WALLHIT_VOLUME;
366#endif
367 m_flNextPrimaryAttack = GetNextAttackDelay(0.25);
368
369 SetThink( &CCrowbar::Smack );
370 pev->nextthink = UTIL_WeaponTimeBase() + 0.2;
371
372
373 }
374 return fDidHit;
375}
