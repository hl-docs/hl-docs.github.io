111void CCycler::GenericCyclerSpawn(char *szModel, Vector vecMin, Vector vecMax)
112{
113 if (!szModel || !*szModel)
114 {
115 ALERT(at_error, "cycler at %.0f %.0f %.0f missing modelname", pev->origin.x, pev->origin.y, pev->origin.z);
116 REMOVE_ENTITY(ENT(pev)); // Remove the entity if no model is specified
117 return;
118 }
119
120 pev->classname = MAKE_STRING("cycler"); // Set the classname
121 PRECACHE_MODEL(szModel); // Precache the model
122 SET_MODEL(ENT(pev), szModel); // Set the model for the entity
123
124 CCycler::Spawn(); // Call the spawn function
125
126 UTIL_SetSize(pev, vecMin, vecMax); // Set the size of the entity
127}

135void CCycler::Spawn()
136{
137 InitBoneControllers(); // Initialize bone controllers
138 pev->solid = SOLID_SLIDEBOX; // Set solid type
139 pev->movetype = MOVETYPE_NONE; // Set movement type
140 pev->takedamage = DAMAGE_YES; // Allow the cycler to take damage
141 pev->effects = 0; // No special effects
142 pev->health = 80000; // Set health (no cycler should die)
143 pev->yaw_speed = 5; // Set yaw speed
144 pev->ideal_yaw = pev->angles.y; // Set ideal yaw
145 ChangeYaw(360); // Change yaw to 360 degrees
146
147 m_flFrameRate = 75; // Set frame rate for animation
148 m_flGroundSpeed = 0; // Set ground speed to zero
149
150 pev->nextthink += 1.0; // Set the next think time
151
152 ResetSequenceInfo(); // Reset sequence information
153
154 if (pev->sequence != 0 || pev->frame != 0)
155 {
156 m_animate = 0; // Disable animation if sequence or frame is set
157 pev->framerate = 0; // Set frame rate to zero
158 }
159 else
160 {
161 m_animate = 1; // Enable animation
162 }
163}

171void CCycler::Think(void)
172{
173 pev->nextthink = gpGlobals->time + 0.1; // Set the next think time
174
175 if (m_animate)
176 {
177 StudioFrameAdvance(); // Advance the animation frame
178 }
179 if (m_fSequenceFinished && !m_fSequenceLoops)
180 {
181 // Reset sequence information to avoid reloading model every frame
182 pev->animtime = gpGlobals->time; // Set animation time
183 pev->framerate = 1.0; // Set frame rate to 1.0
184 m_fSequenceFinished = FALSE; // Reset sequence finished flag
185 m_flLastEventCheck = gpGlobals->time; // Update last event check time
186 pev->frame = 0; // Reset frame to 0
187 if (!m_animate)
188 pev->framerate = 0.0; // Fix: don't reset frame rate if not animating
189 }
190}

203void CCycler::Use(CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
204{
205 m_animate = !m_animate; // Toggle animation state
206 if (m_animate)
207 pev->framerate = 1.0; // Set frame rate to 1.0 if animating
208 else
209 pev->framerate = 0.0; // Set frame rate to 0.0 if not animating
210}

224int CCycler::TakeDamage(entvars_t* pevInflictor, entvars_t* pevAttacker, float flDamage, int bitsDamageType)
225{
226 if (m_animate)
227 {
228 pev->sequence++; // Advance to the next sequence
229 ResetSequenceInfo(); // Reset sequence information
230
231 if (m_flFrameRate == 0.0)
232 {
233 pev->sequence = 0; // Reset sequence if frame rate is zero
234 ResetSequenceInfo(); // Reset sequence information
235 }
236 pev->frame = 0; // Reset frame to 0
237 }
238 else
239 {
240 pev->framerate = 1.0; // Set frame rate to 1.0
241 StudioFrameAdvance(0.1); // Advance the animation frame
242 pev->framerate = 0; // Reset frame rate to 0
243 ALERT(at_console, "sequence: %d, frame %.0f\n", pev->sequence, pev->frame); // Log current sequence and frame
244 }
245
246 return 0; // Return 0 damage taken
247}

571void CWreckage::Think(void)
572{
573 StudioFrameAdvance(); // Advance the animation frame
574 pev->nextthink = gpGlobals->time + 0.2; // Set the next think time
575
576 if (pev->dmgtime) // Check if damage time is set
577 {
578 if (pev->dmgtime < gpGlobals->time) // Check if the damage time has expired
579 {
580 UTIL_Remove(this); // Remove the wreckage entity
581 return; // Exit the function
582 }
583 else if (RANDOM_FLOAT(0, pev->dmgtime - m_flStartTime) > pev->dmgtime - gpGlobals->time)
584 {
585 return; // Exit if the random float condition is met
586 }
587 }
588
589 Vector VecSrc; // Create a vector for the smoke effect source
590
591 VecSrc.x = RANDOM_FLOAT(pev->absmin.x, pev->absmax.x); // Random x position within bounds
592 VecSrc.y = RANDOM_FLOAT(pev->absmin.y, pev->absmax.y); // Random y position within bounds
593 VecSrc.z = RANDOM_FLOAT(pev->absmin.z, pev->absmax.z); // Random z position within bounds
594
595 MESSAGE_BEGIN(MSG_PVS, SVC_TEMPENTITY, VecSrc); // Begin a message for a temporary entity
596 WRITE_BYTE(TE_SMOKE); // Specify the smoke effect
597 WRITE_COORD(VecSrc.x); // Write the x coordinate
598 WRITE_COORD(VecSrc.y); // Write the y coordinate
599 WRITE_COORD(VecSrc.z); // Write the z coordinate
600 WRITE_SHORT(g_sModelIndexSmoke); // Write the smoke model index
601 WRITE_BYTE(RANDOM_LONG(0, 49) + 50); // Write the scale (randomized)
602 WRITE_BYTE(RANDOM_LONG(0, 3) + 8); // Write the framerate (randomized)
603 MESSAGE_END(); // End the message
604}
