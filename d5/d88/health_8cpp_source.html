158int CHudHealth:: MsgFunc_Damage(const char *pszName, int iSize, void *pbuf )
159{
160 BEGIN_READ( pbuf, iSize );
161
162 int armor = READ_BYTE(); // armor
163 int damageTaken = READ_BYTE(); // health
164 long bitsDamage = READ_LONG(); // damage bits
165
166 vec3_t vecFrom;
167
168 for ( int i = 0 ; i < 3 ; i++)
169 vecFrom[i] = READ_COORD();
170
171 UpdateTiles(gHUD.m_flTime, bitsDamage);
172
173 // Actually took damage?
174 if ( damageTaken > 0 || armor > 0 )
175 CalcDamageDirection(vecFrom);
176
177 return 1;

223int CHudHealth::Draw(float flTime)
224{
225 int r, g, b;
226 int a = 0, x, y;
227 int HealthWidth;
228
229 if ( (gHUD.m_iHideHUDDisplay & HIDEHUD_HEALTH) || gEngfuncs.IsSpectateOnly() )
230 return 1;
231
232 if ( !m_hSprite )
233 m_hSprite = LoadSprite(PAIN_NAME);
234
235 // Has health changed? Flash the health #
236 if (m_fFade)
237 {
238 m_fFade -= (gHUD.m_flTimeDelta * 20);
239 if (m_fFade <= 0)
240 {
241 a = MIN_ALPHA;
242 m_fFade = 0;
243 }
244
245 // Fade the health number back to dim
246
247 a = MIN_ALPHA + (m_fFade/FADE_TIME) * 128;
248
249 }
250 else
251 a = MIN_ALPHA;
252
253 // If health is getting low, make it bright red
254 if (m_iHealth <= 15)
255 a = 255;
256
257 GetPainColor( r, g, b );
258 ScaleColors(r, g, b, a );
259
260 // Only draw health if we have the suit.
261 if (gHUD.m_iWeaponBits & (1<<(WEAPON_SUIT)))
262 {
263 HealthWidth = gHUD.GetSpriteRect(gHUD.m_HUD_number_0).right - gHUD.GetSpriteRect(gHUD.m_HUD_number_0).left;
264 int CrossWidth = gHUD.GetSpriteRect(m_HUD_cross).right - gHUD.GetSpriteRect(m_HUD_cross).left;
265
266 y = ScreenHeight - gHUD.m_iFontHeight - gHUD.m_iFontHeight / 2;
267 x = CrossWidth /2;
268
269 SPR_Set(gHUD.GetSprite(m_HUD_cross), r, g, b);
270 SPR_DrawAdditive(0, x, y, &gHUD.GetSpriteRect(m_HUD_cross));
271
272 x = CrossWidth + HealthWidth / 2;
273
274 x = gHUD.DrawHudNumber(x, y, DHN_3DIGITS | DHN_DRAWZERO, m_iHealth, r, g, b);
275
276 x += HealthWidth/2;
277
278 int iHeight = gHUD.m_iFontHeight;
279 int iWidth = HealthWidth/10;
280 FillRGBA(x, y, iWidth, iHeight, 255, 160, 0, a);
281 }
282
283 DrawDamage(flTime);
284 return DrawPain(flTime);
285}

294void CHudHealth::CalcDamageDirection(vec3_t vecFrom)
295{
296 vec3_t forward, right, up;
297 float side, front;
298 vec3_t vecOrigin, vecAngles;
299
300 if (!vecFrom[0] && !vecFrom[1] && !vecFrom[2])
301 {
302 m_fAttackFront = m_fAttackRear = m_fAttackRight = m_fAttackLeft = 0;
303 return;
304 }
305
306
307 memcpy(vecOrigin, gHUD.m_vecOrigin, sizeof(vec3_t));
308 memcpy(vecAngles, gHUD.m_vecAngles, sizeof(vec3_t));
309
310
311 VectorSubtract (vecFrom, vecOrigin, vecFrom);
312
313 float flDistToTarget = vecFrom.Length();
314
315 vecFrom = vecFrom.Normalize();
316 AngleVectors (vecAngles, forward, right, up);
317
318 front = DotProduct (vecFrom, right);
319 side = DotProduct (vecFrom, forward);
320
321 if (flDistToTarget <= 50)
322 {
323 m_fAttackFront = m_fAttackRear = m_fAttackRight = m_fAttackLeft = 1;
324 }
325 else
326 {
327 if (side > 0)
328 {
329 if (side > 0.3)
330 m_fAttackFront = max(m_fAttackFront, side);
331 }
332 else
333 {
334 float f = fabs(side);
335 if (f > 0.3)
336 m_fAttackRear = max(m_fAttackRear, f);
337 }
338
339 if (front > 0)
340 {
341 if (front > 0.3)
342 m_fAttackRight = max(m_fAttackRight, front);
343 }
344 else
345 {
346 float f = fabs(front);
347 if (f > 0.3)
348 m_fAttackLeft = max(m_fAttackLeft, f);
349 }
350 }

443int CHudHealth::DrawDamage(float flTime)
444{
445 int r, g, b, a;
446 DAMAGE_IMAGE *pdmg;
447
448 if (!m_bitsDamage)
449 return 1;
450
451 UnpackRGB(r,g,b, RGB_YELLOWISH);
452
453 a = (int)( fabs(sin(flTime*2)) * 256.0);
454
455 ScaleColors(r, g, b, a);
456
457 // Draw all the items
458 int i;
459 for ( i = 0; i < NUM_DMG_TYPES; i++)
460 {
461 if (m_bitsDamage & giDmgFlags[i])
462 {
463 pdmg = &m_dmg[i];
464 SPR_Set(gHUD.GetSprite(m_HUD_dmg_bio + i), r, g, b );
465 SPR_DrawAdditive(0, pdmg->x, pdmg->y, &gHUD.GetSpriteRect(m_HUD_dmg_bio + i));
466 }
467 }
468
469
470 // check for bits that should be expired
471 for ( i = 0; i < NUM_DMG_TYPES; i++ )
472 {
473 DAMAGE_IMAGE *pdmg = &m_dmg[i];
474
475 if ( m_bitsDamage & giDmgFlags[i] )
476 {
477 pdmg->fExpire = min( flTime + DMG_IMAGE_LIFE, pdmg->fExpire );
478
479 if ( pdmg->fExpire <= flTime // when the time has expired
480 && a < 40 ) // and the flash is at the low point of the cycle
481 {
482 pdmg->fExpire = 0;
483
484 int y = pdmg->y;
485 pdmg->x = pdmg->y = 0;
486
487 // move everyone above down
488 for (int j = 0; j < NUM_DMG_TYPES; j++)
489 {
490 pdmg = &m_dmg[j];
491 if ((pdmg->y) && (pdmg->y < y))
492 pdmg->y += giDmgHeight;
493
494 }
495
496 m_bitsDamage &= ~giDmgFlags[i]; // clear the bits
497 }
498 }
499 }
500
501 return 1;
502}

513void CHudHealth::UpdateTiles(float flTime, long bitsDamage)
514{
515 DAMAGE_IMAGE *pdmg;
516
517 // Which types are new?
518 long bitsOn = ~m_bitsDamage & bitsDamage;
519
520 for (int i = 0; i < NUM_DMG_TYPES; i++)
521 {
522 pdmg = &m_dmg[i];
523
524 // Is this one already on?
525 if (m_bitsDamage & giDmgFlags[i])
526 {
527 pdmg->fExpire = flTime + DMG_IMAGE_LIFE; // extend the duration
528 if (!pdmg->fBaseline)
529 pdmg->fBaseline = flTime;
530 }
531
532 // Are we just turning it on?
533 if (bitsOn & giDmgFlags[i])
534 {
535 // put this one at the bottom
536 pdmg->x = giDmgWidth/8;
537 pdmg->y = ScreenHeight - giDmgHeight * 2;
538 pdmg->fExpire=flTime + DMG_IMAGE_LIFE;
539
540 // move everyone else up
541 for (int j = 0; j < NUM_DMG_TYPES; j++)
542 {
543 if (j == i)
544 continue;
545
546 pdmg = &m_dmg[j];
547 if (pdmg->y)
548 pdmg->y -= giDmgHeight;
549
550 }
551 pdmg = &m_dmg[i];
552 }
553 }
554
555 // damage bits are only turned on here; they are turned off when the draw time has expired (in DrawDamage())
556 m_bitsDamage |= bitsDamage;
557}
