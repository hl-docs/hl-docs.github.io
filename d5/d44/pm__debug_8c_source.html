121void PM_ParticleLine(vec3_t start, vec3_t end, int pcolor, float life, float vert)
122{
123 float linestep = 2.0f;
124 float curdist;
125 float len;
126 vec3_t curpos;
127 vec3_t diff;
128 int i;
129 // Determine distance;
130
131 VectorSubtract(end, start, diff);
132
133 len = VectorNormalize(diff);
134
135 curdist = 0;
136 while (curdist <= len)
137 {
138 for (i = 0; i < 3; i++)
139 curpos[i] = start[i] + curdist * diff[i];
140
141 pmove->PM_Particle( curpos, pcolor, life, 0, vert);
142 curdist += linestep;
143 }
144
145}

178void PM_DrawPhysEntBBox(int num, int pcolor, float life)
179{
180 physent_t *pe;
181 vec3_t org;
182 int j;
183 vec3_t tmp;
184 vec3_t p[8];
185 float gap = BOX_GAP;
186 vec3_t modelmins, modelmaxs;
187
188 if (num >= pmove->numphysent ||
189 num <= 0)
190 return;
191
192 pe = &pmove->physents[num];
193
194 if (pe->model)
195 {
196 VectorCopy(pe->origin, org);
197
198 pmove->PM_GetModelBounds( pe->model, modelmins, modelmaxs );
199 for (j = 0; j < 8; j++)
200 {
201 tmp[0] = (j & 1) ? modelmins[0] - gap : modelmaxs[0] + gap;
202 tmp[1] = (j & 2) ? modelmins[1] - gap : modelmaxs[1] + gap;
203 tmp[2] = (j & 4) ? modelmins[2] - gap : modelmaxs[2] + gap;
204
205 VectorCopy(tmp, p[j]);
206 }
207
208 // If the bbox should be rotated, do that
209 if (pe->angles[0] || pe->angles[1] || pe->angles[2])
210 {
211 vec3_t forward, right, up;
212
213 AngleVectorsTranspose(pe->angles, forward, right, up);
214 for (j = 0; j < 8; j++)
215 {
216 VectorCopy(p[j], tmp);
217 p[j][0] = DotProduct ( tmp, forward );
218 p[j][1] = DotProduct ( tmp, right );
219 p[j][2] = DotProduct ( tmp, up );
220 }
221 }
222
223 // Offset by entity origin, if any.
224 for (j = 0; j < 8; j++)
225 VectorAdd(p[j], org, p[j]);
226
227 for (j = 0; j < 6; j++)
228 {
229 PM_DrawRectangle(
230 p[PM_boxpnt[j][1]],
231 p[PM_boxpnt[j][0]],
232 p[PM_boxpnt[j][2]],
233 p[PM_boxpnt[j][3]],
234 pcolor, life);
235 }
236 }
237 else
238 {
239 for (j = 0; j < 8; j++)
240 {
241 tmp[0] = (j & 1) ? pe->mins[0] : pe->maxs[0];
242 tmp[1] = (j & 2) ? pe->mins[1] : pe->maxs[1];
243 tmp[2] = (j & 4) ? pe->mins[2] : pe->maxs[2];
244
245 VectorAdd(tmp, pe->origin, tmp);
246 VectorCopy(tmp, p[j]);
247 }
248
249 for (j = 0; j < 6; j++)
250 {
251 PM_DrawRectangle(
252 p[PM_boxpnt[j][1]],
253 p[PM_boxpnt[j][0]],
254 p[PM_boxpnt[j][2]],
255 p[PM_boxpnt[j][3]],
256 pcolor, life);
257 }
258
259 }
260}
