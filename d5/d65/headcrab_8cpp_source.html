244// HandleAnimEvent - catches the monster-specific messages
245// that occur when tagged animation frames are played.
246//=========================================================
256void CHeadCrab :: HandleAnimEvent( MonsterEvent_t *pEvent )
257{
258 switch( pEvent->event )
259 {
260 case HC_AE_JUMPATTACK:
261 {
262 ClearBits( pev->flags, FL_ONGROUND );
263
264 UTIL_SetOrigin (pev, pev->origin + Vector ( 0 , 0 , 1) );// take him off ground so engine doesn't instantly reset onground
265 UTIL_MakeVectors ( pev->angles );
266
267 Vector vecJumpDir;
268 if (m_hEnemy != NULL)
269 {
270 float gravity = g_psv_gravity->value;
271 if (gravity <= 1)
272 gravity = 1;
273
274 // How fast does the headcrab need to travel to reach that height given gravity?
275 float height = (m_hEnemy->pev->origin.z + m_hEnemy->pev->view_ofs.z - pev->origin.z);
276 if (height < 16)
277 height = 16;
278 float speed = sqrt( 2 * gravity * height );
279 float time = speed / gravity;
280
281 // Scale the sideways velocity to get there at the right time
282 vecJumpDir = (m_hEnemy->pev->origin + m_hEnemy->pev->view_ofs - pev->origin);
283 vecJumpDir = vecJumpDir * ( 1.0 / time );
284
285 // Speed to offset gravity at the desired height
286 vecJumpDir.z = speed;
287
288 // Don't jump too far/fast
289 float distance = vecJumpDir.Length();
290
291 if (distance > 650)
292 {
293 vecJumpDir = vecJumpDir * ( 650.0 / distance );
294 }
295 }
296 else
297 {
298 // jump hop, don't care where
299 vecJumpDir = Vector( gpGlobals->v_forward.x, gpGlobals->v_forward.y, gpGlobals->v_up.z ) * 350;
300 }
301
302 int iSound = RANDOM_LONG(0,2);
303 if ( iSound != 0 )
304 EMIT_SOUND_DYN( edict(), CHAN_VOICE, pAttackSounds[iSound], GetSoundVolue(), ATTN_IDLE, 0, GetVoicePitch() );
305
306 pev->velocity = vecJumpDir;
307 m_flNextAttack = gpGlobals->time + 2;
308 }
309 break;
310
311 default:
312 CBaseMonster::HandleAnimEvent( pEvent );
313 break;
314 }
315}

402// LeapTouch - this is the headcrab's touch function when it
403// is in the air
404//=========================================================
413void CHeadCrab :: LeapTouch ( CBaseEntity *pOther )
414{
415 if ( !pOther->pev->takedamage )
416 {
417 return;
418 }
419
420 if ( pOther->Classify() == Classify() )
421 {
422 return;
423 }
424
425 // Don't hit if back on ground
426 if ( !FBitSet( pev->flags, FL_ONGROUND ) )
427 {
428 EMIT_SOUND_DYN( edict(), CHAN_WEAPON, RANDOM_SOUND_ARRAY(pBiteSounds), GetSoundVolue(), ATTN_IDLE, 0, GetVoicePitch() );
429
430 pOther->TakeDamage( pev, pev, GetDamageAmount(), DMG_SLASH );
431 }
432
433 SetTouch( NULL );
434}

695BOOL CBabyCrab::CheckRangeAttack1(float flDot, float flDist)
696{
697 if (pev->flags & FL_ONGROUND) // Check if the baby headcrab is on the ground
698 {
699 if (pev->groundentity && (pev->groundentity->v.flags & (FL_CLIENT | FL_MONSTER))) // Check if the ground entity is a client or monster
