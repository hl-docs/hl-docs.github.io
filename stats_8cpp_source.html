36float AmmoDamage(const char *pName)
37{
38 if (!pName)
39 return 0; // Return 0 if the ammunition name is NULL
40
41 // Check the ammunition name and return the corresponding damage value
42 if (!strcmp(pName, "9mm"))
43 return gSkillData.plrDmg9MM;
44 if (!strcmp(pName, "357"))
45 return gSkillData.plrDmg357;
46 if (!strcmp(pName, "ARgrenades"))
47 return gSkillData.plrDmgM203Grenade;
48 if (!strcmp(pName, "buckshot"))
49 return gSkillData.plrDmgBuckshot;
50 if (!strcmp(pName, "bolts"))
51 return gSkillData.plrDmgCrossbowMonster;
52 if (!strcmp(pName, "rockets"))
53 return gSkillData.plrDmgRPG;
54 if (!strcmp(pName, "uranium"))
55 return gSkillData.plrDmgGauss;
56 if (!strcmp(pName, "Hand Grenade"))
57 return gSkillData.plrDmgHandGrenade;
58 if (!strcmp(pName, "Satchel Charge"))
59 return gSkillData.plrDmgSatchel;
60 if (!strcmp(pName, "Trip Mine"))
61 return gSkillData.plrDmgTripmine;
62
63 return 0; // Return 0 if the ammunition name is not recognized
64}

138void UpdateStats(CBasePlayer *pPlayer)
139{
140 int i;
141
142 int ammoCount[MAX_AMMO_SLOTS];
143 memcpy(ammoCount, pPlayer->m_rgAmmo, MAX_AMMO_SLOTS * sizeof(int));
144
145 // Keep a running time, so the graph doesn't overlap
146 if (gpGlobals->time < gStats.lastGameTime) // Changed level or died, don't b0rk
147 {
148 gStats.lastGameTime = gpGlobals->time;
149 gStats.dataTime = gStats.gameTime;
150 }
151
152 gStats.gameTime += gpGlobals->time - gStats.lastGameTime;
153 gStats.lastGameTime = gpGlobals->time;
154
155 for (i = 0; i < MAX_ITEM_TYPES; i++)
156 {
157 CBasePlayerItem *p = pPlayer->m_rgpPlayerItems[i];
158 while (p)
159 {
160 ItemInfo II;
161
162 memset(&II, 0, sizeof(II));
163 p->GetItemInfo(&II);
164
165 int index = pPlayer->GetAmmoIndex(II.pszAmmo1);
166 if (index >= 0)
167 ammoCount[index] += ((CBasePlayerWeapon *)p)->m_iClip;
168
169 p = p->m_pNext;
170 }
171 }
172
173 float ammo = 0;
174 for (i = 1; i < MAX_AMMO_SLOTS; i++)
175 {
176 ammo += ammoCount[i] * AmmoDamage(CBasePlayerItem::AmmoInfoArray[i].pszName);
177 }
178
179 float health = pPlayer->pev->health + pPlayer->pev->armorvalue * 2; // Armor is 2X health
180 float ammoDelta = fabs(ammo - gStats.lastAmmo);
181 float healthDelta = fabs(health - gStats.lastHealth);
182 int forceWrite = 0;
183 if (health <= 0 && gStats.lastHealth > 0)
184 forceWrite = 1;
185
186 if ((ammoDelta > AMMO_THRESHOLD || healthDelta > HEALTH_THRESHOLD) && !forceWrite)
187 {
188 if (gStats.nextOutputTime == 0)
189 gStats.dataTime = gStats.gameTime;
190
191 gStats.lastAmmo = ammo;
192 gStats.lastHealth = health;
193
194 gStats.nextOutputTime = gStats.gameTime + OUTPUT_LATENCY;
195 }
196 else if ((gStats.nextOutputTime != 0 && gStats.nextOutputTime < gStats.gameTime) || forceWrite)
197 {
198 UpdateStatsFile(gStats.dataTime, (char *)STRING(gpGlobals->mapname), health, ammo, (int)CVAR_GET_FLOAT("skill"));
199
200 gStats.lastAmmo = ammo;
201 gStats.lastHealth = health;
202 gStats.lastOutputTime = gStats.gameTime;
203 gStats.nextOutputTime = 0;
204 }
205}
