65void CBaseSpectator::SpectatorImpulseCommand(void)
66{
67 static edict_t *pGoal = NULL;
68 edict_t *pPreviousGoal;
69 edict_t *pCurrentGoal;
70 BOOL bFound;
71
72 switch (pev->impulse)
73 {
74 case 1:
75 // teleport the spectator to the next spawn point
76 // note that if the spectator is tracking, this doesn't do
77 // much
78 pPreviousGoal = pGoal;
79 pCurrentGoal = pGoal;
80 // Start at the current goal, skip the world, and stop if we looped
81 // back around
82
83 bFound = FALSE;
84 while (1)
85 {
86 pCurrentGoal = FIND_ENTITY_BY_CLASSNAME(pCurrentGoal, "info_player_deathmatch");
87 // Looped around, failure
88 if (pCurrentGoal == pPreviousGoal)
89 {
90 ALERT(at_console, "Could not find a spawn spot.\n");
91 break;
92 }
93 // Found a non-world entity, set success, otherwise, look for the next one.
94 if (!FNullEnt(pCurrentGoal))
95 {
96 bFound = TRUE;
97 break;
98 }
99 }
100
101 if (!bFound) // Didn't find a good spot.
102 break;
103
104 pGoal = pCurrentGoal;
105 UTIL_SetOrigin( pev, pGoal->v.origin );
106 pev->angles = pGoal->v.angles;
107 pev->fixangle = FALSE;
108 break;
109 default:
110 ALERT(at_console, "Unknown spectator impulse\n");
111 break;
112 }
113
114 pev->impulse = 0;
115}
