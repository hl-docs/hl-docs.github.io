66void CGrenade::Explode(TraceResult *pTrace, int bitsDamageType)
67{
68 float flRndSound; // Sound randomizer
69
70 pev->model = iStringNull; // Make the grenade invisible
71 pev->solid = SOLID_NOT; // Set to intangible
72
73 pev->takedamage = DAMAGE_NO; // Disable damage
74
75 // Pull out of the wall a bit
76 if (pTrace->flFraction != 1.0)
77 {
78 pev->origin = pTrace->vecEndPos + (pTrace->vecPlaneNormal * (pev->dmg - 24) * 0.6); // Adjust position
79 }
80
81 int iContents = UTIL_PointContents(pev->origin); // Get contents at the explosion point
82
83 // Send explosion message
84 MESSAGE_BEGIN(MSG_PAS, SVC_TEMPENTITY, pev->origin);
85 WRITE_BYTE(TE_EXPLOSION); // Explosion effect
86 WRITE_COORD(pev->origin.x); // X coordinate
87 WRITE_COORD(pev->origin.y); // Y coordinate
88 WRITE_COORD(pev->origin.z); // Z coordinate
89 if (iContents != CONTENTS_WATER)
90 {
91 WRITE_SHORT(g_sModelIndexFireball); // Fireball model
92 }
93 else
94 {
95 WRITE_SHORT(g_sModelIndexWExplosion); // Water explosion model
96 }
97 WRITE_BYTE((pev->dmg - 50) * .60); // Scale damage
98 WRITE_BYTE(15); // Framerate
99 WRITE_BYTE(TE_EXPLFLAG_NONE); // No special flags
100 MESSAGE_END();
101
102 CSoundEnt::InsertSound(bits_SOUND_COMBAT, pev->origin, NORMAL_EXPLOSION_VOLUME, 3.0); // Insert sound for the explosion
103 entvars_t *pevOwner;
104 if (pev->owner)
105 pevOwner = VARS(pev->owner); // Get the owner of the grenade
106 else
107 pevOwner = NULL;
108
109 pev->owner = NULL; // Clear owner to prevent tracing back to the owner
110
111 RadiusDamage(pev, pevOwner, pev->dmg, CLASS_NONE, bitsDamageType); // Apply damage in radius
112
113 // Create scorch marks
114 if (RANDOM_FLOAT(0, 1) < 0.5)
115 {
116 UTIL_DecalTrace(pTrace, DECAL_SCORCH1);
117 }
118 else
119 {
120 UTIL_DecalTrace(pTrace, DECAL_SCORCH2);
121 }
122
123 flRndSound = RANDOM_FLOAT(0, 1); // Random sound selection
124
125 // Play explosion sound
126 switch (RANDOM_LONG(0, 2))
127 {
128 case 0: EMIT_SOUND(ENT(pev), CHAN_VOICE, "weapons/debris1.wav", 0.55, ATTN_NORM); break;
129 case 1: EMIT_SOUND(ENT(pev), CHAN_VOICE, "weapons/debris2.wav", 0.55, ATTN_NORM); break;
130 case 2: EMIT_SOUND(ENT(pev), CHAN_VOICE, "weapons/debris3.wav", 0.55, ATTN_NORM); break;
131 }
132
133 pev->effects |= EF_NODRAW; // Set to not draw
134 SetThink(&CGrenade::Smoke); // Set the next think function to smoke
135 pev->velocity = g_vecZero; // Stop movement
136 pev->nextthink = gpGlobals->time + 0.3; // Set next think time
137
138 // Create sparks if not in water
139 if (iContents != CONTENTS_WATER)
140 {
141 int sparkCount = RANDOM_LONG(0, 3);
142 for (int i = 0; i < sparkCount; i++)
143 Create("spark_shower", pev->origin, pTrace->vecPlaneNormal, NULL); // Create spark shower
144 }
145}

154{
155 if (UTIL_PointContents(pev->origin) == CONTENTS_WATER)
156 {
157 UTIL_Bubbles(pev->origin - Vector(64, 64, 64), pev->origin + Vector(64, 64, 64), 100); // Create bubbles in water

162 WRITE_BYTE(TE_SMOKE); // Smoke effect
163 WRITE_COORD(pev->origin.x); // X coordinate
164 WRITE_COORD(pev->origin.y); // Y coordinate
165 WRITE_COORD(pev->origin.z); // Z coordinate
166 WRITE_SHORT(g_sModelIndexSmoke); // Smoke model
167 WRITE_BYTE((pev->dmg - 50) * 0.80); // Scale damage
168 WRITE_BYTE(12); // Framerate
169 MESSAGE_END();
170 }
171 UTIL_Remove(this); // Remove the grenade entity
172}

288void CGrenade::BounceTouch(CBaseEntity *pOther)
289{
290 // Don't hit the guy that launched this grenade
291 if (pOther->edict() == pev->owner)
292 return;
293
294 // Only do damage if we're moving fairly fast
295 if (m_flNextAttack < gpGlobals->time && pev->velocity.Length() > 100)
296 {
297 entvars_t *pevOwner = VARS(pev->owner);
298 if (pevOwner)
299 {
300 TraceResult tr = UTIL_GetGlobalTrace();
301 ClearMultiDamage();
302 pOther->TraceAttack(pevOwner, 1, gpGlobals->v_forward, &tr, DMG_CLUB); // Apply damage
303 ApplyMultiDamage(pev, pevOwner); // Apply multi damage
304 }
305 m_flNextAttack = gpGlobals->time + 1.0; // Debounce
306 }
307
308 Vector vecTestVelocity;
309
310 vecTestVelocity = pev->velocity;
311 vecTestVelocity.z *= 0.45; // Adjust Z velocity
312
313 if (!m_fRegisteredSound && vecTestVelocity.Length() <= 60)
314 {
315 // Grenade is moving really slow. Emit danger sound.
316 CSoundEnt::InsertSound(bits_SOUND_DANGER, pev->origin, pev->dmg / 0.4, 0.3); // Register danger sound
317 m_fRegisteredSound = TRUE; // Mark sound as registered
318 }
319
320 if (pev->flags & FL_ONGROUND)
321 {
322 // Add a bit of static friction
323 pev->velocity = pev->velocity * 0.8; // Reduce velocity
324
325 pev->sequence = RANDOM_LONG(1, 1); // Set sequence
326 }
327 else
328 {
329 // Play bounce sound
330 BounceSound();
331 }
332 pev->framerate = pev->velocity.Length() / 200.0; // Set framerate based on velocity
333 if (pev->framerate > 1.0)
334 pev->framerate = 1; // Cap framerate
335 else if (pev->framerate < 0.5)
336 pev->framerate = 0; // Minimum framerate
337}

391void CGrenade::TumbleThink(void)
392{
393 if (!IsInWorld())
394 {
395 UTIL_Remove(this); // Remove the grenade if not in the world
396 return;
397 }
398
399 StudioFrameAdvance(); // Advance the animation frame
400 pev->nextthink = gpGlobals->time + 0.1; // Set the next think time
401
402 if (pev->dmgtime - 1 < gpGlobals->time)
403 {
404 CSoundEnt::InsertSound(bits_SOUND_DANGER, pev->origin + pev->velocity * (pev->dmgtime - gpGlobals->time), 400, 0.1); // Insert danger sound
405 }
406
407 if (pev->dmgtime <= gpGlobals->time)
408 {
409 SetThink(&CGrenade::Detonate); // Set think function to detonate
410 }
411 if (pev->waterlevel != 0)
412 {
413 pev->velocity = pev->velocity * 0.5; // Reduce velocity in water
414 pev->framerate = 0.2; // Set framerate in water
415 }

568void CGrenade::UseSatchelCharges(entvars_t *pevOwner, SATCHELCODE code)
569{
570 edict_t *pentFind;
571 edict_t *pentOwner;
572
573 if (!pevOwner)
574 return; // Exit if no owner
575
576 CBaseEntity *pOwner = CBaseEntity::Instance(pevOwner); // Get the owner entity
577
578 pentOwner = pOwner->edict(); // Get the owner's edict
579
580 pentFind = FIND_ENTITY_BY_CLASSNAME(NULL, "grenade"); // Find the first grenade entity
581 while (!FNullEnt(pentFind))
582 {
583 CBaseEntity *pEnt = Instance(pentFind); // Get the entity instance
584 if (pEnt)
585 {
586 if (FBitSet(pEnt->pev->spawnflags, SF_DETONATE) && pEnt->pev->owner == pentOwner) // Check if it's a detonating satchel owned by the player
587 {
588 if (code == SATCHEL_DETONATE)
589 pEnt->Use(pOwner, pOwner, USE_ON, 0); // Detonate the satchel
590 else // SATCHEL_RELEASE
591 pEnt->pev->owner = NULL; // Release the satchel
592 }
593 }
594 pentFind = FIND_ENTITY_BY_CLASSNAME(pentFind, "grenade"); // Find the next grenade entity
595 }
596}
