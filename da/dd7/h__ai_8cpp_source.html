75BOOL FBoxVisible(entvars_t *pevLooker, entvars_t *pevTarget, Vector &vecTargetOrigin, float flSize)
76{
77 // Don't look through water
78 if ((pevLooker->waterlevel != 3 && pevTarget->waterlevel == 3)
79 || (pevLooker->waterlevel == 3 && pevTarget->waterlevel == 0))
80 return FALSE;
81
82 TraceResult tr;
83 Vector vecLookerOrigin = pevLooker->origin + pevLooker->view_ofs; // Look through the monster's 'eyes'
84 for (int i = 0; i < 5; i++)
85 {
86 Vector vecTarget = pevTarget->origin;
87 vecTarget.x += RANDOM_FLOAT(pevTarget->mins.x + flSize, pevTarget->maxs.x - flSize);
88 vecTarget.y += RANDOM_FLOAT(pevTarget->mins.y + flSize, pevTarget->maxs.y - flSize);
89 vecTarget.z += RANDOM_FLOAT(pevTarget->mins.z + flSize, pevTarget->maxs.z - flSize);
90
91 UTIL_TraceLine(vecLookerOrigin, vecTarget, ignore_monsters, ignore_glass, ENT(pevLooker), &tr);
92
93 if (tr.flFraction == 1.0)
94 {
95 vecTargetOrigin = vecTarget; // Set the target origin if visible
96 return TRUE; // Line of sight is valid
97 }
98 }
99 return FALSE; // Line of sight is not established
100}

115Vector VecCheckToss(entvars_t *pev, const Vector &vecSpot1, Vector vecSpot2, float flGravityAdj)
116{
117 TraceResult tr;
118 Vector vecMidPoint; // Halfway point between Spot1 and Spot2
119 Vector vecApex; // Highest point
120 Vector vecGrenadeVel; // Calculated grenade velocity
121 float flGravity = g_psv_gravity->value * flGravityAdj; // Adjusted gravity
122
123 if (vecSpot2.z - vecSpot1.z > 500)
124 {
125 // Too high, fail
126 return g_vecZero;
127 }
128
129 UTIL_MakeVectors(pev->angles); // Make vectors based on entity's angles
130
131 // Toss a little bit to the left or right, not directly on the enemy's head
132 vecSpot2 = vecSpot2 + gpGlobals->v_right * (RANDOM_FLOAT(-8, 8) + RANDOM_FLOAT(-16, 16));
133 vecSpot2 = vecSpot2 + gpGlobals->v_forward * (RANDOM_FLOAT(-8, 8) + RANDOM_FLOAT(-16, 16));
134
135 // Calculate the midpoint and apex of the 'triangle'
136 vecMidPoint = vecSpot1 + (vecSpot2 - vecSpot1) * 0.5; // Midpoint between the two spots
137 UTIL_TraceLine(vecMidPoint, vecMidPoint + Vector(0, 0, 500), ignore_monsters, ENT(pev), &tr);
138 vecMidPoint = tr.vecEndPos; // Adjust midpoint based on trace result
139 vecMidPoint.z -= 15; // Subtract 15 to avoid hitting the ceiling
140
141 if (vecMidPoint.z < vecSpot1.z || vecMidPoint.z < vecSpot2.z)
142 {
143 // Not enough space, fail
144 return g_vecZero;
145 }
146
147 // Calculate the height the grenade should travel to reach the apex
148 float distance1 = (vecMidPoint.z - vecSpot1.z);
149 float distance2 = (vecMidPoint.z - vecSpot2.z);
150
151 // Calculate the time it will take for the grenade to travel this distance
152 float time1 = sqrt(distance1 / (0.5 * flGravity));
153 float time2 = sqrt(distance2 / (0.5 * flGravity));
154
155 if (time1 < 0.1)
156 {
157 // Too close
158 return g_vecZero;
159 }
160
161 // Calculate the velocity needed to throw the grenade
162 vecGrenadeVel = (vecSpot2 - vecSpot1) / (time1 + time2); // Sideways velocity
163 vecGrenadeVel.z = flGravity * time1; // Upward velocity to reach the apex
164
165 // Find the apex
166 vecApex = vecSpot1 + vecGrenadeVel * time1;
167 vecApex.z = vecMidPoint.z; // Set apex height
168
169 UTIL_TraceLine(vecSpot1, vecApex, dont_ignore_monsters, ENT(pev), &tr);
170 if (tr.flFraction != 1.0)
171 {
172 // Fail!
173 return g_vecZero;
174 }
175
176 // Check if the toss is feasible
177 UTIL_TraceLine(vecSpot2, vecApex, ignore_monsters, ENT(pev), &tr);
178 if (tr.flFraction != 1.0)
179 {
180 // Fail!
181 return g_vecZero;
182 }
183
184 return vecGrenadeVel; // Return the calculated velocity
185}

200Vector VecCheckThrow(entvars_t *pev, const Vector &vecSpot1, Vector vecSpot2, float flSpeed, float flGravityAdj)
201{
202 float flGravity = g_psv_gravity->value * flGravityAdj; // Adjusted gravity
203
204 Vector vecGrenadeVel = (vecSpot2 - vecSpot1); // Calculate the vector to the target
205
206 // Throw at a constant time
207 float time = vecGrenadeVel.Length() / flSpeed; // Calculate time to reach the target
208 vecGrenadeVel = vecGrenadeVel * (1.0 / time); // Normalize the velocity vector
209
210 // Adjust upward toss to compensate for gravity loss
211 vecGrenadeVel.z += flGravity * time * 0.5; // Adjust for gravity
212
213 Vector vecApex = vecSpot1 + (vecSpot2 - vecSpot1) * 0.5; // Calculate the apex
214 vecApex.z += 0.5 * flGravity * (time * 0.5) * (time * 0.5); // Adjust apex height
215
216 TraceResult tr;
217 UTIL_TraceLine(vecSpot1, vecApex, dont_ignore_monsters, ENT(pev), &tr);
218 if (tr.flFraction != 1.0)
219 {
220 // Fail!
221 return g_vecZero;
222 }
223
224 UTIL_TraceLine(vecSpot2, vecApex, ignore_monsters, ENT(pev), &tr);
225 if (tr.flFraction != 1.0)
226 {
227 // Fail!
228 return g_vecZero;
229 }
230
231 return vecGrenadeVel; // Return the calculated velocity
232}
