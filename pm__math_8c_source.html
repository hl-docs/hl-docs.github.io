65void AngleVectors (const vec3_t angles, vec3_t forward, vec3_t right, vec3_t up)
66{
67 float angle;
68 float sr, sp, sy, cr, cp, cy;
69
70 angle = angles[YAW] * (M_PI*2 / 360);
71 sy = sin(angle);
72 cy = cos(angle);
73 angle = angles[PITCH] * (M_PI*2 / 360);
74 sp = sin(angle);
75 cp = cos(angle);
76 angle = angles[ROLL] * (M_PI*2 / 360);
77 sr = sin(angle);
78 cr = cos(angle);
79
80 if (forward)
81 {
82 forward[0] = cp*cy;
83 forward[1] = cp*sy;
84 forward[2] = -sp;
85 }
86 if (right)
87 {
88 right[0] = (-1*sr*sp*cy+-1*cr*-sy);
89 right[1] = (-1*sr*sp*sy+-1*cr*cy);
90 right[2] = -1*sr*cp;
91 }
92 if (up)
93 {
94 up[0] = (cr*sp*cy+-sr*-sy);
95 up[1] = (cr*sp*sy+-sr*cy);
96 up[2] = cr*cp;
97 }
98}

262void InterpolateAngles( float *start, float *end, float *output, float frac )
263{
264 int i;
265 float ang1, ang2;
266 float d;
267
268 NormalizeAngles( start );
269 NormalizeAngles( end );
270
271 for ( i = 0 ; i < 3 ; i++ )
272 {
273 ang1 = start[i];
274 ang2 = end[i];
275
276 d = ang2 - ang1;
277 if ( d > 180 )
278 {
279 d -= 360;
280 }
281 else if ( d < -180 )
282 {
283 d += 360;
284 }
285
286 output[i] = ang1 + d * frac;
287 }
288
289 NormalizeAngles( output );
290}

303float AngleBetweenVectors( const vec3_t v1, const vec3_t v2 )
304{
305 float angle;
306 float l1 = Length( v1 );
307 float l2 = Length( v2 );
308
309 if ( !l1 || !l2 )
310 return 0.0f;
311
312 angle = acos( DotProduct( v1, v2 ) ) / (l1*l2);
313 angle = ( angle * 180.0f ) / M_PI;
314
315 return angle;
316}

591void VectorMatrix( vec3_t forward, vec3_t right, vec3_t up)

596 {
