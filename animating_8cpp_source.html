55float CBaseAnimating::StudioFrameAdvance(float flInterval)
56{
57 if (flInterval == 0.0)
58 {
59 flInterval = (gpGlobals->time - pev->animtime); // Calculate time since last update
60 if (flInterval <= 0.001)
61 {
62 pev->animtime = gpGlobals->time; // Reset animation time
63 return 0.0; // No advancement
64 }
65 }
66 if (!pev->animtime)
67 flInterval = 0.0; // No animation time set
68
69 pev->frame += flInterval * m_flFrameRate * pev->framerate; // Update the frame based on the interval
70 pev->animtime = gpGlobals->time; // Update the animation time
71
72 // Handle frame wrapping
73 if (pev->frame < 0.0 || pev->frame >= 256.0)
74 {
75 if (m_fSequenceLoops)
76 pev->frame -= (int)(pev->frame / 256.0) * 256.0; // Loop the frame
77 else
78 pev->frame = (pev->frame < 0.0) ? 0 : 255; // Clamp the frame
79 m_fSequenceFinished = TRUE; // Mark the sequence as finished
80 }
81
82 return flInterval; // Return the interval used
83}

174void CBaseAnimating::DispatchAnimEvents(float flInterval)
175{
176 MonsterEvent_t event; // Event structure
177
178 void *pmodel = GET_MODEL_PTR(ENT(pev)); // Get the model pointer
179
180 if (!pmodel)
181 {
182 ALERT(at_aiconsole, "Gibbed monster is thinking!\n"); // Alert if the model is missing
183 return; // Exit if no model
184 }
185
186 // Set a fixed interval to ensure events are processed
187 flInterval = 0.1;
188
189 // Calculate the start and end frames for event checking
190 float flStart = pev->frame + (m_flLastEventCheck - pev->animtime) * m_flFrameRate * pev->framerate;
191 float flEnd = pev->frame + flInterval * m_flFrameRate * pev->framerate;
192 m_flLastEventCheck = pev->animtime + flInterval; // Update last event check time
193
194 m_fSequenceFinished = FALSE; // Reset sequence finished flag
195 if (flEnd >= 256 || flEnd <= 0.0)
196 m_fSequenceFinished = TRUE; // Mark sequence as finished if out of bounds
197
198 int index = 0; // Initialize event index
199
200 // Process animation events within the specified frame range
201 while ((index = GetAnimationEvent(pmodel, pev, &event, flStart, flEnd, index)) != 0)
202 {
203 HandleAnimEvent(&event); // Handle the animation event
204 }
205}

373void CBaseAnimating::SetSequenceBox(void)
374{
375 Vector mins, maxs;
376
377 // Get sequence bounding box
378 if (ExtractBbox(pev->sequence, mins, maxs))
379 {
380 // Expand box for rotation
381 // Find min/max for rotations
382 float yaw = pev->angles.y * (M_PI / 180.0); // Convert yaw to radians
383
384 Vector xvector, yvector;
385 xvector.x = cos(yaw); // Calculate x vector
386 xvector.y = sin(yaw); // Calculate y vector
387 yvector.x = -sin(yaw); // Calculate y vector (perpendicular)
388 yvector.y = cos(yaw); // Calculate x vector (perpendicular)
389 Vector bounds[2];
390
391 bounds[0] = mins; // Set minimum bounds
392 bounds[1] = maxs; // Set maximum bounds
393
394 Vector rmin(9999, 9999, 9999); // Initialize rotated minimum bounds
395 Vector rmax(-9999, -9999, -9999); // Initialize rotated maximum bounds
396 Vector base, transformed;
397
398 for (int i = 0; i <= 1; i++)
399 {
400 base.x = bounds[i].x; // Set base x
401 for (int j = 0; j <= 1; j++)
402 {
403 base.y = bounds[j].y; // Set base y
404 for (int k = 0; k <= 1; k++)
405 {
406 base.z = bounds[k].z; // Set base z
407
408 // Transform the point
409 transformed.x = xvector.x * base.x + yvector.x * base.y; // Calculate transformed x
410 transformed.y = xvector.y * base.x + yvector.y * base.y; // Calculate transformed y
411 transformed.z = base.z; // Keep z unchanged
412
413 // Update rotated minimum and maximum bounds
414 if (transformed.x < rmin.x)
415 rmin.x = transformed.x;
416 if (transformed.x > rmax.x)
417 rmax.x = transformed.x;
418 if (transformed.y < rmin.y)
419 rmin.y = transformed.y;
420 if (transformed.y > rmax.y)
421 rmax.y = transformed.y;
422 if (transformed.z < rmin.z)
423 rmin.z = transformed.z;
424 if (transformed.z > rmax.z)
425 rmax.z = transformed.z;
426 }
427 }
428 }
429 rmin.z = 0; // Set minimum z to 0
430 rmax.z = rmin.z + 1; // Set maximum z to 1 above minimum
431 UTIL_SetSize(pev, rmin, rmax); // Set the size of the entity
432 }
433}
