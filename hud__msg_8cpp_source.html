52int CHud::MsgFunc_ResetHUD(const char *pszName, int iSize, void *pbuf)
53{
54 ASSERT( iSize == 0 );
55
56 // clear all hud data
57 HUDLIST *pList = m_pHudList;

60 {

65

105{

108

163int CHud::MsgFunc_Damage(const char *pszName, int iSize, void *pbuf)
164{
165 int armor, blood;

167 int i;
168 float count;
169
170 BEGIN_READ( pbuf, iSize );
171 armor = READ_BYTE();
172 blood = READ_BYTE();
173
174 for (i=0 ; i<3 ; i++)
175 from[i] = READ_COORD();
176
177 count = (blood * 0.5) + (armor * 0.5);
178
179 if (count < 10)
180 count = 10;
181
182 // TODO: kick viewangles, show damage visually
