106void CHudStatusBar::ParseStatusString( int line_num )
107{
108 char szBuffer[MAX_STATUSTEXT_LENGTH];
109 memset( szBuffer, 0, sizeof szBuffer );
110 gHUD.m_TextMessage.LocaliseTextString( m_szStatusText[line_num], szBuffer, MAX_STATUSTEXT_LENGTH );
111
112 memset( m_szStatusBar[line_num], 0, MAX_STATUSTEXT_LENGTH );
113 char *src = szBuffer;
114 char *dst = m_szStatusBar[line_num];
115
116 char *src_start = src, *dst_start = dst;
117
118 while ( *src != 0 )
119 {
120 while ( *src == '\n' )
121 src++; // skip newlines
122
123 if ( ((src - src_start) >= MAX_STATUSTEXT_LENGTH) || ((dst - dst_start) >= MAX_STATUSTEXT_LENGTH) )
124 break;
125
126 int index = atoi( src );
127 if ( (index >= 0 && index < MAX_STATUSBAR_VALUES) && (m_iStatusValues[index] != 0) )
128 {
129 while ( *src >= '0' && *src <= '9' )
130 src++;
131
132 if ( *src == '\n' || *src == 0 )
133 continue;
134
135 while ( *src != '\n' && *src != 0 )
136 {
137 if ( *src != '%' )
138 {
139 *dst = *src;
140 dst++, src++;
141 }
142 else
143 {
144 char valtype = *(++src);
145
146 if ( valtype == '%' )
147 {
148 *dst = valtype;
149 dst++, src++;
150 continue;
151 }
152
153 index = atoi( ++src );
154 while ( *src >= '0' && *src <= '9' )
155 src++;
156
157 if ( index >= 0 && index < MAX_STATUSBAR_VALUES )
158 {
159 int indexval = m_iStatusValues[index];
160 char szRepString[MAX_PLAYER_NAME_LENGTH];
161
162 switch ( valtype )
163 {
164 case 'p':
165 gEngfuncs.pfnGetPlayerInfo( indexval, &g_PlayerInfoList[indexval] );
166 if ( g_PlayerInfoList[indexval].name != NULL )
167 {
168 strncpy( szRepString, g_PlayerInfoList[indexval].name, MAX_PLAYER_NAME_LENGTH );
169 m_pflNameColors[line_num] = GetClientColor( indexval );
170 }
171 else
172 {
173 strcpy( szRepString, "******" );
174 }
175 break;
176 case 'i':
177 sprintf( szRepString, "%d", indexval );
178 break;
179 default:
180 szRepString[0] = 0;
181 }
182
183 for ( char *cp = szRepString; *cp != 0 && ((dst - dst_start) < MAX_STATUSTEXT_LENGTH); cp++, dst++ )
184 *dst = *cp;
185 }
186 }
187 }
188 }
189 else
190 {
191 while ( *src != 0 && *src != '\n' )
192 src++;
193 }
194 }
195}

205int CHudStatusBar::Draw( float fTime )
206{
207 if ( m_bReparseString )
208 {
209 for ( int i = 0; i < MAX_STATUSBAR_LINES; i++ )
210 {
211 m_pflNameColors[i] = g_ColorYellow;
212 ParseStatusString( i );
213 }
214 m_bReparseString = FALSE;
215 }
216
217 int Y_START = ScreenHeight - 52;
218
219 for ( int i = 0; i < MAX_STATUSBAR_LINES; i++ )
220 {
221 int TextHeight, TextWidth;
222 GetConsoleStringSize( m_szStatusBar[i], &TextWidth, &TextHeight );
223
224 int x = 8;
225 int y = Y_START - ( 4 + TextHeight * i );
226
227 if ( (i == STATUSBAR_ID_LINE) && CVAR_GET_FLOAT("hud_centerid") )
228 {
229 x = max( 0, max(2, (ScreenWidth - TextWidth)) / 2 );
230 y = (ScreenHeight / 2) + (TextHeight*CVAR_GET_FLOAT("hud_centerid"));
231 }
232
233 if ( m_pflNameColors[i] )
234 gEngfuncs.pfnDrawSetTextColor( m_pflNameColors[i][0], m_pflNameColors[i][1], m_pflNameColors[i][2] );
235
236 DrawConsoleString( x, y, m_szStatusBar[i] );
237 }
238
239 return 1;
240}

252int CHudStatusBar::MsgFunc_StatusText( const char *pszName, int iSize, void *pbuf )
253{
254 BEGIN_READ( pbuf, iSize );
255
256 int line = READ_BYTE();
257
258 if ( line < 0 || line > MAX_STATUSBAR_LINES )
259 return 1;
260
261 strncpy( m_szStatusText[line], READ_STRING(), MAX_STATUSTEXT_LENGTH );
262 m_szStatusText[line][MAX_STATUSTEXT_LENGTH-1] = 0;
263
264 m_iFlags |= HUD_ACTIVE;
265 m_bReparseString = TRUE;
266
267 return 1;
268}
