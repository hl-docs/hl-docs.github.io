333int CBullsquid::TakeDamage(entvars_t *pevInflictor, entvars_t *pevAttacker, float flDamage, int bitsDamageType)
334{
335 float flDist;
336 Vector vecApex;
337
338 // if the squid is running, has an enemy, was hurt by the enemy, hasn't been hurt in the last 3 seconds, and isn't too close to the enemy,
339 // it will swerve. (whew).
340 if (m_hEnemy != NULL && IsMoving() && pevAttacker == m_hEnemy->pev && gpGlobals->time - m_flLastHurtTime > 3)
341 {
342 flDist = (pev->origin - m_hEnemy->pev->origin).Length2D();
343
344 if (flDist > SQUID_SPRINT_DIST)
345 {
346 flDist = (pev->origin - m_Route[m_iRouteIndex].vecLocation).Length2D(); // reusing flDist.
347
348 if (FTriangulate(pev->origin, m_Route[m_iRouteIndex].vecLocation, flDist * 0.5, m_hEnemy, &vecApex))
349 {
350 InsertWaypoint(vecApex, bits_MF_TO_DETOUR | bits_MF_DONT_SIMPLIFY);
351 }
352 }
353 }
354
355 if (!FClassnameIs(pevAttacker, "monster_headcrab"))
356 {
357 // don't forget about headcrabs if it was a headcrab that hurt the squid.
358 m_flLastHurtTime = gpGlobals->time;
359 }
360
361 return CBaseMonster::TakeDamage(pevInflictor, pevAttacker, flDamage, bitsDamageType);

374BOOL CBullsquid::CheckRangeAttack1(float flDot, float flDist)
375{
376 if (IsMoving() && flDist >= 512)
377 {
378 // squid will far too far behind if he stops running to spit at this distance from the enemy.
379 return FALSE;
380 }
381
382 if (flDist > 64 && flDist <= 784 && flDot >= 0.5 && gpGlobals->time >= m_flNextSpitTime)
383 {
384 if (m_hEnemy != NULL)
385 {
386 if (fabs(pev->origin.z - m_hEnemy->pev->origin.z) > 256)
387 {
388 // don't try to spit at someone up really high or down really low.
389 return FALSE;
390 }
391 }
392
393 if (IsMoving())
394 {
395 // don't spit again for a long time, resume chasing enemy.
396 m_flNextSpitTime = gpGlobals->time + 5;
397 }
398 else
399 {
400 // not moving, so spit again pretty soon.
401 m_flNextSpitTime = gpGlobals->time + 0.5;
402 }
403
404 return TRUE;
405 }
406
407 return FALSE;
408}

620void CBullsquid::HandleAnimEvent(MonsterEvent_t *pEvent)
621{
622 switch (pEvent->event)
623 {
624 case BSQUID_AE_SPIT:
625 {
626 Vector vecSpitOffset;
627 Vector vecSpitDir;
628
629 UTIL_MakeVectors(pev->angles);
630
631 // !!!HACKHACK - the spot at which the spit originates (in front of the mouth) was measured in 3ds and hardcoded here.
632 // we should be able to read the position of bones at runtime for this info.
633 vecSpitOffset = (gpGlobals->v_right * 8 + gpGlobals->v_forward * 37 + gpGlobals->v_up * 23);
634 vecSpitOffset = (pev->origin + vecSpitOffset);
635 vecSpitDir = ((m_hEnemy->pev->origin + m_hEnemy->pev->view_ofs) - vecSpitOffset).Normalize();
636
637 vecSpitDir.x += RANDOM_FLOAT(-0.05, 0.05);
638 vecSpitDir.y += RANDOM_FLOAT(-0.05, 0.05);
639 vecSpitDir.z += RANDOM_FLOAT(-0.05, 0);
640
641 // do stuff for this event.
642 AttackSound();
643
644 // spew the spittle temporary ents.
645 MESSAGE_BEGIN(MSG_PVS, SVC_TEMPENTITY, vecSpitOffset);
646 WRITE_BYTE(TE_SPRITE_SPRAY);
647 WRITE_COORD(vecSpitOffset.x); // pos
648 WRITE_COORD(vecSpitOffset.y);
649 WRITE_COORD(vecSpitOffset.z);
650 WRITE_COORD(vecSpitDir.x); // dir
651 WRITE_COORD(vecSpitDir.y);
652 WRITE_COORD(vecSpitDir.z);
653 WRITE_SHORT(iSquidSpitSprite); // model
654 WRITE_BYTE(15); // count
655 WRITE_BYTE(210); // speed
656 WRITE_BYTE(25); // noise (client will divide by 100)
657 MESSAGE_END();
658
659 CSquidSpit::Shoot(pev, vecSpitOffset, vecSpitDir * 900);
660 }
661 break;
662
663 case BSQUID_AE_BITE:
664 {
665 // SOUND HERE!
666 CBaseEntity *pHurt = CheckTraceHullAttack(70, gSkillData.bullsquidDmgBite, DMG_SLASH);
667
668 if (pHurt)
669 {
670 //pHurt->pev->punchangle.z = -15;
671 //pHurt->pev->punchangle.x = -45;
672 pHurt->pev->velocity = pHurt->pev->velocity - gpGlobals->v_forward * 100;
673 pHurt->pev->velocity = pHurt->pev->velocity + gpGlobals->v_up * 100;
674 }
675 }
676 break;
677
678 case BSQUID_AE_TAILWHIP:
679 {
680 CBaseEntity *pHurt = CheckTraceHullAttack(70, gSkillData.bullsquidDmgWhip, DMG_CLUB | DMG_ALWAYSGIB);

682 {
683 pHurt->pev->punchangle.z = -20;
684 pHurt->pev->punchangle.x = 20;
685 pHurt->pev->velocity = pHurt->pev->velocity + gpGlobals->v_right * 200;
686 pHurt->pev->velocity = pHurt->pev->velocity + gpGlobals->v_up * 100;
687 }
688 }
689 break;
690
691 case BSQUID_AE_BLINK:
692 {
693 // close eye.
694 pev->skin = 1;
695 }
696 break;
697
698 case BSQUID_AE_HOP:
699 {
700 float flGravity = g_psv_gravity->value;
701
702 // throw the squid up into the air on this frame.
703 if (FBitSet(pev->flags, FL_ONGROUND))
704 {
705 pev->flags -= FL_ONGROUND;
706 }
707
708 // jump into air for 0.8 (24/30) seconds
709 // pev->velocity.z += (0.875 * flGravity) * 0.5;
710 pev->velocity.z += (0.625 * flGravity) * 0.5;
711 }
712 break;
713
714 case BSQUID_AE_THROW:
715 {
716 int iPitch;
717
718 // squid throws its prey IF the prey is a client.
719 CBaseEntity *pHurt = CheckTraceHullAttack(70, 0, 0);

728 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, "bullchicken/bc_bite2.wav", 1, ATTN_NORM, 0, iPitch);
729 break;

731 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, "bullchicken/bc_bite3.wav", 1, ATTN_NORM, 0, iPitch);
732 break;
733 }
734
735 //pHurt->pev->punchangle.x = RANDOM_LONG(0,34) - 5;
736 //pHurt->pev->punchangle.z = RANDOM_LONG(0,49) - 25;
737 //pHurt->pev->punchangle.y = RANDOM_LONG(0,89) - 45;
738
739 // screeshake transforms the viewmodel as well as the viewangle. No problems with seeing the ends of the viewmodels.
740 UTIL_ScreenShake(pHurt->pev->origin, 25.0, 1.5, 0.7, 2);
741
742 if (pHurt->IsPlayer())
743 {
744 UTIL_MakeVectors(pev->angles);
