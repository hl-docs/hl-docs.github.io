56BOOL CSquadMonster::OccupySlot(int iDesiredSlots)
57{
58 int i;
59 int iMask;
60 int iSquadSlots;
61
62 if (!InSquad())
63 {
64 return TRUE; // If not in a squad, always succeed
65 }
66
67 if (SquadEnemySplit())
68 {
69 // If squad members aren't all fighting the same enemy, disable slots
70 m_iMySlot = bits_SLOT_SQUAD_SPLIT;
71 return TRUE;
72 }
73
74 CSquadMonster *pSquadLeader = MySquadLeader();
75
76 if (!(iDesiredSlots ^ pSquadLeader->m_afSquadSlots))
77 {
78 // None of the desired slots are available
79 return FALSE;
80 }
81
82 iSquadSlots = pSquadLeader->m_afSquadSlots;
83
84 for (i = 0; i < NUM_SLOTS; i++)
85 {
86 iMask = 1 << i;
87 if (iDesiredSlots & iMask) // Check if looking for this bit
88 {
89 if (!(iSquadSlots & iMask)) // Check if it's already taken
90 {
91 // Use this bit
92 pSquadLeader->m_afSquadSlots |= iMask;
93 m_iMySlot = iMask;
94 return TRUE; // Successfully occupied the slot
95 }
96 }
97 }
98
99 return FALSE; // No slots available
100}

166void CSquadMonster::SquadRemove(CSquadMonster *pRemove)
167{
168 ASSERT(pRemove != NULL);
169 ASSERT(this->IsLeader());
170 ASSERT(pRemove->m_hSquadLeader == this);
171
172 // If I'm the leader, get rid of my squad
173 if (pRemove == MySquadLeader())
174 {
175 for (int i = 0; i < MAX_SQUAD_MEMBERS - 1; i++)
176 {
177 CSquadMonster *pMember = MySquadMember(i);
178 if (pMember)
179 {
180 pMember->m_hSquadLeader = NULL; // Clear leader reference
181 m_hSquadMember[i] = NULL; // Clear squad member reference
182 }
183 }
184 }
185 else
186 {
187 CSquadMonster *pSquadLeader = MySquadLeader();
188 if (pSquadLeader)
189 {
190 for (int i = 0; i < MAX_SQUAD_MEMBERS - 1; i++)
191 {
192 if (pSquadLeader->m_hSquadMember[i] == this)
193 {
194 pSquadLeader->m_hSquadMember[i] = NULL; // Remove from leader's member list
195 break;
196 }
197 }
198 }
199 }
200
201 pRemove->m_hSquadLeader = NULL; // Clear the removed member's leader reference
202}

265void CSquadMonster::SquadMakeEnemy(CBaseEntity *pEnemy)
266{
267 if (!InSquad())
268 return;
269
270 if (!pEnemy)
271 {
272 ALERT(at_console, "ERROR: SquadMakeEnemy() - pEnemy is NULL!\n");
273 return; // Exit if no enemy specified
274 }
275
276 CSquadMonster *pSquadLeader = MySquadLeader();
277 for (int i = 0; i < MAX_SQUAD_MEMBERS; i++)
278 {
279 CSquadMonster *pMember = pSquadLeader->MySquadMember(i);
280 if (pMember)
281 {
282 // Reset members who aren't actively engaged in fighting
283 if (pMember->m_hEnemy != pEnemy && !pMember->HasConditions(bits_COND_SEE_ENEMY))
284 {
285 if (pMember->m_hEnemy != NULL)
286 {
287 // Remember their current enemy
288 pMember->PushEnemy(pMember->m_hEnemy, pMember->m_vecEnemyLKP);
289 }
290 // Give them a new enemy
291 pMember->m_hEnemy = pEnemy;
292 pMember->m_vecEnemyLKP = pEnemy->pev->origin; // Update enemy location
293 pMember->SetConditions(bits_COND_NEW_ENEMY); // Set new enemy condition
294 }
295 }
296 }

332int CSquadMonster::SquadRecruit(int searchRadius, int maxMembers)
333{
334 int squadCount;
335 int iMyClass = Classify(); // Cache this monster's class
336
337 // Don't recruit if already in a group
338 if (InSquad())
339 return 0;
340
341 if (maxMembers < 2)
342 return 0; // Require at least 2 members to form a squad
343
344 // I am my own leader
345 m_hSquadLeader = this;
346 squadCount = 1; // Start with self
347
348 CBaseEntity *pEntity = NULL;
349
350 if (!FStringNull(pev->netname))
351 {
352 // Recruit everyone else with the same netname
353 pEntity = UTIL_FindEntityByString(pEntity, "netname", STRING(pev->netname));
354 while (pEntity)
355 {
356 CSquadMonster *pRecruit = pEntity->MySquadMonsterPointer();
357
358 if (pRecruit)
359 {
360 if (!pRecruit->InSquad() && pRecruit->Classify() == iMyClass && pRecruit != this)
361 {
362 // Minimum protection against user error in worldcraft
363 if (!SquadAdd(pRecruit))
364 break; // Stop if unable to add
365 squadCount++; // Increment squad count
366 }
367 }
368
369 pEntity = UTIL_FindEntityByString(pEntity, "netname", STRING(pev->netname)); // Find next entity
370 }
371 }
372 else
373 {
374 // Search for nearby squad members
375 while ((pEntity = UTIL_FindEntityInSphere(pEntity, pev->origin, searchRadius)) != NULL)
376 {
377 CSquadMonster *pRecruit = pEntity->MySquadMonsterPointer();
378
379 if (pRecruit && pRecruit != this && pRecruit->IsAlive() && !pRecruit->m_pCine)
380 {
381 // Can we recruit this guy?
382 if (!pRecruit->InSquad() && pRecruit->Classify() == iMyClass &&
383 ((iMyClass != CLASS_ALIEN_MONSTER) || FStrEq(STRING(pev->classname), STRING(pRecruit->pev->classname))) &&
384 FStringNull(pRecruit->pev->netname))
385 {
386 TraceResult tr;
387 UTIL_TraceLine(pev->origin + pev->view_ofs, pRecruit->pev->origin + pev->view_ofs, ignore_monsters, pRecruit->edict(), &tr); // Try to hit recruit with a trace line
388 if (tr.flFraction == 1.0)
389 {
390 if (!SquadAdd(pRecruit))
391 break; // Stop if unable to add
392
393 squadCount++; // Increment squad count
394 }
395 }
396 }
397 }
398 }
399
400 // No single member squads
401 if (squadCount == 1)
402 {
403 m_hSquadLeader = NULL; // Clear leader reference
404 }
405
406 return squadCount; // Return the size of the squad
407}

418int CSquadMonster::CheckEnemy(CBaseEntity *pEnemy)
419{
420 int iUpdatedLKP;
421
422 iUpdatedLKP = CBaseMonster::CheckEnemy(m_hEnemy);
423
424 // Communicate with squad members about the enemy if this individual has the same enemy as the squad leader.
425 if (InSquad() && (CBaseEntity *)m_hEnemy == MySquadLeader()->m_hEnemy)
426 {
427 if (iUpdatedLKP)
428 {
429 // Have new enemy information, so paste to the squad.
430 SquadPasteEnemyInfo();
431 }
432 else
433 {
434 // Enemy unseen, copy from the squad knowledge.
435 SquadCopyEnemyInfo();
436 }
437 }
438
439 return iUpdatedLKP; // Return the updated last known position
440}

447void CSquadMonster::StartMonster(void)
448{
449 CBaseMonster::StartMonster();
450
451 if ((m_afCapability & bits_CAP_SQUAD) && !InSquad())
452 {
453 if (!FStringNull(pev->netname))
454 {
455 // If I have a group name, I can only recruit if I'm flagged as leader
456 if (!(pev->spawnflags & SF_SQUADMONSTER_LEADER))
457 {
458 return; // Exit if not a leader

464

471 {
472 SetBodygroup(1, 1); // Set body group for grunt
473 pev->skin = 0; // Reset skin
474 }

486BOOL CSquadMonster::NoFriendlyFire(void)
487{
488 if (!InSquad())
489 {
490 return TRUE; // No squad, no friendly fire
491 }
492
493 CPlane backPlane;
494 CPlane leftPlane;
495 CPlane rightPlane;
496
497 Vector vecLeftSide;
498 Vector vecRightSide;
499 Vector v_left;
500
501 if (m_hEnemy != NULL)
502 {
503 UTIL_MakeVectors(UTIL_VecToAngles(m_hEnemy->Center() - pev->origin)); // Set direction to enemy
504 }
505 else
506 {
507 // If there's no enemy, pretend there's a friendly in the way
508 return FALSE;
509 }
510
511 vecLeftSide = pev->origin - (gpGlobals->v_right * (pev->size.x * 1.5));
512 vecRightSide = pev->origin + (gpGlobals->v_right * (pev->size.x * 1.5));
513 v_left = gpGlobals->v_right * -1;
514
515 leftPlane.InitializePlane(gpGlobals->v_right, vecLeftSide); // Initialize left plane
516 rightPlane.InitializePlane(v_left, vecRightSide); // Initialize right plane
517 backPlane.InitializePlane(gpGlobals->v_forward, pev->origin); // Initialize back plane
518
519 CSquadMonster *pSquadLeader = MySquadLeader();
520 for (int i = 0; i < MAX_SQUAD_MEMBERS; i++)
521 {
522 CSquadMonster *pMember = pSquadLeader->MySquadMember(i);
523 if (pMember && pMember != this)
524 {
525 if (backPlane.PointInFront(pMember->pev->origin) &&
526 leftPlane.PointInFront(pMember->pev->origin) &&
527 rightPlane.PointInFront(pMember->pev->origin))
528 {
529 // This member is in the check volume! Don't shoot!
530 return FALSE;
531 }
532 }
533 }
534
535 return TRUE; // No friendly fire possible
536}

603
604 CSquadMonster *pSquadLeader = MySquadLeader();

606

608 {

611 {
612 return TRUE; // Found a member fighting a different enemy
613 }
614 }
615 return FALSE; // All members are fighting the same enemy
616}
