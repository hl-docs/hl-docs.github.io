514int CHudBenchmark::Bench_ScoreForValue( int stage, float raw )
515{
516 int score = 100.0;
517 int power_play = Bench_GetPowerPlay() ? 1 : 0;
518
519 switch ( stage )
520 {
521 case 1: // ping
522 score = 100.0 * ( m_StoredLatency );
523 score = max( score, 0 );
524 score = min( score, 100 );
525
526 // score is inverted
527 score = 100 - score;
528
529 break;
530 case 2: // framerate/performance
531 score = (int)( 100 * m_fAvgFrameRate ) / 72;
532 score = min( score, 100 );
533 score = max( score, 0 );
534
535 score *= BENCH_RANGE/100.0;
536 if ( power_play )
537 {
538 score += ( 100 - BENCH_RANGE );
539 }
540 break;
541 case 3: // tracking
542 score = (100 * m_fAvgScore) / 40;
543 score = max( score, 0 );
544 score = min( score, 100 );
545
546 // score is inverted
547 score = 100 - score;
548
549 score *= BENCH_RANGE/100.0;
550 if ( power_play )
551 {
552 score += ( 100 - BENCH_RANGE );
553 }
554 break;
555 }
556
557 return score;
558}

565void CHudBenchmark::SetCompositeScore( void )
566{
567 int tracking_score = Bench_ScoreForValue( THIRD_STAGE, m_fAvgScore );
568 int ping_score = Bench_ScoreForValue( FIRST_STAGE, m_StoredLatency );
569 int frame_score = Bench_ScoreForValue( SECOND_STAGE, m_fAvgFrameRate );
570
571 int composite = ( ping_score * weights[ 0 ] + frame_score * weights[ 1 ] + tracking_score * weights[ 2 ] );
572
573 composite = min( 100, composite );
574 composite = max( 0, composite );
575
576 m_nCompositeScore = composite;
577}

890 centerspot[2] -= 512;
