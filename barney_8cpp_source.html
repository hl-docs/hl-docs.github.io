343//=========================================================
344// CheckRangeAttack1
345//=========================================================
355BOOL CBarney::CheckRangeAttack1(float flDot, float flDist)
356{
357 if (flDist <= 1024 && flDot >= 0.5)
358 {
359 if (gpGlobals->time > m_checkAttackTime)
360 {
361 TraceResult tr;
362
363 Vector shootOrigin = pev->origin + Vector(0, 0, 55); // Set shooting origin
364 CBaseEntity *pEnemy = m_hEnemy;
365 Vector shootTarget = ((pEnemy->BodyTarget(shootOrigin) - pEnemy->pev->origin) + m_vecEnemyLKP);
366 UTIL_TraceLine(shootOrigin, shootTarget, dont_ignore_monsters, ENT(pev), &tr); // Trace line for attack
367 m_checkAttackTime = gpGlobals->time + 1; // Update check attack time
368 if (tr.flFraction == 1.0 || (tr.pHit != NULL && CBaseEntity::Instance(tr.pHit) == pEnemy))
369 m_lastAttackCheck = TRUE; // Attack is valid
370 else
371 m_lastAttackCheck = FALSE; // Attack is not valid
372 m_checkAttackTime = gpGlobals->time + 1.5; // Update check attack time
373 }
374 return m_lastAttackCheck; // Return last attack check result
375 }
376 return FALSE; // Attack not possible
377}

380//=========================================================
381// BarneyFirePistol - shoots one round from the pistol at
382// the enemy barney is facing.
383//=========================================================
389void CBarney::BarneyFirePistol(void)
390{
391 Vector vecShootOrigin;
392
393 UTIL_MakeVectors(pev->angles); // Make vectors based on current angles
394 vecShootOrigin = pev->origin + Vector(0, 0, 55); // Set shooting origin
395 Vector vecShootDir = ShootAtEnemy(vecShootOrigin); // Get direction to shoot at enemy
396
397 Vector angDir = UTIL_VecToAngles(vecShootDir); // Convert direction to angles
398 SetBlending(0, angDir.x); // Set blending for animation
399 pev->effects = EF_MUZZLEFLASH; // Set muzzle flash effect
400
401 FireBullets(1, vecShootOrigin, vecShootDir, VECTOR_CONE_2DEGREES, 1024, BULLET_MONSTER_9MM); // Fire bullets
402
403 int pitchShift = RANDOM_LONG(0, 20); // Random pitch shift
404
405 // Only shift about half the time
406 if (pitchShift > 10)
407 pitchShift = 0;
408 else
409 pitchShift -= 5;
410 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, "barney/ba_attack2.wav", 1, ATTN_NORM, 0, 100 + pitchShift); // Play shooting sound
411
412 CSoundEnt::InsertSound(bits_SOUND_COMBAT, pev->origin, 384, 0.3); // Insert sound for combat
413
414 m_cAmmoLoaded--; // Decrease ammo count
415}

560static BOOL IsFacing(entvars_t *pevTest, const Vector &reference)
561{
562 Vector vecDir = (reference - pevTest->origin); // Calculate direction to reference point
563 vecDir.z = 0; // Ignore vertical component
564 vecDir = vecDir.Normalize(); // Normalize the direction vector
565
566 Vector forward, angle;
567 angle = pevTest->v_angle; // Get the entity's current angle
568 angle.x = 0; // Ignore pitch for facing check
569 UTIL_MakeVectorsPrivate(angle, forward, NULL, NULL); // Get forward vector based on angle
570
571 // Check if the entity is facing the reference point within +/- 15 degrees
572 if (DotProduct(forward, vecDir) > 0.96) // Dot product threshold for facing
573 {
574 return TRUE; // Entity is facing the reference point
575 }
576 return FALSE; // Entity is not facing the reference point
577}

590int CBarney::TakeDamage(entvars_t* pevInflictor, entvars_t* pevAttacker, float flDamage, int bitsDamageType)
591{
592 int ret = CTalkMonster::TakeDamage(pevInflictor, pevAttacker, flDamage, bitsDamageType); // Call base class method
593 if (!IsAlive() || pev->deadflag == DEAD_DYING)
594 return ret; // Return if dead or dying
595
596 if (m_MonsterState != MONSTERSTATE_PRONE && (pevAttacker->flags & FL_CLIENT))
597 {
598 m_flPlayerDamage += flDamage; // Track damage from player
599
600 if (m_hEnemy == NULL) // Determine if player intended to harm
601 {
602 if ((m_afMemory & bits_MEMORY_SUSPICIOUS) || IsFacing(pevAttacker, pev->origin))
603 {
604 PlaySentence("BA_MAD", 4, VOL_NORM, ATTN_NORM); // Get mad
605 Remember(bits_MEMORY_PROVOKED); // Remember provocation
606 StopFollowing(TRUE); // Stop following
607 }
608 else
609 {
610 PlaySentence("BA_SHOT", 4, VOL_NORM, ATTN_NORM); // Warn player
611 Remember(bits_MEMORY_SUSPICIOUS); // Remember suspicion
612 }
613 }
614 else if (!(m_hEnemy->IsPlayer()) && pev->deadflag == DEAD_NO)
615 {
616 PlaySentence("BA_SHOT", 4, VOL_NORM, ATTN_NORM); // Warn about being shot
617 }
618 }
619
620 return ret; // Return result of damage processing
621}

676void CBarney::TraceAttack(entvars_t *pevAttacker, float flDamage, Vector vecDir, TraceResult *ptr, int bitsDamageType)
677{
678 switch(ptr->iHitgroup) // Check hit group
679 {
680 case HITGROUP_CHEST:
681 case HITGROUP_STOMACH:
682 if (bitsDamageType & (DMG_BULLET | DMG_SLASH | DMG_BLAST))
683 {
684 flDamage = flDamage / 2; // Reduce damage for chest/stomach hits
685 }
686 break;
687 case 10: // Head hit
688 if (bitsDamageType & (DMG_BULLET | DMG_SLASH | DMG_CLUB))
689 {
690 flDamage -= 20; // Reduce damage for head hits
691 if (flDamage <= 0)
692 {
693 UTIL_Ricochet(ptr->vecEndPos, 1.0); // Handle ricochet
694 flDamage = 0.01; // Set minimum damage
695 }
696 }
697 ptr->iHitgroup = HITGROUP_HEAD; // Always register as head hit
698 break;
699 }
700
701 CTalkMonster::TraceAttack(pevAttacker, flDamage, vecDir, ptr, bitsDamageType); // Call base class method
702}

742Schedule_t* CBarney::GetScheduleOfType(int Type)
743{
744 Schedule_t *psched;
745
746 switch (Type)
747 {
748 case SCHED_ARM_WEAPON: // Schedule to arm weapon
749 if (m_hEnemy != NULL)
750 {
751 return slBarneyEnemyDraw; // Face enemy and draw weapon
752 }
753 break;
754
755 // Hook these to make a looping schedule
756 case SCHED_TARGET_FACE:
757 psched = CTalkMonster::GetScheduleOfType(Type); // Call base class method
758 if (psched == slIdleStand)
759 return slBaFaceTarget; // Override for different target face behavior
760 else
761 return psched;
762
763 case SCHED_TARGET_CHASE:
764 return slBaFollow; // Return follow schedule
765
766 case SCHED_IDLE_STAND:
767 psched = CTalkMonster::GetScheduleOfType(Type); // Call base class method
768 if (psched == slIdleStand)
769 {
770 return slIdleBaStand; // Look straight ahead
771 }
772 else
773 return psched;
774 }
775
776 return CTalkMonster::GetScheduleOfType(Type); // Default to base class method
777}

779//=========================================================
780// GetSchedule - Decides which type of schedule best suits
781// the monster's current state and conditions. Then calls
782// monster's member function to get a pointer to a schedule
783// of the proper type.
784//=========================================================
794Schedule_t *CBarney::GetSchedule(void)
795{
796 if (HasConditions(bits_COND_HEAR_SOUND)) // Check if the entity hears a sound
797 {
798 CSound *pSound;
799 pSound = PBestSound(); // Get the best sound
800
801 ASSERT(pSound != NULL); // Ensure the sound is valid
802 if (pSound && (pSound->m_iType & bits_SOUND_DANGER)) // Check if the sound is dangerous
803 return GetScheduleOfType(SCHED_TAKE_COVER_FROM_BEST_SOUND); // Take cover from the best sound
804 }
805 if (HasConditions(bits_COND_ENEMY_DEAD) && FOkToSpeak()) // Check if the enemy is dead and it's okay to speak
806 {
807 PlaySentence("BA_KILL", 4, VOL_NORM, ATTN_NORM); // Play kill sentence
808 }
809
810 switch (m_MonsterState) // Evaluate the monster's state
811 {
812 case MONSTERSTATE_COMBAT:
813 {
814 if (HasConditions(bits_COND_ENEMY_DEAD)) // Check if the enemy is dead
815 {
816 // Call base class to handle dead enemies
817 return CBaseMonster::GetSchedule();
818 }
819
820 // Act surprised with a new enemy
821 if (HasConditions(bits_COND_NEW_ENEMY) && HasConditions(bits_COND_LIGHT_DAMAGE))
822 return GetScheduleOfType(SCHED_SMALL_FLINCH); // Return flinch schedule
823
824 // Wait for one schedule to draw the gun
825 if (!m_fGunDrawn)
826 return GetScheduleOfType(SCHED_ARM_WEAPON); // Return arm weapon schedule
827
828 if (HasConditions(bits_COND_HEAVY_DAMAGE))
829 return GetScheduleOfType(SCHED_TAKE_COVER_FROM_ENEMY); // Take cover from enemy
830 }
831 break;
832
833 case MONSTERSTATE_ALERT:
834 case MONSTERSTATE_IDLE:
835 if (HasConditions(bits_COND_LIGHT_DAMAGE | bits_COND_HEAVY_DAMAGE)) // Check for damage
836 {
837 // Flinch if hurt
838 return GetScheduleOfType(SCHED_SMALL_FLINCH); // Return flinch schedule
839 }
840
841 if (m_hEnemy == NULL && IsFollowing()) // Check if following and no enemy
842 {
843 if (!m_hTargetEnt->IsAlive()) // Check if the target is alive
844 {
845 // Stop following if the target is dead
846 StopFollowing(FALSE);
847 break;
848 }
849 else
850 {
851 if (HasConditions(bits_COND_CLIENT_PUSH)) // Check for client push
852 {
853 return GetScheduleOfType(SCHED_MOVE_AWAY_FOLLOW); // Move away from the client
854 }
855 return GetScheduleOfType(SCHED_TARGET_FACE); // Face the target
856 }
857 }
858
859 if (HasConditions(bits_COND_CLIENT_PUSH)) // Check for client push
860 {
861 return GetScheduleOfType(SCHED_MOVE_AWAY); // Move away
862 }
863
864 // Try to say something about smells
865 TrySmellTalk(); // Attempt to talk about smells
866 break;
867 }
868
869 return CTalkMonster::GetSchedule(); // Default to base class schedule
870}
