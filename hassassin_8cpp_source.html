219//=========================================================
220// Shoot
221//=========================================================
228void CHAssassin :: Shoot ( void )
229{
230 if (m_hEnemy == NULL)
231 {
232 return;
233 }
234
235 Vector vecShootOrigin = GetGunPosition();
236 Vector vecShootDir = ShootAtEnemy( vecShootOrigin );
237
238 if (m_flLastShot + 2 < gpGlobals->time)
239 {
240 m_flDiviation = 0.10;
241 }
242 else
243 {
244 m_flDiviation -= 0.01;
245 if (m_flDiviation < 0.02)
246 m_flDiviation = 0.02;
247 }
248 m_flLastShot = gpGlobals->time;
249
250 UTIL_MakeVectors ( pev->angles );
251
252 Vector vecShellVelocity = gpGlobals->v_right * RANDOM_FLOAT(40,90) + gpGlobals->v_up * RANDOM_FLOAT(75,200) + gpGlobals->v_forward * RANDOM_FLOAT(-40, 40);
253 EjectBrass ( pev->origin + gpGlobals->v_up * 32 + gpGlobals->v_forward * 12, vecShellVelocity, pev->angles.y, m_iShell, TE_BOUNCE_SHELL);
254 FireBullets(1, vecShootOrigin, vecShootDir, Vector( m_flDiviation, m_flDiviation, m_flDiviation ), 2048, BULLET_MONSTER_9MM ); // shoot +-8 degrees
255
256 switch(RANDOM_LONG(0,1))
257 {
258 case 0:
259 EMIT_SOUND(ENT(pev), CHAN_WEAPON, "weapons/pl_gun1.wav", RANDOM_FLOAT(0.6, 0.8), ATTN_NORM);
260 break;
261 case 1:
262 EMIT_SOUND(ENT(pev), CHAN_WEAPON, "weapons/pl_gun2.wav", RANDOM_FLOAT(0.6, 0.8), ATTN_NORM);
263 break;
264 }
265
266 pev->effects |= EF_MUZZLEFLASH;
267
268 Vector angDir = UTIL_VecToAngles( vecShootDir );
269 SetBlending( 0, angDir.x );
270
271 m_cAmmoLoaded--;
272}

275//=========================================================
276// HandleAnimEvent - catches the monster-specific messages
277// that occur when tagged animation frames are played.
278//
279// Returns number of events handled, 0 if none.
280//=========================================================
289void CHAssassin :: HandleAnimEvent( MonsterEvent_t *pEvent )
290{
291 switch( pEvent->event )
292 {
293 case ASSASSIN_AE_SHOOT1:
294 Shoot( );
295 break;
296 case ASSASSIN_AE_TOSS1:
297 {
298 UTIL_MakeVectors( pev->angles );
299 CGrenade::ShootTimed( pev, pev->origin + gpGlobals->v_forward * 34 + Vector (0, 0, 32), m_vecTossVelocity, 2.0 );
300
301 m_flNextGrenadeCheck = gpGlobals->time + 6;// wait six seconds before even looking again to see if a grenade can be thrown.
302 m_fThrowGrenade = FALSE;
303 // !!!LATER - when in a group, only try to throw grenade if ordered.
304 }
305 break;
306 case ASSASSIN_AE_JUMP:
307 {
308 // ALERT( at_console, "jumping");
309 UTIL_MakeAimVectors( pev->angles );
310 pev->movetype = MOVETYPE_TOSS;
311 pev->flags &= ~FL_ONGROUND;
312 pev->velocity = m_vecJumpVelocity;
313 m_flNextJump = gpGlobals->time + 3.0;
314 }
315 return;
316 default:
317 CBaseMonster::HandleAnimEvent( pEvent );
318 break;
319 }
320}

679//=========================================================
680// CheckMeleeAttack1 - jump like crazy if the enemy gets too close.
681//=========================================================
691BOOL CHAssassin :: CheckMeleeAttack1 ( float flDot, float flDist )
692{
693 if ( m_flNextJump < gpGlobals->time && (flDist <= 128 || HasMemory( bits_MEMORY_BADJUMP )) && m_hEnemy != NULL )
694 {
695 TraceResult tr;
696
697 Vector vecDest = pev->origin + Vector( RANDOM_FLOAT( -64, 64), RANDOM_FLOAT( -64, 64 ), 160 );
698
699 UTIL_TraceHull( pev->origin + Vector( 0, 0, 36 ), vecDest + Vector( 0, 0, 36 ), dont_ignore_monsters, human_hull, ENT(pev), &tr);
700
701 if ( tr.fStartSolid || tr.flFraction < 1.0)
702 {
703 return FALSE;
704 }
705
706 float flGravity = g_psv_gravity->value;
707
708 float time = sqrt( 160 / (0.5 * flGravity));
709 float speed = flGravity * time / 160;
710 m_vecJumpVelocity = (vecDest - pev->origin) * speed;
711
712 return TRUE;
713 }
714 return FALSE;
715}

717//=========================================================
718// CheckRangeAttack1 - drop a cap in their ass
719//
720//=========================================================
730BOOL CHAssassin :: CheckRangeAttack1 ( float flDot, float flDist )
731{
732 if ( !HasConditions( bits_COND_ENEMY_OCCLUDED ) && flDist > 64 && flDist <= 2048 /* && flDot >= 0.5 */ /* && NoFriendlyFire() */ )
733 {
734 TraceResult tr;
735
736 Vector vecSrc = GetGunPosition();
737
738 // verify that a bullet fired from the gun will hit the enemy before the world.
739 UTIL_TraceLine( vecSrc, m_hEnemy->BodyTarget(vecSrc), dont_ignore_monsters, ENT(pev), &tr);
740
741 if ( tr.flFraction == 1 || tr.pHit == m_hEnemy->edict() )
742 {
743 return TRUE;
744 }
745 }
746 return FALSE;
747}

749//=========================================================
750// CheckRangeAttack2 - toss grenade is enemy gets in the way and is too close.
751//=========================================================
761BOOL CHAssassin :: CheckRangeAttack2 ( float flDot, float flDist )
762{
763 m_fThrowGrenade = FALSE;
764 if ( !FBitSet ( m_hEnemy->pev->flags, FL_ONGROUND ) )
765 {
766 // don't throw grenades at anything that isn't on the ground!
767 return FALSE;
768 }
769
770 // don't get grenade happy unless the player starts to piss you off
771 if ( m_iFrustration <= 2)
772 return FALSE;
773
774 if ( m_flNextGrenadeCheck < gpGlobals->time && !HasConditions( bits_COND_ENEMY_OCCLUDED ) && flDist <= 512 /* && flDot >= 0.5 */ /* && NoFriendlyFire() */ )
775 {
776 Vector vecToss = VecCheckThrow( pev, GetGunPosition( ), m_hEnemy->Center(), flDist, 0.5 ); // use dist as speed to get there in 1 second
777
778 if ( vecToss != g_vecZero )
779 {
780 m_vecTossVelocity = vecToss;
781
782 // throw a hand grenade
783 m_fThrowGrenade = TRUE;
784
785 return TRUE;
786 }
787 }
788
789 return FALSE;
790}

793//=========================================================
794// RunAI
795//=========================================================
801void CHAssassin :: RunAI( void )
802{
803 CBaseMonster :: RunAI();
804
805 // always visible if moving
806 // always visible is not on hard
807 if (g_iSkillLevel != SKILL_HARD || m_hEnemy == NULL || pev->deadflag != DEAD_NO || m_Activity == ACT_RUN || m_Activity == ACT_WALK || !(pev->flags & FL_ONGROUND))
808 m_iTargetRanderamt = 255;
809 else
810 m_iTargetRanderamt = 20;
811
812 if (pev->renderamt > m_iTargetRanderamt)
813 {
814 if (pev->renderamt == 255)
815 {
816 EMIT_SOUND (ENT(pev), CHAN_BODY, "debris/beamstart1.wav", 0.2, ATTN_NORM );
817 }
818
819 pev->renderamt = max( pev->renderamt - 50, m_iTargetRanderamt );
820 pev->rendermode = kRenderTransTexture;
821 }

930//=========================================================
931// GetSchedule - Decides which type of schedule best suits
932// the monster's current state and conditions. Then calls
933// monster's member function to get a pointer to a schedule
934// of the proper type.
935//=========================================================
943Schedule_t *CHAssassin :: GetSchedule ( void )
944{
945 switch ( m_MonsterState )
946 {
947 case MONSTERSTATE_IDLE:
948 case MONSTERSTATE_ALERT:
949 {
950 if ( HasConditions ( bits_COND_HEAR_SOUND ))
951 {
952 CSound *pSound;
953 pSound = PBestSound();
954
955 ASSERT( pSound != NULL );
956 if ( pSound && (pSound->m_iType & bits_SOUND_DANGER) )
957 {
958 return GetScheduleOfType( SCHED_TAKE_COVER_FROM_BEST_SOUND );
959 }
960 if ( pSound && (pSound->m_iType & bits_SOUND_COMBAT) )
961 {
962 return GetScheduleOfType( SCHED_INVESTIGATE_SOUND );
963 }
964 }
965 }
966 break;
967
968 case MONSTERSTATE_COMBAT:
969 {
970// dead enemy
971 if ( HasConditions( bits_COND_ENEMY_DEAD ) )
972 {
973 // call base class, all code to handle dead enemies is centralized there.
974 return CBaseMonster :: GetSchedule();
975 }
976
977 // flying?
978 if ( pev->movetype == MOVETYPE_TOSS)
979 {
980 if (pev->flags & FL_ONGROUND)
981 {
982 // ALERT( at_console, "landed\n");
983 // just landed
984 pev->movetype = MOVETYPE_STEP;
985 return GetScheduleOfType ( SCHED_ASSASSIN_JUMP_LAND );
986 }
987 else
988 {
989 // ALERT( at_console, "jump\n");
990 // jump or jump/shoot
991 if ( m_MonsterState == MONSTERSTATE_COMBAT )
992 return GetScheduleOfType ( SCHED_ASSASSIN_JUMP );
993 else
994 return GetScheduleOfType ( SCHED_ASSASSIN_JUMP_ATTACK );
995 }
996 }
997
998 if ( HasConditions ( bits_COND_HEAR_SOUND ))
999 {
1000 CSound *pSound;
1001 pSound = PBestSound();
1002
1003 ASSERT( pSound != NULL );
1004 if ( pSound && (pSound->m_iType & bits_SOUND_DANGER) )
1005 {
1006 return GetScheduleOfType( SCHED_TAKE_COVER_FROM_BEST_SOUND );
1007 }
1008 }
1009
1010 if ( HasConditions ( bits_COND_LIGHT_DAMAGE ) )
1011 {
1012 m_iFrustration++;
1013 }
1014 if ( HasConditions ( bits_COND_HEAVY_DAMAGE ) )
1015 {
1016 m_iFrustration++;
1017 }
1018
1019 // jump player!
1020 if ( HasConditions ( bits_COND_CAN_MELEE_ATTACK1 ) )
1021 {
1022 // ALERT( at_console, "melee attack 1\n");
1023 return GetScheduleOfType ( SCHED_MELEE_ATTACK1 );
1024 }
1025
1026 // throw grenade
1027 if ( HasConditions ( bits_COND_CAN_RANGE_ATTACK2 ) )
1028 {
1029 // ALERT( at_console, "range attack 2\n");
1030 return GetScheduleOfType ( SCHED_RANGE_ATTACK2 );
1031 }
1032
1033 // spotted
1034 if ( HasConditions ( bits_COND_SEE_ENEMY ) && HasConditions ( bits_COND_ENEMY_FACING_ME ) )
1035 {
1036 // ALERT( at_console, "exposed\n");
1037 m_iFrustration++;
1038 return GetScheduleOfType ( SCHED_ASSASSIN_EXPOSED );
1039 }
1040
1041 // can attack
1042 if ( HasConditions ( bits_COND_CAN_RANGE_ATTACK1 ) )
1043 {
1044 // ALERT( at_console, "range attack 1\n");
1045 m_iFrustration = 0;
1046 return GetScheduleOfType ( SCHED_RANGE_ATTACK1 );
1047 }
1048
1049 if ( HasConditions ( bits_COND_SEE_ENEMY ) )
1050 {
1051 // ALERT( at_console, "face\n");
1052 return GetScheduleOfType ( SCHED_COMBAT_FACE );
1053 }
1054
1055 // new enemy
1056 if ( HasConditions ( bits_COND_NEW_ENEMY ) )
1057 {
1058 // ALERT( at_console, "take cover\n");
1059 return GetScheduleOfType ( SCHED_TAKE_COVER_FROM_ENEMY );
1060 }
1061
1062 // ALERT( at_console, "stand\n");
1063 return GetScheduleOfType ( SCHED_ALERT_STAND );
1064 }
1065 break;
1066 }
1067
1068 return CBaseMonster :: GetSchedule();
1069}
