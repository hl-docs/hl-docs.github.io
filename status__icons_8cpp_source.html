124int CHudStatusIcons::MsgFunc_StatusIcon(const char* pszName, int iSize, void* pbuf)
125{
126 BEGIN_READ(pbuf, iSize);
127
128 int ShouldEnable = READ_BYTE();
129 char* pszIconName = READ_STRING();
130
131 if (ShouldEnable)
132 {
133 int r = READ_BYTE();
134 int g = READ_BYTE();
135 int b = READ_BYTE();
136 EnableIcon(pszIconName, r, g, b);
137 m_iFlags |= HUD_ACTIVE;
138 }
139 else
140 {
141 DisableIcon(pszIconName);
142 }
143
144 return 1;
145}

159void CHudStatusIcons::EnableIcon(char* pszIconName, unsigned char red, unsigned char green, unsigned char blue)
160{
161 int i;
162
163 // Check if icon already exists
164 for (i = 0; i < MAX_ICONSPRITES; i++)
165 {
166 if (!stricmp(m_IconList[i].szSpriteName, pszIconName))
167 break;
168 }
169
170 // Find an empty slot
171 if (i == MAX_ICONSPRITES)
172 {
173 for (i = 0; i < MAX_ICONSPRITES; i++)
174 {
175 if (!m_IconList[i].spr)
176 break;
177 }
178 }
179
180 // If list full, overwrite the first icon
181 if (i == MAX_ICONSPRITES)
182 i = 0;
183
184 int spr_index = gHUD.GetSpriteIndex(pszIconName);
185 m_IconList[i].spr = gHUD.GetSprite(spr_index);
186 m_IconList[i].rc = gHUD.GetSpriteRect(spr_index);
187 m_IconList[i].r = red;
188 m_IconList[i].g = green;
189 m_IconList[i].b = blue;
190 strcpy(m_IconList[i].szSpriteName, pszIconName);
191
192 // Play a timer sound for grenade icons
193 if (strstr(m_IconList[i].szSpriteName, "grenade"))
194 {
195 cl_entity_t* pthisplayer = gEngfuncs.GetLocalPlayer();
196 gEngfuncs.pEventAPI->EV_PlaySound(pthisplayer->index, pthisplayer->origin, CHAN_STATIC, "weapons/timer.wav", 1.0, ATTN_NORM, 0, PITCH_NORM);
197 }
198}
