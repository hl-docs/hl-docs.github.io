99void EV_GetGunPosition(event_args_t *args, float *pos, float *origin)
100{
101 vec3_t view_ofs;
102 int idx = args->entindex;
103
104 VectorClear(view_ofs);
105 view_ofs[2] = DEFAULT_VIEWHEIGHT;
106
107 if (EV_IsPlayer(idx))
108 {
109 if (EV_IsLocal(idx) && !IS_FIRSTPERSON_SPEC)
110 {
111 gEngfuncs.pEventAPI->EV_LocalPlayerViewheight(view_ofs);
112 }
113 else if (args->ducking == 1)
114 {
115 view_ofs[2] = VEC_DUCK_VIEW;
116 }
117 }
118
119 VectorAdd(origin, view_ofs, pos);
120}

157void EV_GetDefaultShellInfo(event_args_t *args, float *origin, float *velocity, float *ShellVelocity, float *ShellOrigin, float *forward, float *right, float *up, float forwardScale, float upScale, float rightScale)
158{
159 vec3_t view_ofs;
160 VectorClear(view_ofs);
161 view_ofs[2] = DEFAULT_VIEWHEIGHT;
162
163 if (EV_IsPlayer(args->entindex))
164 {
165 if (EV_IsLocal(args->entindex))
166 {
167 gEngfuncs.pEventAPI->EV_LocalPlayerViewheight(view_ofs);
168 }
169 else if (args->ducking == 1)
170 {
171 view_ofs[2] = VEC_DUCK_VIEW;
172 }
173 }
174
175 float fR = gEngfuncs.pfnRandomFloat(50, 70);
176 float fU = gEngfuncs.pfnRandomFloat(100, 150);
177
178 for (int i = 0; i < 3; i++)
179 {
180 ShellVelocity[i] = velocity[i] + right[i] * fR + up[i] * fU + forward[i] * 25;
181 ShellOrigin[i] = origin[i] + view_ofs[i] + up[i] * upScale + forward[i] * forwardScale + right[i] * rightScale;
182 }
183}
