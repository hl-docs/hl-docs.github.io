138 SetThink( NULL );
139
140 if (pOther->pev->takedamage)

231void CCrossbowBolt::ExplodeThink( void )
232{
233 int iContents = UTIL_PointContents ( pev->origin );
234 int iScale;

236 pev->dmg = 40;

239 MESSAGE_BEGIN( MSG_PVS, SVC_TEMPENTITY, pev->origin );
240 WRITE_BYTE( TE_EXPLOSION);
241 WRITE_COORD( pev->origin.x );
242 WRITE_COORD( pev->origin.y );
243 WRITE_COORD( pev->origin.z );
244 if (iContents != CONTENTS_WATER)
245 {
246 WRITE_SHORT( g_sModelIndexFireball );
247 }
248 else
249 {
250 WRITE_SHORT( g_sModelIndexWExplosion );
251 }
252 WRITE_BYTE( iScale ); // scale * 10
253 WRITE_BYTE( 15 ); // framerate
254 WRITE_BYTE( TE_EXPLFLAG_NONE );
255 MESSAGE_END();
256
257 entvars_t *pevOwner;
258
259 if ( pev->owner )
260 pevOwner = VARS( pev->owner );
261 else
262 pevOwner = NULL;
263
264 pev->owner = NULL; // can't traceline attack owner if this is set
265
266 ::RadiusDamage( pev->origin, pev, pevOwner, pev->dmg, 128, CLASS_NONE, DMG_BLAST | DMG_ALWAYSGIB );

268 UTIL_Remove(this);

494void CCrossbow::FireBolt()
495{
496 TraceResult tr;
497
498 if (m_iClip == 0)
499 {
500 PlayEmptySound( );
501 return;
502 }
503
504 m_pPlayer->m_iWeaponVolume = QUIET_GUN_VOLUME;
505
506 m_iClip--;
507
508 int flags;
509#if defined( CLIENT_WEAPONS )
510 flags = FEV_NOTHOST;
511#else
512 flags = 0;
513#endif
514
515 PLAYBACK_EVENT_FULL( flags, m_pPlayer->edict(), m_usCrossbow, 0.0, (float *)&g_vecZero, (float *)&g_vecZero, 0, 0, m_iClip, m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType], 0, 0 );
516
517 // player "shoot" animation
518 m_pPlayer->SetAnimation( PLAYER_ATTACK1 );
519
520 Vector anglesAim = m_pPlayer->pev->v_angle + m_pPlayer->pev->punchangle;
521 UTIL_MakeVectors( anglesAim );
522
523 anglesAim.x = -anglesAim.x;
524 Vector vecSrc = m_pPlayer->GetGunPosition( ) - gpGlobals->v_up * 2;
525 Vector vecDir = gpGlobals->v_forward;
526
527#ifndef CLIENT_DLL
528 CCrossbowBolt *pBolt = CCrossbowBolt::BoltCreate();
529 pBolt->pev->origin = vecSrc;
530 pBolt->pev->angles = anglesAim;
531 pBolt->pev->owner = m_pPlayer->edict();
532
533 if (m_pPlayer->pev->waterlevel == 3)
534 {
535 pBolt->pev->velocity = vecDir * BOLT_WATER_VELOCITY;
536 pBolt->pev->speed = BOLT_WATER_VELOCITY;
537 }
538 else
539 {
540 pBolt->pev->velocity = vecDir * BOLT_AIR_VELOCITY;
541 pBolt->pev->speed = BOLT_AIR_VELOCITY;
542 }
543 pBolt->pev->avelocity.z = 10;
544#endif
545
546 if (!m_iClip && m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0)
547 // HEV suit - indicate out of ammo condition
548 m_pPlayer->SetSuitUpdate("!HEV_AMO0", FALSE, 0);
549
550 m_flNextPrimaryAttack = GetNextAttackDelay(0.75);
551
552 m_flNextSecondaryAttack = UTIL_WeaponTimeBase() + 0.75;
553
554 if (m_iClip != 0)
555 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 5.0;
556 else
557 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 0.75;
558}

610void CCrossbow::WeaponIdle( void )
611{
612 m_pPlayer->GetAutoaimVector( AUTOAIM_2DEGREES ); // get the autoaim vector but ignore it; used for autoaim crosshair in DM
613
614 ResetEmptySound( );
615
616 if ( m_flTimeWeaponIdle < UTIL_WeaponTimeBase() )
617 {
618 float flRand = UTIL_SharedRandomFloat( m_pPlayer->random_seed, 0, 1 );
619 if (flRand <= 0.75)
620 {
621 if (m_iClip)
622 {
623 SendWeaponAnim( CROSSBOW_IDLE1 );
624 }
625 else
626 {
627 SendWeaponAnim( CROSSBOW_IDLE2 );
628 }
629 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat( m_pPlayer->random_seed, 10, 15 );
630 }
631 else
632 {
633 if (m_iClip)
634 {
635 SendWeaponAnim( CROSSBOW_FIDGET1 );
636 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 90.0 / 30.0;
637 }
638 else
639 {
640 SendWeaponAnim( CROSSBOW_FIDGET2 );
641 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 80.0 / 30.0;

643 }
