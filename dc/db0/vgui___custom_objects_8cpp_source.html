291void CommandButton::cursorEntered( void )
292{
293 // unarm all the other buttons in this menu
294 CCommandMenu *containingMenu = getParentMenu();
295 if ( containingMenu )
296 {
297 containingMenu->ClearButtonsOfArmedState();
298
299 // make all our higher buttons armed
300 CCommandMenu *pCParent = containingMenu->GetParentMenu();
301 if ( pCParent )
302 {
303 CommandButton *pParentButton = pCParent->FindButtonWithSubmenu( containingMenu );
304
305 pParentButton->cursorEntered();
306 }
307 }
308
309 // arm ourselves
310 setArmed( true );
311}

350int ClassButton::IsNotValid()
351{
352 // If this is the main ChangeClass button, remove it if the player's only able to be civilians
353 if ( m_iPlayerClass == -1 )
354 {
355 if (gViewPort->GetValidClasses(g_iTeamNumber) == -1)
356 return true;
357
358 return false;
359 }
360
361 // Is it an illegal class?
362#ifdef _TFC
363 if ((gViewPort->GetValidClasses(0) & sTFValidClassInts[ m_iPlayerClass ]) || (gViewPort->GetValidClasses(g_iTeamNumber) & sTFValidClassInts[ m_iPlayerClass ]))
364 return true;
365#endif
366
367 // Only check current class if they've got autokill on
368 bool bAutoKill = CVAR_GET_FLOAT( "hud_classautokill" ) != 0;
369 if ( bAutoKill )
370 {
371 // Is it the player's current class?
372 if (
373#ifdef _TFC
374 (gViewPort->IsRandomPC() && m_iPlayerClass == PC_RANDOM) ||
375#endif
376 (!gViewPort->IsRandomPC() && (m_iPlayerClass == g_iPlayerClass)) )
377 return true;
378 }
379
380 return false;
381}

444void CImageLabel::LoadImage(const char * pImageName)
445{
446 if ( m_pTGA )
447 delete m_pTGA;
448
449 // Load the Image
450 m_pTGA = LoadTGAForRes(pImageName);
451
452 if ( m_pTGA == NULL )
453 {
454 // we didn't find a matching image file for this resolution
455 // try to load file resolution independent
456
457 char sz[256];
458 sprintf(sz, "%s/%s",gEngfuncs.pfnGetGameDirectory(), pImageName );
459 FileInputStream* fis = new FileInputStream( sz, false );
460 m_pTGA = new BitmapTGA(fis,true);
461 fis->close();
462 }
463
464 if ( m_pTGA == NULL )
465 return; // unable to load image
466
467 int w,t;
468
469 m_pTGA->getSize( w, t );
470
471 setSize( XRES (w),YRES (t) );
472 setImage( m_pTGA );
473}
