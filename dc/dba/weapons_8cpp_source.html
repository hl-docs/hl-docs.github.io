209 return;
210

217
218 switch (iBulletType)
219 {
220 case BULLET_PLAYER_9MM:
221 case BULLET_MONSTER_9MM:
222 case BULLET_PLAYER_MP5:
223 case BULLET_MONSTER_MP5:
224 case BULLET_PLAYER_BUCKSHOT:

227 // smoke and decal

305void AddAmmoNameToAmmoRegistry(const char* szAmmoname)
306{
307 // make sure it's not already in the registry
308 for (int i = 0; i < MAX_AMMO_SLOTS; i++)

323 CBasePlayerItem::AmmoInfoArray[giAmmoIndex].iId = giAmmoIndex; // yes, this info is redundant

333void UTIL_PrecacheOtherWeapon(const char* szClassname)
334{
335 edict_t* pent;
336
337 pent = CREATE_NAMED_ENTITY(MAKE_STRING(szClassname));
338 if (FNullEnt(pent))
339 {
340 ALERT(at_console, "NULL Ent in UTIL_PrecacheOtherWeapon\n");
341 return;
342 }
343
344 CBaseEntity* pEntity = CBaseEntity::Instance(VARS(pent));
345
346 if (pEntity)
347 {
348 ItemInfo II;
349 pEntity->Precache();
350 memset(&II, 0, sizeof II);
351 if (((CBasePlayerItem*)pEntity)->GetItemInfo(&II))
352 {
353 CBasePlayerItem::ItemInfoArray[II.iId] = II;
354
355 if (II.pszAmmo1 && *II.pszAmmo1)
356 {
357 AddAmmoNameToAmmoRegistry(II.pszAmmo1);
358 }
359
360 if (II.pszAmmo2 && *II.pszAmmo2)
361 {
362 AddAmmoNameToAmmoRegistry(II.pszAmmo2);
363 }
364
365 memset(&II, 0, sizeof II);
366 }
367 }
368
369 REMOVE_ENTITY(pent);
370}

664//=========================================================
665// Respawn- this item is already in the world, but it is
666// invisible and intangible. Make it visible and tangible.
667//=========================================================
676CBaseEntity* CBasePlayerItem::Respawn(void)
677{
678 // make a copy of this weapon that is invisible and inaccessible to players (no touch function). The weapon spawn/respawn code
679 // will decide when to make the weapon visible and touchable.
680 CBaseEntity* pNewWeapon = CBaseEntity::Create((char*)STRING(pev->classname), g_pGameRules->VecWeaponRespawnSpot(this), pev->angles, pev->owner);
681
682 if (pNewWeapon)
683 {
684 pNewWeapon->pev->effects |= EF_NODRAW; // invisible for now
685 pNewWeapon->SetTouch(NULL); // no touch
686 pNewWeapon->SetThink(&CBasePlayerItem::AttemptToMaterialize);
687
688 DROP_TO_FLOOR(ENT(pev));
689
690 // not a typo! We want to know when the weapon the player just picked up should respawn! This new entity we created is the replacement,
691 // but when it should respawn is based on conditions belonging to the weapon that was taken.
692 pNewWeapon->pev->nextthink = g_pGameRules->FlWeaponRespawnTime(this);
693 }
694 else
695 {
696 ALERT(at_console, "Respawn failed to create %s!\n", STRING(pev->classname));
697 }
698
699 return pNewWeapon;
700}

769void CBasePlayerWeapon::ItemPostFrame(void)
770{
771 if ((m_fInReload) && (m_pPlayer->m_flNextAttack <= UTIL_WeaponTimeBase()))
772 {
773 // complete the reload.
774 int j = min(iMaxClip() - m_iClip, m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType]);
775
776 // Add them to the clip
777 m_iClip += j;
778 m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] -= j;
779
780 m_pPlayer->TabulateAmmo();
781
782 m_fInReload = FALSE;
783 }
784
785 if (!(m_pPlayer->pev->button & IN_ATTACK))
786 {
787 m_flLastFireTime = 0.0f;
788 }
789
790 if ((m_pPlayer->pev->button & IN_ATTACK2) && CanAttack(m_flNextSecondaryAttack, gpGlobals->time, UseDecrement()))
791 {
792 if (pszAmmo2() && !m_pPlayer->m_rgAmmo[SecondaryAmmoIndex()])
793 {
794 m_fFireOnEmpty = TRUE;
795 }
796
797 m_pPlayer->TabulateAmmo();
798 SecondaryAttack();
799 m_pPlayer->pev->button &= ~IN_ATTACK2;
800 }
801 else if ((m_pPlayer->pev->button & IN_ATTACK) && CanAttack(m_flNextPrimaryAttack, gpGlobals->time, UseDecrement()))
802 {
803 if ((m_iClip == 0 && pszAmmo1()) || (iMaxClip() == -1 && !m_pPlayer->m_rgAmmo[PrimaryAmmoIndex()]))
804 {
805 m_fFireOnEmpty = TRUE;
806 }
807
808 m_pPlayer->TabulateAmmo();
809 PrimaryAttack();
810 }
811 else if (m_pPlayer->pev->button & IN_RELOAD && iMaxClip() != WEAPON_NOCLIP && !m_fInReload)
812 {
813 // reload when reload is pressed, or if no buttons are down and weapon is empty.
814 Reload();
815 }
816 else if (!(m_pPlayer->pev->button & (IN_ATTACK | IN_ATTACK2)))
817 {
818 // no fire buttons down
819
820 m_fFireOnEmpty = FALSE;
821
822 if (!IsUseable() && m_flNextPrimaryAttack < (UseDecrement() ? 0.0 : gpGlobals->time))
823 {
824 // weapon isn't useable, switch.
825 if (!(iFlags() & ITEM_FLAG_NOAUTOSWITCHEMPTY) && g_pGameRules->GetNextBestWeapon(m_pPlayer, this))
826 {
827 m_flNextPrimaryAttack = (UseDecrement() ? 0.0 : gpGlobals->time) + 0.3;
828 return;
829 }
830 }
831 else
832 {
833 // weapon is useable. Reload if empty and weapon has waited as long as it has to after firing
834 if (m_iClip == 0 && !(iFlags() & ITEM_FLAG_NOAUTORELOAD) && m_flNextPrimaryAttack < (UseDecrement() ? 0.0 : gpGlobals->time))
835 {
836 Reload();
837 return;
838 }
