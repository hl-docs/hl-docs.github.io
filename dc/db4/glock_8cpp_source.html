191void CGlock::GlockFire( float flSpread , float flCycleTime, BOOL fUseAutoAim )
192{
193 if (m_iClip <= 0)
194 {
195 if (m_fFireOnEmpty)
196 {
197 PlayEmptySound();
198 m_flNextPrimaryAttack = gpGlobals->time + 0.2;
199 }
200
201 return;
202 }
203
204 m_iClip--;
205
206 m_pPlayer->pev->effects = (int)(m_pPlayer->pev->effects) | EF_MUZZLEFLASH;
207
208#if defined ( OLD_WEAPONS )
209 if (m_iClip != 0)
210 SendWeaponAnim( GLOCK_SHOOT );
211 else
212 SendWeaponAnim( GLOCK_SHOOT_EMPTY );
213#endif
214
215 if ( fUseAutoAim )
216 {
217 PLAYBACK_EVENT_FULL( 0, m_pPlayer->edict(), m_usFireGlock1, 0.0, (float *)&g_vecZero, (float *)&g_vecZero, 0.0, 0.0, 0, 0, ( m_iClip == 0 ) ? 1 : 0, 0 );
218 }
219 else
220 {
221 PLAYBACK_EVENT_FULL( 0, m_pPlayer->edict(), m_usFireGlock2, 0.0, (float *)&g_vecZero, (float *)&g_vecZero, 0.0, 0.0, 0, 0, ( m_iClip == 0 ) ? 1 : 0, 0 );
222 }
223
224 // player "shoot" animation
225 m_pPlayer->SetAnimation( PLAYER_ATTACK1 );
226
227#if defined ( OLD_WEAPONS )
228 UTIL_MakeVectors( m_pPlayer->pev->v_angle + m_pPlayer->pev->punchangle );
229
230 Vector vecShellVelocity = m_pPlayer->pev->velocity
231 + gpGlobals->v_right * RANDOM_FLOAT(50,70)
232 + gpGlobals->v_up * RANDOM_FLOAT(100,150)
233 + gpGlobals->v_forward * 25;
234 EjectBrass ( pev->origin + m_pPlayer->pev->view_ofs + gpGlobals->v_up * -12 + gpGlobals->v_forward * 32 + gpGlobals->v_right * 6 , vecShellVelocity, pev->angles.y, m_iShell, TE_BOUNCE_SHELL );
235#endif
236
237 // silenced
238 if (pev->body == 1)
239 {
240 m_pPlayer->m_iWeaponVolume = QUIET_GUN_VOLUME;
241 m_pPlayer->m_iWeaponFlash = DIM_GUN_FLASH;
242#if defined ( OLD_WEAPONS )
243 switch(RANDOM_LONG(0,1))
244 {
245 case 0:
246 EMIT_SOUND(ENT(m_pPlayer->pev), CHAN_WEAPON, "weapons/pl_gun1.wav", RANDOM_FLOAT(0.9, 1.0), ATTN_NORM);
247 break;
248 case 1:
249 EMIT_SOUND(ENT(m_pPlayer->pev), CHAN_WEAPON, "weapons/pl_gun2.wav", RANDOM_FLOAT(0.9, 1.0), ATTN_NORM);
250 break;
251 }
252#endif
253 }
254 else
255 {
256 // non-silenced
257 m_pPlayer->m_iWeaponVolume = NORMAL_GUN_VOLUME;
258 m_pPlayer->m_iWeaponFlash = NORMAL_GUN_FLASH;
259#if defined ( OLD_WEAPONS )
260 EMIT_SOUND_DYN(ENT(m_pPlayer->pev), CHAN_WEAPON, "weapons/pl_gun3.wav", RANDOM_FLOAT(0.92, 1.0), ATTN_NORM, 0, 98 + RANDOM_LONG(0,3));
261#endif
262 }
263
264 Vector vecSrc = m_pPlayer->GetGunPosition( );
265 Vector vecAiming;
266
267 if ( fUseAutoAim )
268 {
269 vecAiming = m_pPlayer->GetAutoaimVector( AUTOAIM_10DEGREES );
270 }
271 else
272 {
273 vecAiming = gpGlobals->v_forward;
274 }
275
276 m_pPlayer->FireBullets( 1, vecSrc, vecAiming, Vector( flSpread, flSpread, flSpread ), 8192, BULLET_PLAYER_9MM, 0 );
277
278 m_flNextPrimaryAttack = m_flNextSecondaryAttack = gpGlobals->time + flCycleTime;
279
280 if (!m_iClip && m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0)
281 // HEV suit - indicate out of ammo condition
282 m_pPlayer->SetSuitUpdate("!HEV_AMO0", FALSE, 0);
283
284 m_flTimeWeaponIdle = gpGlobals->time + RANDOM_FLOAT ( 10, 15 );
285
286#if defined ( OLD_WEAPONS )
287 m_pPlayer->pev->punchangle.x -= 2;
288#endif
289}

319void CGlock::WeaponIdle( void )
320{
321 ResetEmptySound( );
322
323 m_pPlayer->GetAutoaimVector( AUTOAIM_10DEGREES );
324
325 if (m_flTimeWeaponIdle > gpGlobals->time)
326 return;
327
328 // only idle if the slid isn't back
329 if (m_iClip != 0)
330 {
331 int iAnim;
332 float flRand = RANDOM_FLOAT(0, 1);
333 if (flRand <= 0.3 + 0 * 0.75)
334 {
335 iAnim = GLOCK_IDLE3;
336 m_flTimeWeaponIdle = gpGlobals->time + 49.0 / 16;
337 }
338 else if (flRand <= 0.6 + 0 * 0.875)
339 {
340 iAnim = GLOCK_IDLE1;
341 m_flTimeWeaponIdle = gpGlobals->time + 60.0 / 16.0;
342 }
343 else
344 {
345 iAnim = GLOCK_IDLE2;
346 m_flTimeWeaponIdle = gpGlobals->time + 40.0 / 16.0;
347 }
348 SendWeaponAnim( iAnim );
349 }
