132void CBaseEntity::UpdateOnRemove(void)
133{
134 int i;
135
136 if ( FBitSet( pev->flags, FL_GRAPHED ) )
137 {
138 // this entity was a LinkEnt in the world node graph, so we must remove it from
139 // the graph since we are removing it from the world.
140 for ( i = 0 ; i < WorldGraph.m_cLinks ; i++ )
141 {
142 if ( WorldGraph.m_pLinkPool [ i ].m_pLinkEnt == pev )
143 {
144 // if this link has a link ent which is the same ent that is removing itself, remove it!
145 WorldGraph.m_pLinkPool [ i ].m_pLinkEnt = NULL;
146 }
147 }
148 }
149 if ( pev->globalname )
150 gGlobalState.EntitySetState( pev->globalname, GLOBAL_DEAD );
151}

255void FireTargets(const char *targetName, CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
256{
257 edict_t *pentTarget = NULL;
258 if (!targetName)
259 return; // Exit if no target name is provided
260
261 ALERT(at_aiconsole, "Firing: (%s)\n", targetName); // Log firing action
262
263 for (;;)
264 {
265 pentTarget = FIND_ENTITY_BY_TARGETNAME(pentTarget, targetName); // Find the target entity
266 if (FNullEnt(pentTarget))
267 break; // Exit if no more targets are found
268
269 CBaseEntity *pTarget = CBaseEntity::Instance(pentTarget);
270 if (pTarget && !(pTarget->pev->flags & FL_KILLME)) // Don't use dying entities
271 {
272 ALERT(at_aiconsole, "Found: %s, firing (%s)\n", STRING(pTarget->pev->classname), targetName);
273 pTarget->Use(pActivator, pCaller, useType, value); // Call the use function on the target
274 }
275 }

291void CBaseDelay::SUB_UseTargets(CBaseEntity *pActivator, USE_TYPE useType, float value)
292{
293 // Exit immediately if there are no targets or kill targets
294 if (FStringNull(pev->target) && !m_iszKillTarget)
295 return;
296
297 // Check for a delay
298 if (m_flDelay != 0)
299 {
300 // Create a temporary object to fire at a later time
301 CBaseDelay *pTemp = GetClassPtr((CBaseDelay *)NULL);
302 pTemp->pev->classname = MAKE_STRING("DelayedUse");
303
304 pTemp->pev->nextthink = gpGlobals->time + m_flDelay; // Set next think time
305
306 pTemp->SetThink(&CBaseDelay::DelayThink); // Set the think function
307
308 // Save the use type
309 pTemp->pev->button = (int)useType;
310 pTemp->m_iszKillTarget = m_iszKillTarget; // Save kill target
311 pTemp->m_flDelay = 0; // Prevent recursion
312 pTemp->pev->target = pev->target; // Save target
313
314 // Save the activator if it's a player
315 if (pActivator && pActivator->IsPlayer())
316 {
317 pTemp->pev->owner = pActivator->edict();
318 }
319 else
320 {
321 pTemp->pev->owner = NULL; // No owner
322 }
323
324 return; // Exit after setting up the delay
325 }
326
327 // Kill the specified kill targets
328 if (m_iszKillTarget)
329 {
330 edict_t *pentKillTarget = NULL;
331
332 ALERT(at_aiconsole, "KillTarget: %s\n", STRING(m_iszKillTarget));
333 pentKillTarget = FIND_ENTITY_BY_TARGETNAME(NULL, STRING(m_iszKillTarget));
334 while (!FNullEnt(pentKillTarget))
335 {
336 UTIL_Remove(CBaseEntity::Instance(pentKillTarget)); // Remove the target entity
337
338 ALERT(at_aiconsole, "killing %s\n", STRING(pentKillTarget->v.classname));
339 pentKillTarget = FIND_ENTITY_BY_TARGETNAME(pentKillTarget, STRING(m_iszKillTarget)); // Find the next target
340 }
341 }
342
343 // Fire the targets
344 if (!FStringNull(pev->target))
345 {
346 FireTargets(STRING(pev->target), pActivator, this, useType, value);
347 }
348}

482void CBaseToggle::LinearMove(Vector vecDest, float flSpeed)
483{
484 ASSERTSZ(flSpeed != 0, "LinearMove: no speed is defined!");
485
486 m_vecFinalDest = vecDest; // Set the final destination
487
488 // Already at the destination?
489 if (vecDest == pev->origin)
490 {
491 LinearMoveDone(); // Call done function
492 return;
493 }
494
495 // Calculate the vector needed to move
496 Vector vecDestDelta = vecDest - pev->origin;
497
498 // Calculate time to reach destination
499 float flTravelTime = vecDestDelta.Length() / flSpeed;
500
501 // Set next think time to trigger LinearMoveDone when destination is reached
502 pev->nextthink = pev->ltime + flTravelTime;
503 SetThink(&CBaseToggle::LinearMoveDone);
504
505 // Calculate velocity based on travel time
506 pev->velocity = vecDestDelta / flTravelTime;
507}
508
515void CBaseToggle::LinearMoveDone(void)
516{
517 Vector delta = m_vecFinalDest - pev->origin;
518 float error = delta.Length();
519 if (error > 0.03125) // Check if still not at the destination
520 {
521 LinearMove(m_vecFinalDest, 100); // Continue moving
522 return;
523 }
524
525 UTIL_SetOrigin(pev, m_vecFinalDest); // Set the exact final destination
526 pev->velocity = g_vecZero; // Stop movement
527 pev->nextthink = -1; // Clear next think time
528 if (m_pfnCallWhenMoveDone)
529 (this->*m_pfnCallWhenMoveDone)(); // Call post-move function
530}

667BOOL FEntIsVisible(entvars_t* pev, entvars_t* pevTarget)
668{
669 Vector vecSpot1 = pev->origin + pev->view_ofs; // Calculate the view origin of the caller
670 Vector vecSpot2 = pevTarget->origin + pevTarget->view_ofs; // Calculate the view origin of the target
671 TraceResult tr;
672
673 UTIL_TraceLine(vecSpot1, vecSpot2, ignore_monsters, ENT(pev), &tr); // Trace line to check visibility
674
675 if (tr.fInOpen && tr.fInWater)
676 return FALSE; // Sight line crossed contents
677
678 if (tr.flFraction == 1)
679 return TRUE; // Target is visible
680
681 return FALSE; // Target is not visible
682}
