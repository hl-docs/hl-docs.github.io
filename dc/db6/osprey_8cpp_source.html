235void COsprey :: FindAllThink( void )
236{
237 CBaseEntity *pEntity = NULL;
238
239 m_iUnits = 0;
240 while (m_iUnits < MAX_CARRY && (pEntity = UTIL_FindEntityByClassname( pEntity, "monster_human_grunt" )) != NULL)
241 {
242 if (pEntity->IsAlive())
243 {
244 m_hGrunt[m_iUnits] = pEntity;
245 m_vecOrigin[m_iUnits] = pEntity->pev->origin;
246 m_iUnits++;
247 }
248 }
249
250 if (m_iUnits == 0)
251 {
252 ALERT( at_console, "osprey error: no grunts to resupply\n");
253 UTIL_Remove( this );
254 return;
255 }
256 SetThink( &COsprey::FlyThink );
257 pev->nextthink = gpGlobals->time + 0.1;
258 m_startTime = gpGlobals->time;
259}

267void COsprey :: DeployThink( void )
268{
269 UTIL_MakeAimVectors( pev->angles );
270
271 Vector vecForward = gpGlobals->v_forward;
272 Vector vecRight = gpGlobals->v_right;
273 Vector vecUp = gpGlobals->v_up;
274
275 Vector vecSrc;
276
277 TraceResult tr;
278 UTIL_TraceLine( pev->origin, pev->origin + Vector( 0, 0, -4096.0), ignore_monsters, ENT(pev), &tr);
279 CSoundEnt::InsertSound ( bits_SOUND_DANGER, tr.vecEndPos, 400, 0.3 );
280
281 vecSrc = pev->origin + vecForward * 32 + vecRight * 100 + vecUp * -96;
282 m_hRepel[0] = MakeGrunt( vecSrc );
283
284 vecSrc = pev->origin + vecForward * -64 + vecRight * 100 + vecUp * -96;
285 m_hRepel[1] = MakeGrunt( vecSrc );
286
287 vecSrc = pev->origin + vecForward * 32 + vecRight * -100 + vecUp * -96;
288 m_hRepel[2] = MakeGrunt( vecSrc );
289
290 vecSrc = pev->origin + vecForward * -64 + vecRight * -100 + vecUp * -96;
291 m_hRepel[3] = MakeGrunt( vecSrc );
292
293 SetThink( &COsprey::HoverThink );
294 pev->nextthink = gpGlobals->time + 0.1;
295}

307BOOL COsprey :: HasDead( )
308{
309 for (int i = 0; i < m_iUnits; i++)
310 {
311 if (m_hGrunt[i] == NULL || !m_hGrunt[i]->IsAlive())
312 {
313 return TRUE;
314 }
315 else
316 {
317 m_vecOrigin[i] = m_hGrunt[i]->pev->origin; // send them to where they died
318 }
319 }
320 return FALSE;

332CBaseMonster *COsprey :: MakeGrunt( Vector vecSrc )
333{
334 CBaseEntity *pEntity;
335 CBaseMonster *pGrunt;
336
337 TraceResult tr;
338 UTIL_TraceLine( vecSrc, vecSrc + Vector( 0, 0, -4096.0), dont_ignore_monsters, ENT(pev), &tr);
339 if ( tr.pHit && Instance( tr.pHit )->pev->solid != SOLID_BSP)
340 return NULL;
341
342 for (int i = 0; i < m_iUnits; i++)
343 {
344 if (m_hGrunt[i] == NULL || !m_hGrunt[i]->IsAlive())
345 {
346 if (m_hGrunt[i] != NULL && m_hGrunt[i]->pev->rendermode == kRenderNormal)
347 {
348 m_hGrunt[i]->SUB_StartFadeOut( );
349 }
350 pEntity = Create( "monster_human_grunt", vecSrc, pev->angles );
351 pGrunt = pEntity->MyMonsterPointer( );
352 pGrunt->pev->movetype = MOVETYPE_FLY;
353 pGrunt->pev->velocity = Vector( 0, 0, RANDOM_FLOAT( -196, -128 ) );
354 pGrunt->SetActivity( ACT_GLIDE );
355
356 CBeam *pBeam = CBeam::BeamCreate( "sprites/rope.spr", 10 );
357 pBeam->PointEntInit( vecSrc + Vector(0,0,112), pGrunt->entindex() );
358 pBeam->SetFlags( BEAM_FSOLID );
359 pBeam->SetColor( 255, 255, 255 );
360 pBeam->SetThink( &CBeam::SUB_Remove );
361 pBeam->pev->nextthink = gpGlobals->time + -4096.0 * tr.flFraction / pGrunt->pev->velocity.z + 0.5;
362
363 // ALERT( at_console, "%d at %.0f %.0f %.0f\n", i, m_vecOrigin[i].x, m_vecOrigin[i].y, m_vecOrigin[i].z );
364 pGrunt->m_vecLastPosition = m_vecOrigin[i];
365 m_hGrunt[i] = pGrunt;
366 return pGrunt;
367 }
368 }
369 // ALERT( at_console, "none dead\n");
370 return NULL;
371}

407void COsprey::UpdateGoal( )
408{
409 if (m_pGoalEnt)
410 {
411 m_pos1 = m_pos2;
412 m_ang1 = m_ang2;
413 m_vel1 = m_vel2;
414 m_pos2 = m_pGoalEnt->pev->origin;
415 m_ang2 = m_pGoalEnt->pev->angles;
416 UTIL_MakeAimVectors( Vector( 0, m_ang2.y, 0 ) );
417 m_vel2 = gpGlobals->v_forward * m_pGoalEnt->pev->speed;
418
419 m_startTime = m_startTime + m_dTime;
420 m_dTime = 2.0 * (m_pos1 - m_pos2).Length() / (m_vel1.Length() + m_pGoalEnt->pev->speed);
421
422 if (m_ang1.y - m_ang2.y < -180)
423 {
424 m_ang1.y += 360;
425 }
426 else if (m_ang1.y - m_ang2.y > 180)
427 {
428 m_ang1.y -= 360;
429 }
430
431 if (m_pGoalEnt->pev->speed < 400)
432 m_flIdealtilt = 0;
433 else
434 m_flIdealtilt = -90;
435 }
436 else
437 {
438 ALERT( at_console, "osprey missing target");
439 }
440}

448void COsprey::FlyThink( void )
449{
450 StudioFrameAdvance( );
451 pev->nextthink = gpGlobals->time + 0.1;
452
453 if ( m_pGoalEnt == NULL && !FStringNull(pev->target) )// this monster has a target
454 {
455 m_pGoalEnt = CBaseEntity::Instance( FIND_ENTITY_BY_TARGETNAME ( NULL, STRING( pev->target ) ) );
456 UpdateGoal( );
457 }
458
459 if (gpGlobals->time > m_startTime + m_dTime)
460 {
461 if (m_pGoalEnt->pev->speed == 0)
462 {
463 SetThink( &COsprey::DeployThink );
464 }
465 do {
466 m_pGoalEnt = CBaseEntity::Instance( FIND_ENTITY_BY_TARGETNAME ( NULL, STRING( m_pGoalEnt->pev->target ) ) );
467 } while (m_pGoalEnt->pev->speed < 400 && !HasDead());
468 UpdateGoal( );
469 }
470
471 Flight( );
472 ShowDamage( );
473}

481void COsprey::Flight( )
482{
483 float t = (gpGlobals->time - m_startTime);
484 float scale = 1.0 / m_dTime;
485
486 float f = UTIL_SplineFraction( t * scale, 1.0 );
487
488 Vector pos = (m_pos1 + m_vel1 * t) * (1.0 - f) + (m_pos2 - m_vel2 * (m_dTime - t)) * f;
489 Vector ang = (m_ang1) * (1.0 - f) + (m_ang2) * f;
490 m_velocity = m_vel1 * (1.0 - f) + m_vel2 * f;
491
492 UTIL_SetOrigin( pev, pos );
493 pev->angles = ang;
494 UTIL_MakeAimVectors( pev->angles );
495 float flSpeed = DotProduct( gpGlobals->v_forward, m_velocity );
496
497 // float flSpeed = DotProduct( gpGlobals->v_forward, pev->velocity );
498
499 float m_flIdealtilt = (160 - flSpeed) / 10.0;
500
501 // ALERT( at_console, "%f %f\n", flSpeed, flIdealtilt );
502 if (m_flRotortilt < m_flIdealtilt)
503 {
504 m_flRotortilt += 0.5;
505 if (m_flRotortilt > 0)
506 m_flRotortilt = 0;
507 }
508 if (m_flRotortilt > m_flIdealtilt)
509 {
510 m_flRotortilt -= 0.5;
511 if (m_flRotortilt < -90)
512 m_flRotortilt = -90;
513 }
514 SetBoneController( 0, m_flRotortilt );
515
516
517 if (m_iSoundState == 0)
518 {
519 EMIT_SOUND_DYN(ENT(pev), CHAN_STATIC, "apache/ap_rotor4.wav", 1.0, 0.15, 0, 110 );
520 // EMIT_SOUND_DYN(ENT(pev), CHAN_STATIC, "apache/ap_whine1.wav", 0.5, 0.2, 0, 110 );
521
522 m_iSoundState = SND_CHANGE_PITCH; // hack for going through level transitions
523 }
524 else
525 {
526 CBaseEntity *pPlayer = NULL;
527
528 pPlayer = UTIL_FindEntityByClassname( NULL, "player" );
529 // UNDONE: this needs to send different sounds to every player for multiplayer.
530 if (pPlayer)
531 {
532 float pitch = DotProduct( m_velocity - pPlayer->pev->velocity, (pPlayer->pev->origin - pev->origin).Normalize() );
533
534 pitch = (int)(100 + pitch / 75.0);
535
536 if (pitch > 250)
537 pitch = 250;
538 if (pitch < 50)
539 pitch = 50;
540
541 if (pitch == 100)
542 pitch = 101;
543
544 if (pitch != m_iPitch)
545 {
546 m_iPitch = pitch;
547 EMIT_SOUND_DYN(ENT(pev), CHAN_STATIC, "apache/ap_rotor4.wav", 1.0, 0.15, SND_CHANGE_PITCH | SND_CHANGE_VOL, pitch);
548 // ALERT( at_console, "%.0f\n", pitch );
549 }
550 }
551 // EMIT_SOUND_DYN(ENT(pev), CHAN_STATIC, "apache/ap_whine1.wav", flVol, 0.2, SND_CHANGE_PITCH | SND_CHANGE_VOL, pitch);
552 }
553
554}
