339int CCineMonster :: FindEntity( void )
340{
341 edict_t *pentTarget;
342
343 pentTarget = FIND_ENTITY_BY_TARGETNAME(NULL, STRING(m_iszEntity));
344 m_hTargetEnt = NULL;
345 CBaseMonster *pTarget = NULL;
346
347 while (!FNullEnt(pentTarget))
348 {
349 if ( FBitSet( VARS(pentTarget)->flags, FL_MONSTER ))
350 {
351 pTarget = GetMonsterPointer( pentTarget );
352 if ( pTarget && pTarget->CanPlaySequence( FCanOverrideState(), SS_INTERRUPT_BY_NAME ) )
353 {
354 m_hTargetEnt = pTarget;
355 return TRUE;
356 }
357 ALERT( at_console, "Found %s, but can't play!\n", STRING(m_iszEntity) );
358 }
359 pentTarget = FIND_ENTITY_BY_TARGETNAME(pentTarget, STRING(m_iszEntity));
360 pTarget = NULL;
361 }
362
363 if ( !pTarget )
364 {
365 CBaseEntity *pEntity = NULL;
366 while ((pEntity = UTIL_FindEntityInSphere( pEntity, pev->origin, m_flRadius )) != NULL)
367 {
368 if (FClassnameIs( pEntity->pev, STRING(m_iszEntity)))
369 {
370 if ( FBitSet( pEntity->pev->flags, FL_MONSTER ))
371 {
372 pTarget = pEntity->MyMonsterPointer( );
373 if ( pTarget && pTarget->CanPlaySequence( FCanOverrideState(), SS_INTERRUPT_IDLE ) )
374 {
375 m_hTargetEnt = pTarget;
376 return TRUE;
377 }
378 }
379 }
380 }
381 }
382 pTarget = NULL;
383 m_hTargetEnt = NULL;
384 return FALSE;
385}

468{

479 {
480 ALERT( at_aiconsole, "(AI)Can't possess entity %s\n", STRING(pTarget->pev->classname) );

483
484 pTarget->m_pGoalEnt = this;
485 pTarget->m_pCine = this;
486 pTarget->m_hTargetEnt = this;
487
488 m_saved_movetype = pTarget->pev->movetype;
489 m_saved_solid = pTarget->pev->solid;
490 m_saved_effects = pTarget->pev->effects;
491 pTarget->pev->effects |= pev->effects;
492
493 switch (m_fMoveTo)
494 {
495 case 0:
496 case 5:
497 pTarget->m_scriptState = SCRIPT_WAIT;
498 break;
499
500 case 1:
501 pTarget->m_scriptState = SCRIPT_WALK_TO_MARK;
502 break;
503
504 case 2:
505 pTarget->m_scriptState = SCRIPT_RUN_TO_MARK;
506 break;
507
508 case 4:
509 // zap the monster instantly to the site of the script entity.
510 UTIL_SetOrigin( pTarget->pev, pev->origin );
511 pTarget->pev->ideal_yaw = pev->angles.y;
512 pTarget->pev->avelocity = Vector( 0, 0, 0 );
513 pTarget->pev->velocity = Vector( 0, 0, 0 );
514 pTarget->pev->effects |= EF_NOINTERP;

519 pTarget->pev->flags &= ~FL_ONGROUND;
520 break;
521 default:
522 ALERT ( at_aiconsole, "aiscript: invalid Move To Position value!" );

524 }
525

542 {
543 pNewSchedule = pTarget->GetScheduleOfType( SCHED_AISCRIPT );
544 pTarget->ChangeSchedule( pNewSchedule );
545 }
546 }

608// lookup a sequence name and setup the target monster to play it
609// overridden for CCineAI because it's ok for them to not have an animation sequence
610// for the monster to play. For a regular Scripted Sequence, that situation is an error.
622BOOL CCineAI :: StartSequence( CBaseMonster *pTarget, int iszSeq, BOOL completeOnEmpty )
623{
624 if ( iszSeq == 0 && completeOnEmpty )
625 {
626 // no sequence was provided. Just let the monster proceed, however, we still have to fire any Sequence target
627 // and remove any non-repeatable CineAI entities here ( because there is code elsewhere that handles those tasks, but
628 // not until the animation sequence is finished. We have to manually take care of these things where there is no sequence.
629
630 SequenceDone ( pTarget );
631
632 return TRUE;
633 }
634
635 pTarget->pev->sequence = pTarget->LookupSequence( STRING( iszSeq ) );
636
637 if (pTarget->pev->sequence == -1)
638 {
639 ALERT( at_error, "%s: unknown aiscripted sequence \"%s\"\n", STRING( pTarget->pev->targetname ), STRING( iszSeq) );
640 pTarget->pev->sequence = 0;
641 // return FALSE;
642 }
643
644 pTarget->pev->frame = 0;
645 pTarget->ResetSequenceInfo( );
646 return TRUE;

649//=========================================================
650// SequenceDone - called when a scripted sequence animation
651// sequence is done playing ( or when an AI Scripted Sequence
652// doesn't supply an animation sequence to play ). Expects
653// the CBaseMonster pointer to the monster that the sequence
654// possesses.
655//=========================================================
665void CCineMonster :: SequenceDone ( CBaseMonster *pMonster )
666{
667 //ALERT( at_aiconsole, "Sequence %s finished\n", STRING( m_pCine->m_iszPlay ) );
668
669 if ( !( pev->spawnflags & SF_SCRIPT_REPEATABLE ) )
670 {
671 SetThink( &CCineMonster::SUB_Remove );
672 pev->nextthink = gpGlobals->time + 0.1;
673 }
674
675 // This is done so that another sequence can take over the monster when triggered by the first
676
677 pMonster->CineCleanup();
678
679 FixScriptMonsterSchedule( pMonster );
680
681 // This may cause a sequence to attempt to grab this guy NOW, so we have to clear him out
682 // of the existing sequence
683 SUB_UseTargets( NULL, USE_TOGGLE, 0 );
684}

686//=========================================================
687// When a monster finishes a scripted sequence, we have to
688// fix up its state and schedule for it to return to a
689// normal AI monster.
690//
691// Scripted sequences just dirty the Schedule and drop the
692// monster in Idle State.
693//=========================================================
702void CCineMonster :: FixScriptMonsterSchedule( CBaseMonster *pMonster )
703{
704 if ( pMonster->m_IdealMonsterState != MONSTERSTATE_DEAD )
705 pMonster->m_IdealMonsterState = MONSTERSTATE_IDLE;
706 pMonster->ClearSchedule();
707}
708
709//=========================================================
710// When a monster finishes a scripted sequence, we have to
711// fix up its state and schedule for it to return to a
712// normal AI monster.
713//
714// AI Scripted sequences will, depending on what the level
715// designer selects:
716//
717// -Dirty the monster's schedule and drop out of the
718// sequence in their current state.
719//
720// -Select a specific AMBUSH schedule, regardless of state.
721//=========================================================
730void CCineAI :: FixScriptMonsterSchedule( CBaseMonster *pMonster )
731{
732 switch ( m_iFinishSchedule )
733 {
734 case SCRIPT_FINISHSCHED_DEFAULT:
735 pMonster->ClearSchedule();
736 break;
737 case SCRIPT_FINISHSCHED_AMBUSH:
738 pMonster->ChangeSchedule( pMonster->GetScheduleOfType( SCHED_AMBUSH ) );
739 break;
740 default:
741 ALERT ( at_aiconsole, "FixScriptMonsterSchedule - no case!\n" );
742 pMonster->ClearSchedule();
743 break;
744 }
745}

874// find all the cinematic entities with my targetname and tell them to wait before starting
875void CCineMonster :: DelayStart( int state )
876{
877 edict_t *pentCine = FIND_ENTITY_BY_TARGETNAME(NULL, STRING(pev->targetname));
878
879 while (!FNullEnt(pentCine))
880 {
881 if (FClassnameIs( pentCine, "scripted_sequence" ))
882 {
883 CCineMonster *pTarget = GetClassPtr((CCineMonster *)VARS(pentCine));
884 if (state)
885 {
886 pTarget->m_iDelay++;
887 }
888 else
889 {
890 pTarget->m_iDelay--;
891 if (pTarget->m_iDelay <= 0)
892 pTarget->m_startTime = gpGlobals->time + 0.05;
893 }
894 }
895 pentCine = FIND_ENTITY_BY_TARGETNAME(pentCine, STRING(pev->targetname));
896 }

989 pev->deadflag = DEAD_DEAD;

995 SetThink( NULL ); // This will probably break some stuff
996 SetTouch( NULL );
997 }

999 SUB_StartFadeOut(); // SetThink( SUB_DoNothing );
1000 // This turns off animation & physics in case their origin ends up stuck in the world or something
1001 StopAnimation();
1002 pev->movetype = MOVETYPE_NONE;
1003 pev->effects |= EF_NOINTERP; // Don't interpolate either, assume the corpse is positioned in its final resting place
1004 return FALSE;
1005 }
1006
1007 // If we actually played a sequence
1008 if ( pOldCine && pOldCine->m_iszPlay )
1009 {
1010 if ( !(pOldCine->pev->spawnflags & SF_SCRIPT_NOSCRIPTMOVEMENT) )
1011 {
1012 // reset position
1013 Vector new_origin, new_angle;
1014 GetBonePosition( 0, new_origin, new_angle );
1015
1016 // Figure out how far they have moved
1017 // We can't really solve this problem because we can't query the movement of the origin relative
1018 // to the sequence. We can get the root bone's position as we do here, but there are
1019 // cases where the root bone is in a different relative position to the entity's origin
1020 // before/after the sequence plays. So we are stuck doing this:
1021
1022 // !!!HACKHACK: Float the origin up and drop to floor because some sequences have
1023 // irregular motion that can't be properly accounted for.
1024
1025 // UNDONE: THIS SHOULD ONLY HAPPEN IF WE ACTUALLY PLAYED THE SEQUENCE.
1026 Vector oldOrigin = pev->origin;
1027
1028 // UNDONE: ugly hack. Don't move monster if they don't "seem" to move
1029 // this really needs to be done with the AX,AY,etc. flags, but that aren't consistantly
1030 // being set, so animations that really do move won't be caught.
1031 if ((oldOrigin - new_origin).Length2D() < 8.0)
1032 new_origin = oldOrigin;
1033
1034 pev->origin.x = new_origin.x;
1035 pev->origin.y = new_origin.y;
1036 pev->origin.z += 1;
1037
1038 pev->flags |= FL_ONGROUND;
1039 int drop = DROP_TO_FLOOR( ENT(pev) );
1040
1041 // Origin in solid? Set to org at the end of the sequence
1042 if ( drop < 0 )
1043 pev->origin = oldOrigin;
1044 else if ( drop == 0 ) // Hanging in air?
1045 {
1046 pev->origin.z = new_origin.z;
1047 pev->flags &= ~FL_ONGROUND;
1048 }
1049 // else entity hit floor, leave there
1050
1051 // pEntity->pev->origin.z = new_origin.z + 5.0; // damn, got to fix this
1052
1053 UTIL_SetOrigin( pev, pev->origin );
1054 pev->effects |= EF_NOINTERP;
1055 }
1056
1057 // We should have some animation to put these guys in, but for now it's idle.
1058 // Due to NOINTERP above, there won't be any blending between this anim & the sequence
1059 m_Activity = ACT_RESET;
1060 }
1061 // set them back into a normal state
1062 pev->enemy = NULL;
1063 if ( pev->health > 0 )
1064 m_IdealMonsterState = MONSTERSTATE_IDLE; // m_previousState;
1065 else
1066 {
1067 // Dropping out because he got killed
1068 // Can't call killed() no attacker and weirdness (late gibbing) may result
1069 m_IdealMonsterState = MONSTERSTATE_DEAD;
1070 SetConditions( bits_COND_LIGHT_DAMAGE );
1071 pev->deadflag = DEAD_DYING;
1072 FCheckAITrigger();
1073 pev->deadflag = DEAD_NO;
1074 }
1075
1076
1077 // SetAnimation( m_MonsterState );
1078 ClearBits(pev->spawnflags, SF_MONSTER_WAIT_FOR_SCRIPT );
1079
1080 return TRUE;
