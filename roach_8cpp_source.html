94void CRoach::Touch(CBaseEntity *pOther)
95{
96 Vector vecSpot;
97 TraceResult tr;
98
99 if (pOther->pev->velocity == g_vecZero || !pOther->IsPlayer())
100 {
101 return; // Ignore if the other entity is not a moving player
102 }
103
104 vecSpot = pev->origin + Vector(0, 0, 8); // Move up a bit, and trace down
105 UTIL_TraceLine(vecSpot, vecSpot + Vector(0, 0, -24), ignore_monsters, ENT(pev), &tr);
106
107 // This isn't really blood. So you don't have to screen it out based on violence levels (UTIL_ShouldShowBlood())
108
109 UTIL_DecalTrace(&tr, DECAL_YBLOOD1 + RANDOM_LONG(0, 5));
110
111 TakeDamage(pOther->pev, pOther->pev, pev->health, DMG_CRUSH); // Apply damage to the roach
112}

209void CRoach::MonsterThink(void)
210{
211 if (FNullEnt(FIND_CLIENT_IN_PVS(edict())))
212 pev->nextthink = gpGlobals->time + RANDOM_FLOAT(1, 1.5); // Delay if no client is in PVS
213 else
214 pev->nextthink = gpGlobals->time + 0.1; // Keep monster thinking
215
216 float flInterval = StudioFrameAdvance(); // Animate the roach
217
218 if (!m_fLightHacked)
219 {
220 // If light value hasn't been collected for the first time yet,
221 // suspend the creature for a second so the world finishes spawning,
222 // then we'll collect the light level.
223 pev->nextthink = gpGlobals->time + 1;
224 m_fLightHacked = TRUE; // Set light hack flag
225 return;
226 }
227 else if (m_flLastLightLevel < 0)
228 {
229 // Collect light level for the first time
230 m_flLastLightLevel = GETENTITYILLUM(ENT(pev));
231 }
232
233 switch (m_iMode)
234 {
235 case ROACH_IDLE:
236 case ROACH_EAT:
237 {
238 // If not moving, sample environment to see if anything scary is around.
239 if (RANDOM_LONG(0, 3) == 1)
240 {
241 Look(150); // Look for entities within a radius
242 if (HasConditions(bits_COND_SEE_FEAR))
243 {
244 // If see something scary
245 Eat(30 + (RANDOM_LONG(0, 14))); // Roach will ignore food for 30 to 45 seconds
246 PickNewDest(ROACH_SCARED_BY_ENT); // Pick a new destination
247 SetActivity(ACT_WALK); // Set activity to walk
248 }
249 else if (RANDOM_LONG(0, 149) == 1)
250 {
251 // If roach doesn't see anything, there's still a chance that it will move (boredom)
252 PickNewDest(ROACH_BORED); // Pick a new destination
253 SetActivity(ACT_WALK); // Set activity to walk
254
255 if (m_iMode == ROACH_EAT)
256 {
257 // Roach will ignore food for 30 to 45 seconds if it got bored while eating.
258 Eat(30 + (RANDOM_LONG(0, 14)));
259 }
260 }
261 }
262
263 // Don't do this stuff if eating!
264 if (m_iMode == ROACH_IDLE)
265 {
266 if (FShouldEat())
267 {
268 Listen(); // Listen for sounds
269 }
270
271 if (GETENTITYILLUM(ENT(pev)) > m_flLastLightLevel)
272 {
273 // Someone turned on lights!
274 PickNewDest(ROACH_SCARED_BY_LIGHT); // Pick a new destination
275 SetActivity(ACT_WALK); // Set activity to walk
276 }
277 else if (HasConditions(bits_COND_SMELL_FOOD))
278 {
279 CSound *pSound;
280
281 pSound = CSoundEnt::SoundPointerForIndex(m_iAudibleList);
282
283 // Roach smells food and is just standing around. Go to food unless food isn't on the same z-plane.
284 if (pSound && abs(pSound->m_vecOrigin.z - pev->origin.z) <= 3)
285 {
286 PickNewDest(ROACH_SMELL_FOOD); // Pick a new destination for food
287 SetActivity(ACT_WALK); // Set activity to walk
288 }
289 }
290 }
291
292 break;
293 }
294 case ROACH_SCARED_BY_LIGHT:
295 {
296 // If roach was scared by light, then stop if we're over a spot at least as dark as where we started!
297 if (GETENTITYILLUM(ENT(pev)) <= m_flLastLightLevel)
298 {
299 SetActivity(ACT_IDLE); // Set activity to idle
300 m_flLastLightLevel = GETENTITYILLUM(ENT(pev)); // Update light level
301 }
302 break;
303 }
304 }
305
306 if (m_flGroundSpeed != 0)
307 {
308 Move(flInterval); // Move the roach
309 }
310}

319void CRoach::PickNewDest(int iCondition)
320{
321 Vector vecNewDir;
322 Vector vecDest;
323 float flDist;
324
325 m_iMode = iCondition; // Set the current mode
326
327 if (m_iMode == ROACH_SMELL_FOOD)
328 {
329 // Find the food and go there.
330 CSound *pSound;
331
332 pSound = CSoundEnt::SoundPointerForIndex(m_iAudibleList);
333
334 if (pSound)
335 {
336 m_Route[0].vecLocation.x = pSound->m_vecOrigin.x + (3 - RANDOM_LONG(0, 5));
337 m_Route[0].vecLocation.y = pSound->m_vecOrigin.y + (3 - RANDOM_LONG(0, 5));
338 m_Route[0].vecLocation.z = pSound->m_vecOrigin.z;
339 m_Route[0].iType = bits_MF_TO_LOCATION; // Set the route type
340 m_movementGoal = RouteClassify(m_Route[0].iType); // Classify the route
341 return; // Exit the function
342 }
343 }
344
345 do
346 {
347 // Picks a random spot, requiring that it be at least 128 units away
348 vecNewDir.x = RANDOM_FLOAT(-1, 1);
349 vecNewDir.y = RANDOM_FLOAT(-1, 1);
350 flDist = 256 + (RANDOM_LONG(0, 255));
351 vecDest = pev->origin + vecNewDir * flDist;
352
353 } while ((vecDest - pev->origin).Length2D() < 128); // Ensure the new destination is far enough away
354
355 m_Route[0].vecLocation.x = vecDest.x; // Set the new destination
356 m_Route[0].vecLocation.y = vecDest.y;
357 m_Route[0].vecLocation.z = pev->origin.z;
358 m_Route[0].iType = bits_MF_TO_LOCATION; // Set the route type
359 m_movementGoal = RouteClassify(m_Route[0].iType); // Classify the route
360
361 if (RANDOM_LONG(0, 9) == 1)
362 {
363 // Occasionally, a roach will play a skitter sound when they decide to run
364 EMIT_SOUND_DYN(ENT(pev), CHAN_BODY, "roach/rch_walk.wav", 1, ATTN_NORM, 0, 80 + RANDOM_LONG(0, 39));
365 }
366}

375void CRoach::Move(float flInterval)
376{
377 float flWaypointDist;
378 Vector vecApex;
379
380 // Local move to waypoint.
381 flWaypointDist = (m_Route[m_iRouteIndex].vecLocation - pev->origin).Length2D();
382 MakeIdealYaw(m_Route[m_iRouteIndex].vecLocation); // Calculate ideal yaw
383
384 ChangeYaw(pev->yaw_speed); // Change yaw based on speed
385 UTIL_MakeVectors(pev->angles); // Update direction vectors
386
387 if (RANDOM_LONG(0, 7) == 1)
388 {
389 // Randomly check for blocked path (more random load balancing)
390 if (!WALK_MOVE(ENT(pev), pev->ideal_yaw, 4, WALKMOVE_NORMAL))
391 {
392 // Stuck, so just pick a new spot to run off to
393 PickNewDest(m_iMode);
394 }
395 }
396
397 WALK_MOVE(ENT(pev), pev->ideal_yaw, m_flGroundSpeed * flInterval, WALKMOVE_NORMAL); // Move the roach
398
399 // If the waypoint is closer than step size, then stop after next step
400 if (flWaypointDist <= m_flGroundSpeed * flInterval)
401 {
402 // Take truncated step and stop
403 SetActivity(ACT_IDLE); // Set activity to idle
404 m_flLastLightLevel = GETENTITYILLUM(ENT(pev)); // Update light level
405
406 if (m_iMode == ROACH_SMELL_FOOD)
407 {
408 m_iMode = ROACH_EAT; // Set mode to eating
409 }
410 else
411 {
412 m_iMode = ROACH_IDLE; // Set mode to idle
413 }
414 }
415
416 if (RANDOM_LONG(0, 149) == 1 && m_iMode != ROACH_SCARED_BY_LIGHT && m_iMode != ROACH_SMELL_FOOD)
417 {
418 // Random skitter while moving as long as not on a direct path to get out of light or going to food
419 PickNewDest(FALSE); // Pick a new destination
420 }

431void CRoach::Look(int iDistance)
432{
433 CBaseEntity *pSightEnt = NULL; // The current visible entity
434 CBaseEntity *pPreviousEnt; // The last entity added to the link list
435 int iSighted = 0; // Variable to track sighted conditions
436
437 // Clear previous visibility conditions
438 ClearConditions(bits_COND_SEE_HATE | bits_COND_SEE_DISLIKE | bits_COND_SEE_ENEMY | bits_COND_SEE_FEAR);
439
440 // Don't let monsters outside of the player's PVS act up
441 if (FNullEnt(FIND_CLIENT_IN_PVS(edict())))
442 {
443 return; // Exit if no client is in PVS
444 }
445
446 m_pLink = NULL; // Initialize link pointer
447 pPreviousEnt = this; // Set previous entity to self
448
449 // Examine all entities within a reasonable radius
450 while ((pSightEnt = UTIL_FindEntityInSphere(pSightEnt, pev->origin, iDistance)) != NULL)
451 {
452 // Only consider entities that can be damaged
453 if (pSightEnt->IsPlayer() || FBitSet(pSightEnt->pev->flags, FL_MONSTER))
454 {
455 if (!FBitSet(pSightEnt->pev->flags, FL_NOTARGET) && pSightEnt->pev->health > 0)
456 {
457 // Add the entity to the link list
458 pPreviousEnt->m_pLink = pSightEnt;
459 pSightEnt->m_pLink = NULL;
460 pPreviousEnt = pSightEnt;
461
462 // Update sighted conditions based on relationships
463 switch (IRelationship(pSightEnt))
464 {
465 case R_FR:
466 iSighted |= bits_COND_SEE_FEAR; // Set fear condition
467 break;
468 case R_NO:
469 break;
470 default:
471 ALERT(at_console, "%s can't assess %s\n", STRING(pev->classname), STRING(pSightEnt->pev->classname));
472 break;
473 }
474 }
475 }
476 }
477 SetConditions(iSighted); // Update the roach's conditions
478}
