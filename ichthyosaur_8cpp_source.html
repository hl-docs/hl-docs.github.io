506void CIchthyosaur :: HandleAnimEvent( MonsterEvent_t *pEvent )
507{
508 int bDidAttack = FALSE;
509 switch( pEvent->event )
510 {
511 case ICHTHYOSAUR_AE_SHAKE_RIGHT:
512 case ICHTHYOSAUR_AE_SHAKE_LEFT:
513 {
514 if (m_hEnemy != NULL && FVisible( m_hEnemy ))
515 {
516 CBaseEntity *pHurt = m_hEnemy;
517
518 if (m_flEnemyTouched < gpGlobals->time - 0.2 && (m_hEnemy->BodyTarget( pev->origin ) - pev->origin).Length() > (32+16+32))
519 break;
520
521 Vector vecShootDir = ShootAtEnemy( pev->origin );
522 UTIL_MakeAimVectors ( pev->angles );
523
524 if (DotProduct( vecShootDir, gpGlobals->v_forward ) > 0.707)
525 {
526 m_bOnAttack = TRUE;
527 pHurt->pev->punchangle.z = -18;
528 pHurt->pev->punchangle.x = 5;
529 pHurt->pev->velocity = pHurt->pev->velocity - gpGlobals->v_right * 300;
530 if (pHurt->IsPlayer())
531 {
532 pHurt->pev->angles.x += RANDOM_FLOAT( -35, 35 );
533 pHurt->pev->angles.y += RANDOM_FLOAT( -90, 90 );
534 pHurt->pev->angles.z = 0;
535 pHurt->pev->fixangle = TRUE;
536 }
537 pHurt->TakeDamage( pev, pev, gSkillData.ichthyosaurDmgShake, DMG_SLASH );
538 }
539 }
540 BiteSound();
541
542 bDidAttack = TRUE;
543 }
544 break;
545 default:
546 CFlyingMonster::HandleAnimEvent( pEvent );
547 break;
548 }
549
550 if (bDidAttack)
551 {
552 Vector vecSrc = pev->origin + gpGlobals->v_forward * 32;
553 UTIL_Bubbles( vecSrc - Vector( 8, 8, 8 ), vecSrc + Vector( 8, 8, 8 ), 16 );
554 }
555}

575 m_flFieldOfView = VIEW_FIELD_WIDE;
576 m_MonsterState = MONSTERSTATE_NONE;
577 SetBits(pev->flags, FL_SWIM);
578 SetFlyingSpeed( ICHTHYOSAUR_SPEED );
579 SetFlyingMomentum( 2.5 ); // Set momentum constant
580
581 m_afCapability = bits_CAP_RANGE_ATTACK1 | bits_CAP_SWIM;
582
583 MonsterInit();
584
585 SetTouch( &CIchthyosaur::BiteTouch );
586 SetUse( &CIchthyosaur::CombatUse );
587
588 m_idealDist = 384;
589 m_flMinSpeed = 80;
590 m_flMaxSpeed = 300;
591 m_flMaxDist = 384;
592
593 Vector Forward;
594 UTIL_MakeVectorsPrivate(pev->angles, Forward, 0, 0);
595 pev->velocity = m_flightSpeed * Forward.Normalize();
596 m_SaveVelocity = pev->velocity;
597}

624Schedule_t* CIchthyosaur::GetSchedule()
625{
626 // ALERT( at_console, "GetSchedule( )\n" );
627 switch(m_MonsterState)
628 {
629 case MONSTERSTATE_IDLE:
630 m_flightSpeed = 80;
631 return GetScheduleOfType( SCHED_IDLE_WALK );
632
633 case MONSTERSTATE_ALERT:
634 m_flightSpeed = 150;
635 return GetScheduleOfType( SCHED_IDLE_WALK );
636
637 case MONSTERSTATE_COMBAT:
638 m_flMaxSpeed = 400;
639 // eat them
640 if ( HasConditions( bits_COND_CAN_MELEE_ATTACK1 ) )
641 {
642 return GetScheduleOfType( SCHED_MELEE_ATTACK1 );
643 }
644 // chase them down and eat them
645 if ( HasConditions( bits_COND_CAN_RANGE_ATTACK1 ) )
646 {
647 return GetScheduleOfType( SCHED_CHASE_ENEMY );
648 }
649 if ( HasConditions( bits_COND_HEAVY_DAMAGE ) )
650 {
651 m_bOnAttack = TRUE;
652 }
653 if ( pev->health < pev->max_health - 20 )
654 {
655 m_bOnAttack = TRUE;
656 }
657
658 return GetScheduleOfType( SCHED_STANDOFF );
659 }
660
661 return CFlyingMonster :: GetSchedule();
662}

741void CIchthyosaur :: RunTask ( Task_t *pTask )
742{
743 switch ( pTask->iTask )
744 {
745 case TASK_ICHTHYOSAUR_CIRCLE_ENEMY:
746 if (m_hEnemy == NULL)
747 {
748 TaskComplete( );
749 }
750 else if (FVisible( m_hEnemy ))
751 {
752 Vector vecFrom = m_hEnemy->EyePosition( );
753
754 Vector vecDelta = (pev->origin - vecFrom).Normalize( );
755 Vector vecSwim = CrossProduct( vecDelta, Vector( 0, 0, 1 ) ).Normalize( );
756
757 if (DotProduct( vecSwim, m_SaveVelocity ) < 0)
758 vecSwim = vecSwim * -1.0;
759
760 Vector vecPos = vecFrom + vecDelta * m_idealDist + vecSwim * 32;
761
762 // ALERT( at_console, "vecPos %.0f %.0f %.0f\n", vecPos.x, vecPos.y, vecPos.z );
763
764 TraceResult tr;
765
766 UTIL_TraceHull( vecFrom, vecPos, ignore_monsters, large_hull, m_hEnemy->edict(), &tr );
767
768 if (tr.flFraction > 0.5)
769 vecPos = tr.vecEndPos;
770
771 m_SaveVelocity = m_SaveVelocity * 0.8 + 0.2 * (vecPos - pev->origin).Normalize() * m_flightSpeed;
772
773 // ALERT( at_console, "m_SaveVelocity %.2f %.2f %.2f\n", m_SaveVelocity.x, m_SaveVelocity.y, m_SaveVelocity.z );
774
775 if (HasConditions( bits_COND_ENEMY_FACING_ME ) && m_hEnemy->FVisible( this ))
776 {
777 m_flNextAlert -= 0.1;
778
779 if (m_idealDist < m_flMaxDist)
780 {
781 m_idealDist += 4;
782 }
783 if (m_flightSpeed > m_flMinSpeed)
784 {
785 m_flightSpeed -= 2;
786 }
787 else if (m_flightSpeed < m_flMinSpeed)
788 {
789 m_flightSpeed += 2;
790 }
791 if (m_flMinSpeed < m_flMaxSpeed)
792 {
793 m_flMinSpeed += 0.5;
794 }
795 }
796 else
797 {
798 m_flNextAlert += 0.1;
799
800 if (m_idealDist > 128)
801 {
802 m_idealDist -= 4;
803 }
804 if (m_flightSpeed < m_flMaxSpeed)
805 {
806 m_flightSpeed += 4;
807 }
808 }
809 // ALERT( at_console, "%.0f\n", m_idealDist );
810 }
811 else
812 {
813 m_flNextAlert = gpGlobals->time + 0.2;
814 }
815
816 if (m_flNextAlert < gpGlobals->time)
817 {
818 // ALERT( at_console, "AlertSound()\n");
819 AlertSound( );
820 m_flNextAlert = gpGlobals->time + RANDOM_FLOAT( 3, 5 );
821 }
822
823 break;
824 case TASK_ICHTHYOSAUR_SWIM:
825 if (m_fSequenceFinished)
826 {
827 TaskComplete( );
828 }
829 break;
830 case TASK_DIE:
831 if ( m_fSequenceFinished )
832 {
833 pev->deadflag = DEAD_DEAD;
834
835 TaskComplete( );
836 }
837 break;
838
839 case TASK_ICHTHYOSAUR_FLOAT:
840 pev->angles.x = UTIL_ApproachAngle( 0, pev->angles.x, 20 );
841 pev->velocity = pev->velocity * 0.8;
842 if (pev->waterlevel > 1 && pev->velocity.z < 64)
843 {
844 pev->velocity.z += 8;
845 }
846 else
847 {
848 pev->velocity.z -= 8;
849 }
850 // ALERT( at_console, "%f\n", pev->velocity.z );
851 break;
852
853 default:
854 CFlyingMonster :: RunTask ( pTask );
855 break;
856 }
857}

868float CIchthyosaur::VectorToPitch( const Vector &vec )
869{
870 float pitch;
871 if (vec.z == 0 && vec.x == 0)
872 pitch = 0;
873 else
874 {
875 pitch = (int) (atan2(vec.z, sqrt(vec.x*vec.x+vec.y*vec.y)) * 180 / M_PI);
876 if (pitch < 0)
877 pitch += 360;
878 }
879 return pitch;
880}

900float CIchthyosaur::FlPitchDiff( void )
901{
902 float flPitchDiff;
903 float flCurrentPitch;
904
905 flCurrentPitch = UTIL_AngleMod( pev->angles.z );
906
907 if ( flCurrentPitch == pev->idealpitch )
908 {
909 return 0;
910 }
911
912 flPitchDiff = pev->idealpitch - flCurrentPitch;
913
914 if ( pev->idealpitch > flCurrentPitch )
915 {
916 if (flPitchDiff >= 180)
917 flPitchDiff = flPitchDiff - 360;
918 }
919 else
920 {
921 if (flPitchDiff <= -180)
922 flPitchDiff = flPitchDiff + 360;
923 }
924 return flPitchDiff;
925}
