262// Purpose: determine if a uchar32 represents a valid Unicode code point
263//-----------------------------------------------------------------------------
273bool Q_IsValidUChar32(uchar32 uVal)
274{
275 // Values > 0x10FFFF are explicitly invalid; ditto for UTF-16 surrogate halves,
276 // values ending in FFFE or FFFF, or values in the 0x00FDD0-0x00FDEF reserved range
277 return (uVal < 0x110000u) && ((uVal - 0x00D800u) > 0x7FFu) && ((uVal & 0xFFFFu) < 0xFFFEu) && ((uVal - 0x00FDD0u) > 0x1Fu);
278}

291int Q_UTF8ToUChar32(const char *pUTF8_, uchar32 &uValueOut, bool &bErrorOut)
292{
293 const uint8 *pUTF8 = (const uint8 *)pUTF8_;
294
295 int nBytes = 1;
296 uint32 uValue = pUTF8[0];
297 uint32 uMinValue = 0;
298
299 // 0....... single byte
300 if (uValue < 0x80)
301 goto decodeFinishedNoCheck;
302
303 // Expecting at least a two-byte sequence with 0xC0 <= first <= 0xF7 (110...... and 11110...)
304 if ((uValue - 0xC0u) > 0x37u || (pUTF8[1] & 0xC0) != 0x80)
305 goto decodeError;
306
307 uValue = (uValue << 6) - (0xC0 << 6) + pUTF8[1] - 0x80;
308 nBytes = 2;
309 uMinValue = 0x80;
310
311 // 110..... two-byte lead byte
312 if (!(uValue & (0x20 << 6)))
313 goto decodeFinished;
314
315 // Expecting at least a three-byte sequence
316 if ((pUTF8[2] & 0xC0) != 0x80)
317 goto decodeError;
318
319 uValue = (uValue << 6) - (0x20 << 12) + pUTF8[2] - 0x80;
320 nBytes = 3;
321 uMinValue = 0x800;
322
323 // 1110.... three-byte lead byte
324 if (!(uValue & (0x10 << 12)))
325 goto decodeFinishedMaybeCESU8;
326
327 // Expecting a four-byte sequence, longest permissible in UTF-8
328 if ((pUTF8[3] & 0xC0) != 0x80)
329 goto decodeError;
330
331 uValue = (uValue << 6) - (0x10 << 18) + pUTF8[3] - 0x80;
332 nBytes = 4;
333 uMinValue = 0x10000;
334
335 // 11110... four-byte lead byte. fall through to finished.
336
337decodeFinished:
338 if (uValue >= uMinValue && Q_IsValidUChar32(uValue))
339 {
340decodeFinishedNoCheck:
341 uValueOut = uValue;
342 bErrorOut = false;
343 return nBytes;
344 }
345decodeError:
346 uValueOut = '?';
347 bErrorOut = true;
348 return nBytes;
349
350decodeFinishedMaybeCESU8:
351 // Do we have a full UTF-16 surrogate pair that's been UTF-8 encoded afterwards?
352 // That is, do we have 0xD800-0xDBFF followed by 0xDC00-0xDFFF? If so, decode it all.
353 if ((uValue - 0xD800u) < 0x400u && pUTF8[3] == 0xED && (uint8)(pUTF8[4] - 0xB0) < 0x10 && (pUTF8[5] & 0xC0) == 0x80)
354 {
355 uValue = 0x10000 + ((uValue - 0xD800u) << 10) + ((uint8)(pUTF8[4] - 0xB0) << 6) + pUTF8[5] - 0x80;
356 nBytes = 6;
357 uMinValue = 0x10000;
358 }
359 goto decodeFinished;
360}

371bool Q_UnicodeValidate(const char *pUTF8)
372{
373 bool bError = false;
374 while (*pUTF8)
375 {
376 uchar32 uVal;
377 // Our UTF-8 decoder silently fixes up 6-byte CESU-8 (improperly re-encoded UTF-16) sequences.
378 // However, these are technically not valid UTF-8. So if we eat 6 bytes at once, it's an error.
379 int nCharSize = Q_UTF8ToUChar32(pUTF8, uVal, bError);
380 if (bError || nCharSize == 6)
381 return false;
382 pUTF8 += nCharSize;
383 }
384 return true;
385}

396void Host_Say(edict_t *pEntity, int teamonly)
397{
398 CBasePlayer *client;
399 int j;
400 char *p;
401 char text[128];
402 char szTemp[256];
403 const char *cpSay = "say";
404 const char *cpSayTeam = "say_team";
405 const char *pcmd = CMD_ARGV(0);
406
407 // We can get a raw string now, without the "say " prepended
408 if ( CMD_ARGC() == 0 )
409 return;
410
411 entvars_t *pev = &pEntity->v;
412 CBasePlayer* player = GetClassPtr((CBasePlayer *)pev);
413
414 //Not yet.
415 if ( player->m_flNextChatTime > gpGlobals->time )
416 return;
417
418 if ( !stricmp( pcmd, cpSay) || !stricmp( pcmd, cpSayTeam ) )
419 {
420 if ( CMD_ARGC() >= 2 )
421 {
422 p = (char *)CMD_ARGS();
423 }
424 else
425 {
426 // say with a blank message, nothing to do
427 return;
428 }
429 }
430 else // Raw text, need to prepend argv[0]
431 {
432 if ( CMD_ARGC() >= 2 )
433 {
434 sprintf( szTemp, "%s %s", ( char * )pcmd, (char *)CMD_ARGS() );
435 }
436 else
437 {
438 // Just a one word command, use the first word...sigh
439 sprintf( szTemp, "%s", ( char * )pcmd );
440 }
441 p = szTemp;
442 }
443
444// remove quotes if present
445 if (*p == '"')
446 {
447 p++;
448 p[strlen(p)-1] = 0;
449 }
450
451// make sure the text has content
452
453 if ( !p || !p[0] || !Q_UnicodeValidate ( p ) )
454 return; // no character found, so say nothing
455
456// turn on color set 2 (color on, no sound)
457 // turn on color set 2 (color on, no sound)
458 if ( player->IsObserver() && ( teamonly ) )
459 sprintf( text, "%c(SPEC) %s: ", 2, STRING( pEntity->v.netname ) );
460 else if ( teamonly )
461 sprintf( text, "%c(TEAM) %s: ", 2, STRING( pEntity->v.netname ) );
462 else
463 sprintf( text, "%c%s: ", 2, STRING( pEntity->v.netname ) );
464
465 j = sizeof(text) - 2 - strlen(text); // -2 for /n and null terminator
466 if ( (int)strlen(p) > j )
467 p[j] = 0;
468
469 strcat( text, p );
470 strcat( text, "\n" );
471
472
473 player->m_flNextChatTime = gpGlobals->time + CHAT_INTERVAL;
474
475 // loop through all players
476 // Start with the first player.
477 // This may return the world in single player if the client types something between levels or during spawn
478 // so check it, or it will infinite loop
479
480 client = NULL;
481 while ( ((client = (CBasePlayer*)UTIL_FindEntityByClassname( client, "player" )) != NULL) && (!FNullEnt(client->edict())) )
482 {
483 if ( !client->pev )
484 continue;
485
486 if ( client->edict() == pEntity )
487 continue;
488
489 if ( !(client->IsNetClient()) ) // Not a client ? (should never be true)
490 continue;
491
492 // can the receiver hear the sender? or has he muted him?
493 if ( g_VoiceGameMgr.PlayerHasBlockedPlayer( client, player ) )
494 continue;
495
496 if ( !player->IsObserver() && teamonly && g_pGameRules->PlayerRelationship(client, CBaseEntity::Instance(pEntity)) != GR_TEAMMATE )
497 continue;
498
499 // Spectators can only talk to other specs
500 if ( player->IsObserver() && teamonly )
501 if ( !client->IsObserver() )
502 continue;
503
504 MESSAGE_BEGIN( MSG_ONE, gmsgSayText, NULL, client->pev );
505 WRITE_BYTE( ENTINDEX(pEntity) );
506 WRITE_STRING( text );
507 MESSAGE_END();
508
509 }
510
511 // print to the sending client
512 MESSAGE_BEGIN( MSG_ONE, gmsgSayText, NULL, &pEntity->v );
513 WRITE_BYTE( ENTINDEX(pEntity) );
514 WRITE_STRING( text );
515 MESSAGE_END();
516
517 // echo to server console
518 g_engfuncs.pfnServerPrint( text );
519
520 char * temp;
521 if ( teamonly )
522 temp = "say_team";
523 else
524 temp = "say";
525
526 // team match?
527 if ( g_teamplay )
528 {
529 UTIL_LogPrintf( "\"%s<%i><%s><%s>\" %s \"%s\"\n",
530 STRING( pEntity->v.netname ),
531 GETPLAYERUSERID( pEntity ),
532 GETPLAYERAUTHID( pEntity ),
533 g_engfuncs.pfnInfoKeyValue( g_engfuncs.pfnGetInfoKeyBuffer( pEntity ), "model" ),
534 temp,
535 p );
536 }
537 else
538 {
539 UTIL_LogPrintf( "\"%s<%i><%s><%i>\" %s \"%s\"\n",
540 STRING( pEntity->v.netname ),
541 GETPLAYERUSERID( pEntity ),
542 GETPLAYERAUTHID( pEntity ),
543 GETPLAYERUSERID( pEntity ),
544 temp,
545 p );
546 }
547}

569void ClientCommand( edict_t *pEntity )
570{
571 const char *pcmd = CMD_ARGV(0);
572 const char *pstr;
573
574 // Is the client spawned yet?
575 if ( !pEntity->pvPrivateData )
576 return;
577
578 entvars_t *pev = &pEntity->v;
579
580 if ( FStrEq(pcmd, "say" ) )
581 {
582 Host_Say( pEntity, 0 );
583 }
584 else if ( FStrEq(pcmd, "say_team" ) )
585 {
586 Host_Say( pEntity, 1 );
587 }
588 else if ( FStrEq(pcmd, "fullupdate" ) )
589 {
590 GetClassPtr((CBasePlayer *)pev)->ForceClientDllUpdate();
591 }
592 else if ( FStrEq(pcmd, "give" ) )
593 {
594 if ( g_flWeaponCheat != 0.0)
595 {
596 int iszItem = ALLOC_STRING( CMD_ARGV(1) ); // Make a copy of the classname
597 GetClassPtr((CBasePlayer *)pev)->GiveNamedItem( STRING(iszItem) );
598 }
599 }
600
601 else if ( FStrEq(pcmd, "drop" ) )
602 {
603 // player is dropping an item.
604 GetClassPtr((CBasePlayer *)pev)->DropPlayerItem((char *)CMD_ARGV(1));
605 }
606 else if ( FStrEq(pcmd, "fov" ) )
607 {
608 if ( g_flWeaponCheat && CMD_ARGC() > 1)
609 {
610 GetClassPtr((CBasePlayer *)pev)->m_iFOV = atoi( CMD_ARGV(1) );
611 }
612 else
613 {
614 CLIENT_PRINTF( pEntity, print_console, UTIL_VarArgs( "\"fov\" is \"%d\"\n", (int)GetClassPtr((CBasePlayer *)pev)->m_iFOV ) );
615 }
616 }
617 else if ( FStrEq(pcmd, "use" ) )
618 {
619 GetClassPtr((CBasePlayer *)pev)->SelectItem((char *)CMD_ARGV(1));
620 }
621 else if (((pstr = strstr(pcmd, "weapon_")) != NULL) && (pstr == pcmd))
622 {
623 GetClassPtr((CBasePlayer *)pev)->SelectItem(pcmd);
624 }
625 else if (FStrEq(pcmd, "lastinv" ))
626 {
627 GetClassPtr((CBasePlayer *)pev)->SelectLastItem();
628 }
629 else if ( FStrEq( pcmd, "spectate" ) ) // clients wants to become a spectator
630 {
631 // always allow proxies to become a spectator
632 if ( (pev->flags & FL_PROXY) || allow_spectators.value )
633 {
634 CBasePlayer * pPlayer = GetClassPtr((CBasePlayer *)pev);
635
636 edict_t *pentSpawnSpot = g_pGameRules->GetPlayerSpawnSpot( pPlayer );
637 pPlayer->StartObserver( pev->origin, VARS(pentSpawnSpot)->angles);
638
639 // notify other clients of player switching to spectator mode
640 UTIL_ClientPrintAll( HUD_PRINTNOTIFY, UTIL_VarArgs( "%s switched to spectator mode\n",
641 ( pev->netname && STRING(pev->netname)[0] != 0 ) ? STRING(pev->netname) : "unconnected" ) );
642 }
643 else
644 ClientPrint( pev, HUD_PRINTCONSOLE, "Spectator mode is disabled.\n" );
645
646 }
647 else if ( FStrEq( pcmd, "specmode" ) ) // new spectator mode
648 {
649 CBasePlayer * pPlayer = GetClassPtr((CBasePlayer *)pev);
650
651 if ( pPlayer->IsObserver() )
652 pPlayer->Observer_SetMode( atoi( CMD_ARGV(1) ) );
653 }
654 else if ( FStrEq(pcmd, "closemenus" ) )
655 {
656 // just ignore it
657 }
658 else if ( FStrEq( pcmd, "follownext" ) ) // follow next player
659 {
660 CBasePlayer * pPlayer = GetClassPtr((CBasePlayer *)pev);
661
662 if ( pPlayer->IsObserver() )
663 pPlayer->Observer_FindNextPlayer( atoi( CMD_ARGV(1) )?true:false );
664 }
665 else if ( g_pGameRules->ClientCommand( GetClassPtr((CBasePlayer *)pev), pcmd ) )
666 {
667 // MenuSelect returns true only if the command is properly handled, so don't print a warning
668 }
669 else
670 {
671 // tell the user they entered an unknown command
672 char command[128];
673
674 // check the length of the command (prevents crash)
675 // max total length is 192 ...and we're adding a string below ("Unknown command: %s\n")
676 strncpy( command, pcmd, 127 );
677 command[127] = '\0';
678
679 // tell the user they entered an unknown command
680 ClientPrint( &pEntity->v, HUD_PRINTCONSOLE, UTIL_VarArgs( "Unknown command: %s\n", command ) );
681 }
682}

706 // Is the client spawned yet?
707 if ( !pEntity->pvPrivateData )
708 return;

710 // msg everyone if someone changes their name, and it isn't the first time (changing no name to current name)

716 sName[ sizeof(sName) - 1 ] = '\0';
717
718 // First parse the name and remove any %'s
719 for ( char *pApersand = sName; pApersand != NULL && *pApersand != 0; pApersand++ )
720 {
721 // Replace it with a space
722 if ( *pApersand == '%' )
723 *pApersand = ' ';
724 }
725
726 // Set the name

728
729 if (gpGlobals->maxClients > 1)
730 {
731 char text[256];
732 sprintf( text, "* %s changed name to %s\n", STRING(pEntity->v.netname), g_engfuncs.pfnInfoKeyValue( infobuffer, "name" ) );

792void ServerActivate(edict_t *pEdictList, int edictCount, int clientMax)
793{
794 int i;
795 CBaseEntity *pClass;
796
797 // Every call to ServerActivate should be matched by a call to ServerDeactivate
798 g_serveractive = 1;
799
800 // Clients have not been initialized yet
801 for ( i = 0; i < edictCount; i++ )
802 {
803 if ( pEdictList[i].free )
804 continue;

817 {
818 ALERT( at_console, "Can't instance %s\n", STRING(pEdictList[i].v.classname) );

1226int AddToFullPack(struct entity_state_s *state, int e, edict_t *ent, edict_t *host, int hostflags, int player, unsigned char *pSet)
1227{
1228 int i;
1229
1230 // don't send if flagged for NODRAW and it's not the host getting the message
1231 if ( ( ent->v.effects & EF_NODRAW ) &&
1232 ( ent != host ) )
1233 return 0;
1234
1235 // Ignore ents without valid / visible models
1236 if ( !ent->v.modelindex || !STRING( ent->v.model ) )
1237 return 0;
1238
1239 // Don't send spectators to other players
1240 if ( ( ent->v.flags & FL_SPECTATOR ) && ( ent != host ) )
1241 {
1242 return 0;
1243 }
1244
1245 // Ignore if not the host and not touching a PVS/PAS leaf
1246 // If pSet is NULL, then the test will always succeed and the entity will be added to the update
1247 if ( ent != host )
1248 {
1249 if ( !ENGINE_CHECK_VISIBILITY( (const struct edict_s *)ent, pSet ) )
1250 {
1251 return 0;
1252 }
1253 }
1254
1255
1256 // Don't send entity to local client if the client says it's predicting the entity itself.
1257 if ( ent->v.flags & FL_SKIPLOCALHOST )
1258 {
1259 if ( ( hostflags & 1 ) && ( ent->v.owner == host ) )
1260 return 0;
1261 }
1262
1263 if ( host->v.groupinfo )
1264 {
1265 UTIL_SetGroupTrace( host->v.groupinfo, GROUP_OP_AND );
1266
1267 // Should always be set, of course
1268 if ( ent->v.groupinfo )
1269 {
1270 if ( g_groupop == GROUP_OP_AND )
1271 {
1272 if ( !(ent->v.groupinfo & host->v.groupinfo ) )
1273 return 0;
1274 }
1275 else if ( g_groupop == GROUP_OP_NAND )
1276 {
1277 if ( ent->v.groupinfo & host->v.groupinfo )
1278 return 0;
1279 }
1280 }
1281
1282 UTIL_UnsetGroupTrace();
1283 }
1284
1285 memset( state, 0, sizeof( *state ) );
1286
1287 // Assign index so we can track this entity from frame to frame and
1288 // delta from it.
1289 state->number = e;
1290 state->entityType = ENTITY_NORMAL;
1291
1292 // Flag custom entities.
1293 if ( ent->v.flags & FL_CUSTOMENTITY )
1294 {
1295 state->entityType = ENTITY_BEAM;
1296 }
1297
1298 //
1299 // Copy state data
1300 //
1301
1302 // Round animtime to nearest millisecond
1303 state->animtime = (int)(1000.0 * ent->v.animtime ) / 1000.0;
1304
1305 memcpy( state->origin, ent->v.origin, 3 * sizeof( float ) );
1306 memcpy( state->angles, ent->v.angles, 3 * sizeof( float ) );
1307 memcpy( state->mins, ent->v.mins, 3 * sizeof( float ) );
1308 memcpy( state->maxs, ent->v.maxs, 3 * sizeof( float ) );
1309
1310 memcpy( state->startpos, ent->v.startpos, 3 * sizeof( float ) );
1311 memcpy( state->endpos, ent->v.endpos, 3 * sizeof( float ) );
1312
1313 state->impacttime = ent->v.impacttime;
1314 state->starttime = ent->v.starttime;
1315
1316 state->modelindex = ent->v.modelindex;
1317
1318 state->frame = ent->v.frame;
1319
1320 state->skin = ent->v.skin;
1321 state->effects = ent->v.effects;
1322
1323 // This non-player entity is being moved by the game .dll and not the physics simulation system
1324 // make sure that we interpolate it's position on the client if it moves
1325 if ( !player &&
1326 ent->v.animtime &&
1327 ent->v.velocity[ 0 ] == 0 &&
1328 ent->v.velocity[ 1 ] == 0 &&
1329 ent->v.velocity[ 2 ] == 0 )
1330 {
1331 state->eflags |= EFLAG_SLERP;
1332 }
1333
1334 state->scale = ent->v.scale;
1335 state->solid = ent->v.solid;
1336 state->colormap = ent->v.colormap;
1337
1338 state->movetype = ent->v.movetype;
1339 state->sequence = ent->v.sequence;
1340 state->framerate = ent->v.framerate;
1341 state->body = ent->v.body;
1342
1343 for (i = 0; i < 4; i++)
1344 {
1345 state->controller[i] = ent->v.controller[i];
1346 }
1347
1348 for (i = 0; i < 2; i++)
1349 {
1350 state->blending[i] = ent->v.blending[i];
1351 }
1352
1353 state->rendermode = ent->v.rendermode;
1354 state->renderamt = ent->v.renderamt;
1355 state->renderfx = ent->v.renderfx;
1356 state->rendercolor.r = ent->v.rendercolor.x;
1357 state->rendercolor.g = ent->v.rendercolor.y;
1358 state->rendercolor.b = ent->v.rendercolor.z;
1359
1360 state->aiment = 0;
1361 if ( ent->v.aiment )
1362 {
1363 state->aiment = ENTINDEX( ent->v.aiment );
1364 }
1365
1366 state->owner = 0;
1367 if ( ent->v.owner )
1368 {
1369 int owner = ENTINDEX( ent->v.owner );
1370
1371 // Only care if owned by a player
1372 if ( owner >= 1 && owner <= gpGlobals->maxClients )
1373 {
1374 state->owner = owner;
1375 }
1376 }
1377
1378 // HACK: Somewhat...
1379 // Class is overridden for non-players to signify a breakable glass object ( sort of a class? )
1380 if ( !player )
1381 {
1382 state->playerclass = ent->v.playerclass;
1383 }
1384
1385 // Special stuff for players only
1386 if ( player )
1387 {
1388 memcpy( state->basevelocity, ent->v.basevelocity, 3 * sizeof( float ) );
1389
1390 state->weaponmodel = MODEL_INDEX( STRING( ent->v.weaponmodel ) );
1391 state->gaitsequence = ent->v.gaitsequence;
1392 state->spectator = ent->v.flags & FL_SPECTATOR;
1393 state->friction = ent->v.friction;
1394
1395 state->gravity = ent->v.gravity;
1396// state->team = ent->v.team;
1397//

1400 }
