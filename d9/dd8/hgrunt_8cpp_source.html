279void CHGrunt::GibMonster(void)
280{
281 Vector vecGunPos;
282 Vector vecGunAngles;
283
284 if (GetBodygroup(2) != 2) // Check if the grunt has a gun
285 {
286 GetAttachment(0, vecGunPos, vecGunAngles); // Get the position and angles for the gun
287
288 CBaseEntity *pGun;
289 if (FBitSet(pev->weapons, HGRUNT_SHOTGUN))
290 {
291 pGun = DropItem("weapon_shotgun", vecGunPos, vecGunAngles); // Drop shotgun
292 }
293 else
294 {
295 pGun = DropItem("weapon_9mmAR", vecGunPos, vecGunAngles); // Drop 9mm AR
296 }
297 if (pGun)
298 {
299 pGun->pev->velocity = Vector(RANDOM_FLOAT(-100, 100), RANDOM_FLOAT(-100, 100), RANDOM_FLOAT(200, 300)); // Set gun velocity
300 pGun->pev->avelocity = Vector(0, RANDOM_FLOAT(200, 400), 0); // Set gun angular velocity
301 }
302
303 if (FBitSet(pev->weapons, HGRUNT_GRENADELAUNCHER))
304 {
305 pGun = DropItem("ammo_ARgrenades", vecGunPos, vecGunAngles); // Drop grenade ammo
306 if (pGun)
307 {
308 pGun->pev->velocity = Vector(RANDOM_FLOAT(-100, 100), RANDOM_FLOAT(-100, 100), RANDOM_FLOAT(200, 300)); // Set ammo velocity

460BOOL CHGrunt :: CheckRangeAttack1 ( float flDot, float flDist )
461{
462 if ( !HasConditions( bits_COND_ENEMY_OCCLUDED ) && flDist <= 2048 && flDot >= 0.5 && NoFriendlyFire() )
463 {
464 TraceResult tr;
465
466 if ( !m_hEnemy->IsPlayer() && flDist <= 64 )
467 {
468 // kick nonclients, but don't shoot at them.
469 return FALSE;
470 }
471
472 Vector vecSrc = GetGunPosition();
473
474 // verify that a bullet fired from the gun will hit the enemy before the world.
475 UTIL_TraceLine( vecSrc, m_hEnemy->BodyTarget(vecSrc), ignore_monsters, ignore_glass, ENT(pev), &tr);
476
477 if ( tr.flFraction == 1.0 )
478 {
479 return TRUE;
480 }
481 }
482
483 return FALSE;
484}

496BOOL CHGrunt :: CheckRangeAttack2 ( float flDot, float flDist )
497{
498 if (! FBitSet(pev->weapons, (HGRUNT_HANDGRENADE | HGRUNT_GRENADELAUNCHER)))
499 {
500 return FALSE;
501 }
502
503 // if the grunt isn't moving, it's ok to check.
504 if ( m_flGroundSpeed != 0 )
505 {
506 m_fThrowGrenade = FALSE;
507 return m_fThrowGrenade;
508 }
509
510 // assume things haven't changed too much since last time
511 if (gpGlobals->time < m_flNextGrenadeCheck )
512 {
513 return m_fThrowGrenade;
514 }
515
516 if ( !FBitSet ( m_hEnemy->pev->flags, FL_ONGROUND ) && m_hEnemy->pev->waterlevel == 0 && m_vecEnemyLKP.z > pev->absmax.z )
517 {
519 // be grenaded.
520 // don't throw grenades at anything that isn't on the ground!
521 m_fThrowGrenade = FALSE;
522 return m_fThrowGrenade;
523 }
524
525 Vector vecTarget;
526
527 if (FBitSet( pev->weapons, HGRUNT_HANDGRENADE))
528 {
529 // find feet
530 if (RANDOM_LONG(0,1))
531 {
532 // magically know where they are
533 vecTarget = Vector( m_hEnemy->pev->origin.x, m_hEnemy->pev->origin.y, m_hEnemy->pev->absmin.z );
534 }
535 else
536 {
537 // toss it to where you last saw them
538 vecTarget = m_vecEnemyLKP;
539 }
540 // vecTarget = m_vecEnemyLKP + (m_hEnemy->BodyTarget( pev->origin ) - m_hEnemy->pev->origin);
541 // estimate position
542 // vecTarget = vecTarget + m_hEnemy->pev->velocity * 2;
543 }
544 else
545 {
546 // find target
547 // vecTarget = m_hEnemy->BodyTarget( pev->origin );
548 vecTarget = m_vecEnemyLKP + (m_hEnemy->BodyTarget( pev->origin ) - m_hEnemy->pev->origin);
549 // estimate position
550 if (HasConditions( bits_COND_SEE_ENEMY))
551 vecTarget = vecTarget + ((vecTarget - pev->origin).Length() / gSkillData.hgruntGrenadeSpeed) * m_hEnemy->pev->velocity;
552 }
553
554 // are any of my squad members near the intended grenade impact area?
555 if ( InSquad() )
556 {
557 if (SquadMemberInRange( vecTarget, 256 ))
558 {
559 // crap, I might blow my own guy up. Don't throw a grenade and don't check again for a while.
560 m_flNextGrenadeCheck = gpGlobals->time + 1; // one full second.
561 m_fThrowGrenade = FALSE;
562 }
563 }
564
565 if ( ( vecTarget - pev->origin ).Length2D() <= 256 )
566 {
567 // crap, I don't want to blow myself up
568 m_flNextGrenadeCheck = gpGlobals->time + 1; // one full second.
569 m_fThrowGrenade = FALSE;
570 return m_fThrowGrenade;
571 }
572
573
574 if (FBitSet( pev->weapons, HGRUNT_HANDGRENADE))
575 {
576 Vector vecToss = VecCheckToss( pev, GetGunPosition(), vecTarget, 0.5 );
577
578 if ( vecToss != g_vecZero )
579 {
580 m_vecTossVelocity = vecToss;
581
582 // throw a hand grenade
583 m_fThrowGrenade = TRUE;
584 // don't check again for a while.
585 m_flNextGrenadeCheck = gpGlobals->time; // 1/3 second.
586 }
587 else
588 {
589 // don't throw
590 m_fThrowGrenade = FALSE;
591 // don't check again for a while.
592 m_flNextGrenadeCheck = gpGlobals->time + 1; // one full second.
593 }
594 }
595 else
596 {
597 Vector vecToss = VecCheckThrow( pev, GetGunPosition(), vecTarget, gSkillData.hgruntGrenadeSpeed, 0.5 );
598
599 if ( vecToss != g_vecZero )
600 {
601 m_vecTossVelocity = vecToss;
602
603 // throw a hand grenade
604 m_fThrowGrenade = TRUE;
605 // don't check again for a while.
606 m_flNextGrenadeCheck = gpGlobals->time + 0.3; // 1/3 second.
607 }
608 else
609 {
610 // don't throw
611 m_fThrowGrenade = FALSE;
612 // don't check again for a while.
613 m_flNextGrenadeCheck = gpGlobals->time + 1; // one full second.
614 }
615 }
616
617
618
619 return m_fThrowGrenade;
620}

695void CHGrunt::IdleSound(void)
696{
697 if (FOkToSpeak() && (g_fGruntQuestion || RANDOM_LONG(0, 1)))
698 {
699 if (!g_fGruntQuestion)
700 {
701 // Ask question or make statement
702 switch (RANDOM_LONG(0, 2))
703 {
704 case 0: // Check in
705 SENTENCEG_PlayRndSz(ENT(pev), "HG_CHECK", HGRUNT_SENTENCE_VOLUME, ATTN_NORM, 0, m_voicePitch);
706 g_fGruntQuestion = 1;
707 break;
708 case 1: // Question
709 SENTENCEG_PlayRndSz(ENT(pev), "HG_QUEST", HGRUNT_SENTENCE_VOLUME, ATTN_NORM, 0, m_voicePitch);
710 g_fGruntQuestion = 2;
711 break;
712 case 2: // Statement
713 SENTENCEG_PlayRndSz(ENT(pev), "HG_IDLE", HGRUNT_SENTENCE_VOLUME, ATTN_NORM, 0, m_voicePitch);
714 break;
715 }
716 }
717 else
718 {
719 switch (g_fGruntQuestion)
720 {
721 case 1: // Check in
722 SENTENCEG_PlayRndSz(ENT(pev), "HG_CLEAR", HGRUNT_SENTENCE_VOLUME, ATTN_NORM, 0, m_voicePitch);
723 break;
724 case 2: // Question
725 SENTENCEG_PlayRndSz(ENT(pev), "HG_ANSWER", HGRUNT_SENTENCE_VOLUME, ATTN_NORM, 0, m_voicePitch);
726 break;
727 }
728 g_fGruntQuestion = 0; // Reset question state
729 }
730 JustSpoke(); // Mark that the grunt has just spoken
731 }

767CBaseEntity *CHGrunt::Kick(void)
768{
769 TraceResult tr;
770
771 UTIL_MakeVectors(pev->angles); // Set up vectors based on angles
772 Vector vecStart = pev->origin;
773 vecStart.z += pev->size.z * 0.5; // Start position for the kick
774 Vector vecEnd = vecStart + (gpGlobals->v_forward * 70); // End position for the kick
775
776 UTIL_TraceHull(vecStart, vecEnd, dont_ignore_monsters, head_hull, ENT(pev), &tr); // Trace for entities
777
778 if (tr.pHit)
779 {
780 CBaseEntity *pEntity = CBaseEntity::Instance(tr.pHit); // Get the entity hit
781 return pEntity; // Return the entity
782 }
783
784 return NULL; // No entity hit
785}

1156 break;

1158 case TASK_GRUNT_FACE_TOSS_DIR:
1159 break; // No specific action for this task
1160
1161 case TASK_FACE_IDEAL:
1162 case TASK_FACE_ENEMY:
1163 CSquadMonster::StartTask(pTask); // Call base class method
1164 if (pev->movetype == MOVETYPE_FLY)
1165 {
1166 m_IdealActivity = ACT_GLIDE; // Set ideal activity to glide if flying
1167 }
1168 break;

1915void CHGrunt :: SetActivity ( Activity NewActivity )
1916{
1917 int iSequence = ACTIVITY_NOT_AVAILABLE;
1918 void *pmodel = GET_MODEL_PTR( ENT(pev) );
1919
1920 switch ( NewActivity)
1921 {
1922 case ACT_RANGE_ATTACK1:

1925 {

1927 {
1928 // get aimable sequence

1931 else

1933 // get crouching shoot
1934 iSequence = LookupSequence( "crouching_mp5" );
1935 }
1936 }
1937 else
1938 {
1939 if ( m_fStanding )

1941 // get aimable sequence
1942 iSequence = LookupSequence( "standing_shotgun" );
1943 }
1944 else
1945 {
1946 // get crouching shoot
1947 iSequence = LookupSequence( "crouching_shotgun" );
1948 }
1949 }
1950 break;
1951 case ACT_RANGE_ATTACK2:

1953 // grenade or fired grenade, we must determine which and pick proper sequence
1954 if ( pev->weapons & HGRUNT_HANDGRENADE )
1955 {
1956 // get toss anim
1957 iSequence = LookupSequence( "throwgrenade" );
1958 }
1959 else
1960 {
1961 // get launch anim
1962 iSequence = LookupSequence( "launchgrenade" );
1963 }
1964 break;
1965 case ACT_RUN:
1966 if ( pev->health <= HGRUNT_LIMP_HEALTH )
1967 {
1968 // limp!
1969 iSequence = LookupActivity ( ACT_RUN_HURT );
1970 }
1971 else
1972 {
1973 iSequence = LookupActivity ( NewActivity );
1974 }
1975 break;
1976 case ACT_WALK:
1977 if ( pev->health <= HGRUNT_LIMP_HEALTH )
1978 {
1979 // limp!
1980 iSequence = LookupActivity ( ACT_WALK_HURT );
1981 }
1982 else
1983 {
1984 iSequence = LookupActivity ( NewActivity );
1985 }
1986 break;
1987 case ACT_IDLE:
1988 if ( m_MonsterState == MONSTERSTATE_COMBAT )
1989 {
1990 NewActivity = ACT_IDLE_ANGRY;
1991 }
1992 iSequence = LookupActivity ( NewActivity );
1993 break;
1994 default:
1995 iSequence = LookupActivity ( NewActivity );
1996 break;
1997 }
1998
1999 m_Activity = NewActivity; // Go ahead and set this so it doesn't keep trying when the anim is not present
2000
2001 // Set to the desired anim, or default anim if the desired is not present
2002 if ( iSequence > ACTIVITY_NOT_AVAILABLE )
2003 {
2004 if ( pev->sequence != iSequence || !m_fSequenceLoops )
2005 {
2006 pev->frame = 0;
2007 }
2008
2009 pev->sequence = iSequence; // Set to the reset anim (if it's there)
2010 ResetSequenceInfo( );
2011 SetYawSpeed();
2012 }
2013 else
2014 {
2015 // Not available try to get default anim
2016 ALERT ( at_console, "%s has no sequence for act:%d\n", STRING(pev->classname), NewActivity );
2017 pev->sequence = 0; // Set to the reset anim (if it's there)
2018 }
