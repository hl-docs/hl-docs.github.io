76int CL_DLLEXPORT HUD_GetHullBounds( int hullnumber, float *mins, float *maxs )
77{
78// RecClGetHullBounds(hullnumber, mins, maxs);
79
80 int iret = 0;
81
82 switch ( hullnumber )
83 {
84 case 0: // Normal player
85 mins = Vector(-16, -16, -36);
86 maxs = Vector(16, 16, 36);
87 iret = 1;
88 break;
89 case 1: // Crouched player
90 mins = Vector(-16, -16, -18 );
91 maxs = Vector(16, 16, 18 );
92 iret = 1;
93 break;
94 case 2: // Point based hull
95 mins = Vector( 0, 0, 0 );
96 maxs = Vector( 0, 0, 0 );
97 iret = 1;
98 break;
99 }
100
101 return iret;
102}

360void CL_LoadParticleMan( void )
361{
362 char szPDir[512];
363
364 if ( gEngfuncs.COM_ExpandFilename( PARTICLEMAN_DLLNAME, szPDir, sizeof( szPDir ) ) == FALSE )
365 {
366 g_pParticleMan = NULL;
367 g_hParticleManModule = NULL;
368 return;
369 }
370
371 g_hParticleManModule = Sys_LoadModule( szPDir );
372 CreateInterfaceFn particleManFactory = Sys_GetFactory( g_hParticleManModule );
373
374 if ( particleManFactory == NULL )
375 {
376 g_pParticleMan = NULL;
377 g_hParticleManModule = NULL;
378 return;
379 }
380
381 g_pParticleMan = (IParticleMan *)particleManFactory( PARTICLEMAN_INTERFACE, NULL);
382
383 if ( g_pParticleMan )
384 {
385 g_pParticleMan->SetUp( &gEngfuncs );
386
387 // Add custom particle classes here BEFORE calling anything else or you will die.
388 g_pParticleMan->AddCustomParticleClassSize ( sizeof ( CBaseParticle ) );
389 }
390}
