48void CHud::Think(void)
49{
50 m_scrinfo.iSize = sizeof(m_scrinfo);
51 GetScreenInfo(&m_scrinfo);
52
53 int newfov;
54 HUDLIST *pList = m_pHudList;
55
56 while (pList)
57 {
58 if (pList->p->m_iFlags & HUD_ACTIVE)
59 pList->p->Think();
60 pList = pList->pNext;
61 }
62
63 newfov = HUD_GetFOV();
64 if ( newfov == 0 )
65 {
66 m_iFOV = default_fov->value;
67 }
68 else
69 {
70 m_iFOV = newfov;
71 }
72
73 // the clients fov is actually set in the client data update section of the hud
74
75 // Set a new sensitivity
76 if ( m_iFOV == default_fov->value )
77 {
78 // reset to saved sensitivity
79 m_flMouseSensitivity = 0;
80 }
81 else
82 {
83 // set a new sensitivity that is proportional to the change from the FOV default
84 m_flMouseSensitivity = sensitivity->value * ((float)newfov / (float)default_fov->value) * CVAR_GET_FLOAT("zoom_sensitivity_ratio");
85 }
86
87 // think about default fov
88 if ( m_iFOV == 0 )
89 { // only let players adjust up in fov, and only if they are not overriden by something else
90 m_iFOV = max( default_fov->value, 90 );
91 }
92
93 if ( gEngfuncs.IsSpectateOnly() )
94 {
95 m_iFOV = gHUD.m_Spectator.GetFOV(); // default_fov->value;
96 }
97
98 Bench_CheckStart();
99}

112int CHud :: Redraw( float flTime, int intermission )
113{
114 m_fOldTime = m_flTime; // save time of previous redraw
115 m_flTime = flTime;
116 m_flTimeDelta = (double)m_flTime - m_fOldTime;
117 static float m_flShotTime = 0;
118
119 // Clock was reset, reset delta
120 if ( m_flTimeDelta < 0 )
121 m_flTimeDelta = 0;
122
123 // Bring up the scoreboard during intermission
124 if (gViewPort)
125 {
126 if ( m_iIntermission && !intermission )
127 {
128 // Have to do this here so the scoreboard goes away
129 m_iIntermission = intermission;
130 gViewPort->HideCommandMenu();
131 gViewPort->HideScoreBoard();
132 gViewPort->UpdateSpectatorPanel();
133 }
134 else if ( !m_iIntermission && intermission )
135 {
136 m_iIntermission = intermission;
137 gViewPort->HideCommandMenu();
138 gViewPort->HideVGUIMenu();
139 gViewPort->ShowScoreBoard();
140 gViewPort->UpdateSpectatorPanel();
141
142 // Take a screenshot if the client's got the cvar set
143 if ( CVAR_GET_FLOAT( "hud_takesshots" ) != 0 )
144 m_flShotTime = flTime + 1.0; // Take a screenshot in a second
145 }
146 }
147
148 if (m_flShotTime && m_flShotTime < flTime)
149 {
150 gEngfuncs.pfnClientCmd("snapshot\n");
151 m_flShotTime = 0;
152 }
153
154 m_iIntermission = intermission;
155
156 // if no redrawing is necessary
157 // return 0;
158
159 // draw all registered HUD elements
160 if ( m_pCvarDraw->value )
161 {
162 HUDLIST *pList = m_pHudList;
163
164 while (pList)
165 {
166 if ( !Bench_Active() )
167 {
168 if ( !intermission )
169 {
170 if ( (pList->p->m_iFlags & HUD_ACTIVE) && !(m_iHideHUDDisplay & HIDEHUD_ALL) )
171 pList->p->Draw(flTime);
172 }
173 else
174 { // it's an intermission, so only draw hud elements that are set to draw during intermissions
175 if ( pList->p->m_iFlags & HUD_INTERMISSION )
176 pList->p->Draw( flTime );
177 }
178 }
179 else
180 {
181 if ( ( pList->p == &m_Benchmark ) &&
182 ( pList->p->m_iFlags & HUD_ACTIVE ) &&
183 !( m_iHideHUDDisplay & HIDEHUD_ALL ) )
184 {
185 pList->p->Draw(flTime);
186 }
187 }
188
189 pList = pList->pNext;
190 }
191 }
192
193 // are we in demo mode? do we need to draw the logo in the top corner?
194 if (m_iLogo)
195 {
196 int x, y, i;
197
198 if (m_hsprLogo == 0)
199 m_hsprLogo = LoadSprite("sprites/%d_logo.spr");
200
201 SPR_Set(m_hsprLogo, 250, 250, 250 );
202
203 x = SPR_Width(m_hsprLogo, 0);
204 x = ScreenWidth - x;
205 y = SPR_Height(m_hsprLogo, 0)/2;
206
207 // Draw the logo at 20 fps
208 int iFrame = (int)(flTime * 20) % MAX_LOGO_FRAMES;
209 i = grgLogoFrame[iFrame] - 1;
210
211 SPR_DrawAdditive(i, x, y, NULL);
212 }
213

344 x += iWidth;

348 //SPR_DrawAdditive( 0, x, y, &rc );

408{
