83int WeaponsResource :: HasAmmo( WEAPON *p )
84{
85 if ( !p )
86 return FALSE;
87
88 // weapons with no max ammo can always be selected
89 if ( p->iMax1 == -1 )
90 return TRUE;
91
92 return (p->iAmmoType == -1) || p->iClip > 0 || CountAmmo(p->iAmmoType)
93 || CountAmmo(p->iAmmo2Type) || ( p->iFlags & WEAPON_FLAGS_SELECTONEMPTY );
94}

104void WeaponsResource :: LoadWeaponSprites( WEAPON *pWeapon )
105{
106 int i, iRes;
107
108 if (ScreenWidth < 640)
109 iRes = 320;
110 else
111 iRes = 640;
112
113 char sz[128];
114
115 if ( !pWeapon )
116 return;
117
118 memset( &pWeapon->rcActive, 0, sizeof(wrect_t) );
119 memset( &pWeapon->rcInactive, 0, sizeof(wrect_t) );
120 memset( &pWeapon->rcAmmo, 0, sizeof(wrect_t) );
121 memset( &pWeapon->rcAmmo2, 0, sizeof(wrect_t) );
122 pWeapon->hInactive = 0;
123 pWeapon->hActive = 0;
124 pWeapon->hAmmo = 0;
125 pWeapon->hAmmo2 = 0;
126
127 sprintf(sz, "sprites/%s.txt", pWeapon->szName);
128 client_sprite_t *pList = SPR_GetList(sz, &i);
129
130 if (!pList)
131 return;
132
133 client_sprite_t *p;
134
135 p = GetSpriteList( pList, "crosshair", iRes, i );
136 if (p)
137 {
138 sprintf(sz, "sprites/%s.spr", p->szSprite);
139 pWeapon->hCrosshair = SPR_Load(sz);
140 pWeapon->rcCrosshair = p->rc;
141 }
142 else
143 pWeapon->hCrosshair = NULL;
144
145 p = GetSpriteList(pList, "autoaim", iRes, i);
146 if (p)
147 {
148 sprintf(sz, "sprites/%s.spr", p->szSprite);
149 pWeapon->hAutoaim = SPR_Load(sz);
150 pWeapon->rcAutoaim = p->rc;
151 }
152 else
153 pWeapon->hAutoaim = 0;
154
155 p = GetSpriteList( pList, "zoom", iRes, i );
156 if (p)
157 {
158 sprintf(sz, "sprites/%s.spr", p->szSprite);
159 pWeapon->hZoomedCrosshair = SPR_Load(sz);
160 pWeapon->rcZoomedCrosshair = p->rc;
161 }
162 else
163 {
164 pWeapon->hZoomedCrosshair = pWeapon->hCrosshair; //default to non-zoomed crosshair
165 pWeapon->rcZoomedCrosshair = pWeapon->rcCrosshair;
166 }
167
168 p = GetSpriteList(pList, "zoom_autoaim", iRes, i);
169 if (p)
170 {
171 sprintf(sz, "sprites/%s.spr", p->szSprite);
172 pWeapon->hZoomedAutoaim = SPR_Load(sz);
173 pWeapon->rcZoomedAutoaim = p->rc;
174 }
175 else
176 {
177 pWeapon->hZoomedAutoaim = pWeapon->hZoomedCrosshair; //default to zoomed crosshair
178 pWeapon->rcZoomedAutoaim = pWeapon->rcZoomedCrosshair;
179 }
180
181 p = GetSpriteList(pList, "weapon", iRes, i);
182 if (p)
183 {
184 sprintf(sz, "sprites/%s.spr", p->szSprite);
185 pWeapon->hInactive = SPR_Load(sz);
186 pWeapon->rcInactive = p->rc;
187
188 gHR.iHistoryGap = max( gHR.iHistoryGap, pWeapon->rcActive.bottom - pWeapon->rcActive.top );
189 }
190 else
191 pWeapon->hInactive = 0;
192
193 p = GetSpriteList(pList, "weapon_s", iRes, i);
194 if (p)
195 {
196 sprintf(sz, "sprites/%s.spr", p->szSprite);
197 pWeapon->hActive = SPR_Load(sz);
198 pWeapon->rcActive = p->rc;
199 }
200 else
201 pWeapon->hActive = 0;
202
203 p = GetSpriteList(pList, "ammo", iRes, i);
204 if (p)
205 {
206 sprintf(sz, "sprites/%s.spr", p->szSprite);
207 pWeapon->hAmmo = SPR_Load(sz);
208 pWeapon->rcAmmo = p->rc;
209
210 gHR.iHistoryGap = max( gHR.iHistoryGap, pWeapon->rcActive.bottom - pWeapon->rcActive.top );
211 }
212 else
213 pWeapon->hAmmo = 0;
214
215 p = GetSpriteList(pList, "ammo2", iRes, i);
216 if (p)
217 {
218 sprintf(sz, "sprites/%s.spr", p->szSprite);
219 pWeapon->hAmmo2 = SPR_Load(sz);
220 pWeapon->rcAmmo2 = p->rc;
221
222 gHR.iHistoryGap = max( gHR.iHistoryGap, pWeapon->rcActive.bottom - pWeapon->rcActive.top );
223 }
224 else
225 pWeapon->hAmmo2 = 0;
226

238WEAPON *WeaponsResource :: GetFirstPos( int iSlot )
239{
240 WEAPON *pret = NULL;
241
242 for (int i = 0; i < MAX_WEAPON_POSITIONS; i++)
243 {
244 if ( rgSlots[iSlot][i] && HasAmmo( rgSlots[iSlot][i] ) )

264WEAPON* WeaponsResource :: GetNextActivePos( int iSlot, int iSlotPos )
265{
266 if ( iSlotPos >= MAX_WEAPON_POSITIONS || iSlot >= MAX_WEAPON_SLOTS )
267 return NULL;

273
274 return p;

421void CHudAmmo::Think(void)
422{
423 if ( gHUD.m_fPlayerDead )
424 return;
425
426 if ( gHUD.m_iWeaponBits != gWR.iOldWeaponBits )
427 {
428 gWR.iOldWeaponBits = gHUD.m_iWeaponBits;
429
430 for (int i = MAX_WEAPONS-1; i > 0; i-- )
431 {
432 WEAPON *p = gWR.GetWeapon(i);
433
434 if ( p )
435 {
436 if ( gHUD.m_iWeaponBits & ( 1 << p->iId ) )
437 gWR.PickupWeapon( p );
438 else
439 gWR.DropWeapon( p );
440 }
441 }
442 }
443
444 if (!gpActiveSel)
445 return;
446
447 // has the player selected one?
448 if (gHUD.m_iKeyBits & IN_ATTACK)
449 {
450 if (gpActiveSel != (WEAPON *)1)
451 {
452 ServerCmd(gpActiveSel->szName);
453 g_weaponselect = gpActiveSel->iId;
454 }
455
456 gpLastSel = gpActiveSel;
457 gpActiveSel = NULL;
458 gHUD.m_iKeyBits &= ~IN_ATTACK;
459
460 PlaySound("common/wpn_select.wav", 1);
461 }

506void WeaponsResource :: SelectSlot( int iSlot, int fAdvance, int iDirection )
507{
508 if ( gHUD.m_Menu.m_fMenuDisplayed && (fAdvance == FALSE) && (iDirection == 1) )
509 { // menu is overriding slot use commands
510 gHUD.m_Menu.SelectMenuItem( iSlot + 1 ); // slots are one off the key numbers
511 return;
512 }
513
514 if ( iSlot > MAX_WEAPON_SLOTS )
515 return;
516
517 if ( gHUD.m_fPlayerDead || gHUD.m_iHideHUDDisplay & ( HIDEHUD_WEAPONS | HIDEHUD_ALL ) )
518 return;
519
520 if (!(gHUD.m_iWeaponBits & (1<<(WEAPON_SUIT)) ))
521 return;
522
523 if ( ! ( gHUD.m_iWeaponBits & ~(1<<(WEAPON_SUIT)) ))
524 return;
525
526 WEAPON *p = NULL;
527 bool fastSwitch = CVAR_GET_FLOAT( "hud_fastswitch" ) != 0;
528
529 if ( (gpActiveSel == NULL) || (gpActiveSel == (WEAPON *)1) || (iSlot != gpActiveSel->iSlot) )
530 {
531 PlaySound( "common/wpn_hudon.wav", 1 );
532 p = GetFirstPos( iSlot );
533
534 if ( p && fastSwitch ) // check for fast weapon switch mode
535 {
536 // if fast weapon switch is on, then weapons can be selected in a single keypress
537 // but only if there is only one item in the bucket
538 WEAPON *p2 = GetNextActivePos( p->iSlot, p->iSlotPos );
539 if ( !p2 )
540 { // only one active item in bucket, so change directly to weapon
541 ServerCmd( p->szName );
542 g_weaponselect = p->iId;
543 return;
544 }
545 }
546 }
547 else
548 {
549 PlaySound("common/wpn_moveselect.wav", 1);
550 if ( gpActiveSel )
551 p = GetNextActivePos( gpActiveSel->iSlot, gpActiveSel->iSlotPos );
552 if ( !p )
553 p = GetFirstPos( iSlot );
554 }
555
556
557 if ( !p ) // no selection found
558 {
559 // just display the weapon list, unless fastswitch is on just ignore it
560 if ( !fastSwitch )
561 gpActiveSel = (WEAPON *)1;
562 else
563 gpActiveSel = NULL;
564 }
565 else
566 gpActiveSel = p;
567}

702int CHudAmmo::MsgFunc_CurWeapon(const char *pszName, int iSize, void *pbuf )
703{
704 static wrect_t nullrc;
705 int fOnTarget = FALSE;
706
707 BEGIN_READ( pbuf, iSize );
708
709 int iState = READ_BYTE();
710 int iId = READ_CHAR();
711 int iClip = READ_CHAR();
712
713 // detect if we're also on target
714 if ( iState > 1 )
715 {
716 fOnTarget = TRUE;
717 }
718
719 if ( iId < 1 )
720 {
721 SetCrosshair(0, nullrc, 0, 0, 0);
722 return 0;
723 }
724
725 if ( g_iUser1 != OBS_IN_EYE )
726 {
727 // Is player dead???
728 if ((iId == -1) && (iClip == -1))
729 {
730 gHUD.m_fPlayerDead = TRUE;
731 gpActiveSel = NULL;
732 return 1;
733 }
734 gHUD.m_fPlayerDead = FALSE;
735 }
736
737 WEAPON *pWeapon = gWR.GetWeapon( iId );
738
739 if ( !pWeapon )
740 return 0;
741
742 if ( iClip < -1 )
743 pWeapon->iClip = abs(iClip);
744 else
745 pWeapon->iClip = iClip;
746
747
748 if ( iState == 0 ) // we're not the current weapon, so update no more
749 return 1;
750
751 m_pWeapon = pWeapon;
752
753 if ( gHUD.m_iFOV >= 90 )
754 { // normal crosshairs
755 if (fOnTarget && m_pWeapon->hAutoaim)
756 SetCrosshair(m_pWeapon->hAutoaim, m_pWeapon->rcAutoaim, 255, 255, 255);
757 else
758 SetCrosshair(m_pWeapon->hCrosshair, m_pWeapon->rcCrosshair, 255, 255, 255);
759 }
760 else
761 { // zoomed crosshairs
762 if (fOnTarget && m_pWeapon->hZoomedAutoaim)
763 SetCrosshair(m_pWeapon->hZoomedAutoaim, m_pWeapon->rcZoomedAutoaim, 255, 255, 255);
764 else
765 SetCrosshair(m_pWeapon->hZoomedCrosshair, m_pWeapon->rcZoomedCrosshair, 255, 255, 255);
766
767 }
768
769 m_fFade = 200.0f;
770 m_iFlags |= HUD_ACTIVE;
771
772 return 1;
773}

940void CHudAmmo::UserCmd_NextWeapon(void)
941{
942 if ( gHUD.m_fPlayerDead || (gHUD.m_iHideHUDDisplay & (HIDEHUD_WEAPONS | HIDEHUD_ALL)) )
943 return;
944
945 if ( !gpActiveSel || gpActiveSel == (WEAPON*)1 )
946 gpActiveSel = m_pWeapon;
947
948 int pos = 0;
949 int slot = 0;
950 if ( gpActiveSel )
951 {
952 pos = gpActiveSel->iSlotPos + 1;
953 slot = gpActiveSel->iSlot;
954 }
955
956 for ( int loop = 0; loop <= 1; loop++ )
957 {
958 for ( ; slot < MAX_WEAPON_SLOTS; slot++ )
959 {
960 for ( ; pos < MAX_WEAPON_POSITIONS; pos++ )
961 {
962 WEAPON *wsp = gWR.GetWeaponSlot( slot, pos );
963
964 if ( wsp && gWR.HasAmmo(wsp) )
965 {
966 gpActiveSel = wsp;
967 return;
968 }
969 }
970
971 pos = 0;
972 }
973
974 slot = 0; // start looking from the first slot again
975 }
976
977 gpActiveSel = NULL;
978}
