107static int g_ResArray[] =
108{
109 320,
110 400,
111 512,
112 640,

119static int g_NumReses = sizeof(g_ResArray) / sizeof(int);

131static byte *LoadFileByResolution( const char *filePrefix, int xRes, const char *filePostfix )
132{
133 // find our resolution in the res array
134 int resNum = g_NumReses - 1;
135 while ( g_ResArray[resNum] > xRes )
136 {
137 resNum--;
138
139 if ( resNum < 0 )
140 return NULL;
141 }
142
143 // try open the file
144 byte *pFile = NULL;
145 while ( 1 )
146 {
147
148 // try load
149 char fname[256];
150 sprintf( fname, "%s%d%s", filePrefix, g_ResArray[resNum], filePostfix );
151 pFile = gEngfuncs.COM_LoadFile( fname, 5, NULL );
152
153 if ( pFile )
154 break;
155
156 if ( resNum == 0 )
157 return NULL;
158
159 resNum--;
160 };
161
162 return pFile;
163}

173static void ParseRGBAFromString( byte colorArray[4], const char *colorVector )
174{
175 int r, g, b, a;
176 sscanf( colorVector, "%d %d %d %d", &r, &g, &b, &a );
177 colorArray[0] = r;
178 colorArray[1] = g;
179 colorArray[2] = b;
180 colorArray[3] = a;
181}

191CSchemeManager::CSchemeManager( int xRes, int yRes )
192{
193 // basic setup
194 m_pSchemeList = NULL;
195 m_iNumSchemes = 0;
196
197 // find the closest matching scheme file to our resolution
198 char token[1024];
199 char *pFile = (char*)LoadFileByResolution( "", xRes, "_textscheme.txt" );
200 m_xRes = xRes;
201
202 char *pFileStart = pFile;
203
204 byte *pFontData;
205 int fontFileLength;
206 char fontFilename[512];
207
208 //
209 // Read the scheme descriptions from the text file, into a temporary array
210 // format is simply:
211 // <paramName name> = <paramValue>
212 //
213 // a <paramName name> of "SchemeName" signals a new scheme is being described
214 //
215
216 const static int numTmpSchemes = 64;
217 static CScheme tmpSchemes[numTmpSchemes];
218 memset( tmpSchemes, 0, sizeof(tmpSchemes) );
219 int currentScheme = -1;
220 CScheme *pScheme = NULL;
221
222 if ( !pFile )
223 {
224 gEngfuncs.Con_DPrintf( "Unable to find *_textscheme.txt\n");
225 goto buildDefaultFont;
226 }
227
228 // record what has been entered so we can create defaults from the different values
229 bool hasFgColor, hasBgColor, hasArmedFgColor, hasArmedBgColor, hasMouseDownFgColor, hasMouseDownBgColor;
230
231 pFile = gEngfuncs.COM_ParseFile( pFile, token );
232 while ( strlen(token) > 0 && (currentScheme < numTmpSchemes) )
233 {
234 // get the paramName name
235 static const int tokenSize = 64;
236 char paramName[tokenSize], paramValue[tokenSize];
237
238 strncpy( paramName, token, tokenSize );
239 paramName[tokenSize-1] = 0; // ensure null termination
240
241 // get the '=' character
242 pFile = gEngfuncs.COM_ParseFile( pFile, token );
243 if ( stricmp( token, "=" ) )
244 {
245 if ( currentScheme < 0 )
246 {
247 gEngfuncs.Con_Printf( "error parsing font scheme text file at file start - expected '=', found '%s''\n", token );
248 }
249 else
250 {
251 gEngfuncs.Con_Printf( "error parsing font scheme text file at scheme '%s' - expected '=', found '%s''\n", tmpSchemes[currentScheme].schemeName, token );
252 }
253 break;
254 }
255
256 // get paramValue
257 pFile = gEngfuncs.COM_ParseFile( pFile, token );
258 strncpy( paramValue, token, tokenSize );
259 paramValue[tokenSize-1] = 0; // ensure null termination
260
261 // is this a new scheme?
262 if ( !stricmp(paramName, "SchemeName") )
263 {
264 // setup the defaults for the current scheme
265 if ( pScheme )
266 {
267 // foreground color defaults (normal -> armed -> mouse down)
268 if ( !hasFgColor )
269 {
270 pScheme->fgColor[0] = pScheme->fgColor[1] = pScheme->fgColor[2] = pScheme->fgColor[3] = 255;
271 }
272 if ( !hasArmedFgColor )
273 {
274 memcpy( pScheme->armedFgColor, pScheme->fgColor, sizeof(pScheme->armedFgColor) );
275 }
276 if ( !hasMouseDownFgColor )
277 {
278 memcpy( pScheme->mousedownFgColor, pScheme->armedFgColor, sizeof(pScheme->mousedownFgColor) );
279 }
280
281 // background color (normal -> armed -> mouse down)
282 if ( !hasBgColor )
283 {
284 pScheme->bgColor[0] = pScheme->bgColor[1] = pScheme->bgColor[2] = pScheme->bgColor[3] = 0;
285 }
286 if ( !hasArmedBgColor )
287 {
288 memcpy( pScheme->armedBgColor, pScheme->bgColor, sizeof(pScheme->armedBgColor) );
289 }
290 if ( !hasMouseDownBgColor )
291 {
292 memcpy( pScheme->mousedownBgColor, pScheme->armedBgColor, sizeof(pScheme->mousedownBgColor) );
293 }
294
295 // font size
296 if ( !pScheme->fontSize )
297 {
298 pScheme->fontSize = 17;
299 }
300 if ( !pScheme->fontName[0] )
301 {
302 strcpy( pScheme->fontName, "Arial" );
303 }
304 }
305
306 // create the new scheme
307 currentScheme++;
308 pScheme = &tmpSchemes[currentScheme];
309 hasFgColor = hasBgColor = hasArmedFgColor = hasArmedBgColor = hasMouseDownFgColor = hasMouseDownBgColor = false;
310
311 strncpy( pScheme->schemeName, paramValue, CScheme::SCHEME_NAME_LENGTH );
312 pScheme->schemeName[CScheme::SCHEME_NAME_LENGTH-1] = '\0'; // ensure null termination of string
313 }
314
315 if ( !pScheme )
316 {
317 gEngfuncs.Con_Printf( "font scheme text file MUST start with a 'SchemeName'\n");
318 break;
319 }
320
321 // pull the data out into the scheme
322 if ( !stricmp(paramName, "FontName") )
323 {
324 strncpy( pScheme->fontName, paramValue, CScheme::FONT_NAME_LENGTH );
325 pScheme->fontName[CScheme::FONT_NAME_LENGTH-1] = 0;
326 }
327 else if ( !stricmp(paramName, "FontSize") )
328 {
329 pScheme->fontSize = atoi( paramValue );
330 }
331 else if ( !stricmp(paramName, "FontWeight") )
332 {
333 pScheme->fontWeight = atoi( paramValue );
334 }
335 else if ( !stricmp(paramName, "FgColor") )
336 {
337 ParseRGBAFromString( pScheme->fgColor, paramValue );
338 hasFgColor = true;
339 }
340 else if ( !stricmp(paramName, "BgColor") )
341 {
342 ParseRGBAFromString( pScheme->bgColor, paramValue );
343 hasBgColor = true;
344 }
345 else if ( !stricmp(paramName, "FgColorArmed") )
346 {
347 ParseRGBAFromString( pScheme->armedFgColor, paramValue );
348 hasArmedFgColor = true;
349 }
350 else if ( !stricmp(paramName, "BgColorArmed") )
351 {
352 ParseRGBAFromString( pScheme->armedBgColor, paramValue );
353 hasArmedBgColor = true;
354 }
355 else if ( !stricmp(paramName, "FgColorMousedown") )
356 {
357 ParseRGBAFromString( pScheme->mousedownFgColor, paramValue );
358 hasMouseDownFgColor = true;
359 }
360 else if ( !stricmp(paramName, "BgColorMousedown") )
361 {
362 ParseRGBAFromString( pScheme->mousedownBgColor, paramValue );
363 hasMouseDownBgColor = true;
364 }
365 else if ( !stricmp(paramName, "BorderColor") )
366 {
367 ParseRGBAFromString( pScheme->borderColor, paramValue );
368 hasMouseDownBgColor = true;
369 }
370
371 // get the new token last, so we now if the loop needs to be continued or not
372 pFile = gEngfuncs.COM_ParseFile( pFile, token );
373 }
374
375 // free the file
376 gEngfuncs.COM_FreeFile( pFileStart );
377
378
379buildDefaultFont:
380
381 // make sure we have at least 1 valid font
382 if ( currentScheme < 0 )
383 {
384 currentScheme = 0;
385 strcpy( tmpSchemes[0].schemeName, "Default Scheme" );
386 strcpy( tmpSchemes[0].fontName, "Arial" );
387 tmpSchemes[0].fontSize = 0;
388 tmpSchemes[0].fgColor[0] = tmpSchemes[0].fgColor[1] = tmpSchemes[0].fgColor[2] = tmpSchemes[0].fgColor[3] = 255;
389 tmpSchemes[0].armedFgColor[0] = tmpSchemes[0].armedFgColor[1] = tmpSchemes[0].armedFgColor[2] = tmpSchemes[0].armedFgColor[3] = 255;
390 tmpSchemes[0].mousedownFgColor[0] = tmpSchemes[0].mousedownFgColor[1] = tmpSchemes[0].mousedownFgColor[2] = tmpSchemes[0].mousedownFgColor[3] = 255;
391 }
392
393 // we have the full list of schemes in the tmpSchemes array
394 // now allocate the correct sized list
395 m_iNumSchemes = currentScheme + 1; // 0-based index
396 m_pSchemeList = new CScheme[ m_iNumSchemes ];
397
398 // copy in the data
399 memcpy( m_pSchemeList, tmpSchemes, sizeof(CScheme) * m_iNumSchemes );
400
401 // create the fonts
402 for ( int i = 0; i < m_iNumSchemes; i++ )
403 {
404 m_pSchemeList[i].font = NULL;
405
406 // see if the current font values exist in a previously loaded font
407 for ( int j = 0; j < i; j++ )
408 {
409 // check if the font name, size, and weight are the same
410 if ( !stricmp(m_pSchemeList[i].fontName, m_pSchemeList[j].fontName)
411 && m_pSchemeList[i].fontSize == m_pSchemeList[j].fontSize
412 && m_pSchemeList[i].fontWeight == m_pSchemeList[j].fontWeight )
413 {
414 // copy the pointer, but mark i as not owning it
415 m_pSchemeList[i].font = m_pSchemeList[j].font;
416 m_pSchemeList[i].ownFontPointer = false;
417 }
418 }
419
420 // if we haven't found the font already, load it ourselves
421 if ( !m_pSchemeList[i].font )
422 {
423 fontFileLength = -1;
424 pFontData = NULL;
425
426 if(g_CV_BitmapFonts && g_CV_BitmapFonts->value)
427 {
428 int fontRes = 640;
429 if ( m_xRes >= 1600 )
430 fontRes = 1600;
431 else if ( m_xRes >= 1280 )
432 fontRes = 1280;
433 else if ( m_xRes >= 1152 )
434 fontRes = 1152;
435 else if ( m_xRes >= 1024 )
436 fontRes = 1024;
437 else if ( m_xRes >= 800 )
438 fontRes = 800;
439
440 sprintf(fontFilename, "gfx\\vgui\\fonts\\%d_%s.tga", fontRes, m_pSchemeList[i].schemeName);
441 pFontData = gEngfuncs.COM_LoadFile( fontFilename, 5, &fontFileLength );
442 if(!pFontData)
443 gEngfuncs.Con_Printf("Missing bitmap font: %s\n", fontFilename);
444 }
445
446 m_pSchemeList[i].font = new vgui::Font(
447 m_pSchemeList[i].fontName,
448 pFontData,
449 fontFileLength,
450 m_pSchemeList[i].fontSize,
451 0,
452 0,
453 m_pSchemeList[i].fontWeight,
454 false,
455 false,
456 false,
457 false);
458
459 m_pSchemeList[i].ownFontPointer = true;
460 }
461
462 // fix up alpha values; VGUI uses 1-A (A=0 being solid, A=255 transparent)
463 m_pSchemeList[i].fgColor[3] = 255 - m_pSchemeList[i].fgColor[3];
464 m_pSchemeList[i].bgColor[3] = 255 - m_pSchemeList[i].bgColor[3];
465 m_pSchemeList[i].armedFgColor[3] = 255 - m_pSchemeList[i].armedFgColor[3];
466 m_pSchemeList[i].armedBgColor[3] = 255 - m_pSchemeList[i].armedBgColor[3];
467 m_pSchemeList[i].mousedownFgColor[3] = 255 - m_pSchemeList[i].mousedownFgColor[3];
468 m_pSchemeList[i].mousedownBgColor[3] = 255 - m_pSchemeList[i].mousedownBgColor[3];
469 }
470}
