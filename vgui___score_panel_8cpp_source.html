343void ScorePanel::SortTeams()
344{
345 // clear out team scores
346 int i;
347 for ( i = 1; i <= m_iNumTeams; i++ )
348 {
349 if ( !g_TeamInfo[i].scores_overriden )
350 g_TeamInfo[i].frags = g_TeamInfo[i].deaths = 0;
351 g_TeamInfo[i].ping = g_TeamInfo[i].packetloss = 0;
352 }
353
354 // recalc the team scores, then draw them
355 for ( i = 1; i < MAX_PLAYERS; i++ )
356 {
357 if ( g_PlayerInfoList[i].name == NULL )
358 continue; // empty player slot, skip
359
360 if ( g_PlayerExtraInfo[i].teamname[0] == 0 )
361 continue; // skip over players who are not in a team
362
363 // find what team this player is in
364 int j;
365 for ( j = 1; j <= m_iNumTeams; j++ )
366 {
367 if ( !stricmp( g_PlayerExtraInfo[i].teamname, g_TeamInfo[j].name ) )
368 break;
369 }
370 if ( j > m_iNumTeams ) // player is not in a team, skip to the next guy
371 continue;
372
373 if ( !g_TeamInfo[j].scores_overriden )
374 {
375 g_TeamInfo[j].frags += g_PlayerExtraInfo[i].frags;
376 g_TeamInfo[j].deaths += g_PlayerExtraInfo[i].deaths;
377 }
378
379 g_TeamInfo[j].ping += g_PlayerInfoList[i].ping;
380 g_TeamInfo[j].packetloss += g_PlayerInfoList[i].packetloss;
381
382 if ( g_PlayerInfoList[i].thisplayer )
383 g_TeamInfo[j].ownteam = TRUE;
384 else
385 g_TeamInfo[j].ownteam = FALSE;
386
387 // Set the team's number (used for team colors)
388 g_TeamInfo[j].teamnumber = g_PlayerExtraInfo[i].teamnumber;
389 }
390
391 // find team ping/packetloss averages
392 for ( i = 1; i <= m_iNumTeams; i++ )
393 {
394 g_TeamInfo[i].already_drawn = FALSE;
395
396 if ( g_TeamInfo[i].players > 0 )
397 {
398 g_TeamInfo[i].ping /= g_TeamInfo[i].players; // use the average ping of all the players in the team as the teams ping
399 g_TeamInfo[i].packetloss /= g_TeamInfo[i].players; // use the average ping of all the players in the team as the teams ping
400 }
401 }
402
403 // Draw the teams
404 while ( 1 )
405 {
406 int highest_frags = -99999; int lowest_deaths = 99999;
407 int best_team = 0;
408
409 for ( i = 1; i <= m_iNumTeams; i++ )
410 {
411 if ( g_TeamInfo[i].players < 1 )
412 continue;
413
414 if ( !g_TeamInfo[i].already_drawn && g_TeamInfo[i].frags >= highest_frags )
415 {
416 if ( g_TeamInfo[i].frags > highest_frags || g_TeamInfo[i].deaths < lowest_deaths )
417 {
418 best_team = i;
419 lowest_deaths = g_TeamInfo[i].deaths;
420 highest_frags = g_TeamInfo[i].frags;
421 }
422 }
423 }
424
425 // draw the best team on the scoreboard
426 if ( !best_team )
427 break;
428
429 // Put this team in the sorted list
430 m_iSortedRows[ m_iRows ] = best_team;
431 m_iIsATeam[ m_iRows ] = TEAM_YES;
432 g_TeamInfo[best_team].already_drawn = TRUE; // set the already_drawn to be TRUE, so this team won't get sorted again
433 m_iRows++;
434
435 // Now sort all the players on this team
436 SortPlayers( 0, g_TeamInfo[best_team].name );
437 }
438
439 // Add all the players who aren't in a team yet into spectators
440 SortPlayers( TEAM_SPECTATORS, NULL );
441}

453void ScorePanel::SortPlayers( int iTeam, char *team )
454{
455 bool bCreatedTeam = false;
456
457 // draw the players, in order, and restricted to team if set
458 while ( 1 )
459 {
460 // Find the top ranking player
461 int highest_frags = -99999; int lowest_deaths = 99999;
462 int best_player;
463 best_player = 0;
464
465 for ( int i = 1; i < MAX_PLAYERS; i++ )
466 {
467 if ( m_bHasBeenSorted[i] == false && g_PlayerInfoList[i].name && g_PlayerExtraInfo[i].frags >= highest_frags )
468 {
469 cl_entity_t *ent = gEngfuncs.GetEntityByIndex( i );
470
471 if ( ent && !(team && stricmp(g_PlayerExtraInfo[i].teamname, team)) )
472 {
473 extra_player_info_t *pl_info = &g_PlayerExtraInfo[i];
474 if ( pl_info->frags > highest_frags || pl_info->deaths < lowest_deaths )
475 {
476 best_player = i;
477 lowest_deaths = pl_info->deaths;
478 highest_frags = pl_info->frags;
479 }
480 }
481 }
482 }
483
484 if ( !best_player )
485 break;
486
487 // If we haven't created the Team yet, do it first
488 if (!bCreatedTeam && iTeam)
489 {
490 m_iIsATeam[ m_iRows ] = iTeam;
491 m_iRows++;
492
493 bCreatedTeam = true;
494 }
495
496 // Put this player in the sorted list
497 m_iSortedRows[ m_iRows ] = best_player;
498 m_bHasBeenSorted[ best_player ] = true;
499 m_iRows++;
500 }
501
502 if (team)
503 {
504 m_iIsATeam[m_iRows++] = TEAM_BLANK;
505 }
506}

515void ScorePanel::RebuildTeams()
516{
517 // clear out player counts from teams
518 int i;
519 for ( i = 1; i <= m_iNumTeams; i++ )
520 {
521 g_TeamInfo[i].players = 0;
522 }
523
524 // rebuild the team list
525 gViewPort->GetAllPlayersInfo();
526 m_iNumTeams = 0;
527 for ( i = 1; i < MAX_PLAYERS; i++ )
528 {
529 if ( g_PlayerInfoList[i].name == NULL )
530 continue;
531
532 if ( g_PlayerExtraInfo[i].teamname[0] == 0 )
533 continue; // skip over players who are not in a team
534
535 // is this player in an existing team?
536 int j;
537 for ( j = 1; j <= m_iNumTeams; j++ )
538 {
539 if ( g_TeamInfo[j].name[0] == '\0' )
540 break;
541
542 if ( !stricmp( g_PlayerExtraInfo[i].teamname, g_TeamInfo[j].name ) )
543 break;
544 }
545
546 if ( j > m_iNumTeams )
547 { // they aren't in a listed team, so make a new one
548 // search through for an empty team slot
549 for ( int j = 1; j <= m_iNumTeams; j++ )
550 {
551 if ( g_TeamInfo[j].name[0] == '\0' )
552 break;
553 }
554 m_iNumTeams = max( j, m_iNumTeams );
555
556 strncpy( g_TeamInfo[j].name, g_PlayerExtraInfo[i].teamname, MAX_TEAM_NAME );
557 g_TeamInfo[j].players = 0;
558 }
559
560 g_TeamInfo[j].players++;
561 }
562
563 // clear out any empty teams
564 for ( i = 1; i <= m_iNumTeams; i++ )
565 {
566 if ( g_TeamInfo[i].players < 1 )
567 memset( &g_TeamInfo[i], 0, sizeof(team_info_t) );
568 }
569
570 // Update the scoreboard
571 Update();
572}
