403void CL_DLLEXPORT HUD_TempEntUpdate (
404 double frametime, // Simulation time
405 double client_time, // Absolute time on client
406 double cl_gravity, // True gravity on client
407 TEMPENTITY **ppTempEntFree, // List of freed temporary ents
408 TEMPENTITY **ppTempEntActive, // List
409 int ( *Callback_AddVisibleEntity )( cl_entity_t *pEntity ),
410 void ( *Callback_TempEntPlaySound )( TEMPENTITY *pTemp, float damp ) )
411{
412// RecClTempEntUpdate(frametime, client_time, cl_gravity, ppTempEntFree, ppTempEntActive, Callback_AddVisibleEntity, Callback_TempEntPlaySound);
413
414 static int gTempEntFrame = 0;
415 int i;
416 TEMPENTITY *pTemp, *pnext, *pprev;
417 float freq, gravity, gravitySlow, life, fastFreq;
418
419 Vector vAngles;
420
421 gEngfuncs.GetViewAngles( (float*)vAngles );
422
423 if ( g_pParticleMan )
424 g_pParticleMan->SetVariables( cl_gravity, vAngles );
425
426 // Nothing to simulate
427 if ( !*ppTempEntActive )
428 return;
429
430 // in order to have tents collide with players, we have to run the player prediction code so
431 // that the client has the player list. We run this code once when we detect any COLLIDEALL
432 // tent, then set this BOOL to true so the code doesn't get run again if there's more than
433 // one COLLIDEALL ent for this update. (often are).
434 gEngfuncs.pEventAPI->EV_SetUpPlayerPrediction( false, true );
435
436 // Store off the old count
437 gEngfuncs.pEventAPI->EV_PushPMStates();
438
439 // Now add in all of the players.
440 gEngfuncs.pEventAPI->EV_SetSolidPlayers ( -1 );
441
442 // !!!BUGBUG -- This needs to be time based
443 gTempEntFrame = (gTempEntFrame+1) & 31;
444
445 pTemp = *ppTempEntActive;
446
447 // !!! Don't simulate while paused.... This is sort of a hack, revisit.
448 if ( frametime <= 0 )
449 {
450 while ( pTemp )
451 {
452 if ( !(pTemp->flags & FTENT_NOMODEL ) )
453 {
454 Callback_AddVisibleEntity( &pTemp->entity );
455 }
456 pTemp = pTemp->next;
457 }
458 goto finish;
459 }
460
461 pprev = NULL;
462 freq = client_time * 0.01;
463 fastFreq = client_time * 5.5;
464 gravity = -frametime * cl_gravity;
465 gravitySlow = gravity * 0.5;
466
467 while ( pTemp )
468 {
469 int active;
470
471 active = 1;
472
473 life = pTemp->die - client_time;
474 pnext = pTemp->next;
475 if ( life < 0 )
476 {
477 if ( pTemp->flags & FTENT_FADEOUT )
478 {
479 if (pTemp->entity.curstate.rendermode == kRenderNormal)
480 pTemp->entity.curstate.rendermode = kRenderTransTexture;
481 pTemp->entity.curstate.renderamt = pTemp->entity.baseline.renderamt * ( 1 + life * pTemp->fadeSpeed );
482 if ( pTemp->entity.curstate.renderamt <= 0 )
483 active = 0;
484
485 }
486 else
487 active = 0;
488 }
489 if ( !active ) // Kill it
490 {
491 pTemp->next = *ppTempEntFree;
492 *ppTempEntFree = pTemp;
493 if ( !pprev ) // Deleting at head of list
494 *ppTempEntActive = pnext;
495 else
496 pprev->next = pnext;
497 }
498 else
499 {
500 pprev = pTemp;
501
502 VectorCopy( pTemp->entity.origin, pTemp->entity.prevstate.origin );
503
504 if ( pTemp->flags & FTENT_SPARKSHOWER )
505 {
506 // Adjust speed if it's time
507 // Scale is next think time
508 if ( client_time > pTemp->entity.baseline.scale )
509 {
510 // Show Sparks
511 gEngfuncs.pEfxAPI->R_SparkEffect( pTemp->entity.origin, 8, -200, 200 );
512
513 // Reduce life
514 pTemp->entity.baseline.framerate -= 0.1;
515
516 if ( pTemp->entity.baseline.framerate <= 0.0 )
517 {
518 pTemp->die = client_time;
519 }
520 else
521 {
522 // So it will die no matter what
523 pTemp->die = client_time + 0.5;
524
525 // Next think
526 pTemp->entity.baseline.scale = client_time + 0.1;
527 }
528 }
529 }
530 else if ( pTemp->flags & FTENT_PLYRATTACHMENT )
531 {
532 cl_entity_t *pClient;
533
534 pClient = gEngfuncs.GetEntityByIndex( pTemp->clientIndex );
535
536 VectorAdd( pClient->origin, pTemp->tentOffset, pTemp->entity.origin );
537 }
538 else if ( pTemp->flags & FTENT_SINEWAVE )
539 {
540 pTemp->x += pTemp->entity.baseline.origin[0] * frametime;
541 pTemp->y += pTemp->entity.baseline.origin[1] * frametime;
542
543 pTemp->entity.origin[0] = pTemp->x + sin( pTemp->entity.baseline.origin[2] + client_time * pTemp->entity.prevstate.frame ) * (10*pTemp->entity.curstate.framerate);
544 pTemp->entity.origin[1] = pTemp->y + sin( pTemp->entity.baseline.origin[2] + fastFreq + 0.7 ) * (8*pTemp->entity.curstate.framerate);
545 pTemp->entity.origin[2] += pTemp->entity.baseline.origin[2] * frametime;
546 }
547 else if ( pTemp->flags & FTENT_SPIRAL )
548 {
549 float s, c;
550 s = sin( pTemp->entity.baseline.origin[2] + fastFreq );
551 c = cos( pTemp->entity.baseline.origin[2] + fastFreq );
552
553 pTemp->entity.origin[0] += pTemp->entity.baseline.origin[0] * frametime + 8 * sin( client_time * 20 + (int)pTemp );
554 pTemp->entity.origin[1] += pTemp->entity.baseline.origin[1] * frametime + 4 * sin( client_time * 30 + (int)pTemp );
555 pTemp->entity.origin[2] += pTemp->entity.baseline.origin[2] * frametime;
556 }
557
558 else
559 {
560 for ( i = 0; i < 3; i++ )
561 pTemp->entity.origin[i] += pTemp->entity.baseline.origin[i] * frametime;
562 }
563
564 if ( pTemp->flags & FTENT_SPRANIMATE )
565 {
566 pTemp->entity.curstate.frame += frametime * pTemp->entity.curstate.framerate;
567 if ( pTemp->entity.curstate.frame >= pTemp->frameMax )
568 {
569 pTemp->entity.curstate.frame = pTemp->entity.curstate.frame - (int)(pTemp->entity.curstate.frame);
570
571 if ( !(pTemp->flags & FTENT_SPRANIMATELOOP) )
572 {
573 // this animating sprite isn't set to loop, so destroy it.
574 pTemp->die = client_time;
575 pTemp = pnext;
576 continue;
577 }
578 }
579 }
580 else if ( pTemp->flags & FTENT_SPRCYCLE )
581 {
582 pTemp->entity.curstate.frame += frametime * 10;
583 if ( pTemp->entity.curstate.frame >= pTemp->frameMax )
584 {
585 pTemp->entity.curstate.frame = pTemp->entity.curstate.frame - (int)(pTemp->entity.curstate.frame);
586 }
587 }
588// Experiment
589#if 0
590 if ( pTemp->flags & FTENT_SCALE )
591 pTemp->entity.curstate.framerate += 20.0 * (frametime / pTemp->entity.curstate.framerate);
592#endif
593
594 if ( pTemp->flags & FTENT_ROTATE )
595 {
596 pTemp->entity.angles[0] += pTemp->entity.baseline.angles[0] * frametime;
597 pTemp->entity.angles[1] += pTemp->entity.baseline.angles[1] * frametime;
598 pTemp->entity.angles[2] += pTemp->entity.baseline.angles[2] * frametime;
599
600 VectorCopy( pTemp->entity.angles, pTemp->entity.latched.prevangles );
601 }
602
603 if ( pTemp->flags & (FTENT_COLLIDEALL | FTENT_COLLIDEWORLD) )
604 {
605 vec3_t traceNormal;
606 float traceFraction = 1;
607
608 if ( pTemp->flags & FTENT_COLLIDEALL )
609 {
610 pmtrace_t pmtrace;
611 physent_t *pe;
612
613 gEngfuncs.pEventAPI->EV_SetTraceHull( 2 );
614
615 gEngfuncs.pEventAPI->EV_PlayerTrace( pTemp->entity.prevstate.origin, pTemp->entity.origin, PM_STUDIO_BOX, -1, &pmtrace );
616
617
618 if ( pmtrace.fraction != 1 )
619 {
620 pe = gEngfuncs.pEventAPI->EV_GetPhysent( pmtrace.ent );
621
622 if ( !pmtrace.ent || ( pe->info != pTemp->clientIndex ) )
623 {
624 traceFraction = pmtrace.fraction;
625 VectorCopy( pmtrace.plane.normal, traceNormal );
626
627 if ( pTemp->hitcallback )
628 {
629 (*pTemp->hitcallback)( pTemp, &pmtrace );
630 }
631 }
632 }
633 }
634 else if ( pTemp->flags & FTENT_COLLIDEWORLD )
635 {
636 pmtrace_t pmtrace;
637
638 gEngfuncs.pEventAPI->EV_SetTraceHull( 2 );
639
640 gEngfuncs.pEventAPI->EV_PlayerTrace( pTemp->entity.prevstate.origin, pTemp->entity.origin, PM_STUDIO_BOX | PM_WORLD_ONLY, -1, &pmtrace );
641
642 if ( pmtrace.fraction != 1 )
643 {
644 traceFraction = pmtrace.fraction;
645 VectorCopy( pmtrace.plane.normal, traceNormal );
646
647 if ( pTemp->flags & FTENT_SPARKSHOWER )
648 {
649 // Chop spark speeds a bit more
650 //
651 VectorScale( pTemp->entity.baseline.origin, 0.6, pTemp->entity.baseline.origin );
652
653 if ( Length( pTemp->entity.baseline.origin ) < 10 )
654 {
655 pTemp->entity.baseline.framerate = 0.0;
656 }
657 }
658
659 if ( pTemp->hitcallback )
660 {
661 (*pTemp->hitcallback)( pTemp, &pmtrace );
662 }
663 }
664 }
665
666 if ( traceFraction != 1 ) // Decent collision now, and damping works
667 {
668 float proj, damp;
669
670 // Place at contact point
671 VectorMA( pTemp->entity.prevstate.origin, traceFraction*frametime, pTemp->entity.baseline.origin, pTemp->entity.origin );
672 // Damp velocity
673 damp = pTemp->bounceFactor;
674 if ( pTemp->flags & (FTENT_GRAVITY|FTENT_SLOWGRAVITY) )
675 {
676 damp *= 0.5;
677 if ( traceNormal[2] > 0.9 ) // Hit floor?
678 {
679 if ( pTemp->entity.baseline.origin[2] <= 0 && pTemp->entity.baseline.origin[2] >= gravity*3 )
680 {
681 damp = 0; // Stop
682 pTemp->flags &= ~(FTENT_ROTATE|FTENT_GRAVITY|FTENT_SLOWGRAVITY|FTENT_COLLIDEWORLD|FTENT_SMOKETRAIL);
683 pTemp->entity.angles[0] = 0;
684 pTemp->entity.angles[2] = 0;
685 }
686 }
687 }
688
689 if (pTemp->hitSound)
690 {
691 Callback_TempEntPlaySound(pTemp, damp);
692 }
693
694 if (pTemp->flags & FTENT_COLLIDEKILL)
695 {
696 // die on impact
697 pTemp->flags &= ~FTENT_FADEOUT;
698 pTemp->die = client_time;
699 }
700 else
701 {
702 // Reflect velocity
703 if ( damp != 0 )
704 {
705 proj = DotProduct( pTemp->entity.baseline.origin, traceNormal );
706 VectorMA( pTemp->entity.baseline.origin, -proj*2, traceNormal, pTemp->entity.baseline.origin );
707 // Reflect rotation (fake)
708
709 pTemp->entity.angles[1] = -pTemp->entity.angles[1];
710 }
711
712 if ( damp != 1 )
713 {
714
715 VectorScale( pTemp->entity.baseline.origin, damp, pTemp->entity.baseline.origin );
716 VectorScale( pTemp->entity.angles, 0.9, pTemp->entity.angles );
717 }
718 }
719 }
720 }
721
722
723 if ( (pTemp->flags & FTENT_FLICKER) && gTempEntFrame == pTemp->entity.curstate.effects )
724 {
725 dlight_t *dl = gEngfuncs.pEfxAPI->CL_AllocDlight (0);
726 VectorCopy (pTemp->entity.origin, dl->origin);
727 dl->radius = 60;
728 dl->color.r = 255;
729 dl->color.g = 120;
730 dl->color.b = 0;
731 dl->die = client_time + 0.01;
732 }
733
734 if ( pTemp->flags & FTENT_SMOKETRAIL )
735 {
736 gEngfuncs.pEfxAPI->R_RocketTrail (pTemp->entity.prevstate.origin, pTemp->entity.origin, 1);
737 }
738
739 if ( pTemp->flags & FTENT_GRAVITY )
740 pTemp->entity.baseline.origin[2] += gravity;
741 else if ( pTemp->flags & FTENT_SLOWGRAVITY )
742 pTemp->entity.baseline.origin[2] += gravitySlow;
743
744 if ( pTemp->flags & FTENT_CLIENTCUSTOM )
745 {
746 if ( pTemp->callback )
747 {
748 ( *pTemp->callback )( pTemp, frametime, client_time );
749 }
750 }
751
752 // Cull to PVS (not frustum cull, just PVS)
753 if ( !(pTemp->flags & FTENT_NOMODEL ) )
754 {
755 if ( !Callback_AddVisibleEntity( &pTemp->entity ) )
756 {
757 if ( !(pTemp->flags & FTENT_PERSIST) )
758 {
759 pTemp->die = client_time; // If we can't draw it this frame, just dump it.
760 pTemp->flags &= ~FTENT_FADEOUT; // Don't fade out, just die
761 }
762 }
763 }
764 }
765 pTemp = pnext;
766 }
767
768finish:
769 // Restore state info
770 gEngfuncs.pEventAPI->EV_PopPMStates();
771}
