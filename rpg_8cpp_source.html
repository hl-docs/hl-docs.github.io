266void CRpgRocket::FollowThink(void)
267{
268 CBaseEntity *pOther = NULL; // Pointer to other entities
269 Vector vecTarget; // Target direction
270 Vector vecDir; // Direction vector
271 float flDist, flMax, flDot; // Distance and dot product values
272 TraceResult tr; // Trace result
273
274 UTIL_MakeAimVectors(pev->angles); // Update aim vectors
275
276 vecTarget = gpGlobals->v_forward; // Set initial target direction
277 flMax = 4096; // Set maximum distance
278
279 // Examine all entities within a reasonable radius
280 while ((pOther = UTIL_FindEntityByClassname(pOther, "laser_spot")) != NULL)
281 {
282 UTIL_TraceLine(pev->origin, pOther->pev->origin, dont_ignore_monsters, ENT(pev), &tr); // Trace line to the laser spot
283 if (tr.flFraction >= 0.90) // Check if the trace hit the laser spot
284 {
285 vecDir = pOther->pev->origin - pev->origin; // Calculate direction to the laser spot
286 flDist = vecDir.Length(); // Get the distance to the laser spot
287 vecDir = vecDir.Normalize(); // Normalize the direction vector
288 flDot = DotProduct(gpGlobals->v_forward, vecDir); // Calculate the dot product
289
290 if ((flDot > 0) && (flDist * (1 - flDot) < flMax)) // Check if the laser spot is within range
291 {
292 flMax = flDist * (1 - flDot); // Update maximum distance
293 vecTarget = vecDir; // Update target direction
294 }
295 }
296 }
297
298 pev->angles = UTIL_VecToAngles(vecTarget); // Update the rocket's angles
299
300 // Adjust rocket speed and direction
301 float flSpeed = pev->velocity.Length();
302 if (gpGlobals->time - m_flIgniteTime < 1.0) // Check if the rocket is within the first second after ignition
303 {
304 pev->velocity = pev->velocity * 0.2 + vecTarget * (flSpeed * 0.8 + 400); // Adjust velocity towards the target
305 if (pev->waterlevel == 3) // Check if the rocket is underwater
306 {
307 if (pev->velocity.Length() > 300) // Limit speed underwater
308 {
309 pev->velocity = pev->velocity.Normalize() * 300; // Cap speed
310 }
311 UTIL_BubbleTrail(pev->origin - pev->velocity * 0.1, pev->origin, 4); // Create bubble trail
312 }
313 else
314 {
315 if (pev->velocity.Length() > 2000) // Cap speed in air
316 {
317 pev->velocity = pev->velocity.Normalize() * 2000; // Cap speed
318 }
319 }
320 }
321 else
322 {
323 if (pev->effects & EF_LIGHT) // Check if the light effect is enabled
324 {
325 pev->effects = 0; // Disable light effect
326 STOP_SOUND(ENT(pev), CHAN_VOICE, "weapons/rocket1.wav"); // Stop the rocket sound
327 }
328 pev->velocity = pev->velocity * 0.2 + vecTarget * flSpeed * 0.798; // Adjust velocity towards the target
329 if (pev->waterlevel == 0 && pev->velocity.Length() < 1500) // Check if the rocket is above water and slow
330 {
331 Detonate(); // Detonate the rocket
332 }
333 }
334
335 pev->nextthink = gpGlobals->time + 0.1; // Schedule the next think
336}

347void CRpg::Reload(void)
348{
349 int iResult;
350
351 if (m_iClip == 1) // Check if the clip is already full
352 {
353 // Don't bother with any of this if there's no need to reload.
354 return;
355 }
356
357 if (m_pPlayer->ammo_rockets <= 0) // Check if the player has rockets
358 return; // Exit if no rockets are available
359
360 // The RPG waits to autoreload when no missiles are active while the LTD is on,
361 // so the weapon code is constantly calling this function, but is often denied because
362 // a) missiles are in flight, but the LTD is on
363 // or
364 // b) the player is totally out of ammo and has nothing to switch to, and should be allowed to
365 // shine the designator around.
366 //
367 // Set the next attack time into the future so that WeaponIdle will get called more often
368 // than reload, allowing the RPG LTD to be updated.
369
370 m_flNextPrimaryAttack = GetNextAttackDelay(0.5); // Set the next primary attack delay
371
372 if (m_cActiveRockets && m_fSpotActive) // Check if there are active rockets and the spot is active
373 {
374 // No reloading when there are active missiles tracking the designator.
375 // Ward off future autoreload attempts by setting the next attack time into the future for a bit.
376 return; // Exit if reloading is not allowed
377 }
378
379#ifndef CLIENT_DLL
380 if (m_pSpot && m_fSpotActive) // Check if the spot is active
381 {
382 m_pSpot->Suspend(2.1); // Suspend the laser spot for a short duration
383 m_flNextSecondaryAttack = UTIL_WeaponTimeBase() + 2.1; // Set the next secondary attack time
384 }
385#endif
386
387 if (m_iClip == 0) // Check if the clip is empty
388 iResult = DefaultReload(RPG_MAX_CLIP, RPG_RELOAD, 2); // Perform the default reload
389
390 if (iResult) // Check if the reload was successful
391 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat(m_pPlayer->random_seed, 10, 15); // Update idle time
392}

567void CRpg::PrimaryAttack()
568{
569 if (m_iClip) // Check if there is ammo in the clip
570 {
571 m_pPlayer->m_iWeaponVolume = LOUD_GUN_VOLUME; // Set the weapon volume
572 m_pPlayer->m_iWeaponFlash = BRIGHT_GUN_FLASH; // Set the weapon flash
573
574#ifndef CLIENT_DLL
575 // Player "shoot" animation
576 m_pPlayer->SetAnimation(PLAYER_ATTACK1); // Set the player's attack animation
577
578 UTIL_MakeVectors(m_pPlayer->pev->v_angle); // Update aim vectors
579 Vector vecSrc = m_pPlayer->GetGunPosition() + gpGlobals->v_forward * 16 + gpGlobals->v_right * 8 + gpGlobals->v_up * -8; // Calculate the source position
580
581 CRpgRocket *pRocket = CRpgRocket::CreateRpgRocket(vecSrc, m_pPlayer->pev->v_angle, m_pPlayer, this); // Create the RPG rocket
582
583 UTIL_MakeVectors(m_pPlayer->pev->v_angle); // Update aim vectors again
584 pRocket->pev->velocity = pRocket->pev->velocity + gpGlobals->v_forward * DotProduct(m_pPlayer->pev->velocity, gpGlobals->v_forward); // Adjust rocket velocity
585#endif
586
587 // Firing RPG no longer turns on the designator. ALT fire is a toggle switch for the LTD.
588
589 int flags;
590#if defined(CLIENT_WEAPONS)
591 flags = FEV_NOTHOST; // Set flags for client weapons
592#else
593 flags = 0; // No flags for server
594#endif
595
596 PLAYBACK_EVENT(flags, m_pPlayer->edict(), m_usRpg); // Play the RPG firing event
597
598 m_iClip--; // Decrement the clip count
599
600 m_flNextPrimaryAttack = GetNextAttackDelay(1.5); // Set the next primary attack delay
601 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 1.5; // Set the idle time
602 }
603 else
604 {
605 PlayEmptySound(); // Play empty sound if no ammo
606 }
607 UpdateSpot(); // Update the laser spot position
608}

638void CRpg::WeaponIdle(void)
639{
640 UpdateSpot(); // Update the laser spot position
641
642 ResetEmptySound(); // Reset the empty sound state
643
644 if (m_flTimeWeaponIdle > UTIL_WeaponTimeBase()) // Check if the weapon is still idle
645 return; // Exit if still idle
646
647 if (m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType]) // Check if there is primary ammo
648 {
649 int iAnim; // Animation variable
650 float flRand = UTIL_SharedRandomFloat(m_pPlayer->random_seed, 0, 1); // Generate a random float
651 if (flRand <= 0.75 || m_fSpotActive) // Check if to play idle or fidget animation
652 {
653 if (m_iClip == 0)
654 iAnim = RPG_IDLE_UL; // Idle animation for empty clip
655 else
656 iAnim = RPG_IDLE; // Idle animation for full clip
657
658 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 90.0 / 15.0; // Set idle time
659 }
660 else
661 {
662 if (m_iClip == 0)
663 iAnim = RPG_FIDGET_UL; // Fidget animation for empty clip
664 else
665 iAnim = RPG_FIDGET; // Fidget animation for full clip
666
667 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 3.0; // Set idle time
668 }
669
670 SendWeaponAnim(iAnim); // Send the selected animation
671 }
672 else
673 {
674 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 1; // Set idle time if no ammo
675 }
676}
