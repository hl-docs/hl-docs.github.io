111{
112 unsigned int range;
113
114 U_Srand( (int)seed + low + high );
115
116 range = high - low + 1;
117 if ( !(range - 1) )
118 {
119 return low;

122 {
123 int offset;

125

184{

189 UTIL_TraceLine( pev->startpos, pev->startpos + gpGlobals->v_forward * 8192, ignore_monsters, owner, &tr);
190 pev->endpos = tr.vecEndPos;

549{
550 edict_t *pEdict = g_engfuncs.pfnPEntityOfEntIndex( 1 );
551 CBaseEntity *pEntity;

553
554 count = 0;
555
556 if ( !pEdict )

559 for ( int i = 1; i < gpGlobals->maxEntities; i++, pEdict++ )
560 {
561 if ( pEdict->free ) // Not in use

578

580 count++;
581
582 if ( count >= listMax )

584 }
585
586 return count;

601int UTIL_MonstersInSphere( CBaseEntity **pList, int listMax, const Vector &center, float radius )
602{
603 edict_t *pEdict = g_engfuncs.pfnPEntityOfEntIndex( 1 );
604 CBaseEntity *pEntity;
605 int count;
606 float distance, delta;
607
608 count = 0;
609 float radiusSquared = radius * radius;
610
611 if ( !pEdict )
612 return count;
613
614 for ( int i = 1; i < gpGlobals->maxEntities; i++, pEdict++ )
615 {
616 if ( pEdict->free ) // Not in use
617 continue;
618
619 if ( !(pEdict->v.flags & (FL_CLIENT|FL_MONSTER)) ) // Not a client/monster ?
620 continue;
621
622 // Use origin for X & Y since they are centered for all monsters
623 // Now X
624 delta = center.x - pEdict->v.origin.x;//(pEdict->v.absmin.x + pEdict->v.absmax.x)*0.5;
625 delta *= delta;
626
627 if ( delta > radiusSquared )
628 continue;
629 distance = delta;
630
631 // Now Y
632 delta = center.y - pEdict->v.origin.y;//(pEdict->v.absmin.y + pEdict->v.absmax.y)*0.5;
633 delta *= delta;
634
635 distance += delta;
636 if ( distance > radiusSquared )
637 continue;
638
639 // Now Z
640 delta = center.z - (pEdict->v.absmin.z + pEdict->v.absmax.z)*0.5;
641 delta *= delta;
642
643 distance += delta;
644 if ( distance > radiusSquared )
645 continue;
646
647 pEntity = CBaseEntity::Instance(pEdict);
648 if ( !pEntity )
649 continue;
650
651 pList[ count ] = pEntity;
652 count++;
653
654 if ( count >= listMax )
655 return count;
656 }
657
658
659 return count;
660}

946void UTIL_ScreenShake(const Vector &center, float amplitude, float frequency, float duration, float radius)
947{
948 int i;
949 float localAmplitude;
950 ScreenShake shake;
951
952 shake.duration = FixedUnsigned16(duration, 1 << 12); // 4.12 fixed
953 shake.frequency = FixedUnsigned16(frequency, 1 << 8); // 8.8 fixed
954
955 for (i = 1; i <= gpGlobals->maxClients; i++)
956 {
957 CBaseEntity *pPlayer = UTIL_PlayerByIndex(i);
958
959 if (!pPlayer || !(pPlayer->pev->flags & FL_ONGROUND)) // Don't shake if not on ground
960 continue;
961
962 localAmplitude = 0;
963
964 if (radius <= 0)
965 localAmplitude = amplitude;
966 else
967 {
968 Vector delta = center - pPlayer->pev->origin;
969 float distance = delta.Length();
970
971 if (distance < radius)
972 localAmplitude = amplitude; // radius - distance;
973 }
974 if (localAmplitude)
975 {
976 shake.amplitude = FixedUnsigned16(localAmplitude, 1 << 12); // 4.12 fixed
977
978 MESSAGE_BEGIN(MSG_ONE, gmsgShake, NULL, pPlayer->edict()); // use the magic #1 for "one client"
979
980 WRITE_SHORT(shake.amplitude); // shake amount
981 WRITE_SHORT(shake.duration); // shake lasts this long
