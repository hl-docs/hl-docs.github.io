222int UTIL_SharedRandomLong( unsigned int seed, int low, int high )
223{
224 unsigned int range;
225
226 U_Srand( (int)seed + low + high );
227
228 range = high - low + 1;
229 if ( !(range - 1) )
230 {
231 return low;
232 }
233 else
234 {
235 int offset;
236 int rnum;
237
238 rnum = U_Random();
239
240 offset = rnum % range;
241
242 return (low + offset);
243 }
244}

256float UTIL_SharedRandomFloat( unsigned int seed, float low, float high )
257{
258 unsigned int range;
259
260 U_Srand( (int)seed + *(int *)&low + *(int *)&high );
261
262 U_Random();
263 U_Random();
264
265 range = high - low;
266 if ( !range )
267 {
268 return low;
269 }
270 else
271 {
272 int tensixrand;
273 float offset;
274
275 tensixrand = U_Random() & 65535;
276
277 offset = (float)tensixrand / 65536.0;
278
279 return (low + offset * range );
280 }
