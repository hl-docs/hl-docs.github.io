86float CHudMessage::FadeBlend( float fadein, float fadeout, float hold, float localTime )
87{
88 float fadeTime = fadein + hold;
89 float fadeBlend;
90
91 if ( localTime < 0 )
92 return 0;
93
94 if ( localTime < fadein )
95 {
96 fadeBlend = 1 - ((fadein - localTime) / fadein);
97 }
98 else if ( localTime > fadeTime )
99 {
100 if ( fadeout > 0 )
101 fadeBlend = 1 - ((localTime - fadeTime) / fadeout);
102 else
103 fadeBlend = 0;
104 }
105 else
106 fadeBlend = 1;
107
108 return fadeBlend;
109}

176void CHudMessage::MessageScanNextChar( void )
177{
178 int srcRed, srcGreen, srcBlue, destRed, destGreen, destBlue;
179 int blend;
180
181 srcRed = m_parms.pMessage->r1;
182 srcGreen = m_parms.pMessage->g1;
183 srcBlue = m_parms.pMessage->b1;
184 blend = 0; // Pure source
185 destRed = destGreen = destBlue = 0;
186
187 switch( m_parms.pMessage->effect )
188 {
189 // Fade-in / Fade-out
190 case 0:
191 case 1:
192 blend = m_parms.fadeBlend;
193 break;
194
195 case 2:
196 m_parms.charTime += m_parms.pMessage->fadein;
197 if ( m_parms.charTime > m_parms.time )
198 {
199 srcRed = srcGreen = srcBlue = 0;
200 blend = 0; // pure source
201 }
202 else
203 {
204 float deltaTime = m_parms.time - m_parms.charTime;
205
206 if ( m_parms.time > m_parms.fadeTime )
207 {
208 blend = m_parms.fadeBlend;
209 }
210 else if ( deltaTime > m_parms.pMessage->fxtime )
211 blend = 0; // pure dest
212 else
213 {
214 destRed = m_parms.pMessage->r2;
215 destGreen = m_parms.pMessage->g2;
216 destBlue = m_parms.pMessage->b2;
217 blend = 255 - (deltaTime * (1.0/m_parms.pMessage->fxtime) * 255.0 + 0.5);
218 }
219 }
220 break;
221 }
222 if ( blend > 255 )
223 blend = 255;
224 else if ( blend < 0 )
225 blend = 0;
226
227 m_parms.r = ((srcRed * (255-blend)) + (destRed * blend)) >> 8;
228 m_parms.g = ((srcGreen * (255-blend)) + (destGreen * blend)) >> 8;
229 m_parms.b = ((srcBlue * (255-blend)) + (destBlue * blend)) >> 8;
230
231 if ( m_parms.pMessage->effect == 1 && m_parms.charTime != 0 )
232 {
233 if ( m_parms.x >= 0 && m_parms.y >= 0 && (m_parms.x + gHUD.m_scrinfo.charWidths[ m_parms.text ]) <= ScreenWidth )
234 TextMessageDrawChar( m_parms.x, m_parms.y, m_parms.text, m_parms.pMessage->r2, m_parms.pMessage->g2, m_parms.pMessage->b2 );
235 }
236}

241void CHudMessage::MessageScanStart( void )
242{
243 switch( m_parms.pMessage->effect )
244 {
245 // Fade-in / out with flicker
246 case 1:
247 case 0:
248 m_parms.fadeTime = m_parms.pMessage->fadein + m_parms.pMessage->holdtime;
249
250
251 if ( m_parms.time < m_parms.pMessage->fadein )
252 {
253 m_parms.fadeBlend = ((m_parms.pMessage->fadein - m_parms.time) * (1.0/m_parms.pMessage->fadein) * 255);
254 }
255 else if ( m_parms.time > m_parms.fadeTime )
256 {
257 if ( m_parms.pMessage->fadeout > 0 )
258 m_parms.fadeBlend = (((m_parms.time - m_parms.fadeTime) / m_parms.pMessage->fadeout) * 255);
259 else
260 m_parms.fadeBlend = 255; // Pure dest (off)
261 }
262 else
263 m_parms.fadeBlend = 0; // Pure source (on)
264 m_parms.charTime = 0;
265
266 if ( m_parms.pMessage->effect == 1 && (rand()%100) < 10 )
267 m_parms.charTime = 1;
268 break;
269
270 case 2:
271 m_parms.fadeTime = (m_parms.pMessage->fadein * m_parms.length) + m_parms.pMessage->holdtime;
272
273 if ( m_parms.time > m_parms.fadeTime && m_parms.pMessage->fadeout > 0 )
274 m_parms.fadeBlend = (((m_parms.time - m_parms.fadeTime) / m_parms.pMessage->fadeout) * 255);
275 else
276 m_parms.fadeBlend = 0;
277 break;
278 }
279}

287void CHudMessage::MessageDrawScan( client_textmessage_t *pMessage, float time )
288{
289 int i, j, length, width;
290 const char *pText;
291 unsigned char line[80];
292
293 pText = pMessage->pMessage;
294 // Count lines
295 m_parms.lines = 1;
296 m_parms.time = time;
297 m_parms.pMessage = pMessage;
298 length = 0;
299 width = 0;
300 m_parms.totalWidth = 0;
301 while ( *pText )
302 {
303 if ( *pText == '\n' )
304 {
305 m_parms.lines++;
306 if ( width > m_parms.totalWidth )
307 m_parms.totalWidth = width;
308 width = 0;
309 }
310 else
311 width += gHUD.m_scrinfo.charWidths[*pText];
312 pText++;
313 length++;
314 }
315 m_parms.length = length;
316 m_parms.totalHeight = (m_parms.lines * gHUD.m_scrinfo.iCharHeight);
317
318
319 m_parms.y = YPosition( pMessage->y, m_parms.totalHeight );
320 pText = pMessage->pMessage;
321
322 m_parms.charTime = 0;
323
324 MessageScanStart();
325
326 for ( i = 0; i < m_parms.lines; i++ )
327 {
328 m_parms.lineLength = 0;
329 m_parms.width = 0;
330 while ( *pText && *pText != '\n' )
331 {
332 unsigned char c = *pText;
333 line[m_parms.lineLength] = c;
334 m_parms.width += gHUD.m_scrinfo.charWidths[c];
335 m_parms.lineLength++;
336 pText++;
337 }
338 pText++; // Skip LF
339 line[m_parms.lineLength] = 0;
340
341 m_parms.x = XPosition( pMessage->x, m_parms.width, m_parms.totalWidth );
342
343 for ( j = 0; j < m_parms.lineLength; j++ )
344 {
345 m_parms.text = line[j];
346 int next = m_parms.x + gHUD.m_scrinfo.charWidths[ m_parms.text ];
347 MessageScanNextChar();
348
349 if ( m_parms.x >= 0 && m_parms.y >= 0 && next <= ScreenWidth )
350 TextMessageDrawChar( m_parms.x, m_parms.y, m_parms.text, m_parms.r, m_parms.g, m_parms.b );
351 m_parms.x = next;
352 }
353
354 m_parms.y += gHUD.m_scrinfo.iCharHeight;
355 }

364int CHudMessage::Draw( float fTime )
365{
366 int i, drawn;
367 client_textmessage_t *pMessage;
368 float endTime;
369
370 drawn = 0;
371
372 if ( m_gameTitleTime > 0 )
373 {
374 float localTime = gHUD.m_flTime - m_gameTitleTime;
375 float brightness;
376
377 // Maybe timer isn't set yet
378 if ( m_gameTitleTime > gHUD.m_flTime )
379 m_gameTitleTime = gHUD.m_flTime;
380
381 if ( localTime > (m_pGameTitle->fadein + m_pGameTitle->holdtime + m_pGameTitle->fadeout) )
382 m_gameTitleTime = 0;
383 else
384 {
385 brightness = FadeBlend( m_pGameTitle->fadein, m_pGameTitle->fadeout, m_pGameTitle->holdtime, localTime );
386
387 int halfWidth = gHUD.GetSpriteRect(m_HUD_title_half).right - gHUD.GetSpriteRect(m_HUD_title_half).left;
388 int fullWidth = halfWidth + gHUD.GetSpriteRect(m_HUD_title_life).right - gHUD.GetSpriteRect(m_HUD_title_life).left;
389 int fullHeight = gHUD.GetSpriteRect(m_HUD_title_half).bottom - gHUD.GetSpriteRect(m_HUD_title_half).top;
390
391 int x = XPosition( m_pGameTitle->x, fullWidth, fullWidth );
392 int y = YPosition( m_pGameTitle->y, fullHeight );
393
394
395 SPR_Set( gHUD.GetSprite(m_HUD_title_half), brightness * m_pGameTitle->r1, brightness * m_pGameTitle->g1, brightness * m_pGameTitle->b1 );
396 SPR_DrawAdditive( 0, x, y, &gHUD.GetSpriteRect(m_HUD_title_half) );
397
398 SPR_Set( gHUD.GetSprite(m_HUD_title_life), brightness * m_pGameTitle->r1, brightness * m_pGameTitle->g1, brightness * m_pGameTitle->b1 );
399 SPR_DrawAdditive( 0, x + halfWidth, y, &gHUD.GetSpriteRect(m_HUD_title_life) );
400
401 drawn = 1;
402 }
403 }
404 // Fixup level transitions
405 for ( i = 0; i < maxHUDMessages; i++ )
406 {
407 // Assume m_parms.time contains last time
408 if ( m_pMessages[i] )
409 {
410 pMessage = m_pMessages[i];
411 if ( m_startTime[i] > gHUD.m_flTime )
412 m_startTime[i] = gHUD.m_flTime + m_parms.time - m_startTime[i] + 0.2; // Server takes 0.2 seconds to spawn, adjust for this
413 }
414 }
415
416 for ( i = 0; i < maxHUDMessages; i++ )
417 {
418 if ( m_pMessages[i] )
419 {
420 pMessage = m_pMessages[i];
421
422 // This is when the message is over
423 switch( pMessage->effect )
424 {
425 case 0:
426 case 1:
427 endTime = m_startTime[i] + pMessage->fadein + pMessage->fadeout + pMessage->holdtime;
428 break;
429
430 // Fade in is per character in scanning messages
431 case 2:
432 endTime = m_startTime[i] + (pMessage->fadein * strlen( pMessage->pMessage )) + pMessage->fadeout + pMessage->holdtime;
433 break;
434 }
435
436 if ( fTime <= endTime )
437 {
438 float messageTime = fTime - m_startTime[i];
439
440 // Draw the message
441 // effect 0 is fade in/fade out
442 // effect 1 is flickery credits
443 // effect 2 is write out (training room)
444 MessageDrawScan( pMessage, messageTime );
445
446 drawn++;
447 }
448 else
449 {
450 // The message is over
451 m_pMessages[i] = NULL;
452 }
453 }
454 }
455
456 // Remember the time -- to fix up level transitions
457 m_parms.time = gHUD.m_flTime;
458 // Don't call until we get another message
459 if ( !drawn )
460 m_iFlags &= ~HUD_ACTIVE;
461
462 return 1;
463}

471void CHudMessage::MessageAdd( const char *pName, float time )
472{
473 int i,j;
474 client_textmessage_t *tempMessage;
475
476 for ( i = 0; i < maxHUDMessages; i++ )
477 {
478 if ( !m_pMessages[i] )
479 {
480 // Trim off a leading # if it's there
481 if ( pName[0] == '#' )
482 tempMessage = TextMessageGet( pName+1 );
483 else
484 tempMessage = TextMessageGet( pName );
485 // If we couldnt find it in the titles.txt, just create it
486 if ( !tempMessage )
487 {
488 g_pCustomMessage.effect = 2;
489 g_pCustomMessage.r1 = g_pCustomMessage.g1 = g_pCustomMessage.b1 = g_pCustomMessage.a1 = 100;
490 g_pCustomMessage.r2 = 240;
491 g_pCustomMessage.g2 = 110;
492 g_pCustomMessage.b2 = 0;
493 g_pCustomMessage.a2 = 0;
494 g_pCustomMessage.x = -1; // Centered
495 g_pCustomMessage.y = 0.7;
496 g_pCustomMessage.fadein = 0.01;
497 g_pCustomMessage.fadeout = 1.5;
498 g_pCustomMessage.fxtime = 0.25;
499 g_pCustomMessage.holdtime = 5;
500 g_pCustomMessage.pName = g_pCustomName;
501 strcpy( g_pCustomText, pName );
502 g_pCustomMessage.pMessage = g_pCustomText;
503
504 tempMessage = &g_pCustomMessage;
505 }
506
507 for ( j = 0; j < maxHUDMessages; j++ )
508 {
509 if ( m_pMessages[j] )
510 {
511 // is this message already in the list
512 if ( !strcmp( tempMessage->pMessage, m_pMessages[j]->pMessage ) )
513 {
514 return;
515 }
516
517 // get rid of any other messages in same location (only one displays at a time)
518 if ( fabs( tempMessage->y - m_pMessages[j]->y ) < 0.0001 )
519 {
520 if ( fabs( tempMessage->x - m_pMessages[j]->x ) < 0.0001 )
521 {
522 m_pMessages[j] = NULL;
523 }
524 }
525 }
526 }
527
528 m_pMessages[i] = tempMessage;
529 m_startTime[i] = time;
530 return;
531 }
532 }
