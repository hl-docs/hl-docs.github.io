229void CLeech::RecalculateWaterlevel( void )
230{
231 // Calculate boundaries
232 Vector vecTest = pev->origin - Vector(0, 0, 400);
233 TraceResult tr;
234
235 UTIL_TraceLine(pev->origin, vecTest, missile, edict(), &tr);
236 if (tr.flFraction != 1.0)
237 m_bottom = tr.vecEndPos.z + 1;
238 else
239 m_bottom = vecTest.z;
240
241 m_top = UTIL_WaterLevel(pev->origin, pev->origin.z, pev->origin.z + 400) - 1;
242
243 // Chop off 20% of the outside range
244 float newBottom = m_bottom * 0.8 + m_top * 0.2;
245 m_top = m_bottom * 0.2 + m_top * 0.8;
246 m_bottom = newBottom;
247 m_height = RANDOM_FLOAT(m_bottom, m_top);
248 m_waterTime = gpGlobals->time + RANDOM_FLOAT(5, 7);
249}

257void CLeech::SwitchLeechState( void )
258{
259 m_stateTime = gpGlobals->time + RANDOM_FLOAT(3, 6);
260 if (m_MonsterState == MONSTERSTATE_COMBAT)
261 {
262 m_hEnemy = NULL;
263 SetState(MONSTERSTATE_IDLE);
264 m_sideTime = 0;
265 }
266 else
267 {
268 Look(m_flDistLook);
269 CBaseEntity *pEnemy = BestVisibleEnemy();
270 if (pEnemy && pEnemy->pev->waterlevel != 0)
271 {
272 m_hEnemy = pEnemy;
273 SetState(MONSTERSTATE_COMBAT);
274 m_stateTime = gpGlobals->time + RANDOM_FLOAT(18, 25);
275 AlertSound();
276 }
277 }

372void CLeech::HandleAnimEvent(MonsterEvent_t *pEvent)
373{
374 switch( pEvent->event )
375 {
376 case LEECH_AE_ATTACK:
377 AttackSound();
378 CBaseEntity *pEnemy;
379
380 pEnemy = m_hEnemy;
381 if ( pEnemy != NULL )
382 {
383 Vector dir, face;
384
385 UTIL_MakeVectorsPrivate( pev->angles, face, NULL, NULL );
386 face.z = 0;
387 dir = (pEnemy->pev->origin - pev->origin);
388 dir.z = 0;
389 dir = dir.Normalize();
390 face = face.Normalize();
391
392
393 if ( DotProduct(dir, face) > 0.9 ) // Only take damage if the leech is facing the prey
394 pEnemy->TakeDamage( pev, pev, gSkillData.leechDmgBite, DMG_SLASH );
395 }
396 m_stateTime -= 2;
397 break;
398
399 case LEECH_AE_FLOP:
400 // Play flop sound
401 break;
402
403 default:
404 CBaseMonster::HandleAnimEvent( pEvent );
405 break;
406 }
407}

432float CLeech::ObstacleDistance(CBaseEntity *pTarget)
433{
434 TraceResult tr; // Trace result structure
435 Vector vecTest; // Vector for testing
436
437 MakeVectors(); // Create forward vector
438
439 // Check for obstacle ahead
440 vecTest = pev->origin + gpGlobals->v_forward * LEECH_CHECK_DIST; // Calculate test position
441 UTIL_TraceLine(pev->origin, vecTest, missile, edict(), &tr); // Perform trace
442
443 if (tr.fStartSolid)
444 {
445 pev->speed = -LEECH_SWIM_SPEED * 0.5; // Reverse speed if stuck
446 }
447
448 if (tr.flFraction != 1.0)
449 {
450 if ((pTarget == NULL || tr.pHit != pTarget->edict()))
451 {
452 return tr.flFraction; // Return distance to obstacle
453 }
454 else
455 {
456 if (fabs(m_height - pev->origin.z) > 10)
457 return tr.flFraction; // Return distance if height difference is significant
458 }
459 }
460
461 if (m_sideTime < gpGlobals->time)
462 {
463 // Extra wide checks
464 vecTest = pev->origin + gpGlobals->v_right * LEECH_SIZEX * 2 + gpGlobals->v_forward * LEECH_CHECK_DIST;
465 UTIL_TraceLine(pev->origin, vecTest, missile, edict(), &tr);
466 if (tr.flFraction != 1.0)
467 return tr.flFraction; // Return distance if obstacle is detected
468
469 vecTest = pev->origin - gpGlobals->v_right * LEECH_SIZEX * 2 + gpGlobals->v_forward * LEECH_CHECK_DIST;
470 UTIL_TraceLine(pev->origin, vecTest, missile, edict(), &tr);
471 if (tr.flFraction != 1.0)
472 return tr.flFraction; // Return distance if obstacle is detected
473
474 // Didn't hit either side, so stop testing for another 0.5 - 1 seconds
475 m_sideTime = gpGlobals->time + RANDOM_FLOAT(0.5, 1);
476 }
477 return 1.0; // No obstacles detected
478}

485void CLeech::DeadThink(void)
486{
487 if (m_fSequenceFinished)
488 {
489 if (m_Activity == ACT_DIEFORWARD)
490 {
491 SetThink(NULL); // Stop thinking
492 StopAnimation(); // Stop animation
493 return;
494 }
495 else if (pev->flags & FL_ONGROUND)
496 {
497 pev->solid = SOLID_NOT; // Set solid type to not solid
498 SetActivity(ACT_DIEFORWARD); // Set death activity
499 }
500 }
501 StudioFrameAdvance(); // Advance animation frame
502 pev->nextthink = gpGlobals->time + 0.1; // Set next think time
503
504 // Apply damage velocity, but keep out of the walls
505 if (pev->velocity.x != 0 || pev->velocity.y != 0)
506 {
507 TraceResult tr;
508
509 // Look 0.5 seconds ahead
510 UTIL_TraceLine(pev->origin, pev->origin + pev->velocity * 0.5, missile, edict(), &tr);
511 if (tr.flFraction != 1.0)
512 {
513 pev->velocity.x = 0; // Stop x velocity
514 pev->velocity.y = 0; // Stop y velocity
515 }
516 }

525void CLeech::UpdateMotion(void)
526{
527 float flapspeed = (pev->speed - m_flAccelerate) / LEECH_ACCELERATE; // Calculate flapping speed
528 m_flAccelerate = m_flAccelerate * 0.8 + pev->speed * 0.2; // Update acceleration
529
530 if (flapspeed < 0)
531 flapspeed = -flapspeed; // Ensure positive speed
532 flapspeed += 1.0; // Adjust speed
533 if (flapspeed < 0.5)
534 flapspeed = 0.5; // Minimum speed
535 if (flapspeed > 1.9)
536 flapspeed = 1.9; // Maximum speed
537
538 pev->framerate = flapspeed; // Set frame rate
539
540 if (!m_fPathBlocked)
541 pev->avelocity.y = pev->ideal_yaw; // Set angular velocity
542 else
543 pev->avelocity.y = pev->ideal_yaw * m_obstacle; // Adjust angular velocity if path is blocked
544
545 if (pev->avelocity.y > 150)
546 m_IdealActivity = ACT_TURN_LEFT; // Set activity to turn left
547 else if (pev->avelocity.y < -150)
548 m_IdealActivity = ACT_TURN_RIGHT; // Set activity to turn right
549 else
550 m_IdealActivity = ACT_SWIM; // Set activity to swim
551
552 // Lean
553 float targetPitch, delta;
554 delta = m_height - pev->origin.z;
555
556 if (delta < -10)
557 targetPitch = -30; // Lean down
558 else if (delta > 10)
559 targetPitch = 30; // Lean up
560 else
561 targetPitch = 0; // No lean
562
563 pev->angles.x = UTIL_Approach(targetPitch, pev->angles.x, 60 * LEECH_FRAMETIME); // Adjust pitch
564
565 // Bank
566 pev->avelocity.z = -(pev->angles.z + (pev->avelocity.y * 0.25)); // Adjust bank angle
567
568 if (m_MonsterState == MONSTERSTATE_COMBAT && HasConditions(bits_COND_CAN_MELEE_ATTACK1))
569 m_IdealActivity = ACT_MELEE_ATTACK1; // Set melee attack activity
570
571 // Out of water check
572 if (!pev->waterlevel)
573 {
574 pev->movetype = MOVETYPE_TOSS; // Set movement type to toss
575 m_IdealActivity = ACT_TWITCH; // Set activity to twitch
576 pev->velocity = g_vecZero; // Stop movement
577
578 // Animation will intersect the floor if either of these is non-zero
579 pev->angles.z = 0; // Reset z angle
580 pev->angles.x = 0; // Reset x angle
581
582 if (pev->framerate < 1.0)
583 pev->framerate = 1.0; // Minimum frame rate
584 }
585 else if (pev->movetype == MOVETYPE_TOSS)
586 {
587 pev->movetype = MOVETYPE_FLY; // Change movement type to fly
588 pev->flags &= ~FL_ONGROUND; // Clear ground flag
589 RecalculateWaterlevel(); // Recalculate water level
590 m_waterTime = gpGlobals->time + 2; // Set water time
591 }
592
593 if (m_Activity != m_IdealActivity)
594 {
595 SetActivity(m_IdealActivity); // Set activity to ideal activity
596 }
597 float flInterval = StudioFrameAdvance(); // Advance animation frame
598 DispatchAnimEvents(flInterval); // Dispatch animation events
599
600#if DEBUG_BEAMS
601 if ( !m_pb )
602 m_pb = CBeam::BeamCreate( "sprites/laserbeam.spr", 5 );
603 if ( !m_pt )
604 m_pt = CBeam::BeamCreate( "sprites/laserbeam.spr", 5 );
605 m_pb->PointsInit( pev->origin, pev->origin + gpGlobals->v_forward * LEECH_CHECK_DIST );
606 m_pt->PointsInit( pev->origin, pev->origin - gpGlobals->v_right * (pev->avelocity.y*0.25) );
607 if ( m_fPathBlocked )
608 {
609 float color = m_obstacle * 30;
610 if ( m_obstacle == 1.0 )
611 color = 0;
612 if ( color > 255 )
613 color = 255;
614 m_pb->SetColor( 255, (int)color, (int)color );
615 }
616 else
617 m_pb->SetColor( 255, 255, 0 );
618 m_pt->SetColor( 0, 0, 255 );
619#endif
620}

628void CLeech::SwimThink(void)
629{
630 TraceResult tr; // Trace result structure
631 float flLeftSide; // Left side clearance
632 float flRightSide; // Right side clearance
633 float targetSpeed; // Target swimming speed
634 float targetYaw = 0; // Target yaw for turning
635 CBaseEntity *pTarget; // Pointer to the target entity
636
637 if (FNullEnt(FIND_CLIENT_IN_PVS(edict())))
638 {
639 pev->nextthink = gpGlobals->time + RANDOM_FLOAT(1, 1.5); // Set next think time
640 pev->velocity = g_vecZero; // Stop movement
641 return;
642 }
643 else
644 pev->nextthink = gpGlobals->time + 0.1; // Set next think time
645
646 targetSpeed = LEECH_SWIM_SPEED; // Set target speed
647
648 if (m_waterTime < gpGlobals->time)
649 RecalculateWaterlevel(); // Recalculate water level
650
651 if (m_stateTime < gpGlobals->time)
652 SwitchLeechState(); // Switch leech state
653
654 ClearConditions(bits_COND_CAN_MELEE_ATTACK1); // Clear melee attack conditions
655 switch (m_MonsterState)
656 {
657 case MONSTERSTATE_COMBAT:
658 pTarget = m_hEnemy; // Get the current enemy
659 if (!pTarget)
660 SwitchLeechState(); // Switch state if no target
661 else
662 {
663 // Chase the enemy's eyes
664 m_height = pTarget->pev->origin.z + pTarget->pev->view_ofs.z - 5; // Set height based on target
665 // Clip to viable water area
666 if (m_height < m_bottom)
667 m_height = m_bottom; // Adjust height if below bottom
668 else if (m_height > m_top)
669 m_height = m_top; // Adjust height if above top
670 Vector location = pTarget->pev->origin - pev->origin; // Calculate location vector
671 location.z += (pTarget->pev->view_ofs.z); // Adjust for view offset
672 if (location.Length() < 40)
673 SetConditions(bits_COND_CAN_MELEE_ATTACK1); // Set melee attack condition if close enough
674 // Turn towards target entity
675 targetYaw = UTIL_VecToYaw(location); // Calculate target yaw
676
677 targetYaw = UTIL_AngleDiff(targetYaw, UTIL_AngleMod(pev->angles.y)); // Adjust target yaw
678
679 if (targetYaw < (-LEECH_TURN_RATE * 0.75))
680 targetYaw = (-LEECH_TURN_RATE * 0.75); // Limit left turn
681 else if (targetYaw > (LEECH_TURN_RATE * 0.75))
682 targetYaw = (LEECH_TURN_RATE * 0.75); // Limit right turn
683 else
684 targetSpeed *= 2; // Increase speed if turning
685 }
686
687 break;
688
689 default:
690 if (m_zTime < gpGlobals->time)
691 {
692 float newHeight = RANDOM_FLOAT(m_bottom, m_top); // Randomly adjust height
693 m_height = 0.5 * m_height + 0.5 * newHeight; // Smooth height adjustment
694 m_zTime = gpGlobals->time + RANDOM_FLOAT(1, 4); // Set next height adjustment time
695 }
696 if (RANDOM_LONG(0, 100) < 10)
697 targetYaw = RANDOM_LONG(-30, 30); // Randomly adjust yaw
698 pTarget = NULL; // Clear target
699 // Old origin test
700 if ((pev->origin - pev->oldorigin).Length() < 1)
701 {
702 // If leech didn't move, there must be something blocking it, so try to turn
703 m_sideTime = 0; // Reset side time
704 }
705
706 break;
707 }
708
709 m_obstacle = ObstacleDistance(pTarget); // Check for obstacles
710 pev->oldorigin = pev->origin; // Update old origin
711 if (m_obstacle < 0.1)
712 m_obstacle = 0.1; // Ensure minimum obstacle distance
713
714 // Is the way ahead clear?
715 if (m_obstacle == 1.0)
716 {
717 // If the leech is turning, stop the trend.
718 if (m_flTurning != 0)
719 {
720 m_flTurning = 0; // Reset turning
721 }
722
723 m_fPathBlocked = FALSE; // Path is clear
724 pev->speed = UTIL_Approach(targetSpeed, pev->speed, LEECH_SWIM_ACCEL * LEECH_FRAMETIME); // Adjust speed
725 pev->velocity = gpGlobals->v_forward * pev->speed; // Set velocity
726 }
727 else
728 {
729 m_obstacle = 1.0 / m_obstacle; // Invert obstacle distance
730 // If we get this far in the function, the leader's path is blocked!
731 m_fPathBlocked = TRUE; // Set path blocked flag
732
733 if (m_flTurning == 0) // Something in the way and leech is not already turning to avoid
734 {
735 Vector vecTest; // Vector for testing
736 // Measure clearance on left and right to pick the best direction to turn
737 vecTest = pev->origin + (gpGlobals->v_right * LEECH_SIZEX) + (gpGlobals->v_forward * LEECH_CHECK_DIST);
738 UTIL_TraceLine(pev->origin, vecTest, missile, edict(), &tr); // Trace line to the right
739 flRightSide = tr.flFraction; // Get right side clearance
740
741 vecTest = pev->origin + (gpGlobals->v_right * -LEECH_SIZEX) + (gpGlobals->v_forward * LEECH_CHECK_DIST);
742 UTIL_TraceLine(pev->origin, vecTest, missile, edict(), &tr); // Trace line to the left
743 flLeftSide = tr.flFraction; // Get left side clearance
744
745 // Turn left, right, or random depending on clearance ratio
746 float delta = (flRightSide - flLeftSide);
747 if (delta > 0.1 || (delta > -0.1 && RANDOM_LONG(0, 100) < 50))
748 m_flTurning = -LEECH_TURN_RATE; // Turn left
749 else
750 m_flTurning = LEECH_TURN_RATE; // Turn right
751 }
752 pev->speed = UTIL_Approach(-(LEECH_SWIM_SPEED * 0.5), pev->speed, LEECH_SWIM_DECEL * LEECH_FRAMETIME * m_obstacle); // Adjust speed
753 pev->velocity = gpGlobals->v_forward * pev->speed; // Set velocity
754 }
755 pev->ideal_yaw = m_flTurning + targetYaw; // Set ideal yaw
756 UpdateMotion(); // Update motion
757}

769void CLeech::Killed(entvars_t *pevAttacker, int iGib)
770{
771 Vector vecSplatDir; // Direction for splat effect
772 TraceResult tr; // Trace result structure
773
774 // Notify the owner (if any) that the leech is dead. This is mostly for MonsterMaker functionality.
