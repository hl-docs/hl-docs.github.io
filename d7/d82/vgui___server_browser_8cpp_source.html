169 virtual Panel* getCellRenderer(int column,int row,bool columnSelected,bool rowSelected,bool cellSelected)
170 {
171 const char *info;
172 const char *val, *val2;
173 char sz[ 32 ];
174
175 info = ServersGetInfo( row );
176
177 if ( row == m_nMouseOverRow )
178 {
179 m_pLabel->setFgColor( 200, 240, 63, 100 );
180 }
181 else
182 {
183 m_pLabel->setFgColor( 255, 255, 255, 0 );
184 }
185 m_pLabel->setBgColor( 0, 0, 0, 200 );
186 m_pLabel->setContentAlignment( vgui::Label::a_west );
187 m_pLabel->setFont( Scheme::sf_primary2 );
188
189 if ( info )
190 {
191 // Fill out with the correct data
192 switch ( column )
193 {
194 case 0:
195 val = gEngfuncs.pNetAPI->ValueForKey( info, "address" );
196 if ( val )
197 {
198 strncpy( sz, val, 31 );
199 sz[ 31 ] = '\0';
200 // Server Name;
201 m_pLabel->setText( sz );
202 }
203 break;
204 case 1:
205 val = gEngfuncs.pNetAPI->ValueForKey( info, "hostname" );
206 if ( val )
207 {
208 strncpy( sz, val, 31 );
209 sz[ 31 ] = '\0';
210 // Server Map;
211 m_pLabel->setText( sz );
212 }
213 break;
214 case 2:
215 val = gEngfuncs.pNetAPI->ValueForKey( info, "map" );
216 if ( val )
217 {
218 strncpy( sz, val, 31 );
219 sz[ 31 ] = '\0';
220 // Server Name;
221 m_pLabel->setText( sz );
222 }
223 break;
224 case 3:
225 val = gEngfuncs.pNetAPI->ValueForKey( info, "current" );
226 val2 = gEngfuncs.pNetAPI->ValueForKey( info, "max" );
227 if ( val && val2 )
228 {
229 sprintf( sz, "%s/%s", val, val2 );
230 sz[ 31 ] = '\0';
231 // Server Map;
232 m_pLabel->setText( sz );
233 }
234 break;
235 case 4:
236 val = gEngfuncs.pNetAPI->ValueForKey( info, "ping" );
237 if ( val )
238 {
239 strncpy( sz, val, 31 );
240 sz[ 31 ] = '\0';
241 // Server Name;
242 m_pLabel->setText( sz );
243 }
244 break;
245 default:
246 break;
247 }
248 }
249 else
250 {
251 if ( !row && !column )
252 {
253 if ( ServersIsQuerying() )
254 {
255 m_pLabel->setText( "Waiting for servers to respond..." );
256 }
257 else
258 {
259 m_pLabel->setText( "Press 'Refresh' to search for servers..." );
260 }
261 }
262 else
263 {
264 m_pLabel->setText( "" );
265 }
266 }
267
268 return m_pLabel;
269 }
