71void CBreakable::KeyValue(KeyValueData* pkvd)
72{
73 // UNDONE_WC: explicitly ignoring these fields, but they shouldn't be in the map file!
74 if (FStrEq(pkvd->szKeyName, "explosion")) // Check for explosion type
75 {
76 if (!stricmp(pkvd->szValue, "directed"))
77 m_Explosion = expDirected; // Set directed explosion
78 else if (!stricmp(pkvd->szValue, "random"))
79 m_Explosion = expRandom; // Set random explosion
80 else
81 m_Explosion = expRandom; // Default to random explosion
82
83 pkvd->fHandled = TRUE; // Mark as handled
84 }
85 else if (FStrEq(pkvd->szKeyName, "material")) // Check for material type
86 {
87 int i = atoi(pkvd->szValue); // Convert value to integer
88
89 // 0:glass, 1:metal, 2:flesh, 3:wood
90 if ((i < 0) || (i >= matLastMaterial))
91 m_Material = matWood; // Default to wood if out of range
92 else
93 m_Material = (Materials)i; // Set material based on value
94
95 pkvd->fHandled = TRUE; // Mark as handled
96 }
97 else if (FStrEq(pkvd->szKeyName, "deadmodel")) // Check for dead model
98 {
99 pkvd->fHandled = TRUE; // Mark as handled
100 }
101 else if (FStrEq(pkvd->szKeyName, "shards")) // Check for shards
102 {
103 // m_iShards = atof(pkvd->szValue); // Uncomment if shards are to be used
104 pkvd->fHandled = TRUE; // Mark as handled
105 }
106 else if (FStrEq(pkvd->szKeyName, "gibmodel")) // Check for gib model
107 {
108 m_iszGibModel = ALLOC_STRING(pkvd->szValue); // Allocate gib model string
109 pkvd->fHandled = TRUE; // Mark as handled
110 }
111 else if (FStrEq(pkvd->szKeyName, "spawnobject")) // Check for spawn object
112 {
113 int object = atoi(pkvd->szValue); // Convert value to integer
114 if (object > 0 && object < ARRAYSIZE(pSpawnObjects))
115 m_iszSpawnObject = MAKE_STRING(pSpawnObjects[object]); // Set spawn object
116 pkvd->fHandled = TRUE; // Mark as handled
117 }
118 else if (FStrEq(pkvd->szKeyName, "explodemagnitude")) // Check for explosion magnitude
119 {
120 ExplosionSetMagnitude(atoi(pkvd->szValue)); // Set explosion magnitude
121 pkvd->fHandled = TRUE; // Mark as handled
122 }
123 else if (FStrEq(pkvd->szKeyName, "lip")) // Check for lip
124 pkvd->fHandled = TRUE; // Mark as handled
125 else
126 CBaseDelay::KeyValue(pkvd); // Call base class method for other key-value pairs
127}

157void CBreakable::Spawn(void)
158{
159 Precache(); // Precache resources for the breakable entity
160
161 // Set damage properties based on spawn flags
162 if (FBitSet(pev->spawnflags, SF_BREAK_TRIGGER_ONLY))
163 pev->takedamage = DAMAGE_NO; // No damage if only triggered
164 else
165 pev->takedamage = DAMAGE_YES; // Can take damage
166
167 pev->solid = SOLID_BSP; // Set solid type to BSP
168 pev->movetype = MOVETYPE_PUSH; // Set movement type to push
169 m_angle = pev->angles.y; // Store the original yaw angle
170 pev->angles.y = 0; // Reset yaw angle for the entity
171
172 // HACK: matGlass can receive decals, we need the client to know about this
173 // so use class to store the material flag
174 if (m_Material == matGlass)
175 {
176 pev->playerclass = 1; // Set player class for glass material
177 }
178
179 SET_MODEL(ENT(pev), STRING(pev->model)); // Set the model and link into the world
180
181 SetTouch(&CBreakable::BreakTouch); // Set the touch function to handle breaking
182 if (FBitSet(pev->spawnflags, SF_BREAK_TRIGGER_ONLY)) // Only break on trigger
183 SetTouch(NULL); // Disable touch function if only triggered
184
185 // Flag unbreakable glass as "worldbrush" so it will block ALL tracelines
186 if (!IsBreakable() && pev->rendermode != kRenderNormal)
187 pev->flags |= FL_WORLDBRUSH; // Set world brush flag
188}

467void CBreakable::BreakTouch(CBaseEntity *pOther)
468{
469 float flDamage;
470 entvars_t *pevToucher = pOther->pev;
471
472 // Only players can break these right now
473 if (!pOther->IsPlayer() || !IsBreakable())
474 {
475 return; // Exit if not a player or not breakable
476 }
477
478 if (FBitSet(pev->spawnflags, SF_BREAK_TOUCH)) // Can be broken when run into
479 {
480 flDamage = pevToucher->velocity.Length() * 0.01; // Calculate damage based on velocity
481
482 if (flDamage >= pev->health) // Check if damage exceeds health
483 {
484 SetTouch(NULL); // Disable touch function
485 TakeDamage(pevToucher, pevToucher, flDamage, DMG_CRUSH); // Apply damage
486
487 // Do a little damage to player if we broke glass or computer
488 pOther->TakeDamage(pev, pev, flDamage / 4, DMG_SLASH);
489 }
490 }
491
492 if (FBitSet(pev->spawnflags, SF_BREAK_PRESSURE) && pevToucher->absmin.z >= pev->maxs.z - 2) // Can be broken when stood upon
493 {
494 // Play creaking sound here.
495 DamageSound(); // Play damage sound
496
497 SetThink(&CBreakable::Die); // Set the think function to die
498 SetTouch(NULL); // Disable touch function
499
500 if (m_flDelay == 0) // Check for delay
501 {
502 // !!!BUGBUG - why doesn't zero delay work?
503 m_flDelay = 0.1; // Set a small delay
504 }
505
506 pev->nextthink = pev->ltime + m_flDelay; // Schedule the next think
507 }
508}

584int CBreakable::TakeDamage(entvars_t *pevInflictor, entvars_t *pevAttacker, float flDamage, int bitsDamageType)
585{
586 Vector vecTemp;
587
588 // If Attacker == Inflictor, the attack was a melee or other instant-hit attack.
589 // (that is, no actual entity projectile was involved in the attack so use the shooter's origin).
590 if (pevAttacker == pevInflictor)
591 {
592 vecTemp = pevInflictor->origin - (pev->absmin + (pev->size * 0.5)); // Calculate damage vector
593
594 // If a client hit the breakable with a crowbar, and breakable is crowbar-sensitive, break it now.
595 if (FBitSet(pevAttacker->flags, FL_CLIENT) &&
596 FBitSet(pev->spawnflags, SF_BREAK_CROWBAR) && (bitsDamageType & DMG_CLUB))
597 flDamage = pev->health; // Set damage to health
598 }
599 else
600 {
601 // An actual missile was involved.
602 vecTemp = pevInflictor->origin - (pev->absmin + (pev->size * 0.5)); // Calculate damage vector
603 }
604
605 if (!IsBreakable())
606 return 0; // Return if not breakable
607
608 // Breakables take double damage from the crowbar
609 if (bitsDamageType & DMG_CLUB)
610 flDamage *= 2;
611
612 // Boxes / glass / etc. don't take much poison damage, just the impact of the dart - consider that 10%
613 if (bitsDamageType & DMG_POISON)
614 flDamage *= 0.1;
615
616 // This global is still used for glass and other non-monster killables, along with decals.
617 g_vecAttackDir = vecTemp.Normalize(); // Normalize attack direction
618
619 // Do the damage
620 pev->health -= flDamage; // Reduce health
621 if (pev->health <= 0)
622 {
623 Killed(pevAttacker, GIB_NORMAL); // Handle death
624 Die(); // Call die function
625 return 0; // Return if dead
626 }
627
628 // Make a shard noise each time func breakable is hit.
629 // Don't play shard noise if breakable actually died.
630 DamageSound(); // Play damage sound
631
632 return 1; // Return success
633}

641void CBreakable::Die(void)
642{
643 Vector vecSpot; // Shard origin
644 Vector vecVelocity; // Shard velocity
645 CBaseEntity *pEntity = NULL;
646 char cFlag = 0;
647 int pitch;
648 float fvol;
649
650 pitch = 95 + RANDOM_LONG(0, 29); // Random pitch
651
652 if (pitch > 97 && pitch < 103)
653 pitch = 100; // Normalize pitch
654
655 // The more negative pev->health, the louder the sound should be.
656 fvol = RANDOM_FLOAT(0.85, 1.0) + (abs(pev->health) / 100.0); // Calculate volume
657
658 if (fvol > 1.0)
659 fvol = 1.0; // Cap volume
660
661 switch (m_Material)
662 {
663 case matGlass:
664 switch (RANDOM_LONG(0, 1))
665 {
666 case 0: EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "debris/bustglass1.wav", fvol, ATTN_NORM, 0, pitch); break;
667 case 1: EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "debris/bustglass2.wav", fvol, ATTN_NORM, 0, pitch); break;
668 }
669 cFlag = BREAK_GLASS; // Set glass flag
670 break;
671
672 case matWood:
673 switch (RANDOM_LONG(0, 1))
674 {
675 case 0: EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "debris/bustcrate1.wav", fvol, ATTN_NORM, 0, pitch); break;
676 case 1: EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "debris/bustcrate2.wav", fvol, ATTN_NORM, 0, pitch); break;
677 }
678 cFlag = BREAK_WOOD; // Set wood flag
679 break;
680
681 case matComputer:
682 case matMetal:
683 switch (RANDOM_LONG(0, 1))
684 {
685 case 0: EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "debris/bustmetal1.wav", fvol, ATTN_NORM, 0, pitch); break;
686 case 1: EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "debris/bustmetal2.wav", fvol, ATTN_NORM, 0, pitch); break;
687 }
688 cFlag = BREAK_METAL; // Set metal flag
689 break;
690
691 case matFlesh:
692 switch (RANDOM_LONG(0, 1))
693 {
694 case 0: EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "debris/bustflesh1.wav", fvol, ATTN_NORM, 0, pitch); break;
695 case 1: EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "debris/bustflesh2.wav", fvol, ATTN_NORM, 0, pitch); break;
696 }
697 cFlag = BREAK_FLESH; // Set flesh flag
698 break;
699
700 case matRocks:
701 case matCinderBlock:
702 switch (RANDOM_LONG(0, 1))
703 {
704 case 0: EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "debris/bustconcrete1.wav", fvol, ATTN_NORM, 0, pitch); break;
705 case 1: EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "debris/bustconcrete2.wav", fvol, ATTN_NORM, 0, pitch); break;
706 }
707 cFlag = BREAK_CONCRETE; // Set concrete flag
708 break;
709
710 case matCeilingTile:
711 EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "debris/bustceiling.wav", fvol, ATTN_NORM, 0, pitch); // Play ceiling sound
712 break;
713 }
714
715 if (m_Explosion == expDirected)
716 vecVelocity = g_vecAttackDir * 200; // Set velocity for directed explosion
717 else
718 {
719 vecVelocity.x = 0;
720 vecVelocity.y = 0;
721 vecVelocity.z = 0; // No velocity for other explosions
722 }
723
724 vecSpot = pev->origin + (pev->mins + pev->maxs) * 0.5; // Calculate shard origin
725 MESSAGE_BEGIN(MSG_PVS, SVC_TEMPENTITY, vecSpot); // Begin message for shards
726 WRITE_BYTE(TE_BREAKMODEL); // Write break model type
727
728 // Position
729 WRITE_COORD(vecSpot.x);
730 WRITE_COORD(vecSpot.y);
731 WRITE_COORD(vecSpot.z);
732
733 // Size

1010void CPushable::Move(CBaseEntity *pOther, int push)
1011{
1012 entvars_t *pevToucher = pOther->pev; // Get the other entity's variables
1013 int playerTouch = 0; // Flag for player touch
1014
1015 // Is entity standing on this pushable?
1016 if (FBitSet(pevToucher->flags, FL_ONGROUND) && pevToucher->groundentity && VARS(pevToucher->groundentity) == pev)
1017 {
1018 // Only push if floating
1019 if (pev->waterlevel > 0)
1020 pev->velocity.z += pevToucher->velocity.z * 0.1; // Apply upward force if in water
1021
1022 return; // Exit if standing on the pushable
1023 }
1024
1025 if (pOther->IsPlayer()) // Check if the other entity is a player
1026 {
1027 if (push && !(pevToucher->button & (IN_FORWARD | IN_USE))) // Don't push unless the player is pushing forward and NOT using
1028 return;
1029 playerTouch = 1; // Set player touch flag
1030 }
1031
1032 float factor; // Factor for movement calculation
1033
1034 if (playerTouch)
1035 {
1036 if (!(pevToucher->flags & FL_ONGROUND)) // Don't push away from jumping/falling players unless in water
1037 {
1038 if (pev->waterlevel < 1)
1039 return; // Exit if not in water
1040 else
1041 factor = 0.1; // Set factor for water
1042 }
1043 else
1044 factor = 1; // Set factor for grounded player
1045 }
1046 else
1047 factor = 0.25; // Set factor for non-player interaction
1048
1049 pev->velocity.x += pevToucher->velocity.x * factor; // Apply horizontal velocity
1050 pev->velocity.y += pevToucher->velocity.y * factor; // Apply vertical velocity
1051
1052 float length = sqrt(pev->velocity.x * pev->velocity.x + pev->velocity.y * pev->velocity.y); // Calculate velocity length
1053 if (push && (length > MaxSpeed())) // Limit speed if pushing
1054 {
1055 pev->velocity.x = (pev->velocity.x * MaxSpeed() / length); // Normalize velocity
1056 pev->velocity.y = (pev->velocity.y * MaxSpeed() / length);
1057 }
1058 if (playerTouch) // If the player is touching
1059 {
1060 pevToucher->velocity.x = pev->velocity.x; // Set player's velocity to match pushable
1061 pevToucher->velocity.y = pev->velocity.y;
1062 if ((gpGlobals->time - m_soundTime) > 0.7) // Play sound if enough time has passed
1063 {
1064 m_soundTime = gpGlobals->time; // Update sound time
1065 if (length > 0 && FBitSet(pev->flags, FL_ONGROUND)) // Check if on ground
1066 {
1067 m_lastSound = RANDOM_LONG(0, 2); // Select random sound
1068 EMIT_SOUND(ENT(pev), CHAN_WEAPON, m_soundNames[m_lastSound], 0.5, ATTN_NORM); // Emit sound
1069 }
1070 else
1071 STOP_SOUND(ENT(pev), CHAN_WEAPON, m_soundNames[m_lastSound]); // Stop sound if not moving
1072 }
1073 }
1074}
