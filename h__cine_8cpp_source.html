112void CLegacyCineMonster :: CineSpawn( char *szModel )
113{
114 PRECACHE_MODEL(szModel);
115 SET_MODEL(ENT(pev), szModel);
116 UTIL_SetSize(pev, Vector(-16, -16, 0), Vector(16, 16, 64));
117
118 pev->solid = SOLID_SLIDEBOX;
119 pev->movetype = MOVETYPE_STEP;
120 pev->effects = 0;
121 pev->health = 1;
122 pev->yaw_speed = 10;
123
124 // ugly alpha hack, can't set ints from the bsp.
125 pev->sequence = (int)pev->impulse;
126 ResetSequenceInfo( );
127 pev->framerate = 0.0;
128
129 m_bloodColor = BLOOD_COLOR_RED;
130
131 // if no targetname, start now
132 if ( FStringNull(pev->targetname) )
133 {
134 SetThink( &CLegacyCineMonster::CineThink );
135 pev->nextthink += 1.0;
136 }
137}

203void CCineBlood :: BloodGush ( void )
204{
205 Vector vecSplatDir;
206 TraceResult tr;
207 pev->nextthink = gpGlobals->time + 0.1;
208
209 UTIL_MakeVectors(pev->angles);
210 if ( pev->health-- < 0 )
211 REMOVE_ENTITY(ENT(pev));
212// CHANGE_METHOD ( ENT(pev), em_think, SUB_Remove );
213
214 if ( RANDOM_FLOAT ( 0 , 1 ) < 0.7 )// larger chance of globs
215 {
216 UTIL_BloodDrips( pev->origin, UTIL_RandomBloodVector(), BLOOD_COLOR_RED, 10 );
217 }
218 else// slim chance of geyser
219 {
220 UTIL_BloodStream( pev->origin, UTIL_RandomBloodVector(), BLOOD_COLOR_RED, RANDOM_LONG(50, 150) );
221 }
222
223 if ( RANDOM_FLOAT ( 0, 1 ) < 0.75 )
224 {// decals the floor with blood.
225 vecSplatDir = Vector ( 0 , 0 , -1 );
226 vecSplatDir = vecSplatDir + (RANDOM_FLOAT(-1,1) * 0.6 * gpGlobals->v_right) + (RANDOM_FLOAT(-1,1) * 0.6 * gpGlobals->v_forward);// randomize a bit
227 UTIL_TraceLine( pev->origin + Vector ( 0, 0 , 64) , pev->origin + vecSplatDir * 256, ignore_monsters, ENT(pev), &tr);
228 if ( tr.flFraction != 1.0 )
229 {
230 // Decal with a bloodsplat
231 UTIL_BloodDecalTrace( &tr, BLOOD_COLOR_RED );
232 }
233 }
234}
