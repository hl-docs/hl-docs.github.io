144int HistoryResource :: DrawAmmoHistory( float flTime )
145{
146 for ( int i = 0; i < MAX_HISTORY; i++ )
147 {
148 if ( rgAmmoHistory[i].type )
149 {
150 rgAmmoHistory[i].DisplayTime = min( rgAmmoHistory[i].DisplayTime, gHUD.m_flTime + HISTORY_DRAW_TIME );
151
152 if ( rgAmmoHistory[i].DisplayTime <= flTime )
153 { // pic drawing time has expired
154 memset( &rgAmmoHistory[i], 0, sizeof(HIST_ITEM) );
155 CheckClearHistory();
156 }
157 else if ( rgAmmoHistory[i].type == HISTSLOT_AMMO )
158 {
159 wrect_t rcPic;
160 HSPRITE *spr = gWR.GetAmmoPicFromWeapon( rgAmmoHistory[i].iId, rcPic );
161
162 int r, g, b;
163 UnpackRGB(r,g,b, RGB_YELLOWISH);
164 float scale = (rgAmmoHistory[i].DisplayTime - flTime) * 80;
165 ScaleColors(r, g, b, min(scale, 255) );
166
167 // Draw the pic
168 int ypos = ScreenHeight - (AMMO_PICKUP_PICK_HEIGHT + (AMMO_PICKUP_GAP * i));
169 int xpos = ScreenWidth - 24;
170 if ( spr && *spr ) // weapon isn't loaded yet so just don't draw the pic
171 { // the dll has to make sure it has sent info the weapons you need
172 SPR_Set( *spr, r, g, b );
173 SPR_DrawAdditive( 0, xpos, ypos, &rcPic );
174 }
175
176 // Draw the number
177 gHUD.DrawHudNumberString( xpos - 10, ypos, xpos - 100, rgAmmoHistory[i].iCount, r, g, b );
178 }
179 else if ( rgAmmoHistory[i].type == HISTSLOT_WEAP )
180 {
181 WEAPON *weap = gWR.GetWeapon( rgAmmoHistory[i].iId );
182
183 if ( !weap )
184 return 1; // we don't know about the weapon yet, so don't draw anything
185
186 int r, g, b;
187 UnpackRGB(r,g,b, RGB_YELLOWISH);
188
189 if ( !gWR.HasAmmo( weap ) )
190 UnpackRGB(r,g,b, RGB_REDISH); // if the weapon doesn't have ammo, display it as red
191
192 float scale = (rgAmmoHistory[i].DisplayTime - flTime) * 80;
193 ScaleColors(r, g, b, min(scale, 255) );
194
195 int ypos = ScreenHeight - (AMMO_PICKUP_PICK_HEIGHT + (AMMO_PICKUP_GAP * i));
196 int xpos = ScreenWidth - (weap->rcInactive.right - weap->rcInactive.left);
197 SPR_Set( weap->hInactive, r, g, b );
198 SPR_DrawAdditive( 0, xpos, ypos, &weap->rcInactive );
199 }
200 else if ( rgAmmoHistory[i].type == HISTSLOT_ITEM )
201 {
202 int r, g, b;
203
204 if ( !rgAmmoHistory[i].iId )
205 continue; // sprite not loaded
206
207 wrect_t rect = gHUD.GetSpriteRect( rgAmmoHistory[i].iId );
208
209 UnpackRGB(r,g,b, RGB_YELLOWISH);
210 float scale = (rgAmmoHistory[i].DisplayTime - flTime) * 80;
211 ScaleColors(r, g, b, min(scale, 255) );
212
213 int ypos = ScreenHeight - (AMMO_PICKUP_PICK_HEIGHT + (AMMO_PICKUP_GAP * i));
214 int xpos = ScreenWidth - (rect.right - rect.left) - 10;
215
216 SPR_Set( gHUD.GetSprite( rgAmmoHistory[i].iId ), r, g, b );
217 SPR_DrawAdditive( 0, xpos, ypos, &rect );
218 }
219 }
220 }
221
222
223 return 1;
224}
