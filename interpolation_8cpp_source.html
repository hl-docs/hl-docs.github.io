45bool SolveLSE(vec3_t v0, vec3_t v1, vec3_t v2, vec3_t v3, float *x, float *y, float *z);
46{
47 float d = Determinant(v1,v2,v3);
48
49 if (d==0.0f)
50 return false;
51
52 if ( x )
53 *x = Determinant(v0,v2,v3) / d;
54
55 if ( y )
56 *y= Determinant(v1,v0,v3) / d;
57
58 if ( z )
59 *z= Determinant(v1,v2,v0) / d;
60
61 return true;
62}

74bool GetPointBetweenLines(vec3_t &p, vec3_t a1, vec3_t m1, vec3_t a2, vec3_t m2);
75{
76 float x,z;
77
78 vec3_t t1 = CrossProduct(m1, m2);
79 vec3_t t2 = a2 - a1;
80
81 if ( !SolveLSE( t2, m1, t1, m2, &x , NULL, &z ) )
82 return false;
83
84 t1 = a1 + x*m1;
85 t2 = a2 + (-z)*m2;
86
87 p = ( t1 + t2 ) / 2.0f;
88
89 return true;
90}

146void CInterpolation::SetWaypoints( vec3_t * prev, vec3_t start, vec3_t end, vec3_t * next)
147{
148 m_StartPoint = start;
149 m_EndPoint = end;
150
151
152 vec3_t a,b,c,d;
153
154 if ( !prev && !next )
155 {
156 // no direction given, straight linear interpolation
157 m_Center = (m_StartPoint + m_EndPoint) / 2.0f;
158 }
159 else if ( !prev )
160 {
161 a = start - end;
162 float dist = a.Length() / 2.0f;
163 a = a.Normalize();
164 b = *next - end;
165 b = b.Normalize();
166 c = a - b;
167 c = c.Normalize();
168 m_Center = end + c*dist;
169
170 }
171 else if ( !next )
172 {
173 a = *prev - start;
174 a = a.Normalize();
175 b = end - start;
176 float dist = b.Length() / 2.0f;
177 b = b.Normalize();
178 c = b - a;
179 c = c.Normalize();
180 m_Center = start + c*dist;
181 }
182 else
183 {
184 // we have a previous and a next point, great!
185 a = *prev - start;
186 a = a.Normalize();
187 b = end - start;
188 b = b.Normalize();
189 c = b - a;
190
191 a = start - end;
192 a = a.Normalize();
193 b = *next - end;
194 b = b.Normalize();
195 d = a - b;
196
197 GetPointBetweenLines( m_Center, start, c, end, d);
198 }
199}

209void CInterpolation::Interpolate( float t, vec3_t &point, vec3_t &angle, float * fov)
210{
211
212 if ( m_SmoothStart && m_SmoothEnd )
213 {
214 t = (1.0f-t)*(t*t)+t*(1.0f-((t-1.0f)*(t-1.0f)));
215 }
216 else if ( m_SmoothStart )
217 {
218 t = t*t;
219 }
220 else if ( m_SmoothEnd )
221 {
222 t = t - 1.0f;
223 t = -(t*t)+1;
224 }
225
226 if ( point )
227 {
228 BezierInterpolatePoint(t, point);
229 }
230
231 if ( angle )
232 {
233 InterpolateAngle(t, angle);
234 }
235
236 if ( fov )
237 {
238 *fov = m_StartFov + (t * (m_EndFov-m_StartFov));
239 }
240}

274void CInterpolation::InterpolateAngle( float t, vec3_t &angle )
275{
276 int i;
277 float ang1, ang2;
278 float d;
279
280 for ( i = 0 ; i < 3 ; i++ )
281 {
282 ang1 = m_StartAngle[i];
283 ang2 = m_EndAngle[i];
284
285 d = ang2 - ang1;
286 if ( d > 180 )
287 {
288 d -= 360;
289 }
290 else if ( d < -180 )
291 {
292 d += 360;
293 }
294
295 angle[i] = ang1 + d * t;
296 }
297
298 NormalizeAngles( angle );
299}
