131void CLight::Spawn(void)

134 { // Inert light
135 REMOVE_ENTITY(ENT(pev));
136 return;
137 }
138
139 if (m_iStyle >= 32)

168void CLight::Use(CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
169{
170 if (m_iStyle >= 32)
171 {
172 if (!ShouldToggle(useType, !FBitSet(pev->spawnflags, SF_LIGHT_START_OFF)))
173 return;
174
175 if (FBitSet(pev->spawnflags, SF_LIGHT_START_OFF))
176 {
177 if (m_iszPattern)
178 LIGHT_STYLE(m_iStyle, (char *)STRING(m_iszPattern)); // Turn on with pattern
179 else
180 LIGHT_STYLE(m_iStyle, "m"); // Default on
181 ClearBits(pev->spawnflags, SF_LIGHT_START_OFF); // Clear off flag
182 }
183 else
184 {
185 LIGHT_STYLE(m_iStyle, "a"); // Turn off
186 SetBits(pev->spawnflags, SF_LIGHT_START_OFF); // Set off flag
187 }
188 }
189}

244void CEnvLight::KeyValue(KeyValueData* pkvd)
245{
246 if (FStrEq(pkvd->szKeyName, "_light"))
247 {
248 int r, g, b, v, j;
249 char szColor[64];
250 j = sscanf(pkvd->szValue, "%d %d %d %d\n", &r, &g, &b, &v);
251 if (j == 1)
252 {
253 g = b = r; // If only one value is provided, set g and b to r
254 }
255 else if (j == 4)
256 {
257 r = r * (v / 255.0);
258 g = g * (v / 255.0);
259 b = b * (v / 255.0);
260 }
261
262 // Simulate qrad direct, ambient, and gamma adjustments, as well as engine scaling
263 r = pow(r / 114.0, 0.6) * 264;
264 g = pow(g / 114.0, 0.6) * 264;
265 b = pow(b / 114.0, 0.6) * 264;
266
267 pkvd->fHandled = TRUE; // Mark the key as handled
268 sprintf(szColor, "%d", r);
269 CVAR_SET_STRING("sv_skycolor_r", szColor); // Set the red component
270 sprintf(szColor, "%d", g);
271 CVAR_SET_STRING("sv_skycolor_g", szColor); // Set the green component
272 sprintf(szColor, "%d", b);
273 CVAR_SET_STRING("sv_skycolor_b", szColor); // Set the blue component
274 }
275 else
276 {
277 CLight::KeyValue(pkvd); // Call base class method for unrecognized keys
278 }
279}
