197void CFlockingFlyerFlock::SpawnFlock(void)
198{
199 float R = m_flFlockRadius; // Get the flock radius
200 int iCount; // Counter for the number of boids
201 Vector vecSpot; // Position for the boid
202 CFlockingFlyer *pBoid, *pLeader; // Pointers for the boids
203
204 pLeader = pBoid = NULL; // Initialize pointers
205
206 for (iCount = 0; iCount < m_cFlockSize; iCount++)
207 {
208 pBoid = GetClassPtr((CFlockingFlyer *)NULL); // Get a new boid instance
209
210 if (!pLeader) // If no leader yet
211 {
212 pLeader = pBoid; // Make this boid the leader
213 pLeader->m_pSquadLeader = pLeader; // Set leader pointer
214 pLeader->m_pSquadNext = NULL; // No next boid yet
215 }
216
217 // Randomly position the boid within the flock radius
218 vecSpot.x = RANDOM_FLOAT(-R, R);
219 vecSpot.y = RANDOM_FLOAT(-R, R);
220 vecSpot.z = RANDOM_FLOAT(0, 16);
221 vecSpot = pev->origin + vecSpot;
222
223 UTIL_SetOrigin(pBoid->pev, vecSpot); // Set the boid's position
224 pBoid->pev->movetype = MOVETYPE_FLY; // Set movement type
225 pBoid->SpawnCommonCode(); // Initialize common code for the boid
226 pBoid->pev->flags &= ~FL_ONGROUND; // Clear ground flag
227 pBoid->pev->velocity = g_vecZero; // Set initial velocity to zero
228 pBoid->pev->angles = pev->angles; // Set initial angles
229
230 pBoid->pev->frame = 0; // Set initial frame
231 pBoid->pev->nextthink = gpGlobals->time + 0.2; // Schedule next think
232 pBoid->SetThink(&CFlockingFlyer::IdleThink); // Set think function to idle
233
234 if (pBoid != pLeader) // If this boid is not the leader
235 {
236 pLeader->SquadAdd(pBoid); // Add it to the leader's squad
237 }
238 }
239}

305{
306 CFlockingFlyer *pSquad = (CFlockingFlyer *)m_pSquadLeader; // Get the squad leader
307

320 pev->framerate = 0; // Stop animation

324 pev->movetype = MOVETYPE_TOSS; // Set movement type to toss
325

327 pev->nextthink = gpGlobals->time + 0.1; // Schedule next think

381void CFlockingFlyer::BoidAdvanceFrame(void)
382{
383 float flapspeed = (pev->speed - pev->armorvalue) / AFLOCK_ACCELERATE; // Calculate flapping speed
384 pev->armorvalue = pev->armorvalue * .8 + pev->speed * .2; // Update armor value
385
386 if (flapspeed < 0) flapspeed = -flapspeed; // Ensure positive speed
387 if (flapspeed < 0.25) flapspeed = 0.25; // Minimum speed
388 if (flapspeed > 1.9) flapspeed = 1.9; // Maximum speed
389
390 pev->framerate = flapspeed; // Set frame rate
391
392 // Lean and bank adjustments
393 pev->avelocity.x = -(pev->angles.x + flapspeed * 5);
394 pev->avelocity.z = -(pev->angles.z + pev->avelocity.y);
395
396 StudioFrameAdvance(0.1); // Advance the animation frame
397}

448void CFlockingFlyer::FormFlock(void)
449{
450 if (!InSquad()) // If not already in a squad
451 {
452 m_pSquadLeader = this; // Set self as leader
453 m_pSquadNext = NULL; // No next member yet
454 int squadCount = 1; // Initialize squad count
455
456 CBaseEntity *pEntity = NULL; // Pointer for nearby entities
457
458 // Find nearby entities within the recruitment radius
459 while ((pEntity = UTIL_FindEntityInSphere(pEntity, pev->origin, AFLOCK_MAX_RECRUIT_RADIUS)) != NULL)
460 {
461 CBaseMonster *pRecruit = pEntity->MyMonsterPointer(); // Get the monster pointer
462
463 if (pRecruit && pRecruit != this && pRecruit->IsAlive() && !pRecruit->m_pCine) // Check if recruit is valid
464 {
465 // Check if the recruit can be added to the squad
466 if (FClassnameIs(pRecruit->pev, "monster_flyer"))
467 {
468 squadCount++; // Increment squad count
469 SquadAdd((CFlockingFlyer *)pRecruit); // Add recruit to squad
470 }
471 }
472 }
473 }
474
475 SetThink(&CFlockingFlyer::IdleThink); // Go to idle state after forming the flock
476 pev->nextthink = gpGlobals->time; // Schedule next think
477}

485void CFlockingFlyer::SpreadFlock(void)
486{
487 Vector vecDir; // Direction vector
488 float flSpeed; // Holds vector magnitude while adjusting direction
489
490 CFlockingFlyer *pList = m_pSquadLeader; // Start with the squad leader
491 while (pList)
492 {
493 if (pList != this && (pev->origin - pList->pev->origin).Length() <= AFLOCK_TOO_CLOSE) // If too close
494 {
495 // Push the other boid away
496 vecDir = (pList->pev->origin - pev->origin);
497 vecDir = vecDir.Normalize();
498
499 // Store the magnitude of the other boid's velocity
500 flSpeed = pList->pev->velocity.Length();
501 pList->pev->velocity = pList->pev->velocity.Normalize();
502 pList->pev->velocity = (pList->pev->velocity + vecDir) * 0.5; // Adjust velocity
503 pList->pev->velocity = pList->pev->velocity * flSpeed; // Restore original speed
504 }
505
506 pList = pList->m_pSquadNext; // Move to the next boid
507 }
508}

542BOOL CFlockingFlyer::FPathBlocked(void)
543{
544 TraceResult tr; // Trace result for collision detection
545 Vector vecDist; // General measurement vector
546 Vector vecDir; // General measurement vector
547 BOOL fBlocked; // Blocked flag
548
549 if (m_flFakeBlockedTime > gpGlobals->time) // If recently blocked
550 {
551 m_flLastBlockedTime = gpGlobals->time; // Update last blocked time
552 return TRUE; // Path is blocked
553 }
554
555 // Use velocity for direction
556 UTIL_MakeVectors(pev->angles); // Update vectors based on angles
557
558 fBlocked = FALSE; // Assume path is clear
559
560 // Check for obstacles ahead
561 UTIL_TraceLine(pev->origin, pev->origin + gpGlobals->v_forward * AFLOCK_CHECK_DIST, ignore_monsters, ENT(pev), &tr);
562 if (tr.flFraction != 1.0) // If hit something
563 {
564 m_flLastBlockedTime = gpGlobals->time; // Update last blocked time
565 fBlocked = TRUE; // Path is blocked
566 }
567
568 // Extra wide checks
569 UTIL_TraceLine(pev->origin + gpGlobals->v_right * 12, pev->origin + gpGlobals->v_right * 12 + gpGlobals->v_forward * AFLOCK_CHECK_DIST, ignore_monsters, ENT(pev), &tr);
570 if (tr.flFraction != 1.0) // If hit something
571 {
572 m_flLastBlockedTime = gpGlobals->time; // Update last blocked time
573 fBlocked = TRUE; // Path is blocked
574 }
575
576 UTIL_TraceLine(pev->origin - gpGlobals->v_right * 12, pev->origin - gpGlobals->v_right * 12 + gpGlobals->v_forward * AFLOCK_CHECK_DIST, ignore_monsters, ENT(pev), &tr);
577 if (tr.flFraction != 1.0) // If hit something
578 {
579 m_flLastBlockedTime = gpGlobals->time; // Update last blocked time
580 fBlocked = TRUE; // Path is blocked
581 }
582
583 if (!fBlocked && gpGlobals->time - m_flLastBlockedTime > 6) // If not blocked for a while
584 {
585 m_flFakeBlockedTime = gpGlobals->time + RANDOM_LONG(1, 3); // Set fake blocked time
586 }
587
588 return fBlocked; // Return blocked status
589}

597void CFlockingFlyer::FlockLeaderThink(void)
598{
599 TraceResult tr; // Trace result for collision detection
600 Vector vecDist; // General measurement vector
601 Vector vecDir; // General measurement vector
602 int cProcessed = 0; // Count of processed boids
603 float flLeftSide; // Clearance on the left side
604 float flRightSide; // Clearance on the right side
605
606 pev->nextthink = gpGlobals->time + 0.1; // Schedule next think
607
608 UTIL_MakeVectors(pev->angles); // Update vectors based on angles
609
610 // Check if the path ahead is clear
611 if (!FPathBlocked())
612 {
613 // If the boid is turning, stop the trend
614 if (m_fTurning)
615 {
616 m_fTurning = FALSE; // Reset turning flag
617 pev->avelocity.y = 0; // Reset angular velocity
618 }
619
620 m_fPathBlocked = FALSE; // Reset path blocked flag
621
622 if (pev->speed <= AFLOCK_FLY_SPEED)
623 pev->speed += 5; // Increase speed
624
625 pev->velocity = gpGlobals->v_forward * pev->speed; // Set velocity
626
627 BoidAdvanceFrame(); // Advance the animation frame
628
629 return; // Exit function
630 }
631
632 // If the leader's path is blocked
633 m_fPathBlocked = TRUE;
634
635 if (!m_fTurning) // If not already turning
636 {
637 // Measure clearance on left and right to pick the best direction to turn
638 UTIL_TraceLine(pev->origin, pev->origin + gpGlobals->v_right * AFLOCK_CHECK_DIST, ignore_monsters, ENT(pev), &tr);
639 vecDist = (tr.vecEndPos - pev->origin);
640 flRightSide = vecDist.Length();
641
642 UTIL_TraceLine(pev->origin, pev->origin - gpGlobals->v_right * AFLOCK_CHECK_DIST, ignore_monsters, ENT(pev), &tr);
643 vecDist = (tr.vecEndPos - pev->origin);
644 flLeftSide = vecDist.Length();
645
646 // Turn right if more clearance on the right side
647 if (flRightSide > flLeftSide)
648 {
649 pev->avelocity.y = -AFLOCK_TURN_RATE; // Turn right
650 m_fTurning = TRUE; // Set turning flag
651 }
652 // Default to left turn
653 else if (flLeftSide > flRightSide)
654 {
655 pev->avelocity.y = AFLOCK_TURN_RATE; // Turn left
656 m_fTurning = TRUE; // Set turning flag
657 }
658 else
659 {
660 // Equidistant. Pick randomly between left and right.
661 m_fTurning = TRUE; // Set turning flag
662
663 if (RANDOM_LONG(0, 1) == 0)
664 {
665 pev->avelocity.y = AFLOCK_TURN_RATE; // Turn left
666 }
667 else
668 {
669 pev->avelocity.y = -AFLOCK_TURN_RATE; // Turn right
670 }
671 }
672 }
673 SpreadFlock(); // Spread the flock to avoid collisions
674
675 pev->velocity = gpGlobals->v_forward * pev->speed; // Set velocity
676
677 // Check for ground collision
678 UTIL_TraceLine(pev->origin, pev->origin - gpGlobals->v_up * 16, ignore_monsters, ENT(pev), &tr);
679 if (tr.flFraction != 1.0 && pev->velocity.z < 0)
680 pev->velocity.z = 0; // Stop downward velocity
681
682 // If on the ground, adjust position
683 if (FBitSet(pev->flags, FL_ONGROUND))
684 {
685 UTIL_SetOrigin(pev, pev->origin + Vector(0, 0, 1)); // Adjust position
686 pev->velocity.z = 0; // Stop upward velocity
687 }
688
689 if (m_flFlockNextSoundTime < gpGlobals->time) // If it's time to make a sound
690 {
691 MakeSound(); // Make sound
692 m_flFlockNextSoundTime = gpGlobals->time + RANDOM_FLOAT(1, 3); // Schedule next sound
693 }
694
695 BoidAdvanceFrame(); // Advance the animation frame
696
697 return; // Exit function
698}

706void CFlockingFlyer::FlockFollowerThink(void)
707{
708 TraceResult tr; // Trace result for collision detection
709 Vector vecDist; // General measurement vector
710 Vector vecDir; // General measurement vector
711 Vector vecDirToLeader; // Direction to the leader
712 float flDistToLeader; // Distance to the leader
713
714 pev->nextthink = gpGlobals->time + 0.1; // Schedule next think
715
716 if (IsLeader() || !InSquad()) // If this boid is the leader or not in a squad
717 {
718 SetThink(&CFlockingFlyer::FlockLeaderThink); // Set to leader think function
719 return; // Exit function
720 }
721
722 vecDirToLeader = (m_pSquadLeader->pev->origin - pev->origin); // Calculate direction to leader
723 flDistToLeader = vecDirToLeader.Length(); // Calculate distance to leader
724
725 // Match heading with leader
726 pev->angles = m_pSquadLeader->pev->angles;
727
728 // If the leader is visible, try to catch up
729 if (FInViewCone(m_pSquadLeader))
730 {
731 // If too far away, speed up
732 if (flDistToLeader > AFLOCK_TOO_FAR)
733 {
734 m_flGoalSpeed = m_pSquadLeader->pev->velocity.Length() * 1.5; // Increase goal speed
735 }
736 // If too close, slow down
737 else if (flDistToLeader < AFLOCK_TOO_CLOSE)
738 {
739 m_flGoalSpeed = m_pSquadLeader->pev->velocity.Length() * 0.5; // Decrease goal speed
740 }
741 }
742 else
743 {
744 // Wait up! The leader isn't in front, so slow down
745 m_flGoalSpeed = m_pSquadLeader->pev->velocity.Length() * 0.5; // Decrease goal speed
746 }
747
748 SpreadFlock2(); // Spread the flock to avoid collisions
749
750 pev->speed = pev->velocity.Length(); // Update speed
751 pev->velocity = pev->velocity.Normalize(); // Normalize velocity
752
753 // If too far from leader, average a vector towards it
754 if (flDistToLeader > AFLOCK_TOO_FAR)
755 {

822

824 {

845 else // Removing a node

862 }
