67/*
68===============================================================================
69
70KEY BUTTONS
71
72Continuous button event tracking is complicated by the fact that two different
73input sources (say, mouse button 1 and the control key) can both press the
74same button, but the button should only be released when both of the
75pressing key have been released.
76
77When a key event issues a button command (+forward, +attack, etc), it appends
78its key number as a parameter to the command so it can be matched up with
79the release.
80
81state bit 0 is the current state of the key
82state bit 1 is edge triggered on the up to down transition
83state bit 2 is edge triggered on the down to up transition
84
85===============================================================================
86*/

135int KB_ConvertString( char *in, char **ppout )
136{
137 char sz[ 4096 ];
138 char binding[ 64 ];
139 char *p;
140 char *pOut;
141 char *pEnd;
142 const char *pBinding;
143
144 if ( !ppout )
145 return 0;
146
147 *ppout = NULL;
148 p = in;
149 pOut = sz;
150 while ( *p )
151 {
152 if ( *p == '+' )
153 {
154 pEnd = binding;
155 while ( *p && ( isalnum( *p ) || ( pEnd == binding ) ) && ( ( pEnd - binding ) < 63 ) )
156 {
157 *pEnd++ = *p++;
158 }
159
160 *pEnd = '\0';
161
162 pBinding = NULL;
163 if ( strlen( binding + 1 ) > 0 )
164 {
165 // See if there is a binding for binding?
166 pBinding = gEngfuncs.Key_LookupBinding( binding + 1 );
167 }
168
169 if ( pBinding )
170 {
171 *pOut++ = '[';
172 pEnd = (char *)pBinding;
173 }
174 else
175 {
176 pEnd = binding;
177 }
178
179 while ( *pEnd )
180 {
181 *pOut++ = *pEnd++;
182 }
183
184 if ( pBinding )
185 {
186 *pOut++ = ']';
187 }
188 }
189 else
190 {
191 *pOut++ = *p++;
192 }
193 }
194
195 *pOut = '\0';
196
197 pOut = ( char * )malloc( strlen( sz ) + 1 );
198 strcpy( pOut, sz );
199 *ppout = pOut;
200
201 return 1;
202}

297void KeyDown (kbutton_t *b)
298{
299 int k;
300 char *c;
301
302 c = gEngfuncs.Cmd_Argv(1);
303 if (c[0])
304 k = atoi(c);
305 else
306 k = -1; // typed manually at the console for continuous down
307
308 if (k == b->down[0] || k == b->down[1])
309 return; // repeating key
310
311 if (!b->down[0])
312 b->down[0] = k;
313 else if (!b->down[1])
314 b->down[1] = k;
315 else
316 {
317 gEngfuncs.Con_DPrintf ("Three keys down for a button '%c' '%c' '%c'!\n", b->down[0], b->down[1], c);
318 return;
319 }
320
321 if (b->state & 1)
322 return; // still down
323 b->state |= 1 + 2; // down + impulse down
324}

334void KeyUp (kbutton_t *b)
335{
336 int k;
337 char *c;
338
339 c = gEngfuncs.Cmd_Argv(1);
340 if (c[0])
341 k = atoi(c);
342 else
343 { // typed manually at the console, assume for unsticking, so clear all
344 b->down[0] = b->down[1] = 0;
345 b->state = 4; // impulse up
346 return;
347 }
348
349 if (b->down[0] == k)
350 b->down[0] = 0;
351 else if (b->down[1] == k)
352 b->down[1] = 0;
353 else
354 return; // key up without coresponding down (menu pass through)
355 if (b->down[0] || b->down[1])
356 {
357 //Con_Printf ("Keys down for button: '%c' '%c' '%c' (%d,%d,%d)!\n", b->down[0], b->down[1], c, b->down[0], b->down[1], c);
358 return; // some other key is still holding it down
359 }
360
361 if (!(b->state & 1))
362 return; // still up (this should not happen)
363
364 b->state &= ~1; // now up
365 b->state |= 4; // impulse up
366}

663float CL_KeyState (kbutton_t *key)
664{
665 float val = 0.0;
666 int impulsedown, impulseup, down;
667
668 impulsedown = key->state & 2;
669 impulseup = key->state & 4;
670 down = key->state & 1;
671
672 if ( impulsedown && !impulseup )
673 {
674 // pressed and held this frame?
675 val = down ? 0.5 : 0.0;
676 }
677
678 if ( impulseup && !impulsedown )
679 {
680 // released this frame?
681 val = down ? 0.0 : 0.0;
682 }
683
684 if ( !impulsedown && !impulseup )
685 {
686 // held the entire frame?
687 val = down ? 1.0 : 0.0;
688 }
689
690 if ( impulsedown && impulseup )
691 {
692 if ( down )
693 {
694 // released and re-pressed this frame
695 val = 0.75;
696 }
697 else
698 {
699 // pressed and released this frame
700 val = 0.25;
701 }
702 }
703
704 // clear impulses
705 key->state &= 1;
706 return val;
707}

718void CL_AdjustAngles ( float frametime, float *viewangles )
719{
720 float speed;
721 float up, down;
722
723 if (in_speed.state & 1)
724 {
725 speed = frametime * cl_anglespeedkey->value;
726 }
727 else
728 {
729 speed = frametime;
730 }
731
732 if (!(in_strafe.state & 1))
733 {
734 viewangles[YAW] -= speed*cl_yawspeed->value*CL_KeyState (&in_right);
735 viewangles[YAW] += speed*cl_yawspeed->value*CL_KeyState (&in_left);
736 viewangles[YAW] = anglemod(viewangles[YAW]);
737 }
738 if (in_klook.state & 1)
739 {
740 V_StopPitchDrift ();
741 viewangles[PITCH] -= speed*cl_pitchspeed->value * CL_KeyState (&in_forward);
742 viewangles[PITCH] += speed*cl_pitchspeed->value * CL_KeyState (&in_back);
743 }
744
745 up = CL_KeyState (&in_lookup);
746 down = CL_KeyState(&in_lookdown);
747
748 viewangles[PITCH] -= speed*cl_pitchspeed->value * up;
749 viewangles[PITCH] += speed*cl_pitchspeed->value * down;
750
751 if (up || down)
752 V_StopPitchDrift ();
753
754 if (viewangles[PITCH] > cl_pitchdown->value)
755 viewangles[PITCH] = cl_pitchdown->value;
756 if (viewangles[PITCH] < -cl_pitchup->value)
757 viewangles[PITCH] = -cl_pitchup->value;
758
759 if (viewangles[ROLL] > 50)
760 viewangles[ROLL] = 50;
761 if (viewangles[ROLL] < -50)
762 viewangles[ROLL] = -50;
763}

775void CL_DLLEXPORT CL_CreateMove ( float frametime, struct usercmd_s *cmd, int active )
776{
777// RecClCL_CreateMove(frametime, cmd, active);
778
779 float spd;
780 vec3_t viewangles;
781 static vec3_t oldangles;
782
783 if ( active && !Bench_Active() )
784 {
785 //memset( viewangles, 0, sizeof( vec3_t ) );
786 //viewangles[ 0 ] = viewangles[ 1 ] = viewangles[ 2 ] = 0.0;
787 gEngfuncs.GetViewAngles( (float *)viewangles );
788
789 CL_AdjustAngles ( frametime, viewangles );
790
791 memset (cmd, 0, sizeof(*cmd));
792
793 gEngfuncs.SetViewAngles( (float *)viewangles );
794
795 if ( in_strafe.state & 1 )
796 {
797 cmd->sidemove += cl_sidespeed->value * CL_KeyState (&in_right);
798 cmd->sidemove -= cl_sidespeed->value * CL_KeyState (&in_left);
799 }
800
801 cmd->sidemove += cl_sidespeed->value * CL_KeyState (&in_moveright);
802 cmd->sidemove -= cl_sidespeed->value * CL_KeyState (&in_moveleft);
803
804 cmd->upmove += cl_upspeed->value * CL_KeyState (&in_up);
805 cmd->upmove -= cl_upspeed->value * CL_KeyState (&in_down);
806
807 if ( !(in_klook.state & 1 ) )
808 {
809 cmd->forwardmove += cl_forwardspeed->value * CL_KeyState (&in_forward);
810 cmd->forwardmove -= cl_backspeed->value * CL_KeyState (&in_back);
811 }
812
813 // adjust for speed key
814 if ( in_speed.state & 1 )
815 {
816 cmd->forwardmove *= cl_movespeedkey->value;
817 cmd->sidemove *= cl_movespeedkey->value;
818 cmd->upmove *= cl_movespeedkey->value;
819 }
820
821 // clip to maxspeed
822 spd = gEngfuncs.GetClientMaxspeed();
823 if ( spd != 0.0 )
824 {
825 // scale the 3 speeds so that the total velocity is not > cl.maxspeed
826 float fmov = sqrt( (cmd->forwardmove*cmd->forwardmove) + (cmd->sidemove*cmd->sidemove) + (cmd->upmove*cmd->upmove) );
827
828 if ( fmov > spd )
829 {
830 float fratio = spd / fmov;
831 cmd->forwardmove *= fratio;
832 cmd->sidemove *= fratio;
833 cmd->upmove *= fratio;
834 }
835 }
836
837 // Allow mice and other controllers to add their inputs
838 IN_Move ( frametime, cmd );
839 }
840
841 cmd->impulse = in_impulse;
842 in_impulse = 0;
843
844 cmd->weaponselect = g_weaponselect;
845 g_weaponselect = 0;
846 //
847 // set button and flag bits
848 //
849 cmd->buttons = CL_ButtonBits( 1 );
850
851 // If they're in a modal dialog, ignore the attack button.
852 if(GetClientVoiceMgr()->IsInSquelchMode())
853 cmd->buttons &= ~IN_ATTACK;
854
855 // Using joystick?
856 if ( in_joystick->value )
857 {
858 if ( cmd->forwardmove > 0 )
859 {
860 cmd->buttons |= IN_FORWARD;
861 }
862 else if ( cmd->forwardmove < 0 )
863 {
864 cmd->buttons |= IN_BACK;
865 }
866 }
867
868 gEngfuncs.GetViewAngles( (float *)viewangles );
869 // Set current view angles.
870
871 if ( g_iAlive )
872 {
873 VectorCopy( viewangles, cmd->viewangles );
874 VectorCopy( viewangles, oldangles );
875 }
876 else
877 {
878 VectorCopy( oldangles, cmd->viewangles );
879 }
880
881 Bench_SetViewAngles( 1, (float *)&cmd->viewangles, frametime, cmd );
882}
