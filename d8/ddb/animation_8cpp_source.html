237void SequencePrecache(void *pmodel, const char *pSequenceName)
238{
239 int index = LookupSequence(pmodel, pSequenceName);
240 if (index >= 0)
241 {
242 studiohdr_t *pstudiohdr = (studiohdr_t *)pmodel;
243 if (!pstudiohdr || index >= pstudiohdr->numseq)
244 return;
245
246 mstudioseqdesc_t *pseqdesc = (mstudioseqdesc_t *)((byte *)pstudiohdr + pstudiohdr->seqindex) + index;
247 mstudioevent_t *pevent = (mstudioevent_t *)((byte *)pstudiohdr + pseqdesc->eventindex);
248
249 for (int i = 0; i < pseqdesc->numevents; i++)
250 {
251 // Don't send client-side events to the server AI
252 if (pevent[i].event >= EVENT_CLIENT)
253 continue;
254
255 // Pre-cache the sound if it's a sound event
256 if (IsSoundEvent(pevent[i].event))
257 {
258 if (!strlen(pevent[i].options))
259 {
260 ALERT(at_error, "Bad sound event %d in sequence %s :: %s (sound is \"%s\")\n", pevent[i].event, pstudiohdr->name, pSequenceName, pevent[i].options);
261 }
262
263 PRECACHE_SOUND((char *)(gpGlobals->pStringBase + ALLOC_STRING(pevent[i].options))); // Pre-cache the sound
264 }
265 }
266 }
267}

280void GetSequenceInfo(void *pmodel, entvars_t *pev, float *pflFrameRate, float *pflGroundSpeed)
281{
282 studiohdr_t *pstudiohdr = (studiohdr_t *)pmodel;
283
284 if (!pstudiohdr)
285 return;
286
287 mstudioseqdesc_t *pseqdesc;
288
289 if (pev->sequence >= pstudiohdr->numseq)
290 {
291 *pflFrameRate = 0.0;
292 *pflGroundSpeed = 0.0;
293 return;
294 }
295
296 pseqdesc = (mstudioseqdesc_t *)((byte *)pstudiohdr + pstudiohdr->seqindex) + (int)pev->sequence;
297
298 if (pseqdesc->numframes > 1)
299 {
300 *pflFrameRate = 256 * pseqdesc->fps / (pseqdesc->numframes - 1);
301 *pflGroundSpeed = sqrt(pseqdesc->linearmovement[0] * pseqdesc->linearmovement[0] + pseqdesc->linearmovement[1] * pseqdesc->linearmovement[1] + pseqdesc->linearmovement[2] * pseqdesc->linearmovement[2]);
302 *pflGroundSpeed = *pflGroundSpeed * pseqdesc->fps / (pseqdesc->numframes - 1);
303 }
304 else
305 {
306 *pflFrameRate = 256.0;
307 *pflGroundSpeed = 0.0;
308 }

348int GetAnimationEvent(void *pmodel, entvars_t *pev, MonsterEvent_t *pMonsterEvent, float flStart, float flEnd, int index)
349{
350 studiohdr_t *pstudiohdr = (studiohdr_t *)pmodel;
351
352 if (!pstudiohdr || pev->sequence >= pstudiohdr->numseq || !pMonsterEvent)
353 return 0;
354
355 int events = 0;
356
357 mstudioseqdesc_t *pseqdesc;
358 mstudioevent_t *pevent;
359
360 pseqdesc = (mstudioseqdesc_t *)((byte *)pstudiohdr + pstudiohdr->seqindex) + (int)pev->sequence;
361 pevent = (mstudioevent_t *)((byte *)pstudiohdr + pseqdesc->eventindex);
362
363 if (pseqdesc->numevents == 0 || index > pseqdesc->numevents)
364 return 0;
365
366 if (pseqdesc->numframes > 1)
367 {
368 flStart *= (pseqdesc->numframes - 1) / 256.0;
369 flEnd *= (pseqdesc->numframes - 1) / 256.0;
370 }
371 else
372 {
373 flStart = 0;
374 flEnd = 1.0;
375 }
376
377 for (; index < pseqdesc->numevents; index++)
378 {
379 // Don't send client-side events to the server AI
380 if (pevent[index].event >= EVENT_CLIENT)
381 continue;
382
383 if ((pevent[index].frame >= flStart && pevent[index].frame < flEnd) ||
384 ((pseqdesc->flags & STUDIO_LOOPING) && flEnd >= pseqdesc->numframes - 1 && pevent[index].frame < flEnd - pseqdesc->numframes + 1))
385 {
386 pMonsterEvent->event = pevent[index].event;
387 pMonsterEvent->options = pevent[index].options;
388 return index + 1; // Return the next event index
389 }
390 }
391 return 0; // No event found
392}

406float SetController(void *pmodel, entvars_t *pev, int iController, float flValue)
407{
408 studiohdr_t *pstudiohdr = (studiohdr_t *)pmodel;
409 if (!pstudiohdr)
410 return flValue;
411
412 mstudiobonecontroller_t *pbonecontroller = (mstudiobonecontroller_t *)((byte *)pstudiohdr + pstudiohdr->bonecontrollerindex);
413
414 // Find the first controller that matches the index
415 for (int i = 0; i < pstudiohdr->numbonecontrollers; i++, pbonecontroller++)
416 {
417 if (pbonecontroller->index == iController)
418 break;
419 }
420 if (i >= pstudiohdr->numbonecontrollers)
421 return flValue;
422
423 // Wrap 0..360 if it's a rotational controller
424 if (pbonecontroller->type & (STUDIO_XR | STUDIO_YR | STUDIO_ZR))
425 {
426 // Ugly hack, invert value if end < start
427 if (pbonecontroller->end < pbonecontroller->start)
428 flValue = -flValue;
429
430 // Does the controller not wrap?
431 if (pbonecontroller->start + 359.0 >= pbonecontroller->end)
432 {
433 if (flValue > ((pbonecontroller->start + pbonecontroller->end) / 2.0) + 180)
434 flValue = flValue - 360;
435 if (flValue < ((pbonecontroller->start + pbonecontroller->end) / 2.0) - 180)
436 flValue = flValue + 360;
437 }
438 else
439 {
440 if (flValue > 360)
441 flValue = flValue - (int)(flValue / 360.0) * 360.0;
442 else if (flValue < 0)
443 flValue = flValue + (int)((flValue / -360.0) + 1) * 360.0;
444 }
445 }
446
447 int setting = 255 * (flValue - pbonecontroller->start) / (pbonecontroller->end - pbonecontroller->start);
448
449 if (setting < 0) setting = 0;
450 if (setting > 255) setting = 255;
451 pev->controller[iController] = setting;
452
453 return setting * (1.0 / 255.0) * (pbonecontroller->end - pbonecontroller->start) + pbonecontroller->start; // Return the updated value
454}

517int FindTransition(void *pmodel, int iEndingAnim, int iGoalAnim, int *piDir)
518{
519 studiohdr_t *pstudiohdr = (studiohdr_t *)pmodel;
520 if (!pstudiohdr)
521 return iGoalAnim;
522
523 mstudioseqdesc_t *pseqdesc = (mstudioseqdesc_t *)((byte *)pstudiohdr + pstudiohdr->seqindex);
524
525 // Bail if we're going to or from a node 0
526 if (pseqdesc[iEndingAnim].entrynode == 0 || pseqdesc[iGoalAnim].entrynode == 0)
527 {
528 return iGoalAnim;
529 }
530
531 int iEndNode;
532
533 if (*piDir > 0)
534 {
535 iEndNode = pseqdesc[iEndingAnim].exitnode;
536 }
537 else
538 {
539 iEndNode = pseqdesc[iEndingAnim].entrynode;
540 }
541
542 if (iEndNode == pseqdesc[iGoalAnim].entrynode)
543 {
544 *piDir = 1;
545 return iGoalAnim;
546 }
547
548 byte *pTransition = ((byte *)pstudiohdr + pstudiohdr->transitionindex);
549
550 int iInternNode = pTransition[(iEndNode - 1) * pstudiohdr->numtransitions + (pseqdesc[iGoalAnim].entrynode - 1)];
551
552 if (iInternNode == 0)
553 return iGoalAnim;
554
555 for (int i = 0; i < pstudiohdr->numseq; i++)
556 {
557 if (pseqdesc[i].entrynode == iEndNode && pseqdesc[i].exitnode == iInternNode)
558 {
559 *piDir = 1;
560 return i;
561 }
562 if (pseqdesc[i].nodeflags)
563 {
564 if (pseqdesc[i].exitnode == iEndNode && pseqdesc[i].entrynode == iInternNode)
565 {
566 *piDir = -1;
567 return i;
568 }
569 }
570 }
571
572 ALERT(at_console, "error in transition graph");
573 return iGoalAnim; // Return the goal animation if no transition found
574}
