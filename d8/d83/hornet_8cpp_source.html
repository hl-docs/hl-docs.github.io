78void CHornet::Spawn(void)
79{
80 Precache(); // Precache the hornet model and sounds
81
82 pev->movetype = MOVETYPE_FLY; // Set movement type to fly
83 pev->solid = SOLID_BBOX; // Set solid type to bounding box
84 pev->takedamage = DAMAGE_YES; // Allow the hornet to take damage
85 pev->flags |= FL_MONSTER; // Set the hornet as a monster
86 pev->health = 1; // Set health (very weak)
87
88 if (g_pGameRules->IsMultiplayer())
89 {
90 // Hornets don't live as long in multiplayer
91 m_flStopAttack = gpGlobals->time + 3.5; // Set attack duration for multiplayer
92 }
93 else
94 {
95 m_flStopAttack = gpGlobals->time + 5.0; // Set attack duration for single player
96 }
97
98 m_flFieldOfView = 0.9; // Set field of view (Â±25 degrees)
99
100 // Randomly assign hornet type and speed
101 if (RANDOM_LONG(1, 5) <= 2)
102 {
103 m_iHornetType = HORNET_TYPE_RED; // Set hornet type to red
104 m_flFlySpeed = HORNET_RED_SPEED; // Set flying speed for red hornet
105 }
106 else
107 {
108 m_iHornetType = HORNET_TYPE_ORANGE; // Set hornet type to orange
109 m_flFlySpeed = HORNET_ORANGE_SPEED; // Set flying speed for orange hornet
110 }
111
112 SET_MODEL(ENT(pev), "models/hornet.mdl"); // Set the model for the hornet
113 UTIL_SetSize(pev, Vector(-4, -4, -4), Vector(4, 4, 4)); // Set the size of the hornet
114
115 SetTouch(&CHornet::DieTouch); // Set touch function
116 SetThink(&CHornet::StartTrack); // Set think function
117
118 edict_t *pSoundEnt = pev->owner; // Get the owner entity
119 if (!pSoundEnt)
120 pSoundEnt = edict(); // Default to a new entity if no owner
121
122 if (!FNullEnt(pev->owner) && (pev->owner->v.flags & FL_CLIENT))
123 {
124 pev->dmg = gSkillData.plrDmgHornet; // Set damage for player-owned hornets
125 }
126 else
127 {
128 // No real owner, or owner isn't a client.
129 pev->dmg = gSkillData.monDmgHornet; // Set damage for monster-owned hornets
130 }
131
132 pev->nextthink = gpGlobals->time + 0.1; // Set the next think time
133 ResetSequenceInfo(); // Reset sequence information
134}

273void CHornet::TrackTarget(void)
274{
275 Vector vecFlightDir; // Direction the hornet is flying
276 Vector vecDirToEnemy; // Direction to the enemy
277 float flDelta; // Dot product for turning
278
279 StudioFrameAdvance(); // Advance the animation frame
280
281 if (gpGlobals->time > m_flStopAttack)
282 {
283 SetTouch(NULL); // Clear touch function
284 SetThink(&CHornet::SUB_Remove); // Set think function to remove the hornet
285 pev->nextthink = gpGlobals->time + 0.1; // Set the next think time
286 return;
287 }
288
289 // Check if the enemy is dead
290 if (m_hEnemy == NULL)
291 {
292 Look(512); // Look for enemies
293 m_hEnemy = BestVisibleEnemy(); // Find the best visible enemy
294 }
295
296 if (m_hEnemy != NULL && FVisible(m_hEnemy))
297 {
298 m_vecEnemyLKP = m_hEnemy->BodyTarget(pev->origin); // Update the last known position of the enemy
299 }
300 else
301 {
302 m_vecEnemyLKP = m_vecEnemyLKP + pev->velocity * m_flFlySpeed * 0.1; // Update position if enemy is not visible
303 }
304
305 vecDirToEnemy = (m_vecEnemyLKP - pev->origin).Normalize(); // Direction to the enemy
306
307 if (pev->velocity.Length() < 0.1)
308 vecFlightDir = vecDirToEnemy; // If not moving, fly towards the enemy
309 else
310 vecFlightDir = pev->velocity.Normalize(); // Otherwise, continue in the current direction
311
312 // Measure how far the turn is
313 flDelta = DotProduct(vecFlightDir, vecDirToEnemy);
314
315 if (flDelta < 0.5)
316 {
317 // Play sound if turning wide
318 switch (RANDOM_LONG(0, 2))
319 {
320 case 0: EMIT_SOUND(ENT(pev), CHAN_VOICE, "hornet/ag_buzz1.wav", HORNET_BUZZ_VOLUME, ATTN_NORM); break;
321 case 1: EMIT_SOUND(ENT(pev), CHAN_VOICE, "hornet/ag_buzz2.wav", HORNET_BUZZ_VOLUME, ATTN_NORM); break;
322 case 2: EMIT_SOUND(ENT(pev), CHAN_VOICE, "hornet/ag_buzz3.wav", HORNET_BUZZ_VOLUME, ATTN_NORM); break;
323 }
324 }
325
326 if (flDelta <= 0 && m_iHornetType == HORNET_TYPE_RED)
327 {
328 // Prevent flying backwards
329 flDelta = 0.25;
330 }
331
332 pev->velocity = (vecFlightDir + vecDirToEnemy).Normalize(); // Update velocity
333
334 if (pev->owner && (pev->owner->v.flags & FL_MONSTER))
335 {
336 // Random pattern only applies to hornets fired by monsters
337 pev->velocity.x += RANDOM_FLOAT(-0.10, 0.10); // Scramble the flight direction
338 pev->velocity.y += RANDOM_FLOAT(-0.10, 0.10);
339 pev->velocity.z += RANDOM_FLOAT(-0.10, 0.10);
340 }
341
342 switch (m_iHornetType)
343 {
344 case HORNET_TYPE_RED:
345 pev->velocity = pev->velocity * (m_flFlySpeed * flDelta); // Scale the direction by speed
346 pev->nextthink = gpGlobals->time + RANDOM_FLOAT(0.1, 0.3); // Set next think time
347 break;
348 case HORNET_TYPE_ORANGE:
349 pev->velocity = pev->velocity * m_flFlySpeed; // Maintain speed
350 pev->nextthink = gpGlobals->time + 0.1; // Fixed think time
351 break;
352 }
353
354 pev->angles = UTIL_VecToAngles(pev->velocity); // Update angles based on velocity
355
356 pev->solid = SOLID_BBOX; // Set solid type
357
358 // If hornet is close to the enemy, jet in a straight line for a half second
359 if (m_hEnemy != NULL && !g_pGameRules->IsMultiplayer())
360 {
361 if (flDelta >= 0.4 && (pev->origin - m_vecEnemyLKP).Length() <= 300)
362 {
363 MESSAGE_BEGIN(MSG_PVS, SVC_TEMPENTITY, pev->origin); // Create a puff effect
364 WRITE_BYTE(TE_SPRITE); // Specify sprite effect
365 WRITE_COORD(pev->origin.x); // Position
366 WRITE_COORD(pev->origin.y);
367 WRITE_COORD(pev->origin.z);
368 WRITE_SHORT(iHornetPuff); // Model
369 WRITE_BYTE(2); // Size
370 WRITE_BYTE(128); // Brightness
371 MESSAGE_END();
372
373 switch (RANDOM_LONG(0, 2))
374 {
375 case 0: EMIT_SOUND(ENT(pev), CHAN_VOICE, "hornet/ag_buzz1.wav", HORNET_BUZZ_VOLUME, ATTN_NORM); break;
376 case 1: EMIT_SOUND(ENT(pev), CHAN_VOICE, "hornet/ag_buzz2.wav", HORNET_BUZZ_VOLUME, ATTN_NORM); break;
377 case 2: EMIT_SOUND(ENT(pev), CHAN_VOICE, "hornet/ag_buzz3.wav", HORNET_BUZZ_VOLUME, ATTN_NORM); break;
378 }
379 pev->velocity = pev->velocity * 2; // Double the velocity
380 pev->nextthink = gpGlobals->time + 1.0; // Set next think time
381 m_flStopAttack = gpGlobals->time; // Don't attack again
382 }
383 }

394void CHornet::TrackTouch(CBaseEntity *pOther)
395{
396 if (pOther->edict() == pev->owner || pOther->pev->modelindex == pev->modelindex)
397 {
398 // Bumped into the entity that shot it
399 pev->solid = SOLID_NOT; // Set solid type to not solid
400 return; // Exit the function
401 }
402
403 if (IRelationship(pOther) <= R_NO)
404 {
405 // Hit something we don't want to hurt, so turn around
406 pev->velocity = pev->velocity.Normalize(); // Normalize velocity
407
408 pev->velocity.x *= -1; // Reverse x velocity
409 pev->velocity.y *= -1; // Reverse y velocity
410
411 pev->origin = pev->origin + pev->velocity * 4; // Bounce the hornet off a bit
412 pev->velocity = pev->velocity * m_flFlySpeed; // Scale velocity
413
414 return; // Exit the function
415 }
416
417 DieTouch(pOther); // Call the DieTouch function to handle damage
418}

441void CHornet::DieTouch(CBaseEntity *pOther)
442{
443 if (pOther && pOther->pev->takedamage)
444 {
445 // Do the damage
446 switch (RANDOM_LONG(0, 2))
447 {
448 // Buzz when you hit someone
449 case 0: EMIT_SOUND(ENT(pev), CHAN_VOICE, "hornet/ag_hornethit1.wav", 1, ATTN_NORM); break;
450 case 1: EMIT_SOUND(ENT(pev), CHAN_VOICE, "hornet/ag_hornethit2.wav", 1, ATTN_NORM); break;
451 case 2: EMIT_SOUND(ENT(pev), CHAN_VOICE, "hornet/ag_hornethit3.wav", 1, ATTN_NORM); break;
452 }
453
454 pOther->TakeDamage(pev, VARS(pev->owner), pev->dmg, DMG_BULLET); // Inflict damage to the target
455 }
456
457 pev->modelindex = 0; // Set model index to 0 to make it disappear
458 pev->solid = SOLID_NOT; // Set solid type to not solid
459
460 SetThink(&CHornet::SUB_Remove); // Set think function to remove the hornet
461 pev->nextthink = gpGlobals->time + 1; // Stick around long enough for the sound to finish
462}
