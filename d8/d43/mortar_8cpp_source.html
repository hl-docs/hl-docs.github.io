159void CFuncMortarField::FieldUse(CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
160{
161 Vector vecStart; // Starting position for bomb drops
162
163 // Randomly determine the starting position within the defined bounds
164 vecStart.x = RANDOM_FLOAT(pev->mins.x, pev->maxs.x);
165 vecStart.y = RANDOM_FLOAT(pev->mins.y, pev->maxs.y);
166 vecStart.z = pev->maxs.z;
167
168 switch (m_fControl) // Determine the control type
169 {
170 case 0: // Random
171 break;
172 case 1: // Trigger Activator
173 if (pActivator != NULL)
174 {
175 vecStart.x = pActivator->pev->origin.x; // Use activator's X position
176 vecStart.y = pActivator->pev->origin.y; // Use activator's Y position
177 }
178 break;
179 case 2: // Table
180 {
181 CBaseEntity *pController;
182
183 if (!FStringNull(m_iszXController)) // Check if X controller is set
184 {
185 pController = UTIL_FindEntityByTargetname(NULL, STRING(m_iszXController)); // Find the X controller
186 if (pController != NULL)
187 {
188 vecStart.x = pev->mins.x + pController->pev->ideal_yaw * (pev->size.x); // Set X position based on controller
189 }
190 }
191 if (!FStringNull(m_iszYController)) // Check if Y controller is set
192 {
193 pController = UTIL_FindEntityByTargetname(NULL, STRING(m_iszYController)); // Find the Y controller
194 if (pController != NULL)
195 {
196 vecStart.y = pev->mins.y + pController->pev->ideal_yaw * (pev->size.y); // Set Y position based on controller
197 }
198 }
199 }
200 break;
201 }
202
203 int pitch = RANDOM_LONG(95, 124); // Random pitch for sound
204
205 EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "weapons/mortar.wav", 1.0, ATTN_NONE, 0, pitch); // Play mortar sound
206
207 float t = 2.5; // Initial delay for bomb drops
208 for (int i = 0; i < m_iCount; i++) // Loop to drop the specified number of bombs
209 {
210 Vector vecSpot = vecStart; // Set the drop position
211 vecSpot.x += RANDOM_FLOAT(-m_flSpread, m_flSpread); // Add random spread to X
212 vecSpot.y += RANDOM_FLOAT(-m_flSpread, m_flSpread); // Add random spread to Y
213
214 TraceResult tr; // Trace result for the drop
215 UTIL_TraceLine(vecSpot, vecSpot + Vector(0, 0, -1) * 4096, ignore_monsters, ENT(pev), &tr); // Trace downwards to find the ground
216
217 edict_t *pentOwner = NULL; // Owner of the mortar
218 if (pActivator) pentOwner = pActivator->edict(); // Set owner if activator exists
219
220 CBaseEntity *pMortar = Create("monster_mortar", tr.vecEndPos, Vector(0, 0, 0), pentOwner); // Create the mortar entity
221 pMortar->pev->nextthink = gpGlobals->time + t; // Set the next think time for the mortar
222 t += RANDOM_FLOAT(0.2, 0.5); // Increment the delay for the next bomb drop
223
224 if (i == 0) // If this is the first bomb
225 CSoundEnt::InsertSound(bits_SOUND_DANGER, tr.vecEndPos, 400, 0.3); // Insert a danger sound
226 }
227}
