113void CStudioModelRenderer::StudioCalcBoneAdj( float dadt, float *adj, const byte *pcontroller1, const byte *pcontroller2, byte mouthopen )
114{
115 int i, j;
116 float value;
117 mstudiobonecontroller_t *pbonecontroller;
118
119 pbonecontroller = (mstudiobonecontroller_t *)((byte *)m_pStudioHeader + m_pStudioHeader->bonecontrollerindex);
120
121 for (j = 0; j < m_pStudioHeader->numbonecontrollers; j++)
122 {
123 i = pbonecontroller[j].index;
124 if (i <= 3)
125 {
126 // check for 360% wrapping
127 if (pbonecontroller[j].type & STUDIO_RLOOP)
128 {
129 if (abs(pcontroller1[i] - pcontroller2[i]) > 128)
130 {
131 int a, b;
132 a = (pcontroller1[j] + 128) % 256;
133 b = (pcontroller2[j] + 128) % 256;
134 value = ((a * dadt) + (b * (1 - dadt)) - 128) * (360.0/256.0) + pbonecontroller[j].start;
135 }
136 else
137 {
138 value = ((pcontroller1[i] * dadt + (pcontroller2[i]) * (1.0 - dadt))) * (360.0/256.0) + pbonecontroller[j].start;
139 }
140 }
141 else
142 {
143 value = (pcontroller1[i] * dadt + pcontroller2[i] * (1.0 - dadt)) / 255.0;
144 if (value < 0) value = 0;
145 if (value > 1.0) value = 1.0;
146 value = (1.0 - value) * pbonecontroller[j].start + value * pbonecontroller[j].end;
147 }
148 // Con_DPrintf( "%d %d %f : %f\n", m_pCurrentEntity->curstate.controller[j], m_pCurrentEntity->latched.prevcontroller[j], value, dadt );
149 }
150 else
151 {
152 value = mouthopen / 64.0;
153 if (value > 1.0) value = 1.0;
154 value = (1.0 - value) * pbonecontroller[j].start + value * pbonecontroller[j].end;
155 // Con_DPrintf("%d %f\n", mouthopen, value );
156 }
157 switch(pbonecontroller[j].type & STUDIO_TYPES)
158 {
159 case STUDIO_XR:
160 case STUDIO_YR:
161 case STUDIO_ZR:
162 adj[j] = value * (M_PI / 180.0);
163 break;
164 case STUDIO_X:
165 case STUDIO_Y:
166 case STUDIO_Z:
167 adj[j] = value;
168 break;
169 }
170 }

454void CStudioModelRenderer::StudioSetUpTransform (int trivial_accept)
455{
456 int i;
457 vec3_t angles;
458 vec3_t modelpos;
459
460// tweek model origin
461 //for (i = 0; i < 3; i++)
462 // modelpos[i] = m_pCurrentEntity->origin[i];
463
464 VectorCopy( m_pCurrentEntity->origin, modelpos );
465
466// TODO: should really be stored with the entity instead of being reconstructed
467// TODO: should use a look-up table
468// TODO: could cache lazily, stored in the entity
469 angles[ROLL] = m_pCurrentEntity->curstate.angles[ROLL];
470 angles[PITCH] = m_pCurrentEntity->curstate.angles[PITCH];
471 angles[YAW] = m_pCurrentEntity->curstate.angles[YAW];
472
473 //Con_DPrintf("Angles %4.2f prev %4.2f for %i\n", angles[PITCH], m_pCurrentEntity->index);
474 //Con_DPrintf("movetype %d %d\n", m_pCurrentEntity->movetype, m_pCurrentEntity->aiment );
475 if (m_pCurrentEntity->curstate.movetype == MOVETYPE_STEP)
476 {
477 float f = 0;
478 float d;
479
480 // don't do it if the goalstarttime hasn't updated in a while.
481
482 // NOTE: Because we need to interpolate multiplayer characters, the interpolation time limit
483 // was increased to 1.0 s., which is 2x the max lag we are accounting for.
484
485 if ( ( m_clTime < m_pCurrentEntity->curstate.animtime + 1.0f ) &&
486 ( m_pCurrentEntity->curstate.animtime != m_pCurrentEntity->latched.prevanimtime ) )
487 {
488 f = (m_clTime - m_pCurrentEntity->curstate.animtime) / (m_pCurrentEntity->curstate.animtime - m_pCurrentEntity->latched.prevanimtime);
489 //Con_DPrintf("%4.2f %.2f %.2f\n", f, m_pCurrentEntity->curstate.animtime, m_clTime);
490 }
491
492 if (m_fDoInterp)
493 {
494 // ugly hack to interpolate angle, position. current is reached 0.1 seconds after being set
495 f = f - 1.0;
496 }
497 else
498 {
499 f = 0;
500 }
501
502 for (i = 0; i < 3; i++)
503 {
504 modelpos[i] += (m_pCurrentEntity->origin[i] - m_pCurrentEntity->latched.prevorigin[i]) * f;
505 }
506
507 // NOTE: Because multiplayer lag can be relatively large, we don't want to cap
508 // f at 1.5 anymore.
509 //if (f > -1.0 && f < 1.5) {}
510
511// Con_DPrintf("%.0f %.0f\n",m_pCurrentEntity->msg_angles[0][YAW], m_pCurrentEntity->msg_angles[1][YAW] );
512 for (i = 0; i < 3; i++)
513 {
514 float ang1, ang2;
515
516 ang1 = m_pCurrentEntity->angles[i];
517 ang2 = m_pCurrentEntity->latched.prevangles[i];
518
519 d = ang1 - ang2;
520 if (d > 180)
521 {
522 d -= 360;
523 }
524 else if (d < -180)
525 {
526 d += 360;
527 }
528
529 angles[i] += d * f;
530 }
531 //Con_DPrintf("%.3f \n", f );
532 }
533 else if ( m_pCurrentEntity->curstate.movetype != MOVETYPE_NONE )
534 {
535 VectorCopy( m_pCurrentEntity->angles, angles );
536 }
537
538 //Con_DPrintf("%.0f %0.f %0.f\n", modelpos[0], modelpos[1], modelpos[2] );
539 //Con_DPrintf("%.0f %0.f %0.f\n", angles[0], angles[1], angles[2] );
540
541 angles[PITCH] = -angles[PITCH];
542 AngleMatrix (angles, (*m_protationmatrix));
543
544 if ( !IEngineStudio.IsHardware() )
545 {
546 static float viewmatrix[3][4];
547
548 VectorCopy (m_vRight, viewmatrix[0]);
549 VectorCopy (m_vUp, viewmatrix[1]);
550 VectorInverse (viewmatrix[1]);
551 VectorCopy (m_vNormal, viewmatrix[2]);
552
553 (*m_protationmatrix)[0][3] = modelpos[0] - m_vRenderOrigin[0];
554 (*m_protationmatrix)[1][3] = modelpos[1] - m_vRenderOrigin[1];
555 (*m_protationmatrix)[2][3] = modelpos[2] - m_vRenderOrigin[2];
556
557 ConcatTransforms (viewmatrix, (*m_protationmatrix), (*m_paliastransform));
558
559 // do the scaling up of x and y to screen coordinates as part of the transform
560 // for the unclipped case (it would mess up clipping in the clipped case).
561 // Also scale down z, so 1/z is scaled 31 bits for free, and scale down x and y
562 // correspondingly so the projected x and y come out right
563 // FIXME: make this work for clipped case too?
564 if (trivial_accept)
565 {
566 for (i=0 ; i<4 ; i++)
567 {
568 (*m_paliastransform)[0][i] *= m_fSoftwareXScale *
569 (1.0 / (ZISCALE * 0x10000));
570 (*m_paliastransform)[1][i] *= m_fSoftwareYScale *
571 (1.0 / (ZISCALE * 0x10000));
572 (*m_paliastransform)[2][i] *= 1.0 / (ZISCALE * 0x10000);
573
574 }
575 }
576 }
577
578 (*m_protationmatrix)[0][3] = modelpos[0];
579 (*m_protationmatrix)[1][3] = modelpos[1];
580 (*m_protationmatrix)[2][3] = modelpos[2];
581}

1386void CStudioModelRenderer::StudioEstimateGait( entity_state_t *pplayer )
1387{
1388 float dt;
1389 vec3_t est_velocity;
1390
1391 dt = (m_clTime - m_clOldTime);
1392 if (dt < 0)
1393 dt = 0;
1394 else if (dt > 1.0)
1395 dt = 1;
1396
1397 if (dt == 0 || m_pPlayerInfo->renderframe == m_nFrameCount)
1398 {
1399 m_flGaitMovement = 0;
1400 return;
1401 }
1402
1403 // VectorAdd( pplayer->velocity, pplayer->prediction_error, est_velocity );
1404 if ( m_fGaitEstimation )
1405 {
1406 VectorSubtract( m_pCurrentEntity->origin, m_pPlayerInfo->prevgaitorigin, est_velocity );
1407 VectorCopy( m_pCurrentEntity->origin, m_pPlayerInfo->prevgaitorigin );
1408 m_flGaitMovement = Length( est_velocity );
1409 if (dt <= 0 || m_flGaitMovement / dt < 5)
1410 {
1411 m_flGaitMovement = 0;
1412 est_velocity[0] = 0;
1413 est_velocity[1] = 0;
1414 }
1415 }
1416 else
1417 {
1418 VectorCopy( pplayer->velocity, est_velocity );
1419 m_flGaitMovement = Length( est_velocity ) * dt;
1420 }
1421
1422 if (est_velocity[1] == 0 && est_velocity[0] == 0)
1423 {
1424 float flYawDiff = m_pCurrentEntity->angles[YAW] - m_pPlayerInfo->gaityaw;
1425 flYawDiff = flYawDiff - (int)(flYawDiff / 360) * 360;
1426 if (flYawDiff > 180)
1427 flYawDiff -= 360;
1428 if (flYawDiff < -180)
1429 flYawDiff += 360;
1430
1431 if (dt < 0.25)
1432 flYawDiff *= dt * 4;
1433 else
1434 flYawDiff *= dt;
1435
1436 m_pPlayerInfo->gaityaw += flYawDiff;
1437 m_pPlayerInfo->gaityaw = m_pPlayerInfo->gaityaw - (int)(m_pPlayerInfo->gaityaw / 360) * 360;
1438
1439 m_flGaitMovement = 0;
1440 }
1441 else
1442 {
1443 m_pPlayerInfo->gaityaw = (atan2(est_velocity[1], est_velocity[0]) * 180 / M_PI);
1444 if (m_pPlayerInfo->gaityaw > 180)

1462void CStudioModelRenderer::StudioProcessGait( entity_state_t *pplayer )
1463{
1464 mstudioseqdesc_t *pseqdesc;
1465 float dt;
1466 int iBlend;
1467 float flYaw; // view direction relative to movement
1468
1469 if (m_pCurrentEntity->curstate.sequence >= m_pStudioHeader->numseq)
1470 {
1471 m_pCurrentEntity->curstate.sequence = 0;
1472 }
1473
1474 pseqdesc = (mstudioseqdesc_t *)((byte *)m_pStudioHeader + m_pStudioHeader->seqindex) + m_pCurrentEntity->curstate.sequence;
1475
1476 StudioPlayerBlend( pseqdesc, &iBlend, &m_pCurrentEntity->angles[PITCH] );
1477
1478 m_pCurrentEntity->latched.prevangles[PITCH] = m_pCurrentEntity->angles[PITCH];
1479 m_pCurrentEntity->curstate.blending[0] = iBlend;
1480 m_pCurrentEntity->latched.prevblending[0] = m_pCurrentEntity->curstate.blending[0];
1481 m_pCurrentEntity->latched.prevseqblending[0] = m_pCurrentEntity->curstate.blending[0];
1482
1483 // Con_DPrintf("%f %d\n", m_pCurrentEntity->angles[PITCH], m_pCurrentEntity->blending[0] );
1484
1485 dt = (m_clTime - m_clOldTime);
1486 if (dt < 0)
1487 dt = 0;
1488 else if (dt > 1.0)
1489 dt = 1;
1490
1491 StudioEstimateGait( pplayer );
1492
1493 // Con_DPrintf("%f %f\n", m_pCurrentEntity->angles[YAW], m_pPlayerInfo->gaityaw );
1494
1495 // calc side to side turning
1496 flYaw = m_pCurrentEntity->angles[YAW] - m_pPlayerInfo->gaityaw;
1497 flYaw = flYaw - (int)(flYaw / 360) * 360;
1498 if (flYaw < -180)
1499 flYaw = flYaw + 360;
1500 if (flYaw > 180)
1501 flYaw = flYaw - 360;
1502
1503 if (flYaw > 120)
1504 {
1505 m_pPlayerInfo->gaityaw = m_pPlayerInfo->gaityaw - 180;
1506 m_flGaitMovement = -m_flGaitMovement;
1507 flYaw = flYaw - 180;
1508 }
1509 else if (flYaw < -120)
1510 {
1511 m_pPlayerInfo->gaityaw = m_pPlayerInfo->gaityaw + 180;
1512 m_flGaitMovement = -m_flGaitMovement;
1513 flYaw = flYaw + 180;
1514 }
1515
1516 // adjust torso
1517 m_pCurrentEntity->curstate.controller[0] = ((flYaw / 4.0) + 30) / (60.0 / 255.0);
1518 m_pCurrentEntity->curstate.controller[1] = ((flYaw / 4.0) + 30) / (60.0 / 255.0);
1519 m_pCurrentEntity->curstate.controller[2] = ((flYaw / 4.0) + 30) / (60.0 / 255.0);
1520 m_pCurrentEntity->curstate.controller[3] = ((flYaw / 4.0) + 30) / (60.0 / 255.0);
1521 m_pCurrentEntity->latched.prevcontroller[0] = m_pCurrentEntity->curstate.controller[0];
1522 m_pCurrentEntity->latched.prevcontroller[1] = m_pCurrentEntity->curstate.controller[1];
1523 m_pCurrentEntity->latched.prevcontroller[2] = m_pCurrentEntity->curstate.controller[2];
1524 m_pCurrentEntity->latched.prevcontroller[3] = m_pCurrentEntity->curstate.controller[3];
1525
1526 m_pCurrentEntity->angles[YAW] = m_pPlayerInfo->gaityaw;
1527 if (m_pCurrentEntity->angles[YAW] < -0)
1528 m_pCurrentEntity->angles[YAW] += 360;
1529 m_pCurrentEntity->latched.prevangles[YAW] = m_pCurrentEntity->angles[YAW];
1530
1531 if (pplayer->gaitsequence >= m_pStudioHeader->numseq)
1532 {
1533 pplayer->gaitsequence = 0;
1534 }
1535
1536 pseqdesc = (mstudioseqdesc_t *)((byte *)m_pStudioHeader + m_pStudioHeader->seqindex) + pplayer->gaitsequence;
1537
1538 // calc gait frame
1539 if (pseqdesc->linearmovement[0] > 0)
1540 {
1541 m_pPlayerInfo->gaitframe += (m_flGaitMovement / pseqdesc->linearmovement[0]) * pseqdesc->numframes;
1542 }
1543 else
1544 {
1545 m_pPlayerInfo->gaitframe += pseqdesc->fps * dt;
1546 }
1547
1548 // do modulo
1549 m_pPlayerInfo->gaitframe = m_pPlayerInfo->gaitframe - (int)(m_pPlayerInfo->gaitframe / pseqdesc->numframes) * pseqdesc->numframes;
1550 if (m_pPlayerInfo->gaitframe < 0)
1551 m_pPlayerInfo->gaitframe += pseqdesc->numframes;
1552}
