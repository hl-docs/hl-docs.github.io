125void CDecal :: Spawn( void )
126{
127 if ( pev->skin < 0 || (gpGlobals->deathmatch && FBitSet( pev->spawnflags, SF_DECAL_NOTINDEATHMATCH )) )
128 {
129 REMOVE_ENTITY(ENT(pev));
130 return;
131 }
132
133 if ( FStringNull ( pev->targetname ) )
134 {
135 SetThink( &CDecal::StaticDecal );
136 // if there's no targetname, the decal will spray itself on as soon as the world is done spawning.
137 pev->nextthink = gpGlobals->time;
138 }
139 else
140 {
141 // if there IS a targetname, the decal sprays itself on when it is triggered.
142 SetThink ( &CDecal::SUB_DoNothing );
143 SetUse( &CDecal::TriggerDecal);
144 }
145}

159void CDecal :: TriggerDecal ( CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value )
160{
161 // this is set up as a USE function for infodecals that have targetnames, so that the
162 // decal doesn't get applied until it is fired. (usually by a scripted sequence)
163 TraceResult trace;
164 int entityIndex;
165
166 UTIL_TraceLine( pev->origin - Vector(5,5,5), pev->origin + Vector(5,5,5), ignore_monsters, ENT(pev), &trace );
167
168 MESSAGE_BEGIN( MSG_BROADCAST, SVC_TEMPENTITY);
169 WRITE_BYTE( TE_BSPDECAL );
170 WRITE_COORD( pev->origin.x );
171 WRITE_COORD( pev->origin.y );
172 WRITE_COORD( pev->origin.z );
173 WRITE_SHORT( (int)pev->skin );
174 entityIndex = (short)ENTINDEX(trace.pHit);
175 WRITE_SHORT( entityIndex );
176 if ( entityIndex )
177 WRITE_SHORT( (int)VARS(trace.pHit)->modelindex );
178 MESSAGE_END();
179
180 SetThink( &CDecal::SUB_Remove );
181 pev->nextthink = gpGlobals->time + 0.1;
182}
183
192void CDecal :: StaticDecal( void )
193{
194 TraceResult trace;
195 int entityIndex, modelIndex;
196
197 UTIL_TraceLine( pev->origin - Vector(5,5,5), pev->origin + Vector(5,5,5), ignore_monsters, ENT(pev), &trace );
198
199 entityIndex = (short)ENTINDEX(trace.pHit);
200 if ( entityIndex )
201 modelIndex = (int)VARS(trace.pHit)->modelindex;
202 else
203 modelIndex = 0;
204
205 g_engfuncs.pfnStaticDecal( pev->origin, (int)pev->skin, entityIndex, modelIndex );
206
207 SUB_Remove();
208}

248static void InitBodyQue(void)
249{
250 string_t istrClassname = MAKE_STRING("bodyque"); // Create a string for the body queue class name
251
252 g_pBodyQueueHead = CREATE_NAMED_ENTITY(istrClassname); // Create the head entity for the body queue
253 entvars_t *pev = VARS(g_pBodyQueueHead); // Get the entity variables for the head
254
255 // Reserve 3 more slots for dead bodies
256 for (int i = 0; i < 3; i++)
257 {
258 pev->owner = CREATE_NAMED_ENTITY(istrClassname); // Create additional body entities
259 pev = VARS(pev->owner); // Move to the next entity
260 }
261
262 pev->owner = g_pBodyQueueHead; // Link the last entity back to the head
263}

274void CopyToBodyQue(entvars_t *pev)
275{
276 if (pev->effects & EF_NODRAW) // Check if the entity should not be drawn
277 return; // Exit if the entity is not visible
278
279 entvars_t *pevHead = VARS(g_pBodyQueueHead); // Get the head of the body queue
280
281 // Copy properties from the original entity to the body queue head
282 pevHead->angles = pev->angles;
283 pevHead->model = pev->model;
284 pevHead->modelindex = pev->modelindex;
285 pevHead->frame = pev->frame;
286 pevHead->colormap = pev->colormap;
287 pevHead->movetype = MOVETYPE_TOSS; // Set movement type
288 pevHead->velocity = pev->velocity; // Copy velocity
289 pevHead->flags = 0; // Reset flags
290 pevHead->deadflag = pev->deadflag; // Copy dead flag
291 pevHead->renderfx = kRenderFxDeadPlayer; // Set render effect for dead player
292 pevHead->renderamt = ENTINDEX(ENT(pev)); // Set render amount
293
294 pevHead->effects = pev->effects | EF_NOINTERP; // Set effects
295 // Additional properties can be copied here if needed
296 pevHead->sequence = pev->sequence; // Copy animation sequence
297 pevHead->animtime = pev->animtime; // Copy animation time
298
299 UTIL_SetOrigin(pevHead, pev->origin); // Set the origin of the body queue head
300 UTIL_SetSize(pevHead, pev->mins, pev->maxs); // Set the size of the body queue head
301 g_pBodyQueueHead = pevHead->owner; // Update the head of the body queue
302}

333globalentity_t *CGlobalState::Find(string_t globalname)
334{
335 if (!globalname) // Check if the name is valid
336 return NULL; // Return NULL if not valid
337
338 globalentity_t *pTest; // Pointer for testing entities
339 const char *pEntityName = STRING(globalname); // Convert string_t to const char*
340
341 pTest = m_pList; // Start from the head of the list
342 while (pTest)
343 {
344 if (FStrEq(pEntityName, pTest->name)) // Check if names match
345 break; // Found the entity
346
347 pTest = pTest->pNext; // Move to the next entity
348 }
349
350 return pTest; // Return the found entity or NULL
351}

467int CGlobalState::Save(CSave &save)
468{
469 int i;
470 globalentity_t *pEntity;
471
472 if (!save.WriteFields("GLOBAL", this, m_SaveData, ARRAYSIZE(m_SaveData)))
473 return 0; // Return 0 if saving global state fails
474
475 pEntity = m_pList; // Start from the head of the global entity list
476 for (i = 0; i < m_listCount && pEntity; i++)
477 {
478 if (!save.WriteFields("GENT", pEntity, gGlobalEntitySaveData, ARRAYSIZE(gGlobalEntitySaveData)))
479 return 0; // Return 0 if saving entity fails
480
481 pEntity = pEntity->pNext; // Move to the next entity
482 }
483
484 return 1; // Return 1 if saving was successful
485}

496int CGlobalState::Restore(CRestore &restore)
497{
498 int i, listCount;
499 globalentity_t tmpEntity; // Temporary entity for restoration
500
501 ClearStates(); // Clear existing states
502 if (!restore.ReadFields("GLOBAL", this, m_SaveData, ARRAYSIZE(m_SaveData)))
503 return 0; // Return 0 if restoring global state fails
504
505 listCount = m_listCount; // Get the new list count
506 m_listCount = 0; // Clear loaded data
507
508 for (i = 0; i < listCount; i++)
509 {
510 if (!restore.ReadFields("GENT", &tmpEntity, gGlobalEntitySaveData, ARRAYSIZE(gGlobalEntitySaveData)))
511 return 0; // Return 0 if restoring entity fails
512 EntityAdd(MAKE_STRING(tmpEntity.name), MAKE_STRING(tmpEntity.levelName), tmpEntity.state); // Add the restored entity
513 }
514 return 1; // Return 1 if restoring was successful
515}

540void CGlobalState::ClearStates(void)
541{
542 globalentity_t *pFree = m_pList; // Start from the head of the list
543 while (pFree)
544 {
545 globalentity_t *pNext = pFree->pNext; // Store the next entity
546 free(pFree); // Free the current entity

549 Reset(); // Reset the global state

630

786void CWorld::KeyValue(KeyValueData *pkvd)
787{
788 if (FStrEq(pkvd->szKeyName, "skyname"))
789 {
790 CVAR_SET_STRING("sv_skyname", pkvd->szValue); // Set the sky name
791 pkvd->fHandled = TRUE; // Mark as handled
792 }
793 else if (FStrEq(pkvd->szKeyName, "sounds"))
794 {
795 gpGlobals->cdAudioTrack = atoi(pkvd->szValue); // Set the audio track
796 pkvd->fHandled = TRUE; // Mark as handled
797 }
798 else if (FStrEq(pkvd->szKeyName, "WaveHeight"))
799 {
800 pev->scale = atof(pkvd->szValue) * (1.0 / 8.0); // Set wave height scale
801 pkvd->fHandled = TRUE; // Mark as handled
802 CVAR_SET_FLOAT("sv_wateramp", pev->scale); // Set water amplitude
803 }
804 else if (FStrEq(pkvd->szKeyName, "MaxRange"))
805 {
806 pev->speed = atof(pkvd->szValue); // Set maximum range
807 pkvd->fHandled = TRUE; // Mark as handled
808 }
809 else if (FStrEq(pkvd->szKeyName, "chaptertitle"))
810 {
811 pev->netname = ALLOC_STRING(pkvd->szValue); // Set chapter title
812 pkvd->fHandled = TRUE; // Mark as handled
813 }
814 else if (FStrEq(pkvd->szKeyName, "startdark"))
815 {
816 int flag = atoi(pkvd->szValue); // Get start dark flag
817 pkvd->fHandled = TRUE; // Mark as handled
818 if (flag)
819 pev->spawnflags |= SF_WORLD_DARK; // Set dark flag
820 }
821 else if (FStrEq(pkvd->szKeyName, "newunit"))
