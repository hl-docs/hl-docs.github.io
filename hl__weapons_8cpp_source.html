157BOOL CBasePlayerWeapon::DefaultReload(int iClipSize, int iAnim, float fDelay, int body)
158{
159 if (m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0)
160 return FALSE;
161
162 int j = min(iClipSize - m_iClip, m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType]);
163 if (j == 0)
164 return FALSE;
165
166 m_pPlayer->m_flNextAttack = UTIL_WeaponTimeBase() + fDelay;
167 SendWeaponAnim(iAnim, UseDecrement(), body);
168
169 m_fInReload = TRUE;
170 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 3;
171 return TRUE;
172}

179BOOL CBasePlayerWeapon::CanDeploy(void)
180{
181 BOOL bHasAmmo = 0;
182
183 if (!pszAmmo1())
184 return TRUE;
185
186 if (pszAmmo1())
187 bHasAmmo |= (m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] != 0);
188
189 if (pszAmmo2())
190 bHasAmmo |= (m_pPlayer->m_rgAmmo[m_iSecondaryAmmoType] != 0);
191
192 if (m_iClip > 0)
193 bHasAmmo |= 1;
194
195 return bHasAmmo ? TRUE : FALSE;
196}

277Vector CBaseEntity::FireBulletsPlayer(ULONG cShots, Vector vecSrc, Vector vecDirShooting, Vector vecSpread,
278 float flDistance, int iBulletType, int iTracerFreq, int iDamage,
279 entvars_t *pevAttacker, int shared_rand)
280{
281 float x, y, z;
282
283 for (ULONG iShot = 1; iShot <= cShots; iShot++)
284 {
285 if (pevAttacker == NULL)
286 {
287 do {
288 x = RANDOM_FLOAT(-0.5, 0.5) + RANDOM_FLOAT(-0.5, 0.5);
289 y = RANDOM_FLOAT(-0.5, 0.5) + RANDOM_FLOAT(-0.5, 0.5);
290 z = x * x + y * y;
291 } while (z > 1);
292 }
293 else
294 {
295 x = UTIL_SharedRandomFloat(shared_rand + iShot, -0.5, 0.5) +
296 UTIL_SharedRandomFloat(shared_rand + (1 + iShot), -0.5, 0.5);
297 y = UTIL_SharedRandomFloat(shared_rand + (2 + iShot), -0.5, 0.5) +
298 UTIL_SharedRandomFloat(shared_rand + (3 + iShot), -0.5, 0.5);
299 }
300 }
301
302 return Vector(x * vecSpread.x, y * vecSpread.y, 0.0);
303}

311void CBasePlayerWeapon::ItemPostFrame(void)
312{
313 if ((m_fInReload) && (m_pPlayer->m_flNextAttack <= 0.0))
314 {
315#if 0
316 // This part is disabled due to missing ammo info on client side
317 int j = min(iMaxClip() - m_iClip, m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType]);
318 m_iClip += j;
319 m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] -= j;
320#else
321 m_iClip += 10; // Simulate reload
322#endif
323 m_fInReload = FALSE;
324 }
325
326 if ((m_pPlayer->pev->button & IN_ATTACK2) && (m_flNextSecondaryAttack <= 0.0))
327 {
328 if (pszAmmo2() && !m_pPlayer->m_rgAmmo[SecondaryAmmoIndex()])
329 m_fFireOnEmpty = TRUE;
330
331 SecondaryAttack();
332 m_pPlayer->pev->button &= ~IN_ATTACK2;
333 }
334 else if ((m_pPlayer->pev->button & IN_ATTACK) && (m_flNextPrimaryAttack <= 0.0))
335 {
336 if ((m_iClip == 0 && pszAmmo1()) || (iMaxClip() == -1 && !m_pPlayer->m_rgAmmo[PrimaryAmmoIndex()]))
337 m_fFireOnEmpty = TRUE;
338
339 PrimaryAttack();
340 }
341 else if (m_pPlayer->pev->button & IN_RELOAD && iMaxClip() != WEAPON_NOCLIP && !m_fInReload)
342 {
343 Reload();
344 }
345 else if (!(m_pPlayer->pev->button & (IN_ATTACK | IN_ATTACK2)))
346 {
347 m_fFireOnEmpty = FALSE;
348
349 if (m_iClip == 0 && !(iFlags() & ITEM_FLAG_NOAUTORELOAD) && m_flNextPrimaryAttack < 0.0)
350 {
351 Reload();
352 return;
353 }
354
355 WeaponIdle();
356 return;
357 }
358
359 if (ShouldWeaponIdle())
360 {
361 WeaponIdle();
362 }
363}

489 cl_entity_t *player;
490 vec3_t mins, maxs;
491
492 gEngfuncs.pEventAPI->EV_SetUpPlayerPrediction(false, true);
493 gEngfuncs.pEventAPI->EV_PushPMStates();
494
495 player = gEngfuncs.GetLocalPlayer();

497

502 break;
503
504 if (pe->info >= 1 && pe->info <= gEngfuncs.GetMaxClients())
505 {

507 maxs = pe->origin + pe->maxs;
508
509 gEngfuncs.pEfxAPI->R_ParticleBox((float *)&mins, (float *)&maxs, 0, 0, 255, 2.0);

652void HUD_WeaponsPostThink( local_state_s *from, local_state_s *to, usercmd_t *cmd, double time, unsigned int random_seed )
653{
654 int i;
655 int buttonsChanged;
656 CBasePlayerWeapon *pWeapon = NULL;
657 CBasePlayerWeapon *pCurrent;
658 weapon_data_t nulldata, *pfrom, *pto;
659 static int lasthealth;
660
661 memset( &nulldata, 0, sizeof( nulldata ) );
662
663 HUD_InitClientWeapons();
664
665 // Get current clock
666 gpGlobals->time = time;
667
668 // Fill in data based on selected weapon
669 // FIXME, make this a method in each weapon? where you pass in an entity_state_t *?
670 switch ( from->client.m_iId )
671 {
672 case WEAPON_CROWBAR:
673 pWeapon = &g_Crowbar;
674 break;
675
676 case WEAPON_GLOCK:
677 pWeapon = &g_Glock;
678 break;
679
680 case WEAPON_PYTHON:
681 pWeapon = &g_Python;
682 break;
683
684 case WEAPON_MP5:
685 pWeapon = &g_Mp5;
686 break;
687
688 case WEAPON_CROSSBOW:
689 pWeapon = &g_Crossbow;
690 break;
691
692 case WEAPON_SHOTGUN:
693 pWeapon = &g_Shotgun;
694 break;
695
696 case WEAPON_RPG:
697 pWeapon = &g_Rpg;
698 break;
699
700 case WEAPON_GAUSS:
701 pWeapon = &g_Gauss;
702 break;
703
704 case WEAPON_EGON:
705 pWeapon = &g_Egon;
706 break;
707
708 case WEAPON_HORNETGUN:
709 pWeapon = &g_HGun;
710 break;
711
712 case WEAPON_HANDGRENADE:
713 pWeapon = &g_HandGren;
714 break;
715
716 case WEAPON_SATCHEL:
717 pWeapon = &g_Satchel;
718 break;
719
720 case WEAPON_TRIPMINE:
721 pWeapon = &g_Tripmine;
722 break;
723
724 case WEAPON_SNARK:
725 pWeapon = &g_Snark;
726 break;
727 }
728
729 // Store pointer to our destination entity_state_t so we can get our origin, etc. from it
730 // for setting up events on the client
731 g_finalstate = to;
732
733 // If we are running events/etc. go ahead and see if we
734 // managed to die between last frame and this one
735 // If so, run the appropriate player killed or spawn function
736 if ( g_runfuncs )
737 {
738 if ( to->client.health <= 0 && lasthealth > 0 )
739 {
740 player.Killed( NULL, 0 );
741
742 }
743 else if ( to->client.health > 0 && lasthealth <= 0 )
744 {
745 player.Spawn();
746 }
747
748 lasthealth = to->client.health;
749 }
750
751 // We are not predicting the current weapon, just bow out here.
752 if ( !pWeapon )
753 return;
754
755 for ( i = 0; i < 32; i++ )
756 {
757 pCurrent = g_pWpns[ i ];
758 if ( !pCurrent )
759 {
760 continue;
761 }
762
763 pfrom = &from->weapondata[ i ];
764
765 pCurrent->m_fInReload = pfrom->m_fInReload;
766 pCurrent->m_fInSpecialReload = pfrom->m_fInSpecialReload;
767// pCurrent->m_flPumpTime = pfrom->m_flPumpTime;
768 pCurrent->m_iClip = pfrom->m_iClip;
769 pCurrent->m_flNextPrimaryAttack = pfrom->m_flNextPrimaryAttack;
770 pCurrent->m_flNextSecondaryAttack = pfrom->m_flNextSecondaryAttack;
771 pCurrent->m_flTimeWeaponIdle = pfrom->m_flTimeWeaponIdle;
772 pCurrent->pev->fuser1 = pfrom->fuser1;
773 pCurrent->m_flStartThrow = pfrom->fuser2;
774 pCurrent->m_flReleaseThrow = pfrom->fuser3;
775 pCurrent->m_chargeReady = pfrom->iuser1;
776 pCurrent->m_fInAttack = pfrom->iuser2;
777 pCurrent->m_fireState = pfrom->iuser3;
778
779 pCurrent->m_iSecondaryAmmoType = (int)from->client.vuser3[ 2 ];
780 pCurrent->m_iPrimaryAmmoType = (int)from->client.vuser4[ 0 ];
781 player.m_rgAmmo[ pCurrent->m_iPrimaryAmmoType ] = (int)from->client.vuser4[ 1 ];
782 player.m_rgAmmo[ pCurrent->m_iSecondaryAmmoType ] = (int)from->client.vuser4[ 2 ];
783 }
784
785 // For random weapon events, use this seed to seed random # generator
786 player.random_seed = random_seed;
787
788 // Get old buttons from previous state.
789 player.m_afButtonLast = from->playerstate.oldbuttons;
790
791 // Which buttsons chave changed
792 buttonsChanged = (player.m_afButtonLast ^ cmd->buttons); // These buttons have changed this frame
793
794 // Debounced button codes for pressed/released
795 // The changed ones still down are "pressed"
796 player.m_afButtonPressed = buttonsChanged & cmd->buttons;
797 // The ones not down are "released"
798 player.m_afButtonReleased = buttonsChanged & (~cmd->buttons);
799
800 // Set player variables that weapons code might check/alter
801 player.pev->button = cmd->buttons;
802
803 player.pev->velocity = from->client.velocity;
804 player.pev->flags = from->client.flags;
805
806 player.pev->deadflag = from->client.deadflag;
807 player.pev->waterlevel = from->client.waterlevel;
808 player.pev->maxspeed = from->client.maxspeed;
809 player.pev->fov = from->client.fov;
810 player.pev->weaponanim = from->client.weaponanim;
811 player.pev->viewmodel = from->client.viewmodel;
812 player.m_flNextAttack = from->client.m_flNextAttack;
813 player.m_flNextAmmoBurn = from->client.fuser2;
814 player.m_flAmmoStartCharge = from->client.fuser3;
815
816 //Stores all our ammo info, so the client side weapons can use them.
817 player.ammo_9mm = (int)from->client.vuser1[0];
818 player.ammo_357 = (int)from->client.vuser1[1];
819 player.ammo_argrens = (int)from->client.vuser1[2];
820 player.ammo_bolts = (int)from->client.ammo_nails; //is an int anyways...
821 player.ammo_buckshot = (int)from->client.ammo_shells;
822 player.ammo_uranium = (int)from->client.ammo_cells;
823 player.ammo_hornets = (int)from->client.vuser2[0];
824 player.ammo_rockets = (int)from->client.ammo_rockets;
825
826
827 // Point to current weapon object
828 if ( from->client.m_iId )
829 {
830 player.m_pActiveItem = g_pWpns[ from->client.m_iId ];
831 }
832
833 if ( player.m_pActiveItem->m_iId == WEAPON_RPG )
834 {
835 ( ( CRpg * )player.m_pActiveItem)->m_fSpotActive = (int)from->client.vuser2[ 1 ];
836 ( ( CRpg * )player.m_pActiveItem)->m_cActiveRockets = (int)from->client.vuser2[ 2 ];
837 }
838
839 // Don't go firing anything if we have died or are spectating
840 // Or if we don't have a weapon model deployed
841 if ( ( player.pev->deadflag != ( DEAD_DISCARDBODY + 1 ) ) &&
842 !CL_IsDead() && player.pev->viewmodel && !g_iUser1 )
843 {
844 if ( player.m_flNextAttack <= 0 )
845 {
846 pWeapon->ItemPostFrame();
847 }
848 }
849
850 // Assume that we are not going to switch weapons
851 to->client.m_iId = from->client.m_iId;
852
853 // Now see if we issued a changeweapon command ( and we're not dead )
854 if ( cmd->weaponselect && ( player.pev->deadflag != ( DEAD_DISCARDBODY + 1 ) ) )
855 {
856 // Switched to a different weapon?
857 if ( from->weapondata[ cmd->weaponselect ].m_iId == cmd->weaponselect )
858 {
859 CBasePlayerWeapon *pNew = g_pWpns[ cmd->weaponselect ];
860 if ( pNew && ( pNew != pWeapon ) )
861 {
862 // Put away old weapon
863 if (player.m_pActiveItem)
864 player.m_pActiveItem->Holster( );
865
866 player.m_pLastItem = player.m_pActiveItem;
867 player.m_pActiveItem = pNew;
868
869 // Deploy new weapon
870 if (player.m_pActiveItem)
871 {
872 player.m_pActiveItem->Deploy( );
873 }
874
875 // Update weapon id so we can predict things correctly.
876 to->client.m_iId = cmd->weaponselect;
877 }
878 }
879 }
880
881 // Copy in results of prediction code
882 to->client.viewmodel = player.pev->viewmodel;
883 to->client.fov = player.pev->fov;
884 to->client.weaponanim = player.pev->weaponanim;
