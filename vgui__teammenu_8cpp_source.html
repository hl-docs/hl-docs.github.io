187void CTeamMenuPanel::Update( void )
188{
189 int iYPos = TEAMMENU_TOPLEFT_BUTTON_Y;
190
191 // Set the team buttons
192 for (int i = 1; i <= 4; i++)
193 {
194 if (m_pButtons[i])
195 {
196 if ( i <= gViewPort->GetNumberOfTeams() )
197 {
198 m_pButtons[i]->setText( gViewPort->GetTeamName(i) );
199
200 // bound key replacement
201 char sz[32];
202 sprintf( sz, "%d", i );
203 m_pButtons[i]->setBoundKey( sz[0] );
204
205 m_pButtons[i]->setVisible( true );
206 m_pButtons[i]->setPos( TEAMMENU_TOPLEFT_BUTTON_X, iYPos );
207 iYPos += TEAMMENU_BUTTON_SIZE_Y + TEAMMENU_BUTTON_SPACER_Y;
208
209 // Start with the first option up
210 if (!m_iCurrentInfo)
211 SetActiveInfo( i );
212
213 char szPlayerList[ (MAX_PLAYER_NAME_LENGTH + 3) * 31 ]; // name + ", "
214 strcpy(szPlayerList, "\n");
215 // Update the Team Info
216 // Now count the number of teammembers of this class
217 int iTotal = 0;
218 for ( int j = 1; j < MAX_PLAYERS; j++ )
219 {
220 if ( g_PlayerInfoList[j].name == NULL )
221 continue; // empty player slot, skip
222 if ( g_PlayerInfoList[j].thisplayer )
223 continue; // skip this player
224 if ( g_PlayerExtraInfo[j].teamnumber != i )
225 continue; // skip over players in other teams
226
227 iTotal++;
228 if (iTotal > 1)
229 strncat( szPlayerList, ", ", sizeof(szPlayerList) - strlen(szPlayerList) );
230 strncat( szPlayerList, g_PlayerInfoList[j].name, sizeof(szPlayerList) - strlen(szPlayerList) );
231 szPlayerList[ sizeof(szPlayerList) - 1 ] = '\0';
232 }
233
234 if (iTotal > 0)
235 {
236 // Set the text of the info Panel
237 char szText[ ((MAX_PLAYER_NAME_LENGTH + 3) * 31) + 256 ];
238 if (iTotal == 1)
239 sprintf(szText, "%s: %d Player (%d points)", gViewPort->GetTeamName(i), iTotal, g_TeamInfo[i].frags );
240 else
241 sprintf(szText, "%s: %d Players (%d points)", gViewPort->GetTeamName(i), iTotal, g_TeamInfo[i].frags );
242 strncat( szText, szPlayerList, sizeof(szText) - strlen(szText) );
243 szText[ sizeof(szText) - 1 ] = '\0';
244
245 m_pTeamInfoPanel[i]->setText( szText );
246 }
247 else
248 {
249 m_pTeamInfoPanel[i]->setText( "" );
250 }
251 }
252 else
253 {
254 // Hide the button (may be visible from previous maps)
255 m_pButtons[i]->setVisible( false );
256 }
257 }
258 }
259
260 // Move the AutoAssign button into place
261 m_pButtons[5]->setPos( TEAMMENU_TOPLEFT_BUTTON_X, iYPos );
262 iYPos += TEAMMENU_BUTTON_SIZE_Y + TEAMMENU_BUTTON_SPACER_Y;
263
264 // Spectate button
265 if (m_pSpectateButton->IsNotValid())
266 {
267 m_pSpectateButton->setVisible( false );
268 }
269 else
270 {
271 m_pSpectateButton->setPos( TEAMMENU_TOPLEFT_BUTTON_X, iYPos );
272 m_pSpectateButton->setVisible( true );
273 iYPos += TEAMMENU_BUTTON_SIZE_Y + TEAMMENU_BUTTON_SPACER_Y;
274 }
275
276 // If the player is already in a team, make the cancel button visible
277 if ( g_iTeamNumber )
278 {
279 m_pCancelButton->setPos( TEAMMENU_TOPLEFT_BUTTON_X, iYPos );
280 iYPos += TEAMMENU_BUTTON_SIZE_Y + TEAMMENU_BUTTON_SPACER_Y;
281 m_pCancelButton->setVisible( true );
282 }
283 else
284 {
285 m_pCancelButton->setVisible( false );
286 }
287
288 // Set the Map Title
289 if (!m_bUpdatedMapName)
290 {
291 const char *level = gEngfuncs.pfnGetLevelName();
292 if (level && level[0])
293 {
294 char sz[256];
295 char szTitle[256];
296 char *ch;
297
298 // Update the level name
299 strcpy( sz, level );
300 ch = strchr( sz, '/' );
301 if (!ch)
302 ch = strchr( sz, '\\' );
303 strcpy( szTitle, ch+1 );
304 ch = strchr( szTitle, '.' );
305 *ch = '\0';
306 m_pMapTitle->setText( "%s", szTitle );
307 *ch = '.';
308
309 // Update the map briefing
310 strcpy( sz, level );
311 ch = strchr( sz, '.' );
312 *ch = '\0';
313 strcat( sz, ".txt" );
314 char *pfile = (char*)gEngfuncs.COM_LoadFile( sz, 5, NULL );
315 if (pfile)
316 {
317 m_pBriefing->setText( pfile );
318
319 // Get the total size of the Briefing text and resize the text panel
320 int iXSize, iYSize;
321 m_pBriefing->getTextImage()->getTextSize( iXSize, iYSize );
322 m_pBriefing->setSize( iXSize, iYSize );
323 gEngfuncs.COM_FreeFile( pfile );
324 }
325
326 m_bUpdatedMapName = true;
327 }
328 }
329
330 m_pScrollPanel->validate();
331}

339bool CTeamMenuPanel::SlotInput( int iSlot )
340{
341 // Check for AutoAssign
342 if ( iSlot == 5)
343 {
344 m_pButtons[5]->fireActionSignal();
345 return true;
346 }
347
348 // Spectate
349 if ( iSlot == 6)
350 {
351 m_pSpectateButton->fireActionSignal();
352 return true;
353 }
354
355 // Otherwise, see if a particular team is selectable
356 if ( (iSlot < 1) || (iSlot > gViewPort->GetNumberOfTeams()) )
357 return false;
358 if ( !m_pButtons[ iSlot ] )
359 return false;
360
361 // Is the button pushable?
362 if ( m_pButtons[ iSlot ]->isVisible() )
363 {
364 m_pButtons[ iSlot ]->fireActionSignal();
365 return true;
366 }
367
368 return false;
369}
