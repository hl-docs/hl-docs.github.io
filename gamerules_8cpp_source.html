53BOOL CGameRules::CanHaveAmmo( CBasePlayer *pPlayer, const char *pszAmmoName, int iMaxCarry )
54{
55 int iAmmoIndex;
56
57 if ( pszAmmoName )
58 {
59 iAmmoIndex = pPlayer->GetAmmoIndex( pszAmmoName );
60
61 if ( iAmmoIndex > -1 )
62 {
63 if ( pPlayer->AmmoInventory( iAmmoIndex ) < iMaxCarry )
64 {
65 // Player has room for more of this type of ammo
66 return TRUE;
67 }
68 }
69 }
70
71 return FALSE;
72}

107BOOL CGameRules::CanHavePlayerItem( CBasePlayer *pPlayer, CBasePlayerItem *pWeapon )
108{
109 // Only living players can have items
110 if ( pPlayer->pev->deadflag != DEAD_NO )
111 return FALSE;
112
113 if ( pWeapon->pszAmmo1() )
114 {
115 if ( !CanHaveAmmo( pPlayer, pWeapon->pszAmmo1(), pWeapon->iMaxAmmo1() ) )
116 {
117 // Cannot carry more ammo for this weapon. Check if player already has it.
118 if ( pPlayer->HasPlayerItem( pWeapon ) )
119 {
120 return FALSE;
121 }
122 }
123 }
124 else
125 {
126 // Weapon doesn't use ammo, check if player already has it.
127 if ( pPlayer->HasPlayerItem( pWeapon ) )
128 {
129 return FALSE;
130 }
131 }
132
133 // Will fall through to here if GetItemInfo doesn't fill the struct
134 return TRUE;
135}

347CGameRules *InstallGameRules( void )
348{
349 SERVER_COMMAND( "exec game.cfg\n" );
350 SERVER_EXECUTE( );
351
352 if ( !gpGlobals->deathmatch )
353 {
354 // Generic Half-Life rules
355 g_teamplay = 0;
356 return new CHalfLifeRules;
357 }
358 else
359 {
360 if ( teamplay.value > 0 )
361 {
362 // Teamplay rules
363 g_teamplay = 1;
364 return new CHalfLifeTeamplay;
365 }
366 if ((int)gpGlobals->deathmatch == 1)
367 {
368 // Vanilla deathmatch rules
369 g_teamplay = 0;
370 return new CHalfLifeMultiplay;
371 }
372 else
373 {
374 // Vanilla deathmatch (fallback)
375 g_teamplay = 0;
376 return new CHalfLifeMultiplay;
377 }
378 }
