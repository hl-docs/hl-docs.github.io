145void CStomp::Think(void)
146{
147 TraceResult tr;
148
149 pev->nextthink = gpGlobals->time + 0.1;
150
151 // Do damage for this frame
152 Vector vecStart = pev->origin;
153 vecStart.z += 30;
154 Vector vecEnd = vecStart + (pev->movedir * pev->speed * gpGlobals->frametime);
155
156 UTIL_TraceHull(vecStart, vecEnd, dont_ignore_monsters, head_hull, ENT(pev), &tr);
157
158 if (tr.pHit && tr.pHit != pev->owner)
159 {
160 CBaseEntity* pEntity = CBaseEntity::Instance(tr.pHit);
161 entvars_t* pevOwner = pev;
162 if (pev->owner)
163 pevOwner = VARS(pev->owner);
164
165 if (pEntity)
166 pEntity->TakeDamage(pev, pevOwner, gSkillData.gargantuaDmgStomp, DMG_SONIC);
167 }
168
169 // Accelerate the effect
170 pev->speed = pev->speed + (gpGlobals->frametime) * pev->framerate;
171 pev->framerate = pev->framerate + (gpGlobals->frametime) * 1500;
172
173 // Move and spawn trails
174 while (gpGlobals->time - pev->dmgtime > STOMP_INTERVAL)
175 {
176 pev->origin = pev->origin + pev->movedir * pev->speed * STOMP_INTERVAL;
177 for (int i = 0; i < 2; i++)
178 {
179 CSprite* pSprite = CSprite::SpriteCreate(GARG_STOMP_SPRITE_NAME, pev->origin, TRUE);
180 if (pSprite)
181 {
182 UTIL_TraceLine(pev->origin, pev->origin - Vector(0, 0, 500), ignore_monsters, edict(), &tr);
183 pSprite->pev->origin = tr.vecEndPos;
184 pSprite->pev->velocity = Vector(RANDOM_FLOAT(-200, 200), RANDOM_FLOAT(-200, 200), 175);
185 // pSprite->AnimateAndDie( RANDOM_FLOAT( 8.0, 12.0 ) );
186 pSprite->pev->nextthink = gpGlobals->time + 0.3;
187 pSprite->SetThink(&CSprite::SUB_Remove);
188 pSprite->SetTransparency(kRenderTransAdd, 255, 255, 255, 255, kRenderFxFadeFast);
189 }
190 }
191 pev->dmgtime += STOMP_INTERVAL;
192 // Scale has the "life" of this effect
193 pev->scale -= STOMP_INTERVAL * pev->speed;
194 if (pev->scale <= 0)
195 {
196 // Life has run out
197 UTIL_Remove(this);
198 STOP_SOUND(edict(), CHAN_BODY, GARG_STOMP_BUZZ_SOUND);
199 }
200 }

688void CGargantua :: FlameDamage( Vector vecStart, Vector vecEnd, entvars_t *pevInflictor, entvars_t *pevAttacker, float flDamage, int iClassIgnore, int bitsDamageType )
689{
690 CBaseEntity *pEntity = NULL;
691 TraceResult tr;
692 float flAdjustedDamage;
693 Vector vecSpot;
694
695 Vector vecMid = (vecStart + vecEnd) * 0.5;
696
697 float searchRadius = (vecStart - vecMid).Length();
698
699 Vector vecAim = (vecEnd - vecStart).Normalize( );
700
701 // iterate on all entities in the vicinity.
702 while ((pEntity = UTIL_FindEntityInSphere( pEntity, vecMid, searchRadius )) != NULL)
703 {
704 if ( pEntity->pev->takedamage != DAMAGE_NO )
705 {
706 // UNDONE: this should check a damage mask, not an ignore
707 if ( iClassIgnore != CLASS_NONE && pEntity->Classify() == iClassIgnore )
708 {// houndeyes don't hurt other houndeyes with their attack
709 continue;
710 }
711
712 vecSpot = pEntity->BodyTarget( vecMid );
713
714 float dist = DotProduct( vecAim, vecSpot - vecMid );
715 if (dist > searchRadius)
716 dist = searchRadius;
717 else if (dist < -searchRadius)
718 dist = searchRadius;
719
720 Vector vecSrc = vecMid + dist * vecAim;
721
722 UTIL_TraceLine ( vecSrc, vecSpot, dont_ignore_monsters, ENT(pev), &tr );
723
724 if ( tr.flFraction == 1.0 || tr.pHit == pEntity->edict() )
725 {// the explosion can 'see' this entity, so hurt them!
726 // decrease damage for an ent that's farther from the flame.
727 dist = ( vecSrc - tr.vecEndPos ).Length();
728
729 if (dist > 64)
730 {
731 flAdjustedDamage = flDamage - (dist - 64) * 0.4;
732 if (flAdjustedDamage <= 0)
733 continue;
734 }
735 else
736 {
737 flAdjustedDamage = flDamage;
