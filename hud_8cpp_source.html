110void COM_FileBase ( const char *in, char *out)
111{
112 int len, start, end;
113
114 len = strlen( in );
115 end = len - 1;
116 while ( end && in[end] != '.' && in[end] != '/' && in[end] != '\\' )
117 end--;
118 if ( in[end] != '.' )
119 end = len-1;
120 else
121 end--;
122
123 start = len-1;
124 while ( start >= 0 && in[start] != '/' && in[start] != '\\' )
125 start--;
126
127 if ( in[start] != '/' && in[start] != '\\' )
128 start = 0;
129 else
130 start++;
131
132 len = end - start + 1;
133 strncpy( out, &in[start], len );
134 out[len] = 0;
135}

196int CHud::MsgFunc_SetFOV(const char *pszName, int iSize, void *pbuf)
197{
198 BEGIN_READ( pbuf, iSize );
199
200 int newfov = READ_BYTE();
201 int def_fov = CVAR_GET_FLOAT( "default_fov" );
202
203 if ( cl_lw && cl_lw->value )
204 return 1;
205
206 g_lastFOV = newfov;
207
208 if ( newfov == 0 )
209 m_iFOV = def_fov;
210 else
211 m_iFOV = newfov;
212
213 if ( m_iFOV == def_fov )
214 m_flMouseSensitivity = 0;
215 else
216 m_flMouseSensitivity = sensitivity->value * ((float)newfov / (float)def_fov) * CVAR_GET_FLOAT("zoom_sensitivity_ratio");
217
218 return 1;
219}

226void CHud::AddHudElem(CHudBase *phudelem)
227{
228 HUDLIST *pdl, *ptemp;
229
230 if (!phudelem)
231 return;
232
233 pdl = (HUDLIST *)malloc(sizeof(HUDLIST));
234 if (!pdl)
235 return;
236
237 memset(pdl, 0, sizeof(HUDLIST));
238 pdl->p = phudelem;
239
240 if (!m_pHudList)
241 {
242 m_pHudList = pdl;
243 return;
244 }
245
246 ptemp = m_pHudList;
247 while (ptemp->pNext)
248 ptemp = ptemp->pNext;
249
250 ptemp->pNext = pdl;
