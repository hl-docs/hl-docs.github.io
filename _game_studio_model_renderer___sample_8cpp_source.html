95void CGameStudioModelRenderer::StudioSetupBones ( void )
96{
97 int i;
98 double f;
99
100 mstudiobone_t *pbones;
101 mstudioseqdesc_t *pseqdesc;
102 mstudioanim_t *panim;
103
104 static float pos[MAXSTUDIOBONES][3];
105 static vec4_t q[MAXSTUDIOBONES];
106 float bonematrix[3][4];
107
108 static float pos2[MAXSTUDIOBONES][3];
109 static vec4_t q2[MAXSTUDIOBONES];
110 static float pos3[MAXSTUDIOBONES][3];
111 static vec4_t q3[MAXSTUDIOBONES];
112 static float pos4[MAXSTUDIOBONES][3];
113 static vec4_t q4[MAXSTUDIOBONES];
114
115 // Use default bone setup for nonplayers
116 if ( !m_pCurrentEntity->player )
117 {
118 CStudioModelRenderer::StudioSetupBones();
119 return;
120 }
121
122 // Bound sequence number.
123 if ( m_pCurrentEntity->curstate.sequence >= m_pStudioHeader->numseq )
124 {
125 m_pCurrentEntity->curstate.sequence = 0;
126 }
127
128 pseqdesc = (mstudioseqdesc_t *)((byte *)m_pStudioHeader + m_pStudioHeader->seqindex) + m_pCurrentEntity->curstate.sequence;
129
130 if ( m_pPlayerInfo && m_pPlayerInfo->gaitsequence != 0 )
131 {
132 f = m_pPlayerInfo->gaitframe;
133 }
134 else
135 {
136 f = StudioEstimateFrame( pseqdesc );
137 }
138
139 // This game knows how to do three way blending
140 if ( pseqdesc->numblends == 3 )
141 {
142 float s;
143
144 // Get left anim
145 panim = StudioGetAnim( m_pRenderModel, pseqdesc );
146
147 // Blending is 0-127 == Left to Middle, 128 to 255 == Middle to right
148 if ( m_pCurrentEntity->curstate.blending[0] <= 127 )
149 {
150 StudioCalcRotations( pos, q, pseqdesc, panim, f );
151
152 // Scale 0-127 blending up to 0-255
153 s = m_pCurrentEntity->curstate.blending[0];
154 s = ( s * 2.0 );
155 }
156 else
157 {
158
159 // Skip ahead to middle
160 panim += m_pStudioHeader->numbones;
161
162 StudioCalcRotations( pos, q, pseqdesc, panim, f );
163
164 // Scale 127-255 blending up to 0-255
165 s = m_pCurrentEntity->curstate.blending[0];
166 s = 2.0 * ( s - 127.0 );
167 }
168
169 // Normalize interpolant
170 s /= 255.0;
171
172 // Go to middle or right
173 panim += m_pStudioHeader->numbones;
174
175 StudioCalcRotations( pos2, q2, pseqdesc, panim, f );
176
177 // Spherically interpolate the bones
178 StudioSlerpBones( q, pos, q2, pos2, s );
179 }
180 else
181 {
182 panim = StudioGetAnim( m_pRenderModel, pseqdesc );
183 StudioCalcRotations( pos, q, pseqdesc, panim, f );
184 }
185
186 // Are we in the process of transitioning from one sequence to another.
187 if ( m_fDoInterp &&
188 m_pCurrentEntity->latched.sequencetime &&
189 ( m_pCurrentEntity->latched.sequencetime + 0.2 > m_clTime ) &&
190 ( m_pCurrentEntity->latched.prevsequence < m_pStudioHeader->numseq ))
191 {
192 // blend from last sequence
193 static float pos1b[MAXSTUDIOBONES][3];
194 static vec4_t q1b[MAXSTUDIOBONES];
195 float s;
196
197 // Blending value into last sequence
198 unsigned char prevseqblending = m_pCurrentEntity->latched.prevseqblending[ 0 ];
199
200 // Point at previous sequenece
201 pseqdesc = (mstudioseqdesc_t *)((byte *)m_pStudioHeader + m_pStudioHeader->seqindex) + m_pCurrentEntity->latched.prevsequence;
202
203 // Know how to do three way blends
204 if ( pseqdesc->numblends == 3 )
205 {
206 float s;
207
208 // Get left animation
209 panim = StudioGetAnim( m_pRenderModel, pseqdesc );
210
211 if ( prevseqblending <= 127 )
212 {
213 // Set up bones based on final frame of previous sequence
214 StudioCalcRotations( pos1b, q1b, pseqdesc, panim, m_pCurrentEntity->latched.prevframe );
215
216 s = prevseqblending;
217 s = ( s * 2.0 );
218 }
219 else
220 {
221 // Skip to middle blend
222 panim += m_pStudioHeader->numbones;
223
224 StudioCalcRotations( pos1b, q1b, pseqdesc, panim, m_pCurrentEntity->latched.prevframe );
225
226 s = prevseqblending;
227 s = 2.0 * ( s - 127.0 );
228 }
229
230 // Normalize
231 s /= 255.0;
232
233 panim += m_pStudioHeader->numbones;
234 StudioCalcRotations( pos2, q2, pseqdesc, panim, m_pCurrentEntity->latched.prevframe );
235
236 // Interpolate bones
237 StudioSlerpBones( q1b, pos1b, q2, pos2, s );
238 }
239 else
240 {
241 panim = StudioGetAnim( m_pRenderModel, pseqdesc );
242 // clip prevframe
243 StudioCalcRotations( pos1b, q1b, pseqdesc, panim, m_pCurrentEntity->latched.prevframe );
244 }
245
246 // Now blend last frame of previous sequence with current sequence.
247 s = 1.0 - (m_clTime - m_pCurrentEntity->latched.sequencetime) / 0.2;
248 StudioSlerpBones( q, pos, q1b, pos1b, s );
249 }
250 else
251 {
252 m_pCurrentEntity->latched.prevframe = f;
253 }
254
255 // Now convert quaternions and bone positions into matrices
256 pbones = (mstudiobone_t *)((byte *)m_pStudioHeader + m_pStudioHeader->boneindex);
257
258 for (i = 0; i < m_pStudioHeader->numbones; i++)
259 {
260 QuaternionMatrix( q[i], bonematrix );
261
262 bonematrix[0][3] = pos[i][0];
263 bonematrix[1][3] = pos[i][1];
264 bonematrix[2][3] = pos[i][2];
265
266 if (pbones[i].parent == -1)
267 {
268 if ( IEngineStudio.IsHardware() )
269 {
270 ConcatTransforms ((*m_protationmatrix), bonematrix, (*m_pbonetransform)[i]);
271 ConcatTransforms ((*m_protationmatrix), bonematrix, (*m_plighttransform)[i]);
272 }
273 else
274 {
275 ConcatTransforms ((*m_paliastransform), bonematrix, (*m_pbonetransform)[i]);
276 ConcatTransforms ((*m_protationmatrix), bonematrix, (*m_plighttransform)[i]);
277 }
278
279 // Apply client-side effects to the transformation matrix
280 StudioFxTransform( m_pCurrentEntity, (*m_pbonetransform)[i] );
281 }
282 else
283 {
284 ConcatTransforms ((*m_pbonetransform)[pbones[i].parent], bonematrix, (*m_pbonetransform)[i]);
285 ConcatTransforms ((*m_plighttransform)[pbones[i].parent], bonematrix, (*m_plighttransform)[i]);
286 }
287 }

302void CGameStudioModelRenderer::StudioEstimateGait( entity_state_t *pplayer )
303{
304 float dt;
305 vec3_t est_velocity;
306
307 dt = (m_clTime - m_clOldTime);
308 dt = max( 0.0, dt );
309 dt = min( 1.0, dt );
310
311 if (dt == 0 || m_pPlayerInfo->renderframe == m_nFrameCount)
312 {
313 m_flGaitMovement = 0;
314 return;
315 }
316
317 // VectorAdd( pplayer->velocity, pplayer->prediction_error, est_velocity );
318 if ( m_fGaitEstimation )
319 {
320 VectorSubtract( m_pCurrentEntity->origin, m_pPlayerInfo->prevgaitorigin, est_velocity );
321 VectorCopy( m_pCurrentEntity->origin, m_pPlayerInfo->prevgaitorigin );
322 m_flGaitMovement = Length( est_velocity );
323 if (dt <= 0 || m_flGaitMovement / dt < 5)
324 {
325 m_flGaitMovement = 0;
326 est_velocity[0] = 0;
327 est_velocity[1] = 0;
328 }
329 }
330 else
331 {
332 VectorCopy( pplayer->velocity, est_velocity );
333 m_flGaitMovement = Length( est_velocity ) * dt;
334 }
335
336 if (est_velocity[1] == 0 && est_velocity[0] == 0)
337 {
338 float flYawDiff = m_pCurrentEntity->angles[YAW] - m_pPlayerInfo->gaityaw;
339 flYawDiff = flYawDiff - (int)(flYawDiff / 360) * 360;
340 if (flYawDiff > 180)
341 flYawDiff -= 360;
342 if (flYawDiff < -180)
343 flYawDiff += 360;
344
345 if (dt < 0.25)
346 flYawDiff *= dt * 4;
347 else
348 flYawDiff *= dt;
349
350 m_pPlayerInfo->gaityaw += flYawDiff;
351 m_pPlayerInfo->gaityaw = m_pPlayerInfo->gaityaw - (int)(m_pPlayerInfo->gaityaw / 360) * 360;
352
353 m_flGaitMovement = 0;
354 }
355 else
356 {
357 m_pPlayerInfo->gaityaw = (atan2(est_velocity[1], est_velocity[0]) * 180 / M_PI);
358 if (m_pPlayerInfo->gaityaw > 180)
359 m_pPlayerInfo->gaityaw = 180;
360 if (m_pPlayerInfo->gaityaw < -180)
361 m_pPlayerInfo->gaityaw = -180;
362 }
363
364}

376void CGameStudioModelRenderer::StudioProcessGait( entity_state_t *pplayer )
377{
378 mstudioseqdesc_t *pseqdesc;
379 float dt;
380 float flYaw; // view direction relative to movement
381
382 pseqdesc = (mstudioseqdesc_t *)((byte *)m_pStudioHeader + m_pStudioHeader->seqindex) + m_pCurrentEntity->curstate.sequence;
383
384 m_pCurrentEntity->angles[PITCH] = 0;
385 m_pCurrentEntity->latched.prevangles[PITCH] = m_pCurrentEntity->angles[PITCH];
386
387 dt = (m_clTime - m_clOldTime);
388 dt = max( 0.0, dt );
389 dt = min( 1.0, dt );
390
391 StudioEstimateGait( pplayer );
392
393 // calc side to side turning
394 flYaw = m_pCurrentEntity->angles[YAW] - m_pPlayerInfo->gaityaw;
395
396 flYaw = fmod( flYaw, 360.0 );
397
398 if (flYaw < -180)
399 {
400 flYaw = flYaw + 360;
401 }
402 else if (flYaw > 180)
403 {
404 flYaw = flYaw - 360;
405 }
406
407 float maxyaw = 120.0;
408
409 if (flYaw > maxyaw)
410 {
411 m_pPlayerInfo->gaityaw = m_pPlayerInfo->gaityaw - 180;
412 m_flGaitMovement = -m_flGaitMovement;
413 flYaw = flYaw - 180;
414 }
415 else if (flYaw < -maxyaw)
416 {
417 m_pPlayerInfo->gaityaw = m_pPlayerInfo->gaityaw + 180;
418 m_flGaitMovement = -m_flGaitMovement;
419 flYaw = flYaw + 180;
420 }
421
422 float blend_yaw = ( flYaw / 90.0 ) * 128.0 + 127.0;
423 blend_yaw = min( 255.0, blend_yaw );
424 blend_yaw = max( 0.0, blend_yaw );
425
426 blend_yaw = 255.0 - blend_yaw;
427
428 m_pCurrentEntity->curstate.blending[0] = (int)(blend_yaw);
429 m_pCurrentEntity->latched.prevblending[0] = m_pCurrentEntity->curstate.blending[0];
430 m_pCurrentEntity->latched.prevseqblending[0] = m_pCurrentEntity->curstate.blending[0];
431
432 m_pCurrentEntity->angles[YAW] = m_pPlayerInfo->gaityaw;
433 if (m_pCurrentEntity->angles[YAW] < -0)
434 {
435 m_pCurrentEntity->angles[YAW] += 360;
436 }
437 m_pCurrentEntity->latched.prevangles[YAW] = m_pCurrentEntity->angles[YAW];
438
439 pseqdesc = (mstudioseqdesc_t *)((byte *)m_pStudioHeader + m_pStudioHeader->seqindex) + pplayer->gaitsequence;
440
441 // Calc gait frame
442 if (pseqdesc->linearmovement[0] > 0)
443 {
444 m_pPlayerInfo->gaitframe += (m_flGaitMovement / pseqdesc->linearmovement[0]) * pseqdesc->numframes;
445 }
446 else
447 {
448 m_pPlayerInfo->gaitframe += pseqdesc->fps * dt * m_pCurrentEntity->curstate.framerate;
449 }
450
451 // Do modulo
452 m_pPlayerInfo->gaitframe = m_pPlayerInfo->gaitframe - (int)(m_pPlayerInfo->gaitframe / pseqdesc->numframes) * pseqdesc->numframes;
453 if (m_pPlayerInfo->gaitframe < 0)
454 {
455 m_pPlayerInfo->gaitframe += pseqdesc->numframes;
456 }
457}

504void GetSequenceInfo( void *pmodel, client_anim_state_t *pev, float *pflFrameRate, float *pflGroundSpeed )
505{
506 studiohdr_t *pstudiohdr;
507
508 pstudiohdr = (studiohdr_t *)pmodel;
509 if (! pstudiohdr)
510 return;
511
512 mstudioseqdesc_t *pseqdesc;
513
514 if (pev->sequence >= pstudiohdr->numseq)
515 {
516 *pflFrameRate = 0.0;
517 *pflGroundSpeed = 0.0;
518 return;
519 }
520
521 pseqdesc = (mstudioseqdesc_t *)((byte *)pstudiohdr + pstudiohdr->seqindex) + (int)pev->sequence;
522
523 if (pseqdesc->numframes > 1)
524 {
525 *pflFrameRate = 256 * pseqdesc->fps / (pseqdesc->numframes - 1);
526 *pflGroundSpeed = sqrt( pseqdesc->linearmovement[0]*pseqdesc->linearmovement[0]+ pseqdesc->linearmovement[1]*pseqdesc->linearmovement[1]+ pseqdesc->linearmovement[2]*pseqdesc->linearmovement[2] );
527 *pflGroundSpeed = *pflGroundSpeed * pseqdesc->fps / (pseqdesc->numframes - 1);
528 }
529 else
530 {
531 *pflFrameRate = 256.0;
532 *pflGroundSpeed = 0.0;
533 }
534}

570float StudioFrameAdvance ( client_anim_state_t *st, float framerate, float flInterval )
571{
572 if (flInterval == 0.0)
573 {
574 flInterval = (gEngfuncs.GetClientTime() - st->animtime);
575 if (flInterval <= 0.001)
576 {
577 st->animtime = gEngfuncs.GetClientTime();
578 return 0.0;
579 }
580 }
581 if (!st->animtime)
582 flInterval = 0.0;
583
584 st->frame += flInterval * framerate * st->framerate;
585 st->animtime = gEngfuncs.GetClientTime();
586
587 if (st->frame < 0.0 || st->frame >= 256.0)
588 {
589 if ( st->m_fSequenceLoops )
590 st->frame -= (int)(st->frame / 256.0) * 256.0;
591 else
592 st->frame = (st->frame < 0.0) ? 0 : 255;
593 st->m_fSequenceFinished = TRUE; // just in case it wasn't caught in GetEvents
594 }
595
596 return flInterval;
597}

608void CGameStudioModelRenderer::SetupClientAnimation( entity_state_t *pplayer )
609{
610 static double oldtime;
611 double curtime, dt;
612
613 client_anim_state_t *st;
614 float fr, gs;
615
616 cl_entity_t *ent = IEngineStudio.GetCurrentEntity();
617 assert( ent );
618 if ( !ent )
619 return;
620
621 curtime = gEngfuncs.GetClientTime();
622 dt = curtime - oldtime;
623 dt = min( 1.0, max( 0.0, dt ) );
624
625 oldtime = curtime;
626 st = &g_clientstate;
627
628 st->framerate = 1.0;
629
630 int oldseq = st->sequence;
631 Game_GetSequence( &st->sequence, &st->gaitsequence ); //CVAR_GET_FLOAT( "sequence" );
632 Game_GetOrientation( (float *)&st->origin, (float *)&st->angles );
633 st->realangles = st->angles;
634
635 if ( st->sequence != oldseq )
636 {
637 st->frame = 0.0;
638 st->lv.prevsequence = oldseq;
639 st->lv.sequencetime = st->animtime;
640
641 memcpy( st->lv.prevseqblending, st->blending, 2 );
642 memcpy( st->lv.prevcontroller, st->controller, 4 );
643 }
644
645 void *pmodel = (studiohdr_t *)IEngineStudio.Mod_Extradata( ent->model );
646
647 GetSequenceInfo( pmodel, st, &fr, &gs );
648 st->m_fSequenceLoops = ((GetSequenceFlags( pmodel, st ) & STUDIO_LOOPING) != 0);
649 StudioFrameAdvance( st, fr, dt );
650
651// gEngfuncs.Con_Printf( "gs %i frame %f\n", st->gaitsequence, st->frame );
652
653 ent->angles = st->realangles;
654 ent->curstate.angles = st->angles;
655 ent->curstate.origin = st->origin;
656
657 ent->curstate.sequence = st->sequence;
658 pplayer->gaitsequence = st->gaitsequence;
659 ent->curstate.animtime = st->animtime;
660 ent->curstate.frame = st->frame;
661 ent->curstate.framerate = st->framerate;
662 memcpy( ent->curstate.blending, st->blending, 2 );
663 memcpy( ent->curstate.controller, st->controller, 4 );
664
665 ent->latched = st->lv;
666}

728int CGameStudioModelRenderer::StudioDrawPlayer( int flags, entity_state_t *pplayer )
729{
730 int iret = 0;
731
732 bool isLocalPlayer = false;
733
734 // Set up for client?
735 if ( m_bLocal && IEngineStudio.GetCurrentEntity() == gEngfuncs.GetLocalPlayer() )
736 {
737 isLocalPlayer = true;
738 }
739
740 if ( isLocalPlayer )
741 {
742 // Store original data
743 SavePlayerState( pplayer );
744
745 // Copy in client side animation data
746 SetupClientAnimation( pplayer );
747 }
748
749 // Call real draw function
750 iret = _StudioDrawPlayer( flags, pplayer );
751
752 // Restore for client?
753 if ( isLocalPlayer )
754 {
755 // Restore the original data for the player
756 RestorePlayerState( pplayer );
757 }
758
759 return iret;
760}

773int CGameStudioModelRenderer::_StudioDrawPlayer( int flags, entity_state_t *pplayer )
774{
775 alight_t lighting;
776 vec3_t dir;
777
778 m_pCurrentEntity = IEngineStudio.GetCurrentEntity();
779 IEngineStudio.GetTimes( &m_nFrameCount, &m_clTime, &m_clOldTime );
780 IEngineStudio.GetViewInfo( m_vRenderOrigin, m_vUp, m_vRight, m_vNormal );
781 IEngineStudio.GetAliasScale( &m_fSoftwareXScale, &m_fSoftwareYScale );
782
783 m_nPlayerIndex = pplayer->number - 1;
784
785 if (m_nPlayerIndex < 0 || m_nPlayerIndex >= gEngfuncs.GetMaxClients())
786 return 0;
787
788 m_pRenderModel = IEngineStudio.SetupPlayerModel( m_nPlayerIndex );
789 if (m_pRenderModel == NULL)
790 return 0;
791
792 m_pStudioHeader = (studiohdr_t *)IEngineStudio.Mod_Extradata (m_pRenderModel);
793 IEngineStudio.StudioSetHeader( m_pStudioHeader );
794 IEngineStudio.SetRenderModel( m_pRenderModel );
795
796 if (pplayer->gaitsequence)
797 {
798 vec3_t orig_angles;
799 m_pPlayerInfo = IEngineStudio.PlayerInfo( m_nPlayerIndex );
800
801 VectorCopy( m_pCurrentEntity->angles, orig_angles );
802
803 StudioProcessGait( pplayer );
804
805 m_pPlayerInfo->gaitsequence = pplayer->gaitsequence;
806 m_pPlayerInfo = NULL;
807
808 StudioSetUpTransform( 0 );
809 VectorCopy( orig_angles, m_pCurrentEntity->angles );
810 }
811 else
812 {
813 m_pCurrentEntity->curstate.controller[0] = 127;
814 m_pCurrentEntity->curstate.controller[1] = 127;
815 m_pCurrentEntity->curstate.controller[2] = 127;

937void CGameStudioModelRenderer::StudioFxTransform( cl_entity_t *ent, float transform[3][4] )
938{
939 switch( ent->curstate.renderfx )
940 {
941 case kRenderFxDistort:
942 case kRenderFxHologram:
943 if ( gEngfuncs.pfnRandomLong(0,49) == 0 )
944 {
945 int axis = gEngfuncs.pfnRandomLong(0,1);
946 if ( axis == 1 ) // Choose between x & z
947 axis = 2;
948 VectorScale( transform[axis], gEngfuncs.pfnRandomFloat(1,1.484), transform[axis] );
949 }
950 else if ( gEngfuncs.pfnRandomLong(0,49) == 0 )
951 {
952 float offset;
953 int axis = gEngfuncs.pfnRandomLong(0,1);
954 if ( axis == 1 ) // Choose between x & z
955 axis = 2;
956 offset = gEngfuncs.pfnRandomFloat(-10,10);
957 transform[gEngfuncs.pfnRandomLong(0,2)][3] += offset;
958 }
959 break;
960 case kRenderFxExplode:
961 {
962 if ( iRenderStateChanged )
963 {
964 g_flStartScaleTime = m_clTime;
965 iRenderStateChanged = FALSE;
966 }
967
968 // Make the Model continue to shrink
969 float flTimeDelta = m_clTime - g_flStartScaleTime;
970 if ( flTimeDelta > 0 )
971 {
972 float flScale = 0.001;
973 // Goes almost all away
974 if ( flTimeDelta <= 2.0 )
975 flScale = 1.0 - (flTimeDelta / 2.0);
976
977 for (int i = 0; i < 3; i++)
978 {
979 for (int j = 0; j < 3; j++)
980 transform[i][j] *= flScale;
981 }
982 }
983 }
984 break;
985 }
986}
