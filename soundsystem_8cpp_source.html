119int Eng_LoadFunctions( HMODULE hMod )
120{
121 engine_api_func pfnEngineAPI;
122
123 pfnEngineAPI = ( engine_api_func )GetProcAddress( hMod, "Sys_EngineAPI" );
124 if ( !pfnEngineAPI )
125 return 0;
126
127 if ( !(*pfnEngineAPI)( ENGINE_LAUNCHER_API_VERSION, sizeof( engine_api_t ), &engineapi ) )
128 return 0;
129
130 // All is okay
131 return 1;
132}

143void LoadSoundAPIs( void )
144{
145 hEngine = ::LoadLibrary( IEngineStudio.IsHardware() ? "hw.dll" : "sw.dll" );
146 if ( hEngine )
147 {
148 if ( Eng_LoadFunctions( hEngine ) )
149 {
150 if ( engineapi.S_GetDSPointer && engineapi.S_GetWAVPointer )
151 {
152 engineapi.S_GetDSPointer(&lpDS, &lpDSBuf);
153 lpHW = (HWAVEOUT FAR *)engineapi.S_GetWAVPointer();
154 }
155 }
156 }
157}

166void ShutdownSoundAPIs( void )
167{
168 if( hEngine )

170 FreeLibrary( hEngine );

174 lpDS = 0;
175 lpDSBuf = 0;
176 lpHW = 0;
