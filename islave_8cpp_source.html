323{
324 switch (pEvent->event)
325 {
326 case ISLAVE_AE_CLAW:
327 {
328 CBaseEntity *pHurt = CheckTraceHullAttack(70, gSkillData.slaveDmgClaw, DMG_SLASH); // Check for claw attack
329 if (pHurt)
330 {
331 if (pHurt->pev->flags & (FL_MONSTER | FL_CLIENT))
332 {
333 pHurt->pev->punchangle.z = -18; // Apply punch angle
334 pHurt->pev->punchangle.x = 5; // Apply punch angle
335 }
336 // Play a random attack hit sound
337 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, pAttackHitSounds[RANDOM_LONG(0, ARRAYSIZE(pAttackHitSounds) - 1)], 1.0, ATTN_NORM, 0, m_voicePitch);
338 }
339 else
340 {
341 // Play a random attack miss sound
342 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, pAttackMissSounds[RANDOM_LONG(0, ARRAYSIZE(pAttackMissSounds) - 1)], 1.0, ATTN_NORM, 0, m_voicePitch);
343 }
344 }
345 break;
346
347 case ISLAVE_AE_CLAWRAKE:
348 {
349 CBaseEntity *pHurt = CheckTraceHullAttack(70, gSkillData.slaveDmgClawrake, DMG_SLASH); // Check for claw rake attack
350 if (pHurt)
351 {
352 if (pHurt->pev->flags & (FL_MONSTER | FL_CLIENT))
353 {
354 pHurt->pev->punchangle.z = -18; // Apply punch angle
355 pHurt->pev->punchangle.x = 5; // Apply punch angle
356 }
357 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, pAttackHitSounds[RANDOM_LONG(0, ARRAYSIZE(pAttackHitSounds) - 1)], 1.0, ATTN_NORM, 0, m_voicePitch);
358 }
359 else
360 {
361 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, pAttackMissSounds[RANDOM_LONG(0, ARRAYSIZE(pAttackMissSounds) - 1)], 1.0, ATTN_NORM, 0, m_voicePitch);
362 }
363 }
364 break;
365
366 case ISLAVE_AE_ZAP_POWERUP:
367 {
368 // Speed up attack when on hard
369 if (g_iSkillLevel == SKILL_HARD)
370 pev->framerate = 1.5;
371
372 UTIL_MakeAimVectors(pev->angles); // Make aim vectors
373
374 if (m_iBeams == 0)
375 {
376 Vector vecSrc = pev->origin + gpGlobals->v_forward * 2; // Set source position
377 MESSAGE_BEGIN(MSG_PVS, SVC_TEMPENTITY, vecSrc); // Begin message for temporary entity
378 WRITE_BYTE(TE_DLIGHT); // Write light entity
379 WRITE_COORD(vecSrc.x); // X coordinate
380 WRITE_COORD(vecSrc.y); // Y coordinate
381 WRITE_COORD(vecSrc.z); // Z coordinate
382 WRITE_BYTE(12); // Radius
383 WRITE_BYTE(255); // Red
384 WRITE_BYTE(180); // Green
385 WRITE_BYTE(96); // Blue
386 WRITE_BYTE(20 / pev->framerate); // Time
387 WRITE_BYTE(0); // Decay
388 MESSAGE_END();
389
390 }
391 if (m_hDead != NULL)
392 {
393 WackBeam(-1, m_hDead); // Wack beam for dead entity
394 WackBeam(1, m_hDead); // Wack beam for dead entity
395 }
396 else
397 {
398 ArmBeam(-1); // Arm beam
399 ArmBeam(1); // Arm beam
400 BeamGlow(); // Glow beam
401 }
402
403 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, "debris/zap4.wav", 1, ATTN_NORM, 0, 100 + m_iBeams * 10); // Play zap sound
404 pev->skin = m_iBeams / 2; // Update skin based on beams
405 }
406 break;
407
408 case ISLAVE_AE_ZAP_SHOOT:
409 {
410 ClearBeams(); // Clear beams
411
412 if (m_hDead != NULL)
413 {
414 Vector vecDest = m_hDead->pev->origin + Vector(0, 0, 38); // Set destination for dead entity
415 TraceResult trace;
416 UTIL_TraceHull(vecDest, vecDest, dont_ignore_monsters, human_hull, m_hDead->edict(), &trace); // Trace line
417
418 if (!trace.fStartSolid)
419 {
420 CBaseEntity *pNew = Create("monster_alien_slave", m_hDead->pev->origin, m_hDead->pev->angles); // Create new slave
421 CBaseMonster *pNewMonster = pNew->MyMonsterPointer();
422 pNew->pev->spawnflags |= 1; // Set spawn flag
423 WackBeam(-1, pNew); // Wack beam for new entity
424 WackBeam(1, pNew); // Wack beam for new entity
425 UTIL_Remove(m_hDead); // Remove dead entity
426 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, "hassault/hw_shoot1.wav", 1, ATTN_NORM, 0, RANDOM_LONG(130, 160)); // Play shoot sound
427
428 break;
429 }
430 }
431 ClearMultiDamage(); // Clear multi-damage
432
433 UTIL_MakeAimVectors(pev->angles); // Make aim vectors
434
435 ZapBeam(-1); // Zap beam
436 ZapBeam(1); // Zap beam
437
438 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, "hassault/hw_shoot1.wav", 1, ATTN_NORM, 0, RANDOM_LONG(130, 160)); // Play shoot sound
439 ApplyMultiDamage(pev, pev); // Apply multi-damage
440
441 m_flNextAttack = gpGlobals->time + RANDOM_FLOAT(0.5, 4.0); // Set next attack time
442 }
443 break;
444
445 case ISLAVE_AE_ZAP_DONE:
446 {
447 ClearBeams(); // Clear beams
448 }
449 break;
450
451 default:
452 CSquadMonster::HandleAnimEvent(pEvent); // Call base class method for unhandled events
453 break;
454 }
455}

485BOOL CISlave::CheckRangeAttack2(float flDot, float flDist)
486{
487 return FALSE; // Currently not implemented
488
489 if (m_flNextAttack > gpGlobals->time)
490 {
491 return FALSE; // Cannot attack yet
492 }
493
494 m_hDead = NULL; // Reset dead handle
495 m_iBravery = 0; // Reset bravery
496
497 CBaseEntity *pEntity = NULL;
498 while ((pEntity = UTIL_FindEntityByClassname(pEntity, "monster_alien_slave")) != NULL)
499 {
500 TraceResult tr;
501
502 UTIL_TraceLine(EyePosition(), pEntity->EyePosition(), ignore_monsters, ENT(pev), &tr); // Trace line to check visibility
503 if (tr.flFraction == 1.0 || tr.pHit == pEntity->edict())
504 {
505 if (pEntity->pev->deadflag == DEAD_DEAD) // Check if the entity is dead
506 {
507 float d = (pev->origin - pEntity->pev->origin).Length(); // Calculate distance
508 if (d < flDist)
509 {
510 m_hDead = pEntity; // Set dead entity
511 flDist = d; // Update distance
512 }
513 m_iBravery--; // Decrease bravery
514 }
515 else
516 {
517 m_iBravery++; // Increase bravery
518 }
519 }
520 }
521 if (m_hDead != NULL)
522 return TRUE; // Found a dead comrade
523 else
524 return FALSE; // No dead comrades found
525}

690Schedule_t *CISlave::GetSchedule(void)
691{
692 ClearBeams(); // Clear any existing beams
693
694 /*
695 if (pev->spawnflags)
696 {
697 pev->spawnflags = 0;
698 return GetScheduleOfType(SCHED_RELOAD);
699 }
700 */
701
702 if (HasConditions(bits_COND_HEAR_SOUND)) // Check if the entity hears a sound
703 {
704 CSound *pSound;
705 pSound = PBestSound(); // Get the best sound
706
707 ASSERT(pSound != NULL); // Ensure the sound is valid
708
709 if (pSound && (pSound->m_iType & bits_SOUND_DANGER))
710 return GetScheduleOfType(SCHED_TAKE_COVER_FROM_BEST_SOUND); // Take cover from danger sound
711 if (pSound->m_iType & bits_SOUND_COMBAT)
712 m_afMemory |= bits_MEMORY_PROVOKED; // Mark as provoked if combat sound is heard
713 }
714
715 switch (m_MonsterState) // Evaluate the monster's state
716 {
717 case MONSTERSTATE_COMBAT:
718 // Check for dead enemy
719 if (HasConditions(bits_COND_ENEMY_DEAD))
720 {
721 // Call base class to handle dead enemies
722 return CBaseMonster::GetSchedule();
723 }
724
725 if (pev->health < 20 || m_iBravery < 0) // Check health and bravery
726 {
727 if (!HasConditions(bits_COND_CAN_MELEE_ATTACK1))
728 {
729 m_failSchedule = SCHED_CHASE_ENEMY; // Set fail schedule to chase enemy
730 if (HasConditions(bits_COND_LIGHT_DAMAGE | bits_COND_HEAVY_DAMAGE))
731 {
732 return GetScheduleOfType(SCHED_TAKE_COVER_FROM_ENEMY); // Take cover from enemy
733 }
734 if (HasConditions(bits_COND_SEE_ENEMY) && HasConditions(bits_COND_ENEMY_FACING_ME))
735 {
736 // ALERT( at_console, "exposed\n");
737 return GetScheduleOfType(SCHED_TAKE_COVER_FROM_ENEMY); // Take cover if exposed
738 }
739 }
740 }
741 break;
742 }
743 return CSquadMonster::GetSchedule(); // Default to base class schedule
744}

779void CISlave::ArmBeam(int side)
780{
781 TraceResult tr;
782 float flDist = 1.0;
783
784 if (m_iBeams >= ISLAVE_MAX_BEAMS)
785 return; // Limit reached
786
787 UTIL_MakeAimVectors(pev->angles); // Set aim vectors
788 Vector vecSrc = pev->origin + gpGlobals->v_up * 36 + gpGlobals->v_right * side * 16 + gpGlobals->v_forward * 32; // Calculate source position
789
790 for (int i = 0; i < 3; i++)
791 {
792 Vector vecAim = gpGlobals->v_right * side * RANDOM_FLOAT(0, 1) + gpGlobals->v_up * RANDOM_FLOAT(-1, 1); // Random aim direction
793 TraceResult tr1;
794 UTIL_TraceLine(vecSrc, vecSrc + vecAim * 512, dont_ignore_monsters, ENT(pev), &tr1); // Trace line to find geometry
795 if (flDist > tr1.flFraction)
796 {
797 tr = tr1; // Update trace result
798 flDist = tr.flFraction; // Update distance
799 }
800 }
801
802 // Couldn't find anything close enough
803 if (flDist == 1.0)
804 return; // No geometry found
805
806 DecalGunshot(&tr, BULLET_PLAYER_CROWBAR); // Apply gunshot decal
807
808 m_pBeam[m_iBeams] = CBeam::BeamCreate("sprites/lgtning.spr", 30); // Create beam
