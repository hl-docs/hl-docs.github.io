69void VectorAngles( const float *forward, float *angles )
70{
71 float tmp, yaw, pitch;
72
73 if (forward[1] == 0 && forward[0] == 0)
74 {
75 yaw = 0;
76 if (forward[2] > 0)
77 pitch = 90;
78 else
79 pitch = 270;
80 }
81 else
82 {
83 yaw = (atan2(forward[1], forward[0]) * 180 / M_PI);
84 if (yaw < 0)
85 yaw += 360;
86
87 tmp = sqrt (forward[0]*forward[0] + forward[1]*forward[1]);
88 pitch = (atan2(forward[2], tmp) * 180 / M_PI);
89 if (pitch < 0)
90 pitch += 360;
91 }
92
93 angles[0] = pitch;
94 angles[1] = yaw;
95 angles[2] = 0;
96}
