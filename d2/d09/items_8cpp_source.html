100 pEntity->pev->spawnflags = pev->spawnflags;
101 }
102

142void CItem::ItemTouch(CBaseEntity *pOther)
143{
144 // If it's not a player, ignore
145 if (!pOther->IsPlayer())
146 {
147 return; // Exit if the touching entity is not a player
148 }
149
150 CBasePlayer *pPlayer = (CBasePlayer *)pOther; // Cast to player entity
151
152 // Check if the player can have the item
153 if (!g_pGameRules->CanHaveItem(pPlayer, this))
154 {
155 // No? Ignore the touch.
156 return; // Exit if the player cannot have the item
157 }
158
159 if (MyTouch(pPlayer)) // Check if the player can pick up the item
160 {
161 SUB_UseTargets(pOther, USE_TOGGLE, 0); // Trigger any use targets associated with the item
162 SetTouch(NULL); // Remove the touch function
163
164 // Player grabbed the item.
165 g_pGameRules->PlayerGotItem(pPlayer, this); // Notify game rules that the player got the item
166 if (g_pGameRules->ItemShouldRespawn(this) == GR_ITEM_RESPAWN_YES)
167 {
168 Respawn(); // Respawn the item if it should
169 }
170 else
171 {
172 UTIL_Remove(this); // Remove the item if it should not respawn
173 }
174 }
175 else if (gEvilImpulse101) // Check for a specific game condition
176 {
177 UTIL_Remove(this); // Remove the item if the condition is met
178 }
179}

190CBaseEntity* CItem::Respawn(void)
191{
192 SetTouch(NULL); // Remove the touch function
193 pev->effects |= EF_NODRAW; // Set the item to be invisible
194
195 // Move the item to the respawn location
196 UTIL_SetOrigin(pev, g_pGameRules->VecItemRespawnSpot(this));
197
198 SetThink(&CItem::Materialize); // Set the think function to materialize the item
199 pev->nextthink = g_pGameRules->FlItemRespawnTime(this); // Set the next think time based on respawn rules
200 return this; // Return the respawned item
201}

210void CItem::Materialize(void)
211{
212 if (pev->effects & EF_NODRAW) // Check if the item is currently invisible

216 pev->effects &= ~EF_NODRAW; // Remove the invisible effect
217 pev->effects |= EF_MUZZLEFLASH; // Set muzzle flash effect
218 }
219
220 SetTouch(&CItem::ItemTouch); // Set the touch function to allow interaction
221}

281class CItemBattery : public CItem
282{
290 void Spawn(void)
291 {
292 Precache(); // Preload resources
293 SET_MODEL(ENT(pev), "models/w_battery.mdl"); // Set the model for the battery item
294 CItem::Spawn(); // Call the base class spawn method
295 }
303 void Precache(void)
304 {
305 PRECACHE_MODEL("models/w_battery.mdl"); // Precache the battery model
306 PRECACHE_SOUND("items/gunpickup2.wav"); // Precache the sound for picking up the battery
307 }
319 BOOL MyTouch(CBasePlayer *pPlayer)
320 {
321 if (pPlayer->pev->deadflag != DEAD_NO) // Check if the player is alive
322 {
323 return FALSE; // Exit if the player is dead
324 }
325
326 // Check if the player can receive more armor and has the suit
327 if ((pPlayer->pev->armorvalue < MAX_NORMAL_BATTERY) &&
328 (pPlayer->pev->weapons & (1 << WEAPON_SUIT)))
329 {
330 int pct; // Percentage of battery charge
331 char szcharge[64]; // String to hold the suit update message
332
333 // Increase the player's armor value
334 pPlayer->pev->armorvalue += gSkillData.batteryCapacity;
335 pPlayer->pev->armorvalue = min(pPlayer->pev->armorvalue, MAX_NORMAL_BATTERY); // Cap armor value
336
337 EMIT_SOUND(pPlayer->edict(), CHAN_ITEM, "items/gunpickup2.wav", 1, ATTN_NORM); // Play pickup sound
338
339 // Notify the player about the item pickup
340 MESSAGE_BEGIN(MSG_ONE, gmsgItemPickup, NULL, pPlayer->pev);
341 WRITE_STRING(STRING(pev->classname)); // Send the item name
342 MESSAGE_END();
343
344 // Suit reports new power level
345 pct = (int)((float)(pPlayer->pev->armorvalue * 100.0) * (1.0 / MAX_NORMAL_BATTERY) + 0.5); // Calculate percentage
346 pct = (pct / 5); // Round down to nearest 5
347 if (pct > 0)
348 pct--; // Adjust percentage for display
349
350 sprintf(szcharge, "!HEV_%1dP", pct); // Create the suit update message
351
352 pPlayer->SetSuitUpdate(szcharge, FALSE, SUIT_NEXT_IN_30SEC); // Update the suit with the new charge level
353 return TRUE; // Indicate success
354 }
355 return FALSE; // Indicate failure if the player cannot pick up the item
356 }
357};
