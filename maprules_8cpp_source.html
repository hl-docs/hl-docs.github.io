503void CGameTeamMaster::Use(CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
504{
505 if (!CanFireForActivator(pActivator)) // Check if the activator can trigger the action
506 return; // Exit if not allowed
507
508 if (useType == USE_SET) // Check if the use type is SET
509 {
510 if (value < 0) // Check if the value is negative
511 {
512 m_teamIndex = -1; // Set team index to "no team"
513 }
514 else
515 {
516 m_teamIndex = g_pGameRules->GetTeamIndex(pActivator->TeamID()); // Set team index based on activator's team
517 }
518 return; // Exit after setting the team index
519 }
520
521 if (TeamMatch(pActivator)) // Check if the activator's team matches
522 {
523 SUB_UseTargets(pActivator, triggerType, value); // Trigger the targets
524 if (RemoveOnFire()) // Check if the entity should be removed after firing
525 UTIL_Remove(this); // Remove the entity
526 }

709void CGamePlayerZone::Use(CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
710{
711 int playersInCount = 0; // Initialize count of players inside the zone
712 int playersOutCount = 0; // Initialize count of players outside the zone
713
714 if (!CanFireForActivator(pActivator)) // Check if the activator can trigger the action
715 return; // Exit if not allowed
716
717 CBaseEntity *pPlayer = NULL; // Initialize player pointer
718
719 // Loop through all possible player indices
720 for (int i = 1; i <= gpGlobals->maxClients; i++)
721 {
722 pPlayer = UTIL_PlayerByIndex(i); // Get the player by index
723 if (pPlayer) // Check if the player exists
724 {
725 TraceResult trace; // Initialize trace result
726 int hullNumber; // Initialize hull number
727
728 hullNumber = human_hull; // Set hull number for human
729 if (pPlayer->pev->flags & FL_DUCKING) // Check if the player is ducking
730 {
731 hullNumber = head_hull; // Set hull number for ducking
732 }
733
734 // Trace the model to check if the player is inside the zone
735 UTIL_TraceModel(pPlayer->pev->origin, pPlayer->pev->origin, hullNumber, edict(), &trace);
736
737 if (trace.fStartSolid) // Check if the trace hit a solid object
738 {
739 playersInCount++; // Increment count of players inside
740 if (m_iszInTarget) // Check if there is an in-target
741 {
742 FireTargets(STRING(m_iszInTarget), pPlayer, pActivator, useType, value); // Fire in-target
743 }
744 }
745 else // Player is outside the zone
746 {
747 playersOutCount++; // Increment count of players outside
748 if (m_iszOutTarget) // Check if there is an out-target
749 {
750 FireTargets(STRING(m_iszOutTarget), pPlayer, pActivator, useType, value); // Fire out-target
751 }
752 }
753 }
754 }
755
756 // Fire the in-count target if set
757 if (m_iszInCount)
758 {
759 FireTargets(STRING(m_iszInCount), pActivator, this, USE_SET, playersInCount); // Fire in-count target
760 }
761
762 // Fire the out-count target if set
763 if (m_iszOutCount)
764 {
765 FireTargets(STRING(m_iszOutCount), pActivator, this, USE_SET, playersOutCount); // Fire out-count target
766 }
767}
