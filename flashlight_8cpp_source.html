151int CHudFlashlight::Draw(float flTime)
152{
153 if ( gHUD.m_iHideHUDDisplay & ( HIDEHUD_FLASHLIGHT | HIDEHUD_ALL ) )
154 return 1;
155
156 int r, g, b, x, y, a;
157 wrect_t rc;
158
159 if (!(gHUD.m_iWeaponBits & (1<<(WEAPON_SUIT)) ))
160 return 1;
161
162 if (m_fOn)
163 a = 225;
164 else
165 a = MIN_ALPHA;
166
167 if (m_flBat < 0.20)
168 UnpackRGB(r, g, b, RGB_REDISH);
169 else
170 UnpackRGB(r, g, b, RGB_YELLOWISH);
171
172 ScaleColors(r, g, b, a);
173
174 y = (m_prc1->bottom - m_prc2->top) / 2;
175 x = ScreenWidth - m_iWidth - m_iWidth / 2;
176
177 // Draw the flashlight casing
178 SPR_Set(m_hSprite1, r, g, b);
179 SPR_DrawAdditive(0, x, y, m_prc1);
180
181 if (m_fOn)
182 {
183 // Draw the flashlight beam
184 x = ScreenWidth - m_iWidth / 2;
185
186 SPR_Set(m_hBeam, r, g, b);
187 SPR_DrawAdditive(0, x, y, m_prcBeam);
188 }
189
190 // Draw the flashlight energy level
191 x = ScreenWidth - m_iWidth - m_iWidth / 2;
192 int iOffset = m_iWidth * (1.0 - m_flBat);
193 if (iOffset < m_iWidth)
194 {
195 rc = *m_prc2;
196 rc.left += iOffset;
197
198 SPR_Set(m_hSprite2, r, g, b);
199 SPR_DrawAdditive(0, x + iOffset, y, &rc);
200 }
201
202 return 1;
203}
