145void CXenPLight::Think(void)
146{
147 StudioFrameAdvance(); // Advance the animation frame
148 pev->nextthink = gpGlobals->time + 0.1; // Set the next think time
149
150 switch (GetActivity()) // Check the current activity
151 {
152 case ACT_CROUCH:
153 if (m_fSequenceFinished) // Check if the crouch sequence is finished
154 {
155 SetActivity(ACT_CROUCHIDLE); // Set to crouch idle
156 LightOff(); // Turn off the light
157 }
158 break;
159
160 case ACT_CROUCHIDLE:
161 if (gpGlobals->time > pev->dmgtime) // Check if the damage time has elapsed
162 {
163 SetActivity(ACT_STAND); // Set to stand activity
164 LightOn(); // Turn on the light
165 }
166 break;
167
168 case ACT_STAND:
169 if (m_fSequenceFinished) // Check if the stand sequence is finished
170 SetActivity(ACT_IDLE); // Set to idle activity
171 break;
172
173 case ACT_IDLE:
174 default:
175 break; // Default case does nothing
176 }
177}

475void CXenTree::HandleAnimEvent(MonsterEvent_t *pEvent)
476{
477 switch (pEvent->event)
478 {
479 case TREE_AE_ATTACK: // Handle attack event
480 {
481 CBaseEntity *pList[8]; // Array to hold entities in the trigger area
482 BOOL sound = FALSE; // Flag to indicate if a sound should be played
483 int count = UTIL_EntitiesInBox(pList, 8, m_pTrigger->pev->absmin, m_pTrigger->pev->absmax, FL_MONSTER | FL_CLIENT); // Get entities in the trigger area
484 Vector forward; // Forward vector
485
486 UTIL_MakeVectorsPrivate(pev->angles, forward, NULL, NULL); // Calculate the forward vector
487
488 for (int i = 0; i < count; i++) // Loop through the entities
489 {
490 if (pList[i] != this) // Ignore self
491 {
492 if (pList[i]->pev->owner != edict()) // Check if the entity is not the owner
493 {
494 sound = TRUE; // Set sound flag
495 pList[i]->TakeDamage(pev, pev, 25, DMG_CRUSH | DMG_SLASH); // Deal damage to the entity
496 pList[i]->pev->punchangle.x = 15; // Apply a punch angle
497 pList[i]->pev->velocity = pList[i]->pev->velocity + forward * 100; // Apply force to the entity
498 }
499 }
500 }
501
502 if (sound) // If any sound was triggered
503 {
504 EMIT_SOUND_ARRAY_DYN(CHAN_WEAPON, pAttackHitSounds); // Play hit sounds
505 }
506 }
507 return; // Exit the function
508 }
509
510 CActAnimating::HandleAnimEvent(pEvent); // Call the base class event handler
511}
