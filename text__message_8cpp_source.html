66char *CHudTextMessage::LocaliseTextString(const char *msg, char *dst_buffer, int buffer_size)
67{
68 char *dst = dst_buffer; // Pointer to the destination buffer
69 for (char *src = (char*)msg; *src != 0 && buffer_size > 0; buffer_size--)
70 {
71 if (*src == '#') // Check for placeholder character '#'
72 {
73 // Cut message name out of string
74 static char word_buf[255];
75 char *wdst = word_buf, *word_start = src;
76 for (++src; (*src >= 'A' && *src <= 'z') || (*src >= '0' && *src <= '9'); wdst++, src++)
77 {
78 *wdst = *src;
79 }
80 *wdst = 0; // Null-terminate the word buffer
81
82 // Lookup the message name in titles.txt
83 client_textmessage_t *clmsg = TextMessageGet(word_buf);
84 if (!clmsg || !(clmsg->pMessage))
85 {
86 src = word_start;
87 *dst = *src;
88 dst++, src++;
89 continue;
90 }
91
92 // Copy the message into the destination buffer
93 for (char *wsrc = (char*)clmsg->pMessage; *wsrc != 0; wsrc++, dst++)
94 {
95 *dst = *wsrc;
96 }
97 *dst = 0;
98 }
99 else
100 {
101 *dst = *src;
102 dst++, src++;
103 *dst = 0;
104 }
105 }
106
107 dst_buffer[buffer_size - 1] = 0; // Ensure null termination
108 return dst_buffer;
109}

142char *CHudTextMessage::LookupString(const char *msg, int *msg_dest)
143{
144 if (!msg)
145 return "";
146
147 // '#' character indicates this is a reference to a string in titles.txt
148 if (msg[0] == '#')
149 {
150 // Look up the real message in titles.txt
151 client_textmessage_t *clmsg = TextMessageGet(msg + 1);
152
153 if (!clmsg || !(clmsg->pMessage))
154 return (char*)msg; // If lookup fails, return the original string
155
156 if (msg_dest)
157 {
158 // Check if titles.txt info overrides message destination
159 if (clmsg->effect < 0)
160 *msg_dest = -clmsg->effect; // Set the message destination if applicable
161 }
162
163 return (char*)clmsg->pMessage; // Return the localized message
164 }
165 else
166 {
167 // No special processing, return the original message
168 return (char*)msg;
169 }
170}

218int CHudTextMessage::MsgFunc_TextMsg(const char *pszName, int iSize, void *pbuf)
219{
220 BEGIN_READ(pbuf, iSize);
221
222 int msg_dest = READ_BYTE(); // Read the message destination (where to print the message)
223
224#define MSG_BUF_SIZE 128
225 static char szBuf[6][MSG_BUF_SIZE]; // Buffer for message formatting
226 char *msg_text = LookupString(READ_STRING(), &msg_dest);
227 msg_text = safe_strcpy(szBuf[0], msg_text, MSG_BUF_SIZE);
228
229 // Keep reading additional parameters for the message
230 char *sstr1 = LookupString(READ_STRING());
231 sstr1 = safe_strcpy(szBuf[1], sstr1, MSG_BUF_SIZE);
232 StripEndNewlineFromString(sstr1); // Remove any trailing newlines
233
234 char *sstr2 = LookupString(READ_STRING());
235 sstr2 = safe_strcpy(szBuf[2], sstr2, MSG_BUF_SIZE);
236 StripEndNewlineFromString(sstr2);
237
238 char *sstr3 = LookupString(READ_STRING());
239 sstr3 = safe_strcpy(szBuf[3], sstr3, MSG_BUF_SIZE);
240 StripEndNewlineFromString(sstr3);
241
242 char *sstr4 = LookupString(READ_STRING());
243 sstr4 = safe_strcpy(szBuf[4], sstr4, MSG_BUF_SIZE);
244 StripEndNewlineFromString(sstr4);
245
246 char *psz = szBuf[5];
247
248 if (gViewPort && gViewPort->AllowedToPrintText() == FALSE)
249 return 1; // Don't print if not allowed
250
251 // Print the message based on its destination
252 switch (msg_dest)
253 {
254 case HUD_PRINTCENTER:
255 safe_sprintf(psz, MSG_BUF_SIZE, msg_text, sstr1, sstr2, sstr3, sstr4);
256 CenterPrint(ConvertCRtoNL(psz)); // Display in the center of the screen
257 break;
258
259 case HUD_PRINTNOTIFY:
260 psz[0] = 1; // Mark as a notification message
261 safe_sprintf(psz + 1, MSG_BUF_SIZE, msg_text, sstr1, sstr2, sstr3, sstr4);
262 ConsolePrint(ConvertCRtoNL(psz)); // Display in the console
263 break;
264
265 case HUD_PRINTTALK:
266 safe_sprintf(psz, MSG_BUF_SIZE, msg_text, sstr1, sstr2, sstr3, sstr4);
267 gHUD.m_SayText.SayTextPrint(ConvertCRtoNL(psz), 128); // Display in the chat box
268 break;
269
270 case HUD_PRINTCONSOLE:
271 safe_sprintf(psz, MSG_BUF_SIZE, msg_text, sstr1, sstr2, sstr3, sstr4);
272 ConsolePrint(ConvertCRtoNL(psz)); // Display in the console
273 break;
274 }
275
276 return 1;
277}
