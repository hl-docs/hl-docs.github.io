74void SpectatorPanel::ActionSignal(int cmd)
75{
76 switch (cmd)
77 {
78 case SPECTATOR_PANEL_CMD_NONE : break;
79
80 case SPECTATOR_PANEL_CMD_OPTIONS : gViewPort->ShowCommandMenu( gViewPort->m_SpectatorOptionsMenu );

83 case SPECTATOR_PANEL_CMD_NEXTPLAYER : gHUD.m_Spectator.FindNextPlayer(true);

86 case SPECTATOR_PANEL_CMD_PREVPLAYER : gHUD.m_Spectator.FindNextPlayer(false);
87 break;

89 case SPECTATOR_PANEL_CMD_PLAYERS : gViewPort->ShowCommandMenu( gViewPort->m_PlayerMenu );

92 case SPECTATOR_PANEL_CMD_HIDEMENU : ShowMenu(false);
93 break;

95 case SPECTATOR_PANEL_CMD_CAMERA : gViewPort->ShowCommandMenu( gViewPort->m_SpectatorCameraMenu );
96 break;
97
98 case SPECTATOR_PANEL_CMD_TOGGLE_INSET : gHUD.m_Spectator.SetModes( -1,
99 gHUD.m_Spectator.ToggleInset(false) );
100 break;
101
102
