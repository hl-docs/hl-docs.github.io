165void CAmbientGeneric::Spawn(void)
166{
167 /*
168 -1 : "Default"
169 0 : "Everywhere"
170 200 : "Small Radius"
171 125 : "Medium Radius"
172 80 : "Large Radius"
173 */
174
175 if (FBitSet(pev->spawnflags, AMBIENT_SOUND_EVERYWHERE))
176 {
177 m_flAttenuation = ATTN_NONE;
178 }
179 else if (FBitSet(pev->spawnflags, AMBIENT_SOUND_SMALLRADIUS))
180 {
181 m_flAttenuation = ATTN_IDLE;
182 }
183 else if (FBitSet(pev->spawnflags, AMBIENT_SOUND_MEDIUMRADIUS))
184 {
185 m_flAttenuation = ATTN_STATIC;
186 }
187 else if (FBitSet(pev->spawnflags, AMBIENT_SOUND_LARGERADIUS))
188 {
189 m_flAttenuation = ATTN_NORM;
190 }
191 else
192 {
193 // if the designer didn't set a sound attenuation, default to one.
194 m_flAttenuation = ATTN_STATIC;
195 }
196
197 char* szSoundFile = (char*)STRING(pev->message);
198
199 if (FStringNull(pev->message) || strlen(szSoundFile) < 1)
200 {
201 ALERT(at_error, "EMPTY AMBIENT AT: %f, %f, %f\n", pev->origin.x, pev->origin.y, pev->origin.z);
202 pev->nextthink = gpGlobals->time + 0.1;
203 SetThink(&CAmbientGeneric::SUB_Remove);
204 return;
205 }
206 pev->solid = SOLID_NOT;
207 pev->movetype = MOVETYPE_NONE;
208
209 // Set up think function for dynamic modification
210 // of ambient sound's pitch or volume. Don't
211 // start thinking yet.
212
213 SetThink(&CAmbientGeneric::RampThink);
214 pev->nextthink = 0;
215
216 // allow on/off switching via 'use' function.
217
218 SetUse(&CAmbientGeneric::ToggleUse);
219
220 m_fActive = FALSE;
221
222 if (FBitSet(pev->spawnflags, AMBIENT_SOUND_NOT_LOOPING))
223 m_fLooping = FALSE;
224 else
225 m_fLooping = TRUE;
226 Precache();
227}

234void CAmbientGeneric::Precache(void)
235{
236 char* szSoundFile = (char*)STRING(pev->message);
237
238 if (!FStringNull(pev->message) && strlen(szSoundFile) > 1)
239 {
240 if (*szSoundFile != '!')
241 PRECACHE_SOUND(szSoundFile);
242 }
243 // init all dynamic modulation parms
244 InitModulationParms();
245
246 if (!FBitSet(pev->spawnflags, AMBIENT_SOUND_START_SILENT))
247 {
248 // start the sound ASAP
249 if (m_fLooping)
250 m_fActive = TRUE;
251 }
252 if (m_fActive)
253 {
254 UTIL_EmitAmbientSound(ENT(pev), pev->origin, szSoundFile,
255 (m_dpv.vol * 0.01), m_flAttenuation, SND_SPAWNING, m_dpv.pitch);
256
257 pev->nextthink = gpGlobals->time + 0.1;
258 }

266void CAmbientGeneric::RampThink(void)
267{
268 char* szSoundFile = (char*)STRING(pev->message);
269 int pitch = m_dpv.pitch;
270 int vol = m_dpv.vol;
271 int flags = 0;
272 int fChanged = 0; // FALSE if pitch and vol remain unchanged this round
273 int prev;
274
275 if (!m_dpv.spinup && !m_dpv.spindown && !m_dpv.fadein && !m_dpv.fadeout && !m_dpv.lfotype)
276 return; // no ramps or lfo, stop thinking
277
278 // ==============
279 // pitch envelope
280 // ==============
281 if (m_dpv.spinup || m_dpv.spindown)
282 {
283 prev = m_dpv.pitchfrac >> 8;
284
285 if (m_dpv.spinup > 0)
286 m_dpv.pitchfrac += m_dpv.spinup;
287 else if (m_dpv.spindown > 0)
288 m_dpv.pitchfrac -= m_dpv.spindown;
289
290 pitch = m_dpv.pitchfrac >> 8;
291
292 if (pitch > m_dpv.pitchrun)
293 {
294 pitch = m_dpv.pitchrun;
295 m_dpv.spinup = 0; // done with ramp up
296 }
297
298 if (pitch < m_dpv.pitchstart)
299 {
300 pitch = m_dpv.pitchstart;
301 m_dpv.spindown = 0; // done with ramp down
302
303 // shut sound off
304 UTIL_EmitAmbientSound(ENT(pev), pev->origin, szSoundFile,
305 0, 0, SND_STOP, 0);
306
307 // return without setting nextthink
308 return;
309 }
310
311 if (pitch > 255) pitch = 255;
312 if (pitch < 1) pitch = 1;
313
314 m_dpv.pitch = pitch;
315
316 fChanged |= (prev != pitch);
317 flags |= SND_CHANGE_PITCH;
318 }
319
320 // ==================
321 // amplitude envelope
322 // ==================
323 if (m_dpv.fadein || m_dpv.fadeout)
324 {
325 prev = m_dpv.volfrac >> 8;
326
327 if (m_dpv.fadein > 0)
328 m_dpv.volfrac += m_dpv.fadein;
329 else if (m_dpv.fadeout > 0)
330 m_dpv.volfrac -= m_dpv.fadeout;
331
332 vol = m_dpv.volfrac >> 8;
333
334 if (vol > m_dpv.volrun)
335 {
336 vol = m_dpv.volrun;
337 m_dpv.fadein = 0; // done with ramp up
338 }
339
340 if (vol < m_dpv.volstart)
341 {
342 vol = m_dpv.volstart;
343 m_dpv.fadeout = 0; // done with ramp down
344
345 // shut sound off
346 UTIL_EmitAmbientSound(ENT(pev), pev->origin, szSoundFile,
347 0, 0, SND_STOP, 0);
348
349 // return without setting nextthink
350 return;
351 }
352
353 if (vol > 100) vol = 100;
354 if (vol < 1) vol = 1;
355
356 m_dpv.vol = vol;
357
358 fChanged |= (prev != vol);
359 flags |= SND_CHANGE_VOL;
360 }
361
362 // ===================
363 // pitch/amplitude LFO
364 // ===================
365 if (m_dpv.lfotype)
366 {
367 int pos;
368
369 if (m_dpv.lfofrac > 0x6fffffff)
370 m_dpv.lfofrac = 0;
371
372 // update lfo, lfofrac/255 makes a triangle wave 0-255
373 m_dpv.lfofrac += m_dpv.lforate;
374 pos = m_dpv.lfofrac >> 8;
375
376 if (m_dpv.lfofrac < 0)
377 {
378 m_dpv.lfofrac = 0;
379 m_dpv.lforate = abs(m_dpv.lforate);
380 pos = 0;
381 }
382 else if (pos > 255)
383 {
384 pos = 255;
385 m_dpv.lfofrac = (255 << 8);
386 m_dpv.lforate = -abs(m_dpv.lforate);
387 }
388
389 switch (m_dpv.lfotype)
390 {
391 case LFO_SQUARE:
392 if (pos < 128)
393 m_dpv.lfomult = 255;
394 else
395 m_dpv.lfomult = 0;
396
397 break;
398 case LFO_RANDOM:
399 if (pos == 255)
400 m_dpv.lfomult = RANDOM_LONG(0, 255);
401 break;
402 case LFO_TRIANGLE:
403 default:
404 m_dpv.lfomult = pos;
405 break;
406 }
407
408 if (m_dpv.lfomodpitch)
409 {
410 prev = pitch;
411
412 // pitch 0-255
413 pitch += ((m_dpv.lfomult - 128) * m_dpv.lfomodpitch) / 100;
414
415 if (pitch > 255) pitch = 255;
416 if (pitch < 1) pitch = 1;
417
418 fChanged |= (prev != pitch);
419 flags |= SND_CHANGE_PITCH;
420 }
421
422 if (m_dpv.lfomodvol)
423 {
424 // vol 0-100
425 prev = vol;
426
427 vol += ((m_dpv.lfomult - 128) * m_dpv.lfomodvol) / 100;
428
429 if (vol > 100) vol = 100;
430 if (vol < 0) vol = 0;
431
432 fChanged |= (prev != vol);
433 flags |= SND_CHANGE_VOL;
434 }
435
436 }
437
438 // Send update to playing sound only if we actually changed
439 // pitch or volume in this routine.
440
441 if (flags && fChanged)
442 {
443 if (pitch == PITCH_NORM)
444 pitch = PITCH_NORM + 1; // don't send 'no pitch' !
445
446 UTIL_EmitAmbientSound(ENT(pev), pev->origin, szSoundFile,
447 (vol * 0.01), m_flAttenuation, flags, pitch);
448 }
449
450 // update ramps at 5hz
451 pev->nextthink = gpGlobals->time + 0.2;
452 return;
453}

460void CAmbientGeneric::InitModulationParms(void)
461{
462 int pitchinc;
463
464 m_dpv.volrun = pev->health * 10; // 0 - 100
465 if (m_dpv.volrun > 100) m_dpv.volrun = 100;
466 if (m_dpv.volrun < 0) m_dpv.volrun = 0;
467
468 // get presets
469 if (m_dpv.preset != 0 && m_dpv.preset <= CDPVPRESETMAX)
470 {
471 // load preset values
472 m_dpv = rgdpvpreset[m_dpv.preset - 1];
473
474 // fixup preset values, just like
475 // fixups in KeyValue routine.
476 if (m_dpv.spindown > 0)
477 m_dpv.spindown = (101 - m_dpv.spindown) * 64;
478 if (m_dpv.spinup > 0)
479 m_dpv.spinup = (101 - m_dpv.spinup) * 64;
480
481 m_dpv.volstart *= 10;
482 m_dpv.volrun *= 10;
483
484 if (m_dpv.fadein > 0)
485 m_dpv.fadein = (101 - m_dpv.fadein) * 64;
486 if (m_dpv.fadeout > 0)
487 m_dpv.fadeout = (101 - m_dpv.fadeout) * 64;
488
489 m_dpv.lforate *= 256;
490
491 m_dpv.fadeinsav = m_dpv.fadein;
492 m_dpv.fadeoutsav = m_dpv.fadeout;
493 m_dpv.spinupsav = m_dpv.spinup;
494 m_dpv.spindownsav = m_dpv.spindown;
495 }
496
497 m_dpv.fadein = m_dpv.fadeinsav;
498 m_dpv.fadeout = 0;
499
500 if (m_dpv.fadein)
501 m_dpv.vol = m_dpv.volstart;
502 else
503 m_dpv.vol = m_dpv.volrun;
504
505 m_dpv.spinup = m_dpv.spinupsav;
506 m_dpv.spindown = 0;
507
508 if (m_dpv.spinup)
509 m_dpv.pitch = m_dpv.pitchstart;
510 else
511 m_dpv.pitch = m_dpv.pitchrun;
512
513 if (m_dpv.pitch == 0)
514 m_dpv.pitch = PITCH_NORM;
515
516 m_dpv.pitchfrac = m_dpv.pitch << 8;
517 m_dpv.volfrac = m_dpv.vol << 8;
518
519 m_dpv.lfofrac = 0;
520 m_dpv.lforate = abs(m_dpv.lforate);
521
522 m_dpv.cspincount = 1;
523
524 if (m_dpv.cspinup)
525 {
526 pitchinc = (255 - m_dpv.pitchstart) / m_dpv.cspinup;
527
528 m_dpv.pitchrun = m_dpv.pitchstart + pitchinc;
529 if (m_dpv.pitchrun > 255) m_dpv.pitchrun = 255;
530 }
531
532 if ((m_dpv.spinupsav || m_dpv.spindownsav || (m_dpv.lfotype && m_dpv.lfomodpitch))
533 && (m_dpv.pitch == PITCH_NORM))
534 m_dpv.pitch = PITCH_NORM + 1; // must never send 'no pitch' as first pitch
535 // if we intend to pitch shift later!
536}

548void CAmbientGeneric::ToggleUse(CBaseEntity* pActivator, CBaseEntity* pCaller, USE_TYPE useType, float value)
549{
550 char* szSoundFile = (char*)STRING(pev->message);
551 float fraction;
552
553 if (useType != USE_TOGGLE)
554 {
555 if ((m_fActive && useType == USE_ON) || (!m_fActive && useType == USE_OFF))
556 return;
557 }
558 // Directly change pitch if arg passed. Only works if sound is already playing.
559
560 if (useType == USE_SET && m_fActive) // Momentary buttons will pass down a float in here
561 {
562
563 fraction = value;
564
565 if (fraction > 1.0)
566 fraction = 1.0;
567 if (fraction < 0.0)
568 fraction = 0.01;
569
570 m_dpv.pitch = fraction * 255;
571
572 UTIL_EmitAmbientSound(ENT(pev), pev->origin, szSoundFile,
573 0, 0, SND_CHANGE_PITCH, m_dpv.pitch);
574
575 return;
576 }
577
578 // Toggle
579
580 // m_fActive is TRUE only if a looping sound is playing.
581
582 if (m_fActive)
583 {// turn sound off
584
585 if (m_dpv.cspinup)
586 {
587 // Don't actually shut off. Each toggle causes
588 // incremental spinup to max pitch
589
590 if (m_dpv.cspincount <= m_dpv.cspinup)
591 {
592 int pitchinc;
593
594 // start a new spinup
595 m_dpv.cspincount++;
596
597 pitchinc = (255 - m_dpv.pitchstart) / m_dpv.cspinup;
598
599 m_dpv.spinup = m_dpv.spinupsav;
600 m_dpv.spindown = 0;
601
602 m_dpv.pitchrun = m_dpv.pitchstart + pitchinc * m_dpv.cspincount;
603 if (m_dpv.pitchrun > 255) m_dpv.pitchrun = 255;
604
605 pev->nextthink = gpGlobals->time + 0.1;
606 }
607
608 }
609 else
610 {
611 m_fActive = FALSE;
612
613 // HACKHACK - this makes the code in Precache() work properly after a save/restore
614 pev->spawnflags |= AMBIENT_SOUND_START_SILENT;
615
616 if (m_dpv.spindownsav || m_dpv.fadeoutsav)
617 {
618 // spin it down (or fade it) before shutoff if spindown is set
619 m_dpv.spindown = m_dpv.spindownsav;
620 m_dpv.spinup = 0;
621
622 m_dpv.fadeout = m_dpv.fadeoutsav;
623 m_dpv.fadein = 0;
624 pev->nextthink = gpGlobals->time + 0.1;
625 }
626 else
627 UTIL_EmitAmbientSound(ENT(pev), pev->origin, szSoundFile,
628 0, 0, SND_STOP, 0);
629 }
630 }
631 else
632 {// turn sound on
633
634 // only toggle if this is a looping sound. If not looping, each
635 // trigger will cause the sound to play. If the sound is still
636 // playing from a previous trigger press, it will be shut off
637 // and then restarted.
638
639 if (m_fLooping)
640 m_fActive = TRUE;
641 else
642 // shut sound off now - may be interrupting a long non-looping sound
643 UTIL_EmitAmbientSound(ENT(pev), pev->origin, szSoundFile,
644 0, 0, SND_STOP, 0);
645
646 // init all ramp params for startup
647
648 InitModulationParms();
649
650 UTIL_EmitAmbientSound(ENT(pev), pev->origin, szSoundFile,
651 (m_dpv.vol * 0.01), m_flAttenuation, 0, m_dpv.pitch);
652
653 pev->nextthink = gpGlobals->time + 0.1;
654
655 }
656}

665void CAmbientGeneric::KeyValue(KeyValueData* pkvd)
666{
667 // NOTE: changing any of the modifiers in this code
668 // NOTE: also requires changing InitModulationParms code.
669
670 // preset

878BOOL FEnvSoundInRange(entvars_t* pev, entvars_t* pevTarget, float* pflRange)
879{
880 CEnvSound* pSound = GetClassPtr((CEnvSound*)pev);
881 Vector vecSpot1 = pev->origin + pev->view_ofs;
882 Vector vecSpot2 = pevTarget->origin + pevTarget->view_ofs;
883 Vector vecRange;
884 float flRange;
885 TraceResult tr;
886
887 UTIL_TraceLine(vecSpot1, vecSpot2, ignore_monsters, ENT(pev), &tr);
888
889 // check if line of sight crosses water boundary, or is blocked
890 if ((tr.fInOpen && tr.fInWater) || tr.flFraction != 1)
891 return FALSE;
892
893 // calc range from sound entity to player
894 vecRange = tr.vecEndPos - vecSpot1;
895 flRange = vecRange.Length();
896
897 if (pSound->m_flRadius < flRange)
898 return FALSE;
899
900 if (pflRange)
901 *pflRange = flRange;
902
903 return TRUE;
904}

911void CEnvSound::Think(void)
912{
913 // get pointer to client if visible; FIND_CLIENT_IN_PVS will
914 // cycle through visible clients on consecutive calls.
915
916 edict_t* pentPlayer = FIND_CLIENT_IN_PVS(edict());
917 CBasePlayer* pPlayer = NULL;
918
919 if (FNullEnt(pentPlayer))
920 goto env_sound_Think_slow; // no player in pvs of sound entity, slow it down
921
922 pPlayer = GetClassPtr((CBasePlayer*)VARS(pentPlayer));
923 float flRange;
924
925 // check to see if this is the sound entity that is
926 // currently affecting this player
927
928 if (!FNullEnt(pPlayer->m_pentSndLast) && (pPlayer->m_pentSndLast == ENT(pev)))
929 {
930 // this is the entity currently affecting player, check
931 // for validity
932
933 if (pPlayer->m_flSndRoomtype != 0 && pPlayer->m_flSndRange != 0)
934 {
935 // we're looking at a valid sound entity affecting
936 // player, make sure it's still valid, update range
937
938 if (FEnvSoundInRange(pev, VARS(pentPlayer), &flRange))
939 {
940 pPlayer->m_flSndRange = flRange;
941 goto env_sound_Think_fast;
942 }
943 else
944 {
945 // current sound entity affecting player is no longer valid,
946 // flag this state by clearing room_type and range.
947 // NOTE: we do not actually change the player's room_type
948 // NOTE: until we have a new valid room_type to change it to.
949
950 pPlayer->m_flSndRange = 0;
951 pPlayer->m_flSndRoomtype = 0;
952 goto env_sound_Think_slow;
953 }
954 }
955 else
956 {
957 // entity is affecting player but is out of range,
958 // wait passively for another entity to usurp it...
959 goto env_sound_Think_slow;
960 }
961 }
962
963 // if we got this far, we're looking at an entity that is contending
964 // for current player sound. the closest entity to player wins.
965
966 if (FEnvSoundInRange(pev, VARS(pentPlayer), &flRange))
967 {
968 if (flRange < pPlayer->m_flSndRange || pPlayer->m_flSndRange == 0)
969 {
970 // new entity is closer to player, so it wins.
971 pPlayer->m_pentSndLast = ENT(pev);
972 pPlayer->m_flSndRoomtype = m_flRoomtype;
973 pPlayer->m_flSndRange = flRange;
974
975 // send room_type command to player's server.
976 // this should be a rare event - once per change of room_type
977 // only!
978
979 MESSAGE_BEGIN(MSG_ONE, SVC_ROOMTYPE, NULL, pentPlayer); // use the magic #1 for "one client"
980 WRITE_SHORT((short)m_flRoomtype); // sequence number
981 MESSAGE_END();
982
983 // crank up nextthink rate for new active sound entity
984 // by falling through to think_fast...
985 }
986 // player is not closer to the contending sound entity,
987 // just fall through to think_fast. this effectively
988 // cranks up the think_rate of entities near the player.
989 }
990
991 // player is in pvs of sound entity, but either not visible or
992 // not in range. do nothing, fall through to think_fast...
993
994env_sound_Think_fast:
995 pev->nextthink = gpGlobals->time + 0.25;
996 return;
997
998env_sound_Think_slow:
999 pev->nextthink = gpGlobals->time + 0.75;
1000 return;
1001}

1338void SENTENCEG_Init()
1339{
1340 char buffer[512];
1341 char szgroup[64];
1342 int i, j;
1343 int isentencegs;
1344
1345 if (fSentencesInit)
1346 return;
1347
1348 memset(gszallsentencenames, 0, CVOXFILESENTENCEMAX * CBSENTENCENAME_MAX);
1349 gcallsentences = 0;
1350
1351 memset(rgsentenceg, 0, CSENTENCEG_MAX * sizeof(SENTENCEG));
1352 memset(buffer, 0, 512);
1353 memset(szgroup, 0, 64);
1354 isentencegs = -1;
1355
1356 int filePos = 0, fileSize;
1357 byte *pMemFile = g_engfuncs.pfnLoadFileForMe("sound/sentences.txt", &fileSize);
1358 if (!pMemFile)
1359 return;
1360
1361 // for each line in the file...
1362 while (memfgets(pMemFile, fileSize, filePos, buffer, 511) != NULL)
1363 {
1364 // skip whitespace
1365 i = 0;
1366 while (buffer[i] && buffer[i] == ' ')
1367 i++;
1368
1369 if (!buffer[i])
1370 continue;
1371
1372 if (buffer[i] == '/' || !isalpha(buffer[i]))
1373 continue;
1374
1375 // get sentence name
1376 j = i;
1377 while (buffer[j] && buffer[j] != ' ')
1378 j++;
1379
1380 if (!buffer[j])
1381 continue;
1382
1383 if (gcallsentences > CVOXFILESENTENCEMAX)
1384 {
1385 ALERT(at_error, "Too many sentences in sentences.txt!\n");
1386 break;
1387 }
1388
1389 // null-terminate name and save in sentences array
1390 buffer[j] = 0;
1391 const char *pString = buffer + i;
1392
1393 if (strlen(pString) >= CBSENTENCENAME_MAX)
1394 ALERT(at_warning, "Sentence %s longer than %d letters\n", pString, CBSENTENCENAME_MAX - 1);
1395
1396 strcpy(gszallsentencenames[gcallsentences++], pString);
1397
1398 j--;
1399 if (j <= i)
1400 continue;
1401 if (!isdigit(buffer[j]))
1402 continue;
1403
1404 // cut out suffix numbers
1405 while (j > i && isdigit(buffer[j]))
1406 j--;
1407
1408 if (j <= i)
1409 continue;
1410
1411 buffer[j + 1] = 0;
1412
1413 // if new name doesn't match previous group name,
1414 // make a new group.
1415
1416 if (strcmp(szgroup, &(buffer[i])))
1417 {
1418 // name doesn't match with prev name,
1419 // copy name into group, init count to 1
1420 isentencegs++;
1421 if (isentencegs >= CSENTENCEG_MAX)
1422 {
1423 ALERT(at_error, "Too many sentence groups in sentences.txt!\n");
1424 break;
1425 }
1426
1427 strcpy(rgsentenceg[isentencegs].szgroupname, &(buffer[i]));
1428 rgsentenceg[isentencegs].count = 1;
1429
1430 strcpy(szgroup, &(buffer[i]));
1431
1432 continue;
1433 }
1434 else
1435 {
1436 //name matches with previous, increment group count
1437 if (isentencegs >= 0)
1438 rgsentenceg[isentencegs].count++;
1439 }
1440 }
1441
1442 g_engfuncs.pfnFreeFile(pMemFile);
1443
1444 fSentencesInit = TRUE;
1445
1446 // init lru lists
1447
1448 i = 0;
1449
1450 while (rgsentenceg[i].count && i < CSENTENCEG_MAX)
1451 {
1452 USENTENCEG_InitLRU(&(rgsentenceg[i].rgblru[0]), rgsentenceg[i].count);
1453 i++;
1454 }
1455}

1604static char* memfgets(byte* pMemFile, int fileSize, int& filePos, char* pBuffer, int bufferSize)
1605{
1606 // Bullet-proofing
1607 if (!pMemFile || !pBuffer)
1608 return NULL;
1609
1610 if (filePos >= fileSize)
1611 return NULL;
1612
1613 int i = filePos;
1614 int last = fileSize;
1615
1616 // fgets always NULL terminates, so only read bufferSize-1 characters
1617 if (last - filePos > (bufferSize - 1))
1618 last = filePos + (bufferSize - 1);
1619
1620 int stop = 0;
1621
1622 // Stop at the next newline (inclusive) or end of buffer
1623 while (i < last && !stop)
1624 {
1625 if (pMemFile[i] == '\n')
1626 stop = 1;
1627 i++;
1628 }
1629
1630 // If we actually advanced the pointer, copy it over
1631 if (i != filePos)
1632 {
1633 // We read in size bytes
1634 int size = i - filePos;
1635 // copy it out
1636 memcpy(pBuffer, pMemFile + filePos, sizeof(byte) * size);
1637
1638 // If the buffer isn't full, terminate (this is always true)
1639 if (size < bufferSize)
1640 pBuffer[size] = 0;
1641
1642 // Update file pointer
1643 filePos = i;
1644 return pBuffer;
1645 }
1646
1647 // No data read, bail
1648 return NULL;
