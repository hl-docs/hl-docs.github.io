395//=========================================================
396// GetDeathActivity - determines the best type of death
397// anim to play.
398//=========================================================
406Activity CBaseMonster :: GetDeathActivity ( void )
407{
408 Activity deathActivity;
409 BOOL fTriedDirection;
410 float flDot;
411 TraceResult tr;
412 Vector vecSrc;
413
414 if ( pev->deadflag != DEAD_NO )
415 {
416 // don't run this while dying.
417 return m_IdealActivity;
418 }
419
420 vecSrc = Center();
421
422 fTriedDirection = FALSE;
423 deathActivity = ACT_DIESIMPLE;// in case we can't find any special deaths to do.
424
425 UTIL_MakeVectors ( pev->angles );
426 flDot = DotProduct ( gpGlobals->v_forward, g_vecAttackDir * -1 );
427
428 switch ( m_LastHitGroup )
429 {
430 // try to pick a region-specific death.
431 case HITGROUP_HEAD:
432 deathActivity = ACT_DIE_HEADSHOT;
433 break;
434
435 case HITGROUP_STOMACH:
436 deathActivity = ACT_DIE_GUTSHOT;
437 break;
438
439 case HITGROUP_GENERIC:
440 // try to pick a death based on attack direction
441 fTriedDirection = TRUE;
442
443 if ( flDot > 0.3 )
444 {
445 deathActivity = ACT_DIEFORWARD;
446 }
447 else if ( flDot <= -0.3 )
448 {
449 deathActivity = ACT_DIEBACKWARD;
450 }
451 break;
452
453 default:
454 // try to pick a death based on attack direction
455 fTriedDirection = TRUE;
456
457 if ( flDot > 0.3 )
458 {
459 deathActivity = ACT_DIEFORWARD;
460 }
461 else if ( flDot <= -0.3 )
462 {
463 deathActivity = ACT_DIEBACKWARD;
464 }
465 break;
466 }
467
468
469 // can we perform the prescribed death?
470 if ( LookupActivity ( deathActivity ) == ACTIVITY_NOT_AVAILABLE )
471 {
472 // no! did we fail to perform a directional death?
473 if ( fTriedDirection )
474 {
475 // if yes, we're out of options. Go simple.
476 deathActivity = ACT_DIESIMPLE;
477 }
478 else
479 {
480 // cannot perform the ideal region-specific death, so try a direction.
481 if ( flDot > 0.3 )
482 {
483 deathActivity = ACT_DIEFORWARD;
484 }
485 else if ( flDot <= -0.3 )
486 {
487 deathActivity = ACT_DIEBACKWARD;
488 }
489 }
490 }
491
492 if ( LookupActivity ( deathActivity ) == ACTIVITY_NOT_AVAILABLE )
493 {
494 // if we're still invalid, simple is our only option.
495 deathActivity = ACT_DIESIMPLE;
496 }
497
498 if ( deathActivity == ACT_DIEFORWARD )
499 {
500 // make sure there's room to fall forward
501 UTIL_TraceHull ( vecSrc, vecSrc + gpGlobals->v_forward * 64, dont_ignore_monsters, head_hull, edict(), &tr );
502
503 if ( tr.flFraction != 1.0 )
504 {
505 deathActivity = ACT_DIESIMPLE;
506 }
507 }
508
509 if ( deathActivity == ACT_DIEBACKWARD )
510 {
511 // make sure there's room to fall backward
512 UTIL_TraceHull ( vecSrc, vecSrc - gpGlobals->v_forward * 64, dont_ignore_monsters, head_hull, edict(), &tr );
513
514 if ( tr.flFraction != 1.0 )
515 {
516 deathActivity = ACT_DIESIMPLE;
517 }
518 }
519
520 return deathActivity;
521}

660 {
661 pev->health = 0;
662 }
663
664 if ( ShouldFadeOnDeath() && !fade )
665 UTIL_Remove(this);

683{
684 unsigned int cCount = 0;

698 m_IdealMonsterState = MONSTERSTATE_DEAD;
699 // Make sure this condition is fired too (TakeDamage breaks out before this happens on death)
700 SetConditions( bits_COND_LIGHT_DAMAGE );
701
702 // tell owner ( if any ) that we're dead.This is mostly for MonsterMaker functionality.
703 CBaseEntity *pOwner = CBaseEntity::Instance(pev->owner);
704 if ( pOwner )
705 {
706 pOwner->DeathNotice( pev );
707 }
708
709 if ( ShouldGibMonster( iGib ) )
710 {
711 CallGibMonster();
712 return;
713 }
714 else if ( pev->flags & FL_MONSTER )
715 {
716 SetTouch( NULL );
717 BecomeDead();
718 }
719
720 // don't let the status bar glitch for players.with <0 health.
721 if (pev->health < -99)
722 {
723 pev->health = 0;
724 }
725
726 //pev->enemy = ENT( pevAttacker );//why? (sjb)
727
728 m_IdealMonsterState = MONSTERSTATE_DEAD;

1000 // no pain sound during death animation.
1001 PainSound();// "Ouch!"
1002 }
1003

1006
1007 // set damage type sustained
1008 m_bitsDamageType |= bitsDamageType;
1009
1010 // grab the vector of the incoming attack. ( pretend that the inflictor is a little lower than it really is, so the body will tend to fly upward a bit).
1011 vecDir = Vector( 0, 0, 0 );

1017 vecDir = ( pInflictor->Center() - Vector ( 0, 0, 10 ) - Center() ).Normalize();
1018 vecDir = g_vecAttackDir = vecDir.Normalize();
1019 }
1020 }
1021

1023 // message at the end of the frame
1024 // todo: remove after combining shotgun blasts?
1025 if ( IsPlayer() )
1026 {
1027 if ( pevInflictor )
1028 pev->dmg_inflictor = ENT(pevInflictor);
1029
1030 pev->dmg_take += flTake;
1031
1032 // check for godmode or invincibility
1033 if ( pev->flags & FL_GODMODE )
1034 {
1035 return 0;
1036 }
1037 }
1038
1039 // if this is a player, move him around!
1040 if ( ( !FNullEnt( pevInflictor ) ) && (pev->movetype == MOVETYPE_WALK) && (!pevAttacker || pevAttacker->solid != SOLID_TRIGGER) )
1041 {
1042 pev->velocity = pev->velocity + vecDir * -DamageForce( flDamage );
1043 }
1044
1045 // do the damage
1046 pev->health -= flTake;
1047
1048
1049 // HACKHACK Don't kill monsters in a script. Let them break their scripts first
1050 if ( m_MonsterState == MONSTERSTATE_SCRIPT )
1051 {
1052 SetConditions( bits_COND_LIGHT_DAMAGE );
1053 return 0;
1054 }
1055
1056 if ( pev->health <= 0 )
1057 {
1058 g_pevLastInflictor = pevInflictor;
1059
1060 if ( bitsDamageType & DMG_ALWAYSGIB )
1061 {
1062 Killed( pevAttacker, GIB_ALWAYS );
1063 }
1064 else if ( bitsDamageType & DMG_NEVERGIB )
1065 {
1066 Killed( pevAttacker, GIB_NEVER );
1067 }
1068 else
1069 {
1070 Killed( pevAttacker, GIB_NORMAL );
1071 }
1072
1073 g_pevLastInflictor = NULL;
1074
1075 return 0;
1076 }
1077
1078 // react to the damage (get mad)
1079 if ( (pev->flags & FL_MONSTER) && !FNullEnt(pevAttacker) )
1080 {
1081 if ( pevAttacker->flags & (FL_MONSTER | FL_CLIENT) )
1082 {// only if the attack was a monster or client!
1083
1084 // enemy's last known position is somewhere down the vector that the attack came from.
1085 if (pevInflictor)
1086 {
1087 if (m_hEnemy == NULL || pevInflictor == m_hEnemy->pev || !HasConditions(bits_COND_SEE_ENEMY))
1088 {
1089 m_vecEnemyLKP = pevInflictor->origin;
1090 }
1091 }
1092 else

1094 m_vecEnemyLKP = pev->origin + ( g_vecAttackDir * 64 );
1095 }
1096
1097 MakeIdealYaw( m_vecEnemyLKP );

1100 // !!!HACKHACK - fudged for now. Do we want to have a virtual function to determine what is light and
1101 // heavy damage per monster class?
1102 if ( flDamage > 0 )
1103 {
1104 SetConditions(bits_COND_LIGHT_DAMAGE);
1105 }
1106
1107 if ( flDamage >= 20 )
1108 {
1109 SetConditions(bits_COND_HEAVY_DAMAGE);
1110 }
1111 }
1112 }
1113

1283 }
1284 else
1285 {
1286 pEntity->TakeDamage(pevInflictor, pevAttacker, flAdjustedDamage, bitsDamageType);
1287 }
1288 }
1289 }
1290 }
