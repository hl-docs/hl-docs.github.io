97int CHudMOTD::Draw(float fTime)
98{
99 static float sfLastTime;
100 float fElapsed;
101
102 if (m_flActiveRemaining <= 0.0)
103 {
104 // MOTD expired
105 m_szMOTD[0] = 0;
106 m_iLines = 0;
107 m_iFlags &= ~HUD_ACTIVE;
108 m_flActiveRemaining = 0.0;
109 return 1;
110 }
111
112 fElapsed = gHUD.m_flTime - sfLastTime;
113 fElapsed = max(0.0f, fElapsed);
114 fElapsed = min(1.0f, fElapsed);
115 sfLastTime = gHUD.m_flTime;
116 m_flActiveRemaining -= fElapsed;
117
118 int ypos = max(((ScreenHeight - (m_iLines * LINE_HEIGHT)) / 2) - 40, 30);
119 char* ch = m_szMOTD;
120
121 while (*ch)
122 {
123 int line_length = 0;
124 for (char* next_line = ch; *next_line != '\n' && *next_line != 0; next_line++)
125 line_length += gHUD.m_scrinfo.charWidths[*next_line];
126
127 char* top = next_line;
128 if (*top == '\n')
129 *top = 0;
130 else
131 top = NULL;
132
133 int xpos = (ScreenWidth - line_length) / 2;
134 gHUD.DrawHudString(xpos, ypos, ScreenWidth, ch, 255, 180, 0);
135 ypos += LINE_HEIGHT;
136
137 if (top)
138 *top = '\n';
139 ch = next_line;
140 if (*ch == '\n')
141 ch++;
142
143 if (ypos > (ScreenHeight - 20))
144 break;
145 }
146
147 return 1;
148}

163int CHudMOTD::MsgFunc_MOTD(const char* pszName, int iSize, void* pbuf)
164{
165 if (m_iFlags & HUD_ACTIVE)
166 {
167 Reset();
168 }
169
170 BEGIN_READ(pbuf, iSize);
171 int is_finished = READ_BYTE();
172 strcat(m_szMOTD, READ_STRING());
173
174 if (is_finished)
175 {
176 m_iFlags |= HUD_ACTIVE;
177 MOTD_DISPLAY_TIME = max(10, CVAR_GET_FLOAT("motd_display_time"));
178 m_flActiveRemaining = MOTD_DISPLAY_TIME;
179
180 for (char* sz = m_szMOTD; *sz != 0; sz++)
181 {
182 if (*sz == '\n')
183 m_iLines++;
184 }
185 }
186
187 return 1;
188}
