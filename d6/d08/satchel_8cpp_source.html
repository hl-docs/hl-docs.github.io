111void CSatchelCharge::SatchelSlide(CBaseEntity *pOther)
112{
113 entvars_t *pevOther = pOther->pev; // Get the other entity's pev
114
115 // Don't hit the entity that launched this grenade
116 if (pOther->edict() == pev->owner)
117 return;
118
119 pev->gravity = 1; // Set normal gravity
120
121 // Check for ground underneath
122 TraceResult tr;
123 UTIL_TraceLine(pev->origin, pev->origin - Vector(0, 0, 10), ignore_monsters, edict(), &tr);
124
125 if (tr.flFraction < 1.0)
126 {
127 // Add a bit of static friction
128 pev->velocity = pev->velocity * 0.95; // Reduce velocity
129 pev->avelocity = pev->avelocity * 0.9; // Reduce angular velocity
130 }
131
132 if (!(pev->flags & FL_ONGROUND) && pev->velocity.Length2D() > 10)
133 {
134 BounceSound(); // Play bounce sound
135 }
136 StudioFrameAdvance(); // Advance the animation frame
137}

146void CSatchelCharge::SatchelThink(void)
147{
148 StudioFrameAdvance(); // Advance the animation frame
149 pev->nextthink = gpGlobals->time + 0.1; // Set the next think time
150
151 if (!IsInWorld())
152 {
153 UTIL_Remove(this); // Remove the satchel if it's out of the world
154 return;
155 }
156
157 if (pev->waterlevel == 3) // Check if submerged in water
158 {
159 pev->movetype = MOVETYPE_FLY; // Set movement type to fly
160 pev->velocity = pev->velocity * 0.8; // Reduce velocity
161 pev->avelocity = pev->avelocity * 0.9; // Reduce angular velocity
162 pev->velocity.z += 8; // Add upward velocity
163 }
164 else if (pev->waterlevel == 0) // Check if not in water
165 {
166 pev->movetype = MOVETYPE_BOUNCE; // Set movement type to bounce
167 }
168 else
169 {
170 pev->velocity.z -= 8; // Apply downward force
171 }
172}

430void CSatchel::PrimaryAttack()
431{
432 switch (m_chargeReady)
433 {
434 case 0:
435 Throw(); // Throw the satchel
436 break;
437 case 1:
438 SendWeaponAnim(SATCHEL_RADIO_FIRE); // Play fire animation
439
440 edict_t *pPlayer = m_pPlayer->edict(); // Get player entity
441
442 CBaseEntity *pSatchel = NULL;
443
444 while ((pSatchel = UTIL_FindEntityInSphere(pSatchel, m_pPlayer->pev->origin, 4096)) != NULL)
445 {
446 if (FClassnameIs(pSatchel->pev, "monster_satchel"))
447 {
448 if (pSatchel->pev->owner == pPlayer)
449 {
450 pSatchel->Use(m_pPlayer, m_pPlayer, USE_ON, 0); // Detonate the satchel
451 m_chargeReady = 2; // Set charge state
452 }
453 }
454 }
455
456 m_chargeReady = 2; // Set charge state
457 m_flNextPrimaryAttack = GetNextAttackDelay(0.5); // Set next attack delay
458 m_flNextSecondaryAttack = UTIL_WeaponTimeBase() + 0.5; // Set next secondary attack time
459 break;
460
461 case 2:
462 // We're reloading, don't allow fire
463 break;
464 }
465}

487void CSatchel::Throw(void)
488{
489 if (m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType])
490 {
491 Vector vecSrc = m_pPlayer->pev->origin; // Get the player's origin
492
493 Vector vecThrow = gpGlobals->v_forward * 274 + m_pPlayer->pev->velocity; // Calculate throw velocity
494
495#ifndef CLIENT_DLL
496 CBaseEntity *pSatchel = Create("monster_satchel", vecSrc, Vector(0, 0, 0), m_pPlayer->edict()); // Create the satchel entity
497 pSatchel->pev->velocity = vecThrow; // Set the velocity
498 pSatchel->pev->avelocity.y = 400; // Set angular velocity
499
500 m_pPlayer->pev->viewmodel = MAKE_STRING("models/v_satchel_radio.mdl"); // Set view model
501 m_pPlayer->pev->weaponmodel = MAKE_STRING("models/p_satchel_radio.mdl"); // Set weapon model
502#else
503 LoadVModel("models/v_satchel_radio.mdl", m_pPlayer); // Load view model for client
504#endif
505
506 SendWeaponAnim(SATCHEL_RADIO_DRAW); // Play draw animation
507
508 // Player "shoot" animation
509 m_pPlayer->SetAnimation(PLAYER_ATTACK1); // Set player animation
510
511 m_chargeReady = 1; // Set charge state
512
513 m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType]--; // Decrease ammo count
514
515 m_flNextPrimaryAttack = GetNextAttackDelay(1.0); // Set next attack delay
516 m_flNextSecondaryAttack = UTIL_WeaponTimeBase() + 0.5; // Set next secondary attack time
517 }

526void CSatchel::WeaponIdle(void)
527{
528 if (m_flTimeWeaponIdle > UTIL_WeaponTimeBase())
529 return; // Exit if still in idle time
530
531 switch (m_chargeReady)
532 {
533 case 0:
534 SendWeaponAnim(SATCHEL_FIDGET1); // Play fidget animation
535 strcpy(m_pPlayer->m_szAnimExtention, "trip"); // Set animation extension
536 break;
537 case 1:
538 SendWeaponAnim(SATCHEL_RADIO_FIDGET1); // Play radio fidget animation
539 strcpy(m_pPlayer->m_szAnimExtention, "hive"); // Set animation extension
540 break;
541 case 2:
542 if (!m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType])
543 {
544 m_chargeReady = 0; // Reset charge state
545 RetireWeapon(); // Retire the weapon
546 return; // Exit
547 }
548
549#ifndef CLIENT_DLL
550 m_pPlayer->pev->viewmodel = MAKE_STRING("models/v_satchel.mdl"); // Set view model
551 m_pPlayer->pev->weaponmodel = MAKE_STRING("models/p_satchel.mdl"); // Set weapon model
552#else
553 LoadVModel("models/v_satchel.mdl", m_pPlayer); // Load view model for client
554#endif
555
556 SendWeaponAnim(SATCHEL_DRAW); // Play draw animation
557 strcpy(m_pPlayer->m_szAnimExtention, "trip"); // Set animation extension
558
559 m_flNextPrimaryAttack = GetNextAttackDelay(0.5); // Set next attack delay
560 m_flNextSecondaryAttack = UTIL_WeaponTimeBase() + 0.5; // Set next secondary attack time
561 m_chargeReady = 0; // Reset charge state
562 break;
563 }
564 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat(m_pPlayer->random_seed, 10, 15); // Set idle time
565}

591 }
