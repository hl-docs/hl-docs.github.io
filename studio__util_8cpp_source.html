26void AngleMatrix(const float *angles, float (*matrix)[4])
27{
28 float angle;
29 float sr, sp, sy, cr, cp, cy;
30
31 angle = angles[YAW] * (M_PI * 2 / 360);
32 sy = sin(angle);
33 cy = cos(angle);
34 angle = angles[PITCH] * (M_PI * 2 / 360);
35 sp = sin(angle);
36 cp = cos(angle);
37 angle = angles[ROLL] * (M_PI * 2 / 360);
38 sr = sin(angle);
39 cr = cos(angle);
40
41 // matrix = (YAW * PITCH) * ROLL
42 matrix[0][0] = cp * cy;
43 matrix[1][0] = cp * sy;
44 matrix[2][0] = -sp;
45 matrix[0][1] = sr * sp * cy + cr * -sy;
46 matrix[1][1] = sr * sp * sy + cr * cy;
47 matrix[2][1] = sr * cp;
48 matrix[0][2] = (cr * sp * cy + -sr * -sy);
49 matrix[1][2] = (cr * sp * sy + -sr * cy);
50 matrix[2][2] = cr * cp;
51 matrix[0][3] = 0.0;
52 matrix[1][3] = 0.0;
53 matrix[2][3] = 0.0;
54}

152void AngleQuaternion(float *angles, vec4_t quaternion)
153{
154 float angle;
155 float sr, sp, sy, cr, cp, cy;
156
157 // FIXME: rescale the inputs to 1/2 angle
158 angle = angles[2] * 0.5;
159 sy = sin(angle);
160 cy = cos(angle);
161 angle = angles[1] * 0.5;
162 sp = sin(angle);
163 cp = cos(angle);
164 angle = angles[0] * 0.5;
165 sr = sin(angle);
166 cr = cos(angle);
167
168 quaternion[0] = sr * cp * cy - cr * sp * sy; // X
169 quaternion[1] = cr * sp * cy + sr * cp * sy; // Y
170 quaternion[2] = cr * cp * sy - sr * sp * cy; // Z
171 quaternion[3] = cr * cp * cy + sr * sp * sy; // W

185void QuaternionSlerp(vec4_t p, vec4_t q, float t, vec4_t qt)
186{
187 int i;
188 float omega, cosom, sinom, sclp, sclq;
189
190 // decide if one of the quaternions is backwards
191 float a = 0;
192 float b = 0;
193
194 for (i = 0; i < 4; i++)
195 {
196 a += (p[i] - q[i]) * (p[i] - q[i]);
197 b += (p[i] + q[i]) * (p[i] + q[i]);
198 }
199 if (a > b)
200 {
201 for (i = 0; i < 4; i++)
202 {
203 q[i] = -q[i];
204 }
205 }
206
207 cosom = p[0] * q[0] + p[1] * q[1] + p[2] * q[2] + p[3] * q[3];
208
209 if ((1.0 + cosom) > 0.000001)
210 {
211 if ((1.0 - cosom) > 0.000001)
212 {
213 omega = acos(cosom);
214 sinom = sin(omega);
215 sclp = sin((1.0 - t) * omega) / sinom;
216 sclq = sin(t * omega) / sinom;
217 }
218 else
219 {
220 sclp = 1.0 - t;
221 sclq = t;
222 }
223 for (i = 0; i < 4; i++) {
224 qt[i] = sclp * p[i] + sclq * q[i];
225 }
226 }
227 else
228 {
229 qt[0] = -q[1];
230 qt[1] = q[0];
231 qt[2] = -q[3];
232 qt[3] = q[2];
233 sclp = sin((1.0 - t) * (0.5 * M_PI));
234 sclq = sin(t * (0.5 * M_PI));
235 for (i = 0; i < 3; i++)
236 {
237 qt[i] = sclp * p[i] + sclq * qt[i];
238 }
239 }
240}

251void QuaternionMatrix(vec4_t quaternion, float (*matrix)[4])
252{
253 matrix[0][0] = 1.0 - 2.0 * quaternion[1] * quaternion[1] - 2.0 * quaternion[2] * quaternion[2];
254 matrix[1][0] = 2.0 * quaternion[0] * quaternion[1] + 2.0 * quaternion[3] * quaternion[2];
255 matrix[2][0] = 2.0 * quaternion[0] * quaternion[2] - 2.0 * quaternion[3] * quaternion[1];
256
257 matrix[0][1] = 2.0 * quaternion[0] * quaternion[1] - 2.0 * quaternion[3] * quaternion[2];
258 matrix[1][1] = 1.0 - 2.0 * quaternion[0] * quaternion[0] - 2.0 * quaternion[2] * quaternion[2];
259 matrix[2][1] = 2.0 * quaternion[1] * quaternion[2] + 2.0 * quaternion[3] * quaternion[0];
260
261 matrix[0][2] = 2.0 * quaternion[0] * quaternion[2] + 2.0 * quaternion[3] * quaternion[1];
262 matrix[1][2] = 2.0 * quaternion[1] * quaternion[2] - 2.0 * quaternion[3] * quaternion[0];
263 matrix[2][2] = 1.0 - 2.0 * quaternion[0] * quaternion[0] - 2.0 * quaternion[1] * quaternion[1];
