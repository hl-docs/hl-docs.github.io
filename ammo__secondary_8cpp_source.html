86int CHudAmmoSecondary::Draw(float flTime)
87{
88 if ((gHUD.m_iHideHUDDisplay & (HIDEHUD_WEAPONS | HIDEHUD_ALL)))
89 return 1;
90
91 int a, x, y, r, g, b, AmmoWidth;
92 UnpackRGB(r, g, b, RGB_YELLOWISH);
93 a = (int)max(MIN_ALPHA, m_fFade);
94
95 if (m_fFade > 0)
96 m_fFade -= (gHUD.m_flTimeDelta * 20);
97
98 ScaleColors(r, g, b, a);
99
100 AmmoWidth = gHUD.GetSpriteRect(gHUD.m_HUD_number_0).right - gHUD.GetSpriteRect(gHUD.m_HUD_number_0).left;
101 y = ScreenHeight - (gHUD.m_iFontHeight * 4);
102 x = ScreenWidth - AmmoWidth;
103
104 if (m_HUD_ammoicon)
105 {
106 x -= (gHUD.GetSpriteRect(m_HUD_ammoicon).right - gHUD.GetSpriteRect(m_HUD_ammoicon).left);
107 y -= (gHUD.GetSpriteRect(m_HUD_ammoicon).top - gHUD.GetSpriteRect(m_HUD_ammoicon).bottom);
108
109 SPR_Set(gHUD.GetSprite(m_HUD_ammoicon), r, g, b);
110 SPR_DrawAdditive(0, x, y, &gHUD.GetSpriteRect(m_HUD_ammoicon));
111 }
112 else
113 {
114 x -= AmmoWidth;
115 y -= (gHUD.GetSpriteRect(gHUD.m_HUD_number_0).top - gHUD.GetSpriteRect(gHUD.m_HUD_number_0).bottom);
116 }
117
118 for (int i = MAX_SEC_AMMO_VALUES - 1; i >= 0; i--)
119 {
120 if (m_iAmmoAmounts[i] < 0)
121 continue;
122
123 x -= (AmmoWidth / 2);
124 x -= (gHUD.GetNumWidth(m_iAmmoAmounts[i], DHN_DRAWZERO) * AmmoWidth);
125 gHUD.DrawHudNumber(x, y, DHN_DRAWZERO, m_iAmmoAmounts[i], r, g, b);
126
127 if (i != 0)
128 {
129 x -= (AmmoWidth / 2);
130 FillRGBA(x, y, (AmmoWidth / 10), gHUD.m_iFontHeight, r, g, b, a);
131 }
132 }
133
134 return 1;
135}

168int CHudAmmoSecondary::MsgFunc_SecAmmoVal(const char* pszName, int iSize, void* pbuf)
169{
170 BEGIN_READ(pbuf, iSize);
171
172 int index = READ_BYTE();
173 if (index < 0 || index >= MAX_SEC_AMMO_VALUES)
174 return 1;
175
176 m_iAmmoAmounts[index] = READ_BYTE();
177 m_iFlags |= HUD_ACTIVE;
178
179 int count = 0;
180 for (int i = 0; i < MAX_SEC_AMMO_VALUES; i++)
181 {
182 count += max(0, m_iAmmoAmounts[i]);
183 }
184
185 if (count == 0)
186 {
187 m_iFlags &= ~HUD_ACTIVE;
188 return 1;
189 }
190
191 m_fFade = 200.0f;
192
193 return 1;
194}
