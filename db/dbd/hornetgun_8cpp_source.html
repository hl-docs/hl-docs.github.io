193void CHgun::PrimaryAttack()
194{
195 Reload(); // Reload if necessary
196
197 if (m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0) // Check if ammo is available
198 {
199 return; // Exit if no ammo
200 }
201
202#ifndef CLIENT_DLL
203 UTIL_MakeVectors(m_pPlayer->pev->v_angle); // Calculate firing direction
204
205 CBaseEntity *pHornet = CBaseEntity::Create("hornet", m_pPlayer->GetGunPosition() + gpGlobals->v_forward * 16 + gpGlobals->v_right * 8 + gpGlobals->v_up * -12, m_pPlayer->pev->v_angle, m_pPlayer->edict()); // Create hornet entity
206 pHornet->pev->velocity = gpGlobals->v_forward * 300; // Set hornet velocity
207
208 m_flRechargeTime = gpGlobals->time + 0.5; // Set recharge time
209#endif
210
211 m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType]--; // Decrease ammo count
212
213 m_pPlayer->m_iWeaponVolume = QUIET_GUN_VOLUME; // Set weapon volume
214 m_pPlayer->m_iWeaponFlash = DIM_GUN_FLASH; // Set weapon flash
215
216 int flags;
217#if defined(CLIENT_WEAPONS)
218 flags = FEV_NOTHOST; // Set flags for client weapons
219#else
220 flags = 0; // No flags for server
221#endif
222
223 PLAYBACK_EVENT_FULL(flags, m_pPlayer->edict(), m_usHornetFire, 0.0, (float *)&g_vecZero, (float *)&g_vecZero, 0.0, 0.0, FIREMODE_TRACK, 0, 0, 0); // Play firing event
224
225 // Player "shoot" animation
226 m_pPlayer->SetAnimation(PLAYER_ATTACK1);
227
228 m_flNextPrimaryAttack = GetNextAttackDelay(0.25); // Set next primary attack time
229
230 if (m_flNextPrimaryAttack < UTIL_WeaponTimeBase())
231 {
232 m_flNextPrimaryAttack = UTIL_WeaponTimeBase() + 0.25; // Ensure minimum delay
233 }
234
235 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat(m_pPlayer->random_seed, 10, 15); // Set idle time
236}

332void CHgun::Reload(void)
333{
334 if (m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] >= HORNET_MAX_CARRY) // Check if ammo is at max
335 return; // Exit if at max
336
337 while (m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] < HORNET_MAX_CARRY && m_flRechargeTime < gpGlobals->time) // Increment ammo until max or recharge time is up
338 {
339 m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType]++; // Increase ammo count
340 m_flRechargeTime += 0.5; // Increment recharge time
341 }

350void CHgun::WeaponIdle(void)
351{
352 Reload(); // Reload if necessary
353
354 if (m_flTimeWeaponIdle > UTIL_WeaponTimeBase()) // Check if still in idle time
355 return; // Exit if still in idle time
356
357 int iAnim;
358 float flRand = UTIL_SharedRandomFloat(m_pPlayer->random_seed, 0, 1); // Get random value
359 if (flRand <= 0.75)
360 {
361 iAnim = HGUN_IDLE1; // Select idle animation
362 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 30.0 / 16 * (2); // Set idle time
363 }
364 else if (flRand <= 0.875)
365 {
366 iAnim = HGUN_FIDGETSWAY; // Select fidget animation
367 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 40.0 / 16.0; // Set idle time
368 }
369 else
370 {
371 iAnim = HGUN_FIDGETSHAKE; // Select fidget animation
372 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 35.0 / 16.0; // Set idle time
373 }
374 SendWeaponAnim(iAnim); // Play idle animation
375}
