112void PlayLockSounds(entvars_t *pev, locksound_t *pls, int flocked, int fbutton)
113{
114 // Determine sound wait time based on entity type
115 float flsoundwait = fbutton ? BUTTON_SOUNDWAIT : DOOR_SOUNDWAIT;
116
117 if (flocked)
118 {
119 // Locked sound logic
120 int fplaysound = (pls->sLockedSound && gpGlobals->time > pls->flwaitSound);
121 int fplaysentence = (pls->sLockedSentence && !pls->bEOFLocked && gpGlobals->time > pls->flwaitSentence);
122 float fvol = (fplaysound && fplaysentence) ? 0.25 : 1.0;
123
124 if (fplaysound)
125 {
126 // Play 'door locked' sound
127 EMIT_SOUND(ENT(pev), CHAN_ITEM, (char*)STRING(pls->sLockedSound), fvol, ATTN_NORM);
128 pls->flwaitSound = gpGlobals->time + flsoundwait;
129 }
130
131 if (fplaysentence)
132 {
133 // Play next 'door locked' sentence in group
134 int iprev = pls->iLockedSentence;
135 pls->iLockedSentence = SENTENCEG_PlaySequentialSz(ENT(pev), STRING(pls->sLockedSentence),
136 0.85, ATTN_NORM, 0, 100, pls->iLockedSentence, FALSE);
137 pls->iUnlockedSentence = 0;
138 pls->bEOFLocked = (iprev == pls->iLockedSentence);
139 pls->flwaitSentence = gpGlobals->time + DOOR_SENTENCEWAIT;
140 }
141 }
142 else
143 {
144 // Unlocked sound logic
145 int fplaysound = (pls->sUnlockedSound && gpGlobals->time > pls->flwaitSound);
146 int fplaysentence = (pls->sUnlockedSentence && !pls->bEOFUnlocked && gpGlobals->time > pls->flwaitSentence);
147 float fvol = (fplaysound && fplaysentence) ? 0.25 : 1.0;
148
149 if (fplaysound)
150 {
151 // Play 'door unlocked' sound
152 EMIT_SOUND(ENT(pev), CHAN_ITEM, (char*)STRING(pls->sUnlockedSound), fvol, ATTN_NORM);
153 pls->flwaitSound = gpGlobals->time + flsoundwait;
154 }
155
156 if (fplaysentence)
157 {
158 // Play next 'door unlocked' sentence in group
159 int iprev = pls->iUnlockedSentence;
160 pls->iUnlockedSentence = SENTENCEG_PlaySequentialSz(ENT(pev), STRING(pls->sUnlockedSentence),
161 0.85, ATTN_NORM, 0, 100, pls->iUnlockedSentence, FALSE);
162 pls->iLockedSentence = 0;
163 pls->bEOFUnlocked = (iprev == pls->iUnlockedSentence);
164 pls->flwaitSentence = gpGlobals->time + DOOR_SENTENCEWAIT;
165 }
166 }

267void CBaseDoor::Spawn( )
268{
269 Precache();
270 SetMovedir (pev);
271
272 if ( pev->skin == 0 )
273 {//normal door
274 if ( FBitSet (pev->spawnflags, SF_DOOR_PASSABLE) )
275 pev->solid = SOLID_NOT;
276 else
277 pev->solid = SOLID_BSP;
278 }
279 else
280 {// special contents
281 pev->solid = SOLID_NOT;
282 SetBits( pev->spawnflags, SF_DOOR_SILENT ); // water is silent for now
283 }
284
285 pev->movetype = MOVETYPE_PUSH;
286 UTIL_SetOrigin(pev, pev->origin);
287 SET_MODEL( ENT(pev), STRING(pev->model) );
288
289 if (pev->speed == 0)
290 pev->speed = 100;
291
292 m_vecPosition1 = pev->origin;
293 // Subtract 2 from size because the engine expands bboxes by 1 in all directions making the size too big
294 m_vecPosition2 = m_vecPosition1 + (pev->movedir * (fabs( pev->movedir.x * (pev->size.x-2) ) + fabs( pev->movedir.y * (pev->size.y-2) ) + fabs( pev->movedir.z * (pev->size.z-2) ) - m_flLip));
295 ASSERTSZ(m_vecPosition1 != m_vecPosition2, "door start/end positions are equal");
296 if ( FBitSet (pev->spawnflags, SF_DOOR_START_OPEN) )
297 { // swap pos1 and pos2, put door at pos2
298 UTIL_SetOrigin(pev, m_vecPosition2);
299 m_vecPosition2 = m_vecPosition1;
300 m_vecPosition1 = pev->origin;
301 }
302
303 m_toggle_state = TS_AT_BOTTOM;
304
305 // if the door is flagged for USE button activation only, use NULL touch function
306 if ( FBitSet ( pev->spawnflags, SF_DOOR_USE_ONLY ) )
307 {
308 SetTouch ( NULL );
309 }
310 else // touchable button
311 SetTouch( &CBaseDoor::DoorTouch );
312}

483//
484// Doors not tied to anything (e.g. button, another door) can be touched, to make them activate.
485//
494void CBaseDoor::DoorTouch( CBaseEntity *pOther )
495{
496 entvars_t* pevToucher = pOther->pev;
497
498 // Ignore touches by anything but players
499 if (!FClassnameIs(pevToucher, "player"))
500 return;
501
502 // If door has master, and it's not ready to trigger,
503 // play 'locked' sound
504
505 if (m_sMaster && !UTIL_IsMasterTriggered(m_sMaster, pOther))
506 PlayLockSounds(pev, &m_ls, TRUE, FALSE);
507
508 // If door is somebody's target, then touching does nothing.
509 // You have to activate the owner (e.g. button).
510
511 if (!FStringNull(pev->targetname))
512 {
513 // play locked sound
514 PlayLockSounds(pev, &m_ls, TRUE, FALSE);
515 return;
516 }
517
518 m_hActivator = pOther;// remember who activated the door
519
520 if (DoorActivate( ))
521 SetTouch( NULL ); // Temporarily disable the touch function, until movement is finished.
522}

547//
548// Causes the door to "do its thing", i.e. start moving, and cascade activation.
549//
558int CBaseDoor::DoorActivate( )
559{
560 if (!UTIL_IsMasterTriggered(m_sMaster, m_hActivator))
561 return 0;
562
563 if (FBitSet(pev->spawnflags, SF_DOOR_NO_AUTO_RETURN) && m_toggle_state == TS_AT_TOP)
564 {// door should close
565 DoorGoDown();
566 }
567 else
568 {// door should open
569
570 if ( m_hActivator != NULL && m_hActivator->IsPlayer() )
571 {// give health if player opened the door (medikit)
572 // VARS( m_eoActivator )->health += m_bHealthValue;
573
574 m_hActivator->TakeHealth( m_bHealthValue, DMG_GENERIC );
575
576 }
577
578 // play door unlock sounds
579 PlayLockSounds(pev, &m_ls, FALSE, FALSE);
580
581 DoorGoUp();
582 }
583
584 return 1;
585}

589//
590// Starts the door going to its "up" position (simply ToggleData->vecPosition2).
591//
598void CBaseDoor::DoorGoUp( void )
599{
600 entvars_t *pevActivator;
601
602 // It could be going-down, if blocked.
603 ASSERT(m_toggle_state == TS_AT_BOTTOM || m_toggle_state == TS_GOING_DOWN);
604
605 // emit door moving and stop sounds on CHAN_STATIC so that the multicast doesn't
606 // filter them out and leave a client stuck with looping door sounds!
607 if ( !FBitSet( pev->spawnflags, SF_DOOR_SILENT ) )
608 {
609 if ( m_toggle_state != TS_GOING_UP && m_toggle_state != TS_GOING_DOWN )
610 EMIT_SOUND(ENT(pev), CHAN_STATIC, (char*)STRING(pev->noiseMoving), 1, ATTN_NORM);
611 }
612
613 m_toggle_state = TS_GOING_UP;
614
615 SetMoveDone( &CBaseDoor::DoorHitTop );
616 if ( FClassnameIs(pev, "func_door_rotating")) // !!! BUGBUG Triggered doors don't work with this yet
617 {
618 float sign = 1.0;
619
620 if ( m_hActivator != NULL )
621 {
622 pevActivator = m_hActivator->pev;
623
624 if ( !FBitSet( pev->spawnflags, SF_DOOR_ONEWAY ) && pev->movedir.y ) // Y axis rotation, move away from the player
625 {
626 Vector vec = pevActivator->origin - pev->origin;
627 Vector angles = pevActivator->angles;
628 angles.x = 0;
629 angles.z = 0;
630 UTIL_MakeVectors (angles);
631 // Vector vnext = (pevToucher->origin + (pevToucher->velocity * 10)) - pev->origin;
632 UTIL_MakeVectors ( pevActivator->angles );
633 Vector vnext = (pevActivator->origin + (gpGlobals->v_forward * 10)) - pev->origin;
634 if ( (vec.x*vnext.y - vec.y*vnext.x) < 0 )
635 sign = -1.0;
636 }
637 }
638 AngularMove(m_vecAngle2*sign, pev->speed);
639 }
640 else
641 LinearMove(m_vecPosition2, pev->speed);
642}

645//
646// The door has reached the "up" position. Either go back down, or wait for another activation.
647//
654void CBaseDoor::DoorHitTop( void )
655{
656 if ( !FBitSet( pev->spawnflags, SF_DOOR_SILENT ) )
657 {
658 STOP_SOUND(ENT(pev), CHAN_STATIC, (char*)STRING(pev->noiseMoving) );
659 EMIT_SOUND(ENT(pev), CHAN_STATIC, (char*)STRING(pev->noiseArrived), 1, ATTN_NORM);
660 }
661
662 ASSERT(m_toggle_state == TS_GOING_UP);
663 m_toggle_state = TS_AT_TOP;
664
665 // toggle-doors don't come down automatically, they wait for refire.
666 if (FBitSet(pev->spawnflags, SF_DOOR_NO_AUTO_RETURN))
667 {
668 // Re-instate touch method, movement is complete
669 if ( !FBitSet ( pev->spawnflags, SF_DOOR_USE_ONLY ) )
670 SetTouch( &CBaseDoor::DoorTouch );
671 }
672 else
673 {
674 // In flWait seconds, DoorGoDown will fire, unless wait is -1, then door stays open
675 pev->nextthink = pev->ltime + m_flWait;
676 SetThink( &CBaseDoor::DoorGoDown );
677
678 if ( m_flWait == -1 )
679 {
680 pev->nextthink = -1;
681 }
682 }
683
684 // Fire the close target (if startopen is set, then "top" is closed) - netname is the close target
685 if ( pev->netname && (pev->spawnflags & SF_DOOR_START_OPEN) )
686 FireTargets( STRING(pev->netname), m_hActivator, this, USE_TOGGLE, 0 );
687
688 SUB_UseTargets( m_hActivator, USE_TOGGLE, 0 ); // this isn't finished
689}

723//
