105BOOL CHealthKit::MyTouch(CBasePlayer *pPlayer)
106{
107 if (pPlayer->pev->deadflag != DEAD_NO) // Check if the player is dead
108 {
109 return FALSE; // Cannot use health kit if dead
110 }
111
112 if (pPlayer->TakeHealth(gSkillData.healthkitCapacity, DMG_GENERIC)) // Attempt to give health
113 {
114 // Send a pickup message to the player
115 MESSAGE_BEGIN(MSG_ONE, gmsgItemPickup, NULL, pPlayer->pev);
116 WRITE_STRING(STRING(pev->classname)); // Write the item name
117 MESSAGE_END();
118
119 EMIT_SOUND(ENT(pPlayer->pev), CHAN_ITEM, "items/smallmedkit1.wav", 1, ATTN_NORM); // Play pickup sound
120
121 // Check if the health kit should respawn
122 if (g_pGameRules->ItemShouldRespawn(this))
123 {
124 Respawn(); // Respawn the health kit
125 }
126 else
127 {
128 UTIL_Remove(this); // Remove the health kit from the game
129 }
130
131 return TRUE; // Health kit was successfully used
132 }
133
134 return FALSE; // Health kit was not used
135}

269void CWallHealth::Use(CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
270{
271 // Make sure that we have a caller
272 if (!pActivator)
273 return;
274
275 // If it's not a player, ignore
276 if (!pActivator->IsPlayer())
277 return;
278
279 // If there is no juice left, turn it off
280 if (m_iJuice <= 0)
281 {
282 pev->frame = 1; // Set frame to indicate off state
283 Off(); // Call Off function
284 }
285
286 // If the player doesn't have the suit, or there is no juice left, make the deny noise
287 if ((m_iJuice <= 0) || (!(pActivator->pev->weapons & (1 << WEAPON_SUIT))))
288 {
289 if (m_flSoundTime <= gpGlobals->time)
290 {
291 m_flSoundTime = gpGlobals->time + 0.62; // Set sound time for next deny sound
292 EMIT_SOUND(ENT(pev), CHAN_ITEM, "items/medshotno1.wav", 1.0, ATTN_NORM); // Play deny sound
293 }
294 return; // Exit if conditions are not met
295 }
296
297 pev->nextthink = pev->ltime + 0.25; // Set next think time
298 SetThink(&CWallHealth::Off); // Set think function to Off
299
300 // Time to recharge yet?
301 if (m_flNextCharge >= gpGlobals->time)
302 return; // Exit if not time to recharge
303
304 // Play the on sound or the looping charging sound
305 if (!m_iOn)
306 {
307 m_iOn++; // Increment on state
308 EMIT_SOUND(ENT(pev), CHAN_ITEM, "items/medshot4.wav", 1.0, ATTN_NORM); // Play charging sound
309 m_flSoundTime = 0.56 + gpGlobals->time; // Set sound time for next charge sound
310 }
311 if ((m_iOn == 1) && (m_flSoundTime <= gpGlobals->time))
312 {
313 m_iOn++; // Increment on state
314 EMIT_SOUND(ENT(pev), CHAN_STATIC, "items/medcharge4.wav", 1.0, ATTN_NORM); // Play looping charge sound
315 }
316
317 // Charge the player
318 if (pActivator->TakeHealth(1, DMG_GENERIC))
319 {
320 m_iJuice--; // Decrease juice level
321 }
322
323 // Govern the rate of charge
324 m_flNextCharge = gpGlobals->time + 0.1; // Set next charge time
325}

349void CWallHealth::Off(void)
350{
351 // Stop looping sound.
352 if (m_iOn > 1)
353 STOP_SOUND(ENT(pev), CHAN_STATIC, "items/medcharge4.wav"); // Stop charging sound
354
355 m_iOn = 0; // Reset on state
356
357 if ((!m_iJuice) && ((m_iReactivate = g_pGameRules->FlHealthChargerRechargeTime()) > 0))
358 {
359 pev->nextthink = pev->ltime + m_iReactivate; // Set next think time for recharge
360 SetThink(&CWallHealth::Recharge); // Set think function to Recharge
361 }
362 else
363 SetThink(&CWallHealth::SUB_DoNothing); // Set think function to do nothing
364}
