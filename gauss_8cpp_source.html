206{
207 // don't fire underwater

209 {
210 PlayEmptySound( );
211 m_flNextSecondaryAttack = m_flNextPrimaryAttack = GetNextAttackDelay(0.15);

239void CGauss::SecondaryAttack()
240{
241 // don't fire underwater
242 if ( m_pPlayer->pev->waterlevel == 3 )
243 {
244 if ( m_fInAttack != 0 )
245 {
246 EMIT_SOUND_DYN(ENT(m_pPlayer->pev), CHAN_WEAPON, "weapons/electro4.wav", 1.0, ATTN_NORM, 0, 80 + RANDOM_LONG(0,0x3f));
247 SendWeaponAnim( GAUSS_IDLE );
248 m_fInAttack = 0;
249 }
250 else
251 {
252 PlayEmptySound( );
253 }
254
255 m_flNextSecondaryAttack = m_flNextPrimaryAttack = GetNextAttackDelay(0.5);
256 return;
257 }
258
259 if ( m_fInAttack == 0 )
260 {
261 if ( m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0 )
262 {
263 EMIT_SOUND(ENT(m_pPlayer->pev), CHAN_WEAPON, "weapons/357_cock1.wav", 0.8, ATTN_NORM);
264 m_pPlayer->m_flNextAttack = UTIL_WeaponTimeBase() + 0.5;
265 return;
266 }
267
268 m_fPrimaryFire = FALSE;
269
270 m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType]--;// take one ammo just to start the spin
271 m_pPlayer->m_flNextAmmoBurn = UTIL_WeaponTimeBase();
272
273 // spin up
274 m_pPlayer->m_iWeaponVolume = GAUSS_PRIMARY_CHARGE_VOLUME;
275
276 SendWeaponAnim( GAUSS_SPINUP );
277 m_fInAttack = 1;
278 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 0.5;
279 m_pPlayer->m_flStartCharge = gpGlobals->time;
280 m_pPlayer->m_flAmmoStartCharge = UTIL_WeaponTimeBase() + GetFullChargeTime();
281
282 PLAYBACK_EVENT_FULL( FEV_NOTHOST, m_pPlayer->edict(), m_usGaussSpin, 0.0, (float *)&g_vecZero, (float *)&g_vecZero, 0.0, 0.0, 110, 0, 0, 0 );
283
284 m_iSoundState = SND_CHANGE_PITCH;
285 }
286 else if (m_fInAttack == 1)
287 {
288 if (m_flTimeWeaponIdle < UTIL_WeaponTimeBase())
289 {
290 SendWeaponAnim( GAUSS_SPIN );
291 m_fInAttack = 2;
292 }
293 }
294 else
295 {
296 // during the charging process, eat one bit of ammo every once in a while
297 if ( UTIL_WeaponTimeBase() >= m_pPlayer->m_flNextAmmoBurn && m_pPlayer->m_flNextAmmoBurn != 1000 )
298 {
299#ifdef CLIENT_DLL
300 if ( bIsMultiplayer() )
301#else
302 if ( g_pGameRules->IsMultiplayer() )
303#endif
304 {
305 m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType]--;
306 m_pPlayer->m_flNextAmmoBurn = UTIL_WeaponTimeBase() + 0.1;
307 }
308 else
309 {
310 m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType]--;
311 m_pPlayer->m_flNextAmmoBurn = UTIL_WeaponTimeBase() + 0.3;
312 }
313 }
314
315 if ( m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0 )
316 {
317 // out of ammo! force the gun to fire
318 StartFire();
319 m_fInAttack = 0;
320 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 1.0;
321 m_pPlayer->m_flNextAttack = UTIL_WeaponTimeBase() + 1;
322 return;
323 }
324
325 if ( UTIL_WeaponTimeBase() >= m_pPlayer->m_flAmmoStartCharge )
326 {
327 // don't eat any more ammo after gun is fully charged.
328 m_pPlayer->m_flNextAmmoBurn = 1000;
329 }
330
331 int pitch = ( gpGlobals->time - m_pPlayer->m_flStartCharge ) * ( 150 / GetFullChargeTime() ) + 100;
332 if ( pitch > 250 )
333 pitch = 250;
334
335 // ALERT( at_console, "%d %d %d\n", m_fInAttack, m_iSoundState, pitch );
336
337 if ( m_iSoundState == 0 )
338 ALERT( at_console, "sound state %d\n", m_iSoundState );
339
340 PLAYBACK_EVENT_FULL( FEV_NOTHOST, m_pPlayer->edict(), m_usGaussSpin, 0.0, (float *)&g_vecZero, (float *)&g_vecZero, 0.0, 0.0, pitch, 0, ( m_iSoundState == SND_CHANGE_PITCH ) ? 1 : 0, 0 );
341
342 m_iSoundState = SND_CHANGE_PITCH; // hack for going through level transitions
343
344 m_pPlayer->m_iWeaponVolume = GAUSS_PRIMARY_CHARGE_VOLUME;
345
346 // m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 0.1;
347 if ( m_pPlayer->m_flStartCharge < gpGlobals->time - 10 )
348 {
349 // Player charged up too long. Zap him.
350 EMIT_SOUND_DYN(ENT(m_pPlayer->pev), CHAN_WEAPON, "weapons/electro4.wav", 1.0, ATTN_NORM, 0, 80 + RANDOM_LONG(0,0x3f));
351 EMIT_SOUND_DYN(ENT(m_pPlayer->pev), CHAN_ITEM, "weapons/electro6.wav", 1.0, ATTN_NORM, 0, 75 + RANDOM_LONG(0,0x3f));
352
353 m_fInAttack = 0;
354 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 1.0;
355 m_pPlayer->m_flNextAttack = UTIL_WeaponTimeBase() + 1.0;
356
357#ifndef CLIENT_DLL
358 m_pPlayer->TakeDamage( VARS(eoNullEntity), VARS(eoNullEntity), 50, DMG_SHOCK );
359 UTIL_ScreenFade( m_pPlayer, Vector(255,128,0), 2, 0.5, 128, FFADE_IN );
360#endif
361 SendWeaponAnim( GAUSS_IDLE );
362
363 // Player may have been killed and this weapon dropped, don't execute any more code after this!
364 return;
365 }

378void CGauss::StartFire(void)
379{
380 float flDamage;
381
382 UTIL_MakeVectors( m_pPlayer->pev->v_angle + m_pPlayer->pev->punchangle );
383 Vector vecAiming = gpGlobals->v_forward;
384 Vector vecSrc = m_pPlayer->GetGunPosition( ); // + gpGlobals->v_up * -8 + gpGlobals->v_right * 8;
385
386 if ( gpGlobals->time - m_pPlayer->m_flStartCharge > GetFullChargeTime() )
387 {
388 flDamage = 200;
389 }
390 else
391 {
392 flDamage = 200 * (( gpGlobals->time - m_pPlayer->m_flStartCharge) / GetFullChargeTime() );
393 }
394
395 if ( m_fPrimaryFire )
396 {
397 // fixed damage on primary attack
398#ifdef CLIENT_DLL
399 flDamage = 20;
400#else
401 flDamage = gSkillData.plrDmgGauss;
402#endif
403 }
404
405 if (m_fInAttack != 3)
406 {
407 //ALERT ( at_console, "Time:%f Damage:%f\n", gpGlobals->time - m_pPlayer->m_flStartCharge, flDamage );
408
409#ifndef CLIENT_DLL
410 float flZVel = m_pPlayer->pev->velocity.z;
411
412 if ( !m_fPrimaryFire )
413 {
414 m_pPlayer->pev->velocity = m_pPlayer->pev->velocity - gpGlobals->v_forward * flDamage * 5;
415 }
416
417 if ( !g_pGameRules->IsMultiplayer() )
418
419 {
420 // in deathmatch, gauss can pop you up into the air. Not in single play.
421 m_pPlayer->pev->velocity.z = flZVel;
422 }
423#endif
424 // player "shoot" animation
425 m_pPlayer->SetAnimation( PLAYER_ATTACK1 );
426 }
427
428 // time until aftershock 'static discharge' sound
429 m_pPlayer->m_flPlayAftershock = gpGlobals->time + UTIL_SharedRandomFloat( m_pPlayer->random_seed, 0.3, 0.8 );
430
431 Fire( vecSrc, vecAiming, flDamage );
432}

434void CGauss::Fire( Vector vecOrigSrc, Vector vecDir, float flDamage )
435{
436 m_pPlayer->m_iWeaponVolume = GAUSS_PRIMARY_FIRE_VOLUME;
437
438 Vector vecSrc = vecOrigSrc;
439 Vector vecDest = vecSrc + vecDir * 8192;
440 edict_t *pentIgnore;
441 TraceResult tr, beam_tr;
442 float flMaxFrac = 1.0;
443 int nTotal = 0;
444 int fHasPunched = 0;
445 int fFirstBeam = 1;
446 int nMaxHits = 10;
447
448 pentIgnore = ENT( m_pPlayer->pev );
449
450#ifdef CLIENT_DLL
451 if ( m_fPrimaryFire == false )
452 g_irunninggausspred = true;
453#endif
454
455 // The main firing event is sent unreliably so it won't be delayed.
456 PLAYBACK_EVENT_FULL( FEV_NOTHOST, m_pPlayer->edict(), m_usGaussFire, 0.0, (float *)&m_pPlayer->pev->origin, (float *)&m_pPlayer->pev->angles, flDamage, 0.0, 0, 0, m_fPrimaryFire ? 1 : 0, 0 );
457
458 // This reliable event is used to stop the spinning sound
459 // It's delayed by a fraction of second to make sure it is delayed by 1 frame on the client
460 // It's sent reliably anyway, which could lead to other delays
461
462 PLAYBACK_EVENT_FULL( FEV_NOTHOST | FEV_RELIABLE, m_pPlayer->edict(), m_usGaussFire, 0.01, (float *)&m_pPlayer->pev->origin, (float *)&m_pPlayer->pev->angles, 0.0, 0.0, 0, 0, 0, 1 );
463
464
465 /*ALERT( at_console, "%f %f %f\n%f %f %f\n",
466 vecSrc.x, vecSrc.y, vecSrc.z,
467 vecDest.x, vecDest.y, vecDest.z );*/
468
469
470// ALERT( at_console, "%f %f\n", tr.flFraction, flMaxFrac );
471
472#ifndef CLIENT_DLL
473 while (flDamage > 10 && nMaxHits > 0)
474 {
475 nMaxHits--;
476
477 // ALERT( at_console, "." );
478 UTIL_TraceLine(vecSrc, vecDest, dont_ignore_monsters, pentIgnore, &tr);
479
480 if (tr.fAllSolid)
481 break;
482
483 CBaseEntity *pEntity = CBaseEntity::Instance(tr.pHit);
484
485 if (pEntity == NULL)
486 break;
487
488 if ( fFirstBeam )
489 {
490 m_pPlayer->pev->effects |= EF_MUZZLEFLASH;
491 fFirstBeam = 0;
492
493 nTotal += 26;
494 }
495
496 if (pEntity->pev->takedamage)
497 {
498 ClearMultiDamage();
499 pEntity->TraceAttack( m_pPlayer->pev, flDamage, vecDir, &tr, DMG_BULLET );
500 ApplyMultiDamage(m_pPlayer->pev, m_pPlayer->pev);
501 }
502
503 if ( pEntity->ReflectGauss() )
504 {
505 float n;
506
507 pentIgnore = NULL;
508
509 n = -DotProduct(tr.vecPlaneNormal, vecDir);
510
511 if (n < 0.5) // 60 degrees
512 {
513 // ALERT( at_console, "reflect %f\n", n );
514 // reflect
515 Vector r;
516
517 r = 2.0 * tr.vecPlaneNormal * n + vecDir;
518 flMaxFrac = flMaxFrac - tr.flFraction;
519 vecDir = r;
520 vecSrc = tr.vecEndPos + vecDir * 8;
521 vecDest = vecSrc + vecDir * 8192;
522
523 // explode a bit
524 m_pPlayer->RadiusDamage( tr.vecEndPos, pev, m_pPlayer->pev, flDamage * n, CLASS_NONE, DMG_BLAST );
525
526 nTotal += 34;
527
528 // lose energy
529 if (n == 0) n = 0.1;
530 flDamage = flDamage * (1 - n);
531 }
532 else
533 {
534 nTotal += 13;
535
536 // limit it to one hole punch
537 if (fHasPunched)
538 break;
539 fHasPunched = 1;
540
541 // try punching through wall if secondary attack (primary is incapable of breaking through)
542 if ( !m_fPrimaryFire )
543 {
544 UTIL_TraceLine( tr.vecEndPos + vecDir * 8, vecDest, dont_ignore_monsters, pentIgnore, &beam_tr);
545 if (!beam_tr.fAllSolid)
546 {
547 // trace backwards to find exit point
548 UTIL_TraceLine( beam_tr.vecEndPos, tr.vecEndPos, dont_ignore_monsters, pentIgnore, &beam_tr);
549
550 float n = (beam_tr.vecEndPos - tr.vecEndPos).Length( );
551
552 if (n < flDamage)
553 {
554 if (n == 0) n = 1;
555 flDamage -= n;
556
557 // ALERT( at_console, "punch %f\n", n );
558 nTotal += 21;
559
560 // exit blast damage
561 //m_pPlayer->RadiusDamage( beam_tr.vecEndPos + vecDir * 8, pev, m_pPlayer->pev, flDamage, CLASS_NONE, DMG_BLAST );
562 float damage_radius;
563
564
565 if ( g_pGameRules->IsMultiplayer() )
566 {
567 damage_radius = flDamage * 1.75; // Old code == 2.5
568 }
569 else
570 {
571 damage_radius = flDamage * 2.5;
572 }
573
574 ::RadiusDamage( beam_tr.vecEndPos + vecDir * 8, pev, m_pPlayer->pev, flDamage, damage_radius, CLASS_NONE, DMG_BLAST );
575
576 CSoundEnt::InsertSound ( bits_SOUND_COMBAT, pev->origin, NORMAL_EXPLOSION_VOLUME, 3.0 );
577
578 nTotal += 53;
579
580 vecSrc = beam_tr.vecEndPos + vecDir;
581 }
582 }
583 else
584 {
585 //ALERT( at_console, "blocked %f\n", n );
586 flDamage = 0;
587 }
588 }
589 else
590 {
591 //ALERT( at_console, "blocked solid\n" );
592
593 flDamage = 0;
594 }
595
596 }
597 }
598 else
599 {
600 vecSrc = tr.vecEndPos + vecDir;
601 pentIgnore = ENT( pEntity->pev );
602 }
603 }
604#endif
605 // ALERT( at_console, "%d bytes\n", nTotal );
606}

617void CGauss::WeaponIdle( void )
618{
619 ResetEmptySound( );
620
621 // play aftershock static discharge
622 if ( m_pPlayer->m_flPlayAftershock && m_pPlayer->m_flPlayAftershock < gpGlobals->time )
623 {
624 switch (RANDOM_LONG(0,3))
625 {
626 case 0: EMIT_SOUND(ENT(m_pPlayer->pev), CHAN_WEAPON, "weapons/electro4.wav", RANDOM_FLOAT(0.7, 0.8), ATTN_NORM); break;
627 case 1: EMIT_SOUND(ENT(m_pPlayer->pev), CHAN_WEAPON, "weapons/electro5.wav", RANDOM_FLOAT(0.7, 0.8), ATTN_NORM); break;
628 case 2: EMIT_SOUND(ENT(m_pPlayer->pev), CHAN_WEAPON, "weapons/electro6.wav", RANDOM_FLOAT(0.7, 0.8), ATTN_NORM); break;
629 case 3: break; // no sound
630 }
631 m_pPlayer->m_flPlayAftershock = 0.0;
632 }
633
634 if (m_flTimeWeaponIdle > UTIL_WeaponTimeBase())
635 return;
636
637 if (m_fInAttack != 0)
638 {
639 StartFire();
640 m_fInAttack = 0;
641 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 2.0;
642 }
643 else
644 {
645 int iAnim;
646 float flRand = RANDOM_FLOAT(0, 1);
647 if (flRand <= 0.5)
648 {
649 iAnim = GAUSS_IDLE;
650 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat( m_pPlayer->random_seed, 10, 15 );
651 }
652 else if (flRand <= 0.75)
653 {
654 iAnim = GAUSS_IDLE2;
655 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat( m_pPlayer->random_seed, 10, 15 );
656 }
657 else
658 {
659 iAnim = GAUSS_FIDGET;
660 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 3;
661 }
662
663 SendWeaponAnim(iAnim);
664 }
665}
