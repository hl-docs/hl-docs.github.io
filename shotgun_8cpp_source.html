163void CShotgun::PrimaryAttack()
164{
165 // don't fire underwater
166 if (m_pPlayer->pev->waterlevel == 3)
167 {
168 PlayEmptySound( );
169 m_flNextPrimaryAttack = GetNextAttackDelay(0.15);
170 return;
171 }
172
173 if (m_iClip <= 0)
174 {
175 Reload( );
176 if (m_iClip == 0)
177 PlayEmptySound( );
178 return;
179 }
180
181 m_pPlayer->m_iWeaponVolume = LOUD_GUN_VOLUME;
182 m_pPlayer->m_iWeaponFlash = NORMAL_GUN_FLASH;
183
184 m_iClip--;
185
186 int flags;
187#if defined( CLIENT_WEAPONS )
188 flags = FEV_NOTHOST;
189#else
190 flags = 0;
191#endif
192
193
194 m_pPlayer->pev->effects = (int)(m_pPlayer->pev->effects) | EF_MUZZLEFLASH;
195
196 Vector vecSrc = m_pPlayer->GetGunPosition( );
197 Vector vecAiming = m_pPlayer->GetAutoaimVector( AUTOAIM_5DEGREES );
198
199 Vector vecDir;
200
201#ifdef CLIENT_DLL
202 if ( bIsMultiplayer() )
203#else
204 if ( g_pGameRules->IsMultiplayer() )
205#endif
206 {
207 vecDir = m_pPlayer->FireBulletsPlayer( 4, vecSrc, vecAiming, VECTOR_CONE_DM_SHOTGUN, 2048, BULLET_PLAYER_BUCKSHOT, 0, 0, m_pPlayer->pev, m_pPlayer->random_seed );
208 }
209 else
210 {
211 // regular old, untouched spread.
212 vecDir = m_pPlayer->FireBulletsPlayer( 6, vecSrc, vecAiming, VECTOR_CONE_10DEGREES, 2048, BULLET_PLAYER_BUCKSHOT, 0, 0, m_pPlayer->pev, m_pPlayer->random_seed );
213 }
214
215 PLAYBACK_EVENT_FULL( flags, m_pPlayer->edict(), m_usSingleFire, 0.0, (float *)&g_vecZero, (float *)&g_vecZero, vecDir.x, vecDir.y, 0, 0, 0, 0 );
216
217
218 if (!m_iClip && m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0)
219 // HEV suit - indicate out of ammo condition
220 m_pPlayer->SetSuitUpdate("!HEV_AMO0", FALSE, 0);
221
222 if (m_iClip != 0)
223 m_flPumpTime = gpGlobals->time + 0.5;
224
225 m_flNextPrimaryAttack = GetNextAttackDelay(0.75);
226 m_flNextSecondaryAttack = UTIL_WeaponTimeBase() + 0.75;
227 if (m_iClip != 0)
228 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 5.0;
229 else
230 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 0.75;
231 m_fInSpecialReload = 0;
232}

321void CShotgun::Reload( void )
322{
323 if (m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0 || m_iClip == SHOTGUN_MAX_CLIP)
324 return;
325
326 // don't reload until recoil is done
327 if (m_flNextPrimaryAttack > UTIL_WeaponTimeBase())
328 return;
329
330 // check to see if we're ready to reload
331 if (m_fInSpecialReload == 0)
332 {
333 SendWeaponAnim( SHOTGUN_START_RELOAD );
334 m_fInSpecialReload = 1;
335 m_pPlayer->m_flNextAttack = UTIL_WeaponTimeBase() + 0.6;
336 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 0.6;
337 m_flNextPrimaryAttack = GetNextAttackDelay(1.0);
338 m_flNextSecondaryAttack = UTIL_WeaponTimeBase() + 1.0;
339 return;
340 }
341 else if (m_fInSpecialReload == 1)
342 {
343 if (m_flTimeWeaponIdle > UTIL_WeaponTimeBase())
344 return;
345 // was waiting for gun to move to side
346 m_fInSpecialReload = 2;
347
348 if (RANDOM_LONG(0,1))
349 EMIT_SOUND_DYN(ENT(m_pPlayer->pev), CHAN_ITEM, "weapons/reload1.wav", 1, ATTN_NORM, 0, 85 + RANDOM_LONG(0,0x1f));
350 else
351 EMIT_SOUND_DYN(ENT(m_pPlayer->pev), CHAN_ITEM, "weapons/reload3.wav", 1, ATTN_NORM, 0, 85 + RANDOM_LONG(0,0x1f));
352
353 SendWeaponAnim( SHOTGUN_RELOAD );
354
355 m_flNextReload = UTIL_WeaponTimeBase() + 0.5;
356 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 0.5;
357 }
358 else
359 {
360 // Add them to the clip
361 m_iClip += 1;
362 m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] -= 1;
363 m_fInSpecialReload = 1;
364 }
365}

373void CShotgun::WeaponIdle( void )
374{
375 ResetEmptySound( );
376
377 m_pPlayer->GetAutoaimVector( AUTOAIM_5DEGREES );
378
379 if ( m_flPumpTime && m_flPumpTime < gpGlobals->time )
380 {
381 // play pumping sound
382 EMIT_SOUND_DYN(ENT(m_pPlayer->pev), CHAN_ITEM, "weapons/scock1.wav", 1, ATTN_NORM, 0, 95 + RANDOM_LONG(0,0x1f));
383 m_flPumpTime = 0;
384 }
385
386 if (m_flTimeWeaponIdle < UTIL_WeaponTimeBase() )
387 {
388 if (m_iClip == 0 && m_fInSpecialReload == 0 && m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType])
389 {
390 Reload( );
391 }
392 else if (m_fInSpecialReload != 0)
393 {
394 if (m_iClip != 8 && m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType])
395 {
396 Reload( );
397 }
398 else
399 {
400 // reload debounce has timed out
401 SendWeaponAnim( SHOTGUN_PUMP );
402
403 // play cocking sound
404 EMIT_SOUND_DYN(ENT(m_pPlayer->pev), CHAN_ITEM, "weapons/scock1.wav", 1, ATTN_NORM, 0, 95 + RANDOM_LONG(0,0x1f));
405 m_fInSpecialReload = 0;
406 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 1.5;
407 }
408 }
409 else
410 {
411 int iAnim;
412 float flRand = UTIL_SharedRandomFloat( m_pPlayer->random_seed, 0, 1 );
413 if (flRand <= 0.8)
414 {
415 iAnim = SHOTGUN_IDLE_DEEP;
416 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + (60.0/12.0);// * RANDOM_LONG(2, 5);
417 }
418 else if (flRand <= 0.95)
419 {
420 iAnim = SHOTGUN_IDLE;
421 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + (20.0/9.0);
422 }
423 else
424 {
425 iAnim = SHOTGUN_IDLE4;
426 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + (20.0/9.0);
427 }
428 SendWeaponAnim( iAnim );
429 }
430 }
