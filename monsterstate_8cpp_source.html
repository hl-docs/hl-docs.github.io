99void CBaseMonster::RunAI(void)
100{
101 // IDLE sound permitted in ALERT state is because monsters were silent in ALERT state.
102 // Only play IDLE sound in IDLE state once we have sounds for that state.
103 if ((m_MonsterState == MONSTERSTATE_IDLE || m_MonsterState == MONSTERSTATE_ALERT) && RANDOM_LONG(0, 99) == 0 && !(pev->flags & SF_MONSTER_GAG))
104 {
105 IdleSound(); // Play idle sound
106 }
107
108 if (m_MonsterState != MONSTERSTATE_NONE &&
109 m_MonsterState != MONSTERSTATE_PRONE &&
110 m_MonsterState != MONSTERSTATE_DEAD) // Don't bother if monster is prone or dead
111 {
112 // Collect sensory condition information.
113 // Don't let monsters outside of the player's PVS act up.
114 if (!FNullEnt(FIND_CLIENT_IN_PVS(edict())) || (m_MonsterState == MONSTERSTATE_COMBAT))
115 {
116 Look(m_flDistLook); // Look for entities
117 Listen(); // Check for audible sounds
118
119 // Filter conditions
120 ClearConditions(IgnoreConditions());
121
122 GetEnemy(); // Get the current enemy
123 }
124
125 // Do these calculations if the monster has an enemy
126 if (m_hEnemy != NULL)
127 {
128 CheckEnemy(m_hEnemy); // Check the enemy's status
129 }
130
131 CheckAmmo(); // Check the monster's ammo
132 }
133
134 FCheckAITrigger(); // Check for AI triggers
135
136 PrescheduleThink(); // Prepare for the next think
137
138 MaintainSchedule(); // Maintain the current schedule
139
140 // Clear unused conditions
141 m_afConditions &= ~(bits_COND_LIGHT_DAMAGE | bits_COND_HEAVY_DAMAGE);
142}

154MONSTERSTATE CBaseMonster::GetIdealState(void)
155{
156 int iConditions;
157
158 iConditions = IScheduleFlags(); // Get current schedule conditions
159
160 // Determine the ideal state based on the current monster state
161 switch (m_MonsterState)
162 {
163 case MONSTERSTATE_IDLE:
164 // IDLE state transitions
165 if (iConditions & bits_COND_NEW_ENEMY)
166 {
167 m_IdealMonsterState = MONSTERSTATE_COMBAT; // New enemy detected
168 }
169 else if (iConditions & (bits_COND_LIGHT_DAMAGE | bits_COND_HEAVY_DAMAGE))
170 {
171 MakeIdealYaw(m_vecEnemyLKP); // Turn towards the enemy
172 m_IdealMonsterState = MONSTERSTATE_ALERT; // Transition to ALERT state
173 }
174 else if (iConditions & bits_COND_HEAR_SOUND)
175 {
176 CSound *pSound = PBestSound(); // Get the best sound
177 ASSERT(pSound != NULL);
178 if (pSound)
179 {
180 MakeIdealYaw(pSound->m_vecOrigin); // Turn towards the sound
181 if (pSound->m_iType & (bits_SOUND_COMBAT | bits_SOUND_DANGER))
182 m_IdealMonsterState = MONSTERSTATE_ALERT; // Transition to ALERT state
183 }
184 }
185 else if (iConditions & (bits_COND_SMELL | bits_COND_SMELL_FOOD))
186 {
187 m_IdealMonsterState = MONSTERSTATE_ALERT; // Transition to ALERT state
188 }
189 break;
190
191 case MONSTERSTATE_ALERT:
192 // ALERT state transitions
193 if (iConditions & (bits_COND_NEW_ENEMY | bits_COND_SEE_ENEMY))
194 {
195 m_IdealMonsterState = MONSTERSTATE_COMBAT; // See an enemy, must attack
196 }
197 else if (iConditions & bits_COND_HEAR_SOUND)
198 {
199 m_IdealMonsterState = MONSTERSTATE_ALERT; // Stay alert
200 CSound *pSound = PBestSound(); // Get the best sound
201 ASSERT(pSound != NULL);
202 if (pSound)
203 MakeIdealYaw(pSound->m_vecOrigin); // Turn towards the sound
204 }
205 break;
206
207 case MONSTERSTATE_COMBAT:
208 // COMBAT state transitions
209 if (m_hEnemy == NULL)
210 {
211 m_IdealMonsterState = MONSTERSTATE_ALERT; // No enemy, go alert
212 ALERT(at_aiconsole, "***Combat state with no enemy!\n");
213 }
214 break;
215
216 case MONSTERSTATE_HUNT:
217 // HUNT state transitions
218 break;
219
220 case MONSTERSTATE_SCRIPT:
221 if (iConditions & (bits_COND_TASK_FAILED | bits_COND_LIGHT_DAMAGE | bits_COND_HEAVY_DAMAGE))
222 {
223 ExitScriptedSequence(); // Exit scripted sequence, set ideal state
224 }
225 break;
226
227 case MONSTERSTATE_DEAD:
228 m_IdealMonsterState = MONSTERSTATE_DEAD; // Stay dead
229 break;
230 }
231
232 return m_IdealMonsterState; // Return the ideal state
233}
