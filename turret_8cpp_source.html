410void CBaseTurret::Initialize(void)
411{
412 m_iOn = 0;
413 m_fBeserk = 0;
414 m_iSpin = 0;
415
416 SetBoneController( 0, 0 );
417 SetBoneController( 1, 0 );
418
419 if (m_iBaseTurnRate == 0) m_iBaseTurnRate = TURRET_TURNRATE;
420 if (m_flMaxWait == 0) m_flMaxWait = TURRET_MAXWAIT;
421 m_flStartYaw = pev->angles.y;
422 if (m_iOrientation == 1)
423 {
424 pev->idealpitch = 180;
425 pev->angles.x = 180;
426 pev->view_ofs.z = -pev->view_ofs.z;
427 pev->effects |= EF_INVLIGHT;
428 pev->angles.y = pev->angles.y + 180;
429 if (pev->angles.y > 360)
430 pev->angles.y = pev->angles.y - 360;
431 }
432
433 m_vecGoalAngles.x = 0;
434
435 if (m_iAutoStart)
436 {
437 m_flLastSight = gpGlobals->time + m_flMaxWait;
438 SetThink(&CBaseTurret::AutoSearchThink);
439 pev->nextthink = gpGlobals->time + .1;
440 }
441 else
442 SetThink(&CBaseTurret::SUB_DoNothing);
443}

456void CBaseTurret::TurretUse( CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value )
457{
458 if ( !ShouldToggle( useType, m_iOn ) )
459 return;
460
461 if (m_iOn)
462 {
463 m_hEnemy = NULL;
464 pev->nextthink = gpGlobals->time + 0.1;
465 m_iAutoStart = FALSE;// switching off a turret disables autostart
467 SetThink(&CBaseTurret::Retire);
468 }
469 else
470 {
471 pev->nextthink = gpGlobals->time + 0.1; // turn on delay
472
473 // if the turret is flagged as an autoactivate turret, re-enable it's ability open self.
474 if ( pev->spawnflags & SF_MONSTER_TURRET_AUTOACTIVATE )
475 {
476 m_iAutoStart = TRUE;
477 }
478
479 SetThink(&CBaseTurret::Deploy);
480 }

550void CBaseTurret::ActiveThink(void)
551{
552 int fAttack = 0;
553 Vector vecDirToEnemy;
554
555 pev->nextthink = gpGlobals->time + 0.1;
556 StudioFrameAdvance( );
557
558 if ((!m_iOn) || (m_hEnemy == NULL))
559 {
560 m_hEnemy = NULL;
561 m_flLastSight = gpGlobals->time + m_flMaxWait;
562 SetThink(&CBaseTurret::SearchThink);
563 return;
564 }
565
566 // if it's dead, look for something new
567 if ( !m_hEnemy->IsAlive() )
568 {
569 if (!m_flLastSight)
570 {
571 m_flLastSight = gpGlobals->time + 0.5; // continue-shooting timeout
572 }
573 else
574 {
575 if (gpGlobals->time > m_flLastSight)
576 {
577 m_hEnemy = NULL;
578 m_flLastSight = gpGlobals->time + m_flMaxWait;
579 SetThink(&CBaseTurret::SearchThink);
580 return;
581 }
582 }
583 }
584
585 Vector vecMid = pev->origin + pev->view_ofs;
586 Vector vecMidEnemy = m_hEnemy->BodyTarget( vecMid );
587
588 // Look for our current enemy
589 int fEnemyVisible = FBoxVisible(pev, m_hEnemy->pev, vecMidEnemy );
590
591 vecDirToEnemy = vecMidEnemy - vecMid; // calculate dir and dist to enemy
592 float flDistToEnemy = vecDirToEnemy.Length();
593
594 Vector vec = UTIL_VecToAngles(vecMidEnemy - vecMid);
595
596 // Current enmey is not visible.
597 if (!fEnemyVisible || (flDistToEnemy > TURRET_RANGE))
598 {
599 if (!m_flLastSight)
600 m_flLastSight = gpGlobals->time + 0.5;
601 else
602 {
603 // Should we look for a new target?
604 if (gpGlobals->time > m_flLastSight)
605 {
606 m_hEnemy = NULL;
607 m_flLastSight = gpGlobals->time + m_flMaxWait;
608 SetThink(&CBaseTurret::SearchThink);
609 return;
610 }
611 }
612 fEnemyVisible = 0;
613 }
614 else
615 {
616 m_vecLastSight = vecMidEnemy;
617 }
618
619 UTIL_MakeAimVectors(m_vecCurAngles);
620
621 /*
622 ALERT( at_console, "%.0f %.0f : %.2f %.2f %.2f\n",
623 m_vecCurAngles.x, m_vecCurAngles.y,
624 gpGlobals->v_forward.x, gpGlobals->v_forward.y, gpGlobals->v_forward.z );
625 */
626
627 Vector vecLOS = vecDirToEnemy; //vecMid - m_vecLastSight;
628 vecLOS = vecLOS.Normalize();
629
630 // Is the Gun looking at the target
631 if (DotProduct(vecLOS, gpGlobals->v_forward) <= 0.866) // 30 degree slop
632 fAttack = FALSE;
633 else
634 fAttack = TRUE;
635
636 // fire the gun
637 if (m_iSpin && ((fAttack) || (m_fBeserk)))
638 {
639 Vector vecSrc, vecAng;
640 GetAttachment( 0, vecSrc, vecAng );
641 SetTurretAnim(TURRET_ANIM_FIRE);
642 Shoot(vecSrc, gpGlobals->v_forward );
643 }
644 else
645 {
646 SetTurretAnim(TURRET_ANIM_SPIN);
647 }
648
649 //move the gun
650 if (m_fBeserk)
651 {
652 if (RANDOM_LONG(0,9) == 0)
653 {
654 m_vecGoalAngles.y = RANDOM_FLOAT(0,360);
655 m_vecGoalAngles.x = RANDOM_FLOAT(0,90) - 90 * m_iOrientation;
656 TakeDamage(pev,pev,1, DMG_GENERIC); // don't beserk forever
657 return;
658 }
659 }
660 else if (fEnemyVisible)
661 {
662 if (vec.y > 360)
663 vec.y -= 360;
664
665 if (vec.y < 0)
666 vec.y += 360;
667
668 //ALERT(at_console, "[%.2f]", vec.x);
669
670 if (vec.x < -180)
671 vec.x += 360;
672
673 if (vec.x > 180)
674 vec.x -= 360;
675
676 // now all numbers should be in [1...360]
677 // pin to turret limitations to [-90...15]
678
679 if (m_iOrientation == 0)
680 {
681 if (vec.x > 90)
682 vec.x = 90;
683 else if (vec.x < m_iMinPitch)
684 vec.x = m_iMinPitch;
685 }
686 else
687 {
688 if (vec.x < -90)
689 vec.x = -90;
690 else if (vec.x > -m_iMinPitch)
691 vec.x = -m_iMinPitch;
692 }
693
694 // ALERT(at_console, "->[%.2f]\n", vec.x);
695
696 m_vecGoalAngles.y = vec.y;
697 m_vecGoalAngles.x = vec.x;
698
699 }
700
701 SpinUpCall();
702 MoveTurret();
703}

752void CBaseTurret::Deploy(void)
753{
754 pev->nextthink = gpGlobals->time + 0.1;
755 StudioFrameAdvance( );
756
757 if (pev->sequence != TURRET_ANIM_DEPLOY)
758 {
759 m_iOn = 1;
760 SetTurretAnim(TURRET_ANIM_DEPLOY);
761 EMIT_SOUND(ENT(pev), CHAN_BODY, "turret/tu_deploy.wav", TURRET_MACHINE_VOLUME, ATTN_NORM);
762 SUB_UseTargets( this, USE_ON, 0 );
763 }
764
765 if (m_fSequenceFinished)
766 {
767 pev->maxs.z = m_iDeployHeight;
768 pev->mins.z = -m_iDeployHeight;
769 UTIL_SetSize(pev, pev->mins, pev->maxs);
770
771 m_vecCurAngles.x = 0;
772
773 if (m_iOrientation == 1)
774 {
775 m_vecCurAngles.y = UTIL_AngleMod( pev->angles.y + 180 );
776 }
777 else
778 {
779 m_vecCurAngles.y = UTIL_AngleMod( pev->angles.y );
780 }
781
782 SetTurretAnim(TURRET_ANIM_SPIN);
783 pev->framerate = 0;
784 SetThink(&CBaseTurret::SearchThink);
785 }
786
787 m_flLastSight = gpGlobals->time + m_flMaxWait;
788}

796void CBaseTurret::Retire(void)
797{
798 // make the turret level
799 m_vecGoalAngles.x = 0;
800 m_vecGoalAngles.y = m_flStartYaw;
801
802 pev->nextthink = gpGlobals->time + 0.1;
803
804 StudioFrameAdvance( );
805
806 EyeOff( );
807
808 if (!MoveTurret())
809 {
810 if (m_iSpin)
811 {
812 SpinDownCall();
813 }
814 else if (pev->sequence != TURRET_ANIM_RETIRE)

1223int CBaseTurret::MoveTurret(void)
1224{
1225 int state = 0;
1226 // any x movement?
1227
1228 if (m_vecCurAngles.x != m_vecGoalAngles.x)
1229 {
1230 float flDir = m_vecGoalAngles.x > m_vecCurAngles.x ? 1 : -1 ;
1231
1232 m_vecCurAngles.x += 0.1 * m_fTurnRate * flDir;
1233
1234 // if we started below the goal, and now we're past, peg to goal
1235 if (flDir == 1)
1236 {
1237 if (m_vecCurAngles.x > m_vecGoalAngles.x)
1238 m_vecCurAngles.x = m_vecGoalAngles.x;
1239 }
1240 else
1241 {
1242 if (m_vecCurAngles.x < m_vecGoalAngles.x)
1243 m_vecCurAngles.x = m_vecGoalAngles.x;
1244 }
1245
1246 if (m_iOrientation == 0)
1247 SetBoneController(1, -m_vecCurAngles.x);
1248 else
1249 SetBoneController(1, m_vecCurAngles.x);
1250 state = 1;
1251 }
1252
1253 if (m_vecCurAngles.y != m_vecGoalAngles.y)
1254 {
1255 float flDir = m_vecGoalAngles.y > m_vecCurAngles.y ? 1 : -1 ;
1256 float flDist = fabs(m_vecGoalAngles.y - m_vecCurAngles.y);
1257
1258 if (flDist > 180)
1259 {
1260 flDist = 360 - flDist;
1261 flDir = -flDir;
1262 }
1263 if (flDist > 30)
1264 {
1265 if (m_fTurnRate < m_iBaseTurnRate * 10)
1266 {
1267 m_fTurnRate += m_iBaseTurnRate;
1268 }
1269 }
1270 else if (m_fTurnRate > 45)
1271 {
1272 m_fTurnRate -= m_iBaseTurnRate;
1273 }
1274 else
1275 {
1276 m_fTurnRate += m_iBaseTurnRate;
1277 }
1278
1279 m_vecCurAngles.y += 0.1 * m_fTurnRate * flDir;
1280
1281 if (m_vecCurAngles.y < 0)
1282 m_vecCurAngles.y += 360;
1283 else if (m_vecCurAngles.y >= 360)
1284 m_vecCurAngles.y -= 360;
1285
1286 if (flDist < (0.05 * m_iBaseTurnRate))
1287 m_vecCurAngles.y = m_vecGoalAngles.y;
1288
1289 //ALERT(at_console, "%.2f -> %.2f\n", m_vecCurAngles.y, y);
1290 if (m_iOrientation == 0)
1291 SetBoneController(0, m_vecCurAngles.y - pev->angles.y );
1292 else
1293 SetBoneController(0, pev->angles.y - 180 - m_vecCurAngles.y );
1294 state = 1;
1295 }
1296
1297 if (!state)
1298 m_fTurnRate = m_iBaseTurnRate;
1299
1300 //ALERT(at_console, "(%.2f, %.2f)->(%.2f, %.2f)\n", m_vecCurAngles.x,
1301 // m_vecCurAngles.y, m_vecGoalAngles.x, m_vecGoalAngles.y);
1302 return state;
