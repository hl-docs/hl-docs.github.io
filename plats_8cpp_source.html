297void CFuncPlat :: Setup( void )
298{
299 if (m_flTLength == 0)
300 m_flTLength = 80;
301 if (m_flTWidth == 0)
302 m_flTWidth = 10;
303
304 pev->angles = g_vecZero;
305 pev->solid = SOLID_BSP;
306 pev->movetype = MOVETYPE_PUSH;
307
308 UTIL_SetOrigin(pev, pev->origin); // Set size and link into world
309 UTIL_SetSize(pev, pev->mins, pev->maxs);
310 SET_MODEL(ENT(pev), STRING(pev->model));
311
312 // vecPosition1 is the top position, vecPosition2 is the bottom
313 m_vecPosition1 = pev->origin;
314 m_vecPosition2 = pev->origin;
315 if (m_flHeight != 0)
316 m_vecPosition2.z = pev->origin.z - m_flHeight;
317 else
318 m_vecPosition2.z = pev->origin.z - pev->size.z + 8;
319 if (pev->speed == 0)
320 pev->speed = 150;
321
322 if (m_volume == 0)
323 m_volume = 0.85;
324}

385void CPlatTrigger :: SpawnInsideTrigger( CFuncPlat *pPlatform )
386{
387 m_pPlatform = pPlatform;
388 // Create trigger entity, "point" it at the owning platform, give it a touch method
389 pev->solid = SOLID_TRIGGER;
390 pev->movetype = MOVETYPE_NONE;
391 pev->origin = pPlatform->pev->origin;
392
393 // Establish the trigger field's size
394 Vector vecTMin = m_pPlatform->pev->mins + Vector ( 25 , 25 , 0 );
395 Vector vecTMax = m_pPlatform->pev->maxs + Vector ( 25 , 25 , 8 );
396 vecTMin.z = vecTMax.z - ( m_pPlatform->m_vecPosition1.z - m_pPlatform->m_vecPosition2.z + 8 );
397 if (m_pPlatform->pev->size.x <= 50)
398 {
399 vecTMin.x = (m_pPlatform->pev->mins.x + m_pPlatform->pev->maxs.x) / 2;
400 vecTMax.x = vecTMin.x + 1;
401 }
402 if (m_pPlatform->pev->size.y <= 50)
403 {
404 vecTMin.y = (m_pPlatform->pev->mins.y + m_pPlatform->pev->maxs.y) / 2;
405 vecTMax.y = vecTMin.y + 1;
406 }
407 UTIL_SetSize ( pev, vecTMin, vecTMax );
408}

418void CPlatTrigger :: Touch( CBaseEntity *pOther )
419{
420 // Ignore touches by non-players
421 entvars_t* pevToucher = pOther->pev;
422 if ( !FClassnameIs (pevToucher, "player") )
423 return;
424
425 // Ignore touches by corpses
426 if (!pOther->IsAlive()||!m_pPlatform||!m_pPlatform->pev)
427 return;
428
429 // Make linked platform go up/down.
430 if (m_pPlatform->m_toggle_state == TS_AT_BOTTOM)
431 m_pPlatform->GoUp();
432 else if (m_pPlatform->m_toggle_state == TS_AT_TOP)
433 m_pPlatform->pev->nextthink = m_pPlatform->pev->ltime + 1;// delay going down
434}

447void CFuncPlat :: PlatUse( CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value )
448{
449 if ( IsTogglePlat() )
450 {
451 // Top is off, bottom is on
452 BOOL on = (m_toggle_state == TS_AT_BOTTOM) ? TRUE : FALSE;
453
454 if ( !ShouldToggle( useType, on ) )
455 return;
456
457 if (m_toggle_state == TS_AT_TOP)
458 GoDown();
459 else if ( m_toggle_state == TS_AT_BOTTOM )
460 GoUp();
461 }
462 else
463 {
464 SetUse( NULL );
465
466 if (m_toggle_state == TS_AT_TOP)
467 GoDown();
468 }
469}

557{
558 ALERT( at_aiconsole, "%s Blocked by %s\n", STRING(pev->classname), STRING(pOther->pev->classname) );

568 GoDown();
569 else if (m_toggle_state == TS_GOING_DOWN)

793{

795 {

805 pev->target = pev->enemy->v.targetname;

807 pev->velocity = g_vecZero;

819void CFuncTrain::Wait(void)
820{
821 // Fire the pass target if there is one
822 if (m_pevCurrentTarget->message)
823 {
824 FireTargets(STRING(m_pevCurrentTarget->message), this, this, USE_TOGGLE, 0);
825 if (FBitSet(m_pevCurrentTarget->spawnflags, SF_CORNER_FIREONCE))
826 m_pevCurrentTarget->message = 0;
827 }
828
829 // Need pointer to LAST target.
830 if (FBitSet(m_pevCurrentTarget->spawnflags, SF_TRAIN_WAIT_RETRIGGER) || (pev->spawnflags & SF_TRAIN_WAIT_RETRIGGER))
831 {
832 pev->spawnflags |= SF_TRAIN_WAIT_RETRIGGER;
833 // Clear the sound channel.
834 if (pev->noiseMovement)
835 STOP_SOUND(edict(), CHAN_STATIC, (char*)STRING(pev->noiseMovement));
836 if (pev->noiseStopMoving)
837 EMIT_SOUND(ENT(pev), CHAN_VOICE, (char*)STRING(pev->noiseStopMoving), m_volume, ATTN_NORM);
838 pev->nextthink = 0;
839 return;
840 }
841
842 if (m_flWait != 0) // -1 wait will wait forever!
843 {
844 pev->nextthink = pev->ltime + m_flWait;
845 if (pev->noiseMovement)
846 STOP_SOUND(edict(), CHAN_STATIC, (char*)STRING(pev->noiseMovement));
847 if (pev->noiseStopMoving)
848 EMIT_SOUND(ENT(pev), CHAN_VOICE, (char*)STRING(pev->noiseStopMoving), m_volume, ATTN_NORM);
849 SetThink(&CFuncTrain::Next);
850 }
851 else
852 {
853 Next(); // Do it RIGHT now!
854 }
855}

1265void CFuncTrackTrain::Next(void)
1266{
1267 float time = 0.5;
1268
1269 if (!pev->speed)
1270 {
1271 ALERT(at_aiconsole, "TRAIN(%s): Speed is 0\n", STRING(pev->targetname));
1272 StopSound();
1273 return;
1274 }
1275
1276 if (!m_ppath)
1277 {
1278 ALERT(at_aiconsole, "TRAIN(%s): Lost path\n", STRING(pev->targetname));
1279 StopSound();
1280 return;
1281 }
1282
1283 UpdateSound();
1284
1285 Vector nextPos = pev->origin;
1286
1287 nextPos.z -= m_height;
1288 CPathTrack *pnext = m_ppath->LookAhead(&nextPos, pev->speed * 0.1, 1);
1289 nextPos.z += m_height;
1290
1291 pev->velocity = (nextPos - pev->origin) * 10;
1292 Vector nextFront = pev->origin;
1293
1294 nextFront.z -= m_height;
1295 if (m_length > 0)
1296 m_ppath->LookAhead(&nextFront, m_length, 0);
1297 else
1298 m_ppath->LookAhead(&nextFront, 100, 0);
1299 nextFront.z += m_height;
1300
1301 Vector delta = nextFront - pev->origin;
1302 Vector angles = UTIL_VecToAngles(delta);
1303 // The train actually points west
1304 angles.y += 180;
1305
1306 // !!! All of this crap has to be done to make the angles not wrap around, revisit this.
1307 FixupAngles(angles);
1308 FixupAngles(pev->angles);
1309
1310 if (!pnext || (delta.x == 0 && delta.y == 0))
1311 angles = pev->angles;
1312
1313 float vy, vx;
1314 if (!(pev->spawnflags & SF_TRACKTRAIN_NOPITCH))
1315 vx = UTIL_AngleDistance(angles.x, pev->angles.x);
1316 else
1317 vx = 0;
1318 vy = UTIL_AngleDistance(angles.y, pev->angles.y);
1319
1320 pev->avelocity.y = vy * 10;
1321 pev->avelocity.x = vx * 10;
1322
1323 if (m_flBank != 0)
1324 {
1325 if (pev->avelocity.y < -5)
1326 pev->avelocity.z = UTIL_AngleDistance(UTIL_ApproachAngle(-m_flBank, pev->angles.z, m_flBank * 2), pev->angles.z);
1327 else if (pev->avelocity.y > 5)
1328 pev->avelocity.z = UTIL_AngleDistance(UTIL_ApproachAngle(m_flBank, pev->angles.z, m_flBank * 2), pev->angles.z);
1329 else
1330 pev->avelocity.z = UTIL_AngleDistance(UTIL_ApproachAngle(0, pev->angles.z, m_flBank * 4), pev->angles.z) * 4;
1331 }
1332
1333 if (pnext)
1334 {
1335 if (pnext != m_ppath)
1336 {
1337 CPathTrack *pFire;
1338 if (pev->speed >= 0)
1339 pFire = pnext;
1340 else
1341 pFire = m_ppath;
1342
1343 m_ppath = pnext;
1344 // Fire the pass target if there is one
1345 if (pFire->pev->message)
1346 {
1347 FireTargets(STRING(pFire->pev->message), this, this, USE_TOGGLE, 0);
1348 if (FBitSet(pFire->pev->spawnflags, SF_PATH_FIREONCE))
1349 pFire->pev->message = 0;
1350 }
1351
1352 if (pFire->pev->spawnflags & SF_PATH_DISABLE_TRAIN)
1353 pev->spawnflags |= SF_TRACKTRAIN_NOCONTROL;
1354
1355 // Don't override speed if under user control
1356 if (pev->spawnflags & SF_TRACKTRAIN_NOCONTROL)
1357 {
1358 if (pFire->pev->speed != 0)
1359 { // Don't copy speed from target if it is 0 (uninitialized)
1360 pev->speed = pFire->pev->speed;
1361 ALERT(at_aiconsole, "TrackTrain %s speed to %4.2f\n", STRING(pev->targetname), pev->speed);
1362 }
1363 }
1364 }
1365 SetThink(&CFuncTrackTrain::Next);
1366 NextThink(pev->ltime + time, TRUE);
1367 }
1368 else // End of path, stop
1369 {
1370 StopSound();
1371 pev->velocity = (nextPos - pev->origin);
1372 pev->avelocity = g_vecZero;
1373 float distance = pev->velocity.Length();
1374 m_oldSpeed = pev->speed;
1375
1376 pev->speed = 0;
1377
1378 // Move to the dead end
1379
1380 // Are we there yet?
1381 if (distance > 0)
1382 {
1383 // No, how long to get there?
1384 time = distance / m_oldSpeed;
1385 pev->velocity = pev->velocity * (m_oldSpeed / distance);
1386 SetThink(&CFuncTrackTrain::DeadEnd);
1387 NextThink(pev->ltime + time, FALSE);
1388 }
1389 else
1390 {
1391 DeadEnd();
1392 }
1393 }
1394}

1848TRAIN_CODE CFuncTrackChange::EvaluateTrain(CPathTrack *pcurrent)
1849{

1853
1854 if (m_train->m_ppath == pcurrent || (pcurrent->m_pprevious && m_train->m_ppath == pcurrent->m_pprevious) ||
1855 (pcurrent->m_pnext && m_train->m_ppath == pcurrent->m_pnext))

1858 return TRAIN_BLOCKING;

1861 float length = dist.Length2D();

1863 return TRAIN_FOLLOWING;

1880void CFuncTrackChange::UpdateTrain(Vector &dest)
1881{
1882 float time = (pev->nextthink - pev->ltime);
1883
1884 m_train->pev->velocity = pev->velocity;
1885 m_train->pev->avelocity = pev->avelocity;
1886 m_train->NextThink(m_train->pev->ltime + time, FALSE);
1887
1888 // Attempt at getting the train to rotate properly around the origin of the track change
1889 if (time <= 0)
1890 return;
1891
1892 Vector offset = m_train->pev->origin - pev->origin;
1893 Vector delta = dest - pev->angles;
1894 // Transform offset into local coordinates
1895 UTIL_MakeInvVectors(delta, gpGlobals);
1896 Vector local;
1897 local.x = DotProduct(offset, gpGlobals->v_forward);
1898 local.y = DotProduct(offset, gpGlobals->v_right);
1899 local.z = DotProduct(offset, gpGlobals->v_up);
1900
1901 local = local - offset;
1902 m_train->pev->velocity = pev->velocity + (local * (1.0 / time));
