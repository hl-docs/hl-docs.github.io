99int CSqueakGrenade::Classify(void)
100{

230void CSqueakGrenade::HuntThink(void)
231{
232 if (!IsInWorld()) // Check if in world
233 {
234 SetTouch(NULL); // Clear touch function
235 UTIL_Remove(this); // Remove grenade
236 return;
237 }
238
239 StudioFrameAdvance(); // Advance the animation frame
240 pev->nextthink = gpGlobals->time + 0.1; // Schedule next think
241
242 // Explode when ready
243 if (gpGlobals->time >= m_flDie)
244 {
245 g_vecAttackDir = pev->velocity.Normalize();
246 pev->health = -1; // Set health to -1 to trigger death
247 Killed(pev, 0); // Call killed method
248 return;
249 }
250
251 // Adjust movement if in water
252 if (pev->waterlevel != 0)
253 {
254 if (pev->movetype == MOVETYPE_BOUNCE)
255 {
256 pev->movetype = MOVETYPE_FLY; // Change to fly mode
257 }
258 pev->velocity = pev->velocity * 0.9; // Reduce velocity
259 pev->velocity.z += 8.0; // Add upward velocity
260 }
261 else if (pev->movetype == MOVETYPE_FLY)
262 {
263 pev->movetype = MOVETYPE_BOUNCE; // Change to bounce mode
264 }
265
266 // Return if not time to hunt
267 if (m_flNextHunt > gpGlobals->time)
268 return;
269
270 m_flNextHunt = gpGlobals->time + 2.0; // Schedule next hunt
271
272 CBaseEntity *pOther = NULL; // Initialize other entity
273 Vector vecDir; // Initialize direction vector
274 TraceResult tr; // Initialize trace result
275
276 Vector vecFlat = pev->velocity; // Get flat velocity
277 vecFlat.z = 0; // Set z to 0
278 vecFlat = vecFlat.Normalize(); // Normalize the vector
279
280 UTIL_MakeVectors(pev->angles); // Make vectors based on angles
281
282 if (m_hEnemy == NULL || !m_hEnemy->IsAlive()) // Check if there is an enemy
283 {
284 // Find target, bounce a bit towards it
285 Look(512); // Look for enemies
286 m_hEnemy = BestVisibleEnemy(); // Get the best visible enemy
287 }
288
289 // Squeak if it's about time to blow up
290 if ((m_flDie - gpGlobals->time <= 0.5) && (m_flDie - gpGlobals->time >= 0.3))
291 {
292 EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "squeek/sqk_die1.wav", 1, ATTN_NORM, 0, 100 + RANDOM_LONG(0, 0x3F)); // Play squeak sound
293 CSoundEnt::InsertSound(bits_SOUND_COMBAT, pev->origin, 256, 0.25); // Insert sound for combat
294 }
295
296 // Adjust pitch as squeaker gets closer to detonation time
297 float flpitch = 155.0 - 60.0 * ((m_flDie - gpGlobals->time) / SQUEEK_DETONATE_DELAY);
298 if (flpitch < 80)
299 flpitch = 80; // Set minimum pitch
300
301 if (m_hEnemy != NULL) // Check if there is an enemy
302 {
303 if (FVisible(m_hEnemy)) // Check if the enemy is visible
304 {
305 vecDir = m_hEnemy->EyePosition() - pev->origin; // Calculate direction to enemy
306 m_vecTarget = vecDir.Normalize(); // Normalize the target vector
307 }
308
309 float flVel = pev->velocity.Length(); // Get current velocity
310 float flAdj = 50.0 / (flVel + 10.0); // Calculate adjustment factor
311
312 if (flAdj > 1.2)
313 flAdj = 1.2; // Cap adjustment factor
314
315 pev->velocity = pev->velocity * flAdj + m_vecTarget * 300; // Update velocity
316 }
317
318 if (pev->flags & FL_ONGROUND) // Check if on the ground
319 {
320 pev->avelocity = Vector(0, 0, 0); // Reset angular velocity
321 }
322 else
323 {
324 if (pev->avelocity == Vector(0, 0, 0)) // Check if angular velocity is zero
325 {
326 pev->avelocity.x = RANDOM_FLOAT(-100, 100); // Set random angular velocity
327 pev->avelocity.z = RANDOM_FLOAT(-100, 100); // Set random angular velocity
328 }
329 }
330
331 if ((pev->origin - m_posPrev).Length() < 1.0) // Check if moved less than 1 unit
332 {
333 pev->velocity.x = RANDOM_FLOAT(-100, 100); // Set random x velocity
334 pev->velocity.y = RANDOM_FLOAT(-100, 100); // Set random y velocity
335 }
336 m_posPrev = pev->origin; // Update previous position
337
338 pev->angles = UTIL_VecToAngles(pev->velocity); // Update angles based on velocity
339 pev->angles.z = 0; // Reset z angle
340 pev->angles.x = 0; // Reset x angle
341}

351void CSqueakGrenade::SuperBounceTouch(CBaseEntity *pOther)
352{
353 float flpitch; // Variable to hold pitch
354
355 TraceResult tr = UTIL_GetGlobalTrace(); // Get the global trace result
356
357 // Don't hit the entity that launched this grenade
358 if (pev->owner && pOther->edict() == pev->owner)
359 return;
360
361 // At least until we've bounced once
362 pev->owner = NULL; // Clear owner
363
364 pev->angles.x = 0; // Reset x angle
365 pev->angles.z = 0; // Reset z angle
366
367 // Avoid bouncing too much
368 if (m_flNextHit > gpGlobals->time)
369 return;
370
371 // Adjust pitch as squeaker gets closer to detonation time
372 flpitch = 155.0 - 60.0 * ((m_flDie - gpGlobals->time) / SQUEEK_DETONATE_DELAY);
373
374 if (pOther->pev->takedamage && m_flNextAttack < gpGlobals->time) // Check if the other entity can take damage
375 {
376 // Attack!
377 // Make sure it's me who has touched them
378 if (tr.pHit == pOther->edict())
379 {
380 // And it's not another squeak grenade
381 if (tr.pHit->v.modelindex != pev->modelindex)
382 {
383 ClearMultiDamage(); // Clear previous damage
384 pOther->TraceAttack(pev, gSkillData.snarkDmgBite, gpGlobals->v_forward, &tr, DMG_SLASH); // Apply damage
385 if (m_hOwner != NULL)
386 ApplyMultiDamage(pev, m_hOwner->pev); // Apply multi-damage to owner
387 else
388 ApplyMultiDamage(pev, pev); // Apply multi-damage to self
389
390 pev->dmg += gSkillData.snarkDmgPop; // Add more explosion damage
391
392 // Play bite sound
393 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, "squeek/sqk_deploy1.wav", 1.0, ATTN_NORM, 0, (int)flpitch);
394 m_flNextAttack = gpGlobals->time + 0.5; // Set next attack time
395 }
396 }
397 else
398 {
399 // Debug message for being hit
400 }
401 }
402
403 m_flNextHit = gpGlobals->time + 0.1; // Set next hit time
404 m_flNextHunt = gpGlobals->time; // Set next hunt time
405
406 if (g_pGameRules->IsMultiplayer()) // Check if in multiplayer
407 {
408 // Limit how often snarks can make their bounce sounds to prevent overflows
409 if (gpGlobals->time < m_flNextBounceSoundTime)
410 {
411 // Too soon!
412 return;
413 }
414 }
415
416 if (!(pev->flags & FL_ONGROUND)) // Check if not on the ground
417 {
418 // Play bounce sound
419 float flRndSound = RANDOM_FLOAT(0, 1);
420
421 if (flRndSound <= 0.33)
422 EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "squeek/sqk_hunt1.wav", 1, ATTN_NORM, 0, (int)flpitch);
423 else if (flRndSound <= 0.66)
424 EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "squeek/sqk_hunt2.wav", 1, ATTN_NORM, 0, (int)flpitch);
425 else
426 EMIT_SOUND_DYN(ENT(pev), CHAN_VOICE, "squeek/sqk_hunt3.wav", 1, ATTN_NORM, 0, (int)flpitch);
427 CSoundEnt::InsertSound(bits_SOUND_COMBAT, pev->origin, 256, 0.25); // Insert sound for combat
428 }
429 else
430 {
431 // Skittering sound
432 CSoundEnt::InsertSound(bits_SOUND_COMBAT, pev->origin, 100, 0.1); // Insert sound for combat
433 }
434
435 m_flNextBounceSoundTime = gpGlobals->time + 0.5; // Set next bounce sound time
436}

635void CSqueak::WeaponIdle(void)
636{
637 if (m_flTimeWeaponIdle > UTIL_WeaponTimeBase()) // Check if still idle
638 return; // Exit if still idle
639
640 if (m_fJustThrown) // Check if just thrown
641 {
642 m_fJustThrown = 0; // Reset thrown flag
643
644 if (!m_pPlayer->m_rgAmmo[PrimaryAmmoIndex()]) // Check if out of ammo
645 {
646 RetireWeapon(); // Retire the weapon
647 return; // Exit
648 }
649
650 SendWeaponAnim(SQUEAK_UP); // Play idle animation
651 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat(m_pPlayer->random_seed, 10, 15); // Set idle time
652 return; // Exit
653 }
654
655 int iAnim; // Animation variable
656 float flRand = UTIL_SharedRandomFloat(m_pPlayer->random_seed, 0, 1); // Generate random float
657 if (flRand <= 0.75) // Check random value
658 {
659 iAnim = SQUEAK_IDLE1; // Set idle animation
660 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 30.0 / 16 * (2); // Set idle time
661 }
662 else if (flRand <= 0.875) // Check random value
663 {
664 iAnim = SQUEAK_FIDGETFIT; // Set fidget animation
665 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 70.0 / 16.0; // Set idle time
666 }
667 else
668 {
669 iAnim = SQUEAK_FIDGETNIP; // Set fidget animation
670 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 80.0 / 16.0; // Set idle time
671 }
672 SendWeaponAnim(iAnim); // Send the selected animation
673}
