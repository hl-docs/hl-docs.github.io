613void CHoundeye :: SonicAttack ( void )
614{
615 float flAdjustedDamage;
616 float flDist;
617
618 switch ( RANDOM_LONG( 0, 2 ) )
619 {
620 case 0: EMIT_SOUND(ENT(pev), CHAN_WEAPON, "houndeye/he_blast1.wav", 1, ATTN_NORM); break;
621 case 1: EMIT_SOUND(ENT(pev), CHAN_WEAPON, "houndeye/he_blast2.wav", 1, ATTN_NORM); break;
622 case 2: EMIT_SOUND(ENT(pev), CHAN_WEAPON, "houndeye/he_blast3.wav", 1, ATTN_NORM); break;
623 }
624
625 // blast circles
626 MESSAGE_BEGIN( MSG_PAS, SVC_TEMPENTITY, pev->origin );
627 WRITE_BYTE( TE_BEAMCYLINDER );
628 WRITE_COORD( pev->origin.x);
629 WRITE_COORD( pev->origin.y);
630 WRITE_COORD( pev->origin.z + 16);
631 WRITE_COORD( pev->origin.x);
632 WRITE_COORD( pev->origin.y);
633 WRITE_COORD( pev->origin.z + 16 + HOUNDEYE_MAX_ATTACK_RADIUS / .2); // reach damage radius over .3 seconds
634 WRITE_SHORT( m_iSpriteTexture );
635 WRITE_BYTE( 0 ); // startframe
636 WRITE_BYTE( 0 ); // framerate
637 WRITE_BYTE( 2 ); // life
638 WRITE_BYTE( 16 ); // width
639 WRITE_BYTE( 0 ); // noise
640
641 WriteBeamColor();
642
643 WRITE_BYTE( 255 ); //brightness
644 WRITE_BYTE( 0 ); // speed
645 MESSAGE_END();
646
647 MESSAGE_BEGIN( MSG_PAS, SVC_TEMPENTITY, pev->origin );
648 WRITE_BYTE( TE_BEAMCYLINDER );
649 WRITE_COORD( pev->origin.x);
650 WRITE_COORD( pev->origin.y);
651 WRITE_COORD( pev->origin.z + 16);
652 WRITE_COORD( pev->origin.x);
653 WRITE_COORD( pev->origin.y);
654 WRITE_COORD( pev->origin.z + 16 + ( HOUNDEYE_MAX_ATTACK_RADIUS / 2 ) / .2); // reach damage radius over .3 seconds
655 WRITE_SHORT( m_iSpriteTexture );
656 WRITE_BYTE( 0 ); // startframe
657 WRITE_BYTE( 0 ); // framerate
658 WRITE_BYTE( 2 ); // life
659 WRITE_BYTE( 16 ); // width
660 WRITE_BYTE( 0 ); // noise
661
662 WriteBeamColor();
663
664 WRITE_BYTE( 255 ); //brightness
665 WRITE_BYTE( 0 ); // speed
666 MESSAGE_END();
667
668
669 CBaseEntity *pEntity = NULL;
670 // iterate on all entities in the vicinity.
671 while ((pEntity = UTIL_FindEntityInSphere( pEntity, pev->origin, HOUNDEYE_MAX_ATTACK_RADIUS )) != NULL)
672 {
673 if ( pEntity->pev->takedamage != DAMAGE_NO )
674 {
675 if ( !FClassnameIs(pEntity->pev, "monster_houndeye") )
676 {// houndeyes don't hurt other houndeyes with their attack
677
678 // houndeyes do FULL damage if the ent in question is visible. Half damage otherwise.
679 // This means that you must get out of the houndeye's attack range entirely to avoid damage.
680 // Calculate full damage first
681
682 if ( SquadCount() > 1 )
683 {
684 // squad gets attack bonus.
685 flAdjustedDamage = gSkillData.houndeyeDmgBlast + gSkillData.houndeyeDmgBlast * ( HOUNDEYE_SQUAD_BONUS * ( SquadCount() - 1 ) );
686 }
687 else
688 {
689 // solo
690 flAdjustedDamage = gSkillData.houndeyeDmgBlast;
691 }
692
693 flDist = (pEntity->Center() - pev->origin).Length();
694
695 flAdjustedDamage -= ( flDist / HOUNDEYE_MAX_ATTACK_RADIUS ) * flAdjustedDamage;
696
697 if ( !FVisible( pEntity ) )
698 {
699 if ( pEntity->IsPlayer() )
700 {
701 // if this entity is a client, and is not in full view, inflict half damage. We do this so that players still
702 // take the residual damage if they don't totally leave the houndeye's effective radius. We restrict it to clients
703 // so that monsters in other parts of the level don't take the damage and get pissed.
704 flAdjustedDamage *= 0.5;
705 }
706 else if ( !FClassnameIs( pEntity->pev, "func_breakable" ) && !FClassnameIs( pEntity->pev, "func_pushable" ) )
707 {
708 // do not hurt nonclients through walls, but allow damage to be done to breakables
709 flAdjustedDamage = 0;
710 }
711 }
712
713 //ALERT ( at_aiconsole, "Damage: %f\n", flAdjustedDamage );
714
715 if (flAdjustedDamage > 0 )
716 {
717 pEntity->TakeDamage ( pev, pev, flAdjustedDamage, DMG_SONIC | DMG_ALWAYSGIB );
718 }
719 }
720 }
721 }
