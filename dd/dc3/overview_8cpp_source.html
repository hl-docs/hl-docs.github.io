60int CHudOverview::Draw(float flTime)
61{
62 // only draw in overview mode
63 if (!gEngfuncs.Overview_GetOverviewState())
64 return 1;
65
66 // make sure we have player info
67 gViewPort->GetAllPlayersInfo();
68
69 // calculate player size on the overview
70 int x1, y1, x2, y2;
71 float v0[3]={0,0,0}, v1[3]={64,64,0};
72 gEngfuncs.Overview_WorldToScreen(v0, &x1, &y1);
73 gEngfuncs.Overview_WorldToScreen(v1, &x2, &y2);
74 float scale = abs(x2 - x1);
75
76 // loop through all the players and draw them on the map
77 for (int i = 1; i < MAX_PLAYERS; i++)
78 {
79 cl_entity_t *pl = gEngfuncs.GetEntityByIndex(i);
80
81 if (pl && pl->player && pl->curstate.health > 0 && pl->curstate.solid != SOLID_NOT)
82 {
83 int x, y, z = 0;
84 float v[3]={pl->origin[0], pl->origin[1], 0};
85 gEngfuncs.Overview_WorldToScreen(v, &x, &y);
86
87 // hack in some team colors
88 float r, g, bc;
89 if (g_PlayerExtraInfo[i].teamnumber == 1)
90 {
91 r = 0.0f; g = 0.0f; bc = 1.0f;
92 }
93 else if (g_PlayerExtraInfo[i].teamnumber == 2)
94 {
95 r = 1.0f; g = 0.0f; bc = 0.0f;
96 }
97 else
98 {
99 // just use the default orange color if the team isn't set
100 r = 1.0f; g = 0.7f; bc = 0.0f;
101 }
102
103 // set the current texture
104 gEngfuncs.pTriAPI->SpriteTexture((struct model_s *)gEngfuncs.GetSpritePointer(m_hsprPlayer), 0);
105
106 // additive render mode
107 gEngfuncs.pTriAPI->RenderMode(kRenderTransAdd);
108
109 // no culling
110 gEngfuncs.pTriAPI->CullFace(TRI_NONE);
111
112 // draw a square
113 gEngfuncs.pTriAPI->Begin(TRI_QUADS);
114
115 // set the color to be that of the team
116 gEngfuncs.pTriAPI->Color4f(r, g, bc, 1.0f);
117
118 // calculate rotational matrix
119 vec3_t a, b, angles;
120 float rmatrix[3][4]; // transformation matrix
121 VectorCopy(pl->angles, angles);
122 angles[0] = 0.0f;
123 angles[1] += 90.f;
124 angles[1] = -angles[1];
125 angles[2] = 0.0f;
126 AngleMatrix(angles, rmatrix);
127 a[2] = 0;
128
129 a[0] = -scale; a[1] = -scale;
130 VectorTransform(a, rmatrix , b );
131 gEngfuncs.pTriAPI->TexCoord2f( 0, 0 );
132 gEngfuncs.pTriAPI->Vertex3f(x + b[0], y + b[1], z);
133
134 a[0]=-scale; a[1] = scale;
135 VectorTransform(a, rmatrix , b );
136 gEngfuncs.pTriAPI->TexCoord2f( 0, 1 );
137 gEngfuncs.pTriAPI->Vertex3f (x + b[0], y + b[1], z);
138
139 a[0]=scale; a[1] = scale;
140 VectorTransform(a, rmatrix , b );
141 gEngfuncs.pTriAPI->TexCoord2f( 1, 1 );
142 gEngfuncs.pTriAPI->Vertex3f (x + b[0], y + b[1], z);
143
144 a[0]=scale; a[1] = -scale;
145 VectorTransform(a, rmatrix , b );
146 gEngfuncs.pTriAPI->TexCoord2f( 1, 0 );
147 gEngfuncs.pTriAPI->Vertex3f (x + b[0], y + b[1], z);
148
149 // finish up
150 gEngfuncs.pTriAPI->End();
151 gEngfuncs.pTriAPI->RenderMode( kRenderNormal );
152
153 // draw the players name and health underneath
154 char string[256];
155 sprintf(string, "%s (%i%%)", g_PlayerInfoList[i].name, pl->curstate.health);
156 DrawConsoleString(x, y + (1.1 * scale), string);
157 }
158 }
159
160 return 1;
161}
