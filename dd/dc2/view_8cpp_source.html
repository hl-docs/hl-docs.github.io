172float V_CalcBob ( struct ref_params_s *pparams )
173{
174 static double bobtime;
175 static float bob;
176 float cycle;
177 static float lasttime;
178 vec3_t vel;
179
180
181 if ( pparams->onground == -1 ||
182 pparams->time == lasttime )
183 {
184 // just use old value
185 return bob;
186 }
187
188 lasttime = pparams->time;
189
190 bobtime += pparams->frametime;
191 cycle = bobtime - (int)( bobtime / cl_bobcycle->value ) * cl_bobcycle->value;
192 cycle /= cl_bobcycle->value;
193
194 if ( cycle < cl_bobup->value )
195 {
196 cycle = M_PI * cycle / cl_bobup->value;
197 }
198 else
199 {
200 cycle = M_PI + M_PI * ( cycle - cl_bobup->value )/( 1.0 - cl_bobup->value );
201 }
202
203 // bob is proportional to simulated velocity in the xy plane
204 // (don't count Z, or jumping messes it up)
205 VectorCopy( pparams->simvel, vel );
206 vel[2] = 0;
207
208 bob = sqrt( vel[0] * vel[0] + vel[1] * vel[1] ) * cl_bob->value;
209 bob = bob * 0.3 + bob * 0.7 * sin(cycle);
210 bob = min( bob, 4 );
211 bob = max( bob, -7 );
212 return bob;
213
214}

229float V_CalcRoll (vec3_t angles, vec3_t velocity, float rollangle, float rollspeed )
230{
231 float sign;
232 float side;
233 float value;
234 vec3_t forward, right, up;
235
236 AngleVectors ( angles, forward, right, up );
237
238 side = DotProduct (velocity, right);
239 sign = side < 0 ? -1 : 1;
240 side = fabs( side );
241
242 value = rollangle;
243 if (side < rollspeed)
244 {
245 side = side * value / rollspeed;
246 }
247 else
248 {
249 side = value;
250 }
251 return side * sign;
252}

306void V_DriftPitch ( struct ref_params_s *pparams )
307{
308 float delta, move;
309
310 if ( gEngfuncs.IsNoClipping() || !pparams->onground || pparams->demoplayback || pparams->spectator )
311 {
312 pd.driftmove = 0;
313 pd.pitchvel = 0;
314 return;
315 }
316
317 // don't count small mouse motion
318 if ( pd.nodrift)
319 {
320 if ( v_centermove->value > 0 && !(in_mlook.state & 1) )
321 {
322 // this is for lazy players. if they stopped, looked around and then continued
323 // to move the view will be centered automatically if they move more than
324 // v_centermove units.
325
326 if ( fabs( pparams->cmd->forwardmove ) < cl_forwardspeed->value )
327 pd.driftmove = 0;
328 else
329 pd.driftmove += pparams->frametime;
330
331 if ( pd.driftmove > v_centermove->value)
332 {
333 V_StartPitchDrift ();
334 }
335 else
336 {
337 return; // player didn't move enough
338 }
339 }
340
341 return; // don't drift view
342 }
343
344 delta = pparams->idealpitch - pparams->cl_viewangles[PITCH];
345
346 if (!delta)
347 {
348 pd.pitchvel = 0;
349 return;
350 }
351
352 move = pparams->frametime * pd.pitchvel;
353
354 pd.pitchvel *= (1.0f+(pparams->frametime*0.25f)); // get faster by time
355
356 if (delta > 0)
357 {
358 if (move > delta)
359 {
360 pd.pitchvel = 0;
361 move = delta;
362 }
363 pparams->cl_viewangles[PITCH] += move;
364 }
365 else if (delta < 0)
366 {
367 if (move > -delta)
368 {
369 pd.pitchvel = 0;
370 move = -delta;
371 }
372 pparams->cl_viewangles[PITCH] -= move;
373 }
374}

390void V_CalcGunAngle ( struct ref_params_s *pparams )
391{
392 cl_entity_t *viewent;
393
394 viewent = gEngfuncs.GetViewModel();
395 if ( !viewent )
396 return;
397
398 viewent->angles[YAW] = pparams->viewangles[YAW] + pparams->crosshairangle[YAW];
399 viewent->angles[PITCH] = -pparams->viewangles[PITCH] + pparams->crosshairangle[PITCH] * 0.25;
400 viewent->angles[ROLL] -= v_idlescale * sin(pparams->time*v_iroll_cycle.value) * v_iroll_level.value;
401
402 // don't apply all of the v_ipitch to prevent normally unseen parts of viewmodel from coming into view.
403 viewent->angles[PITCH] -= v_idlescale * sin(pparams->time*v_ipitch_cycle.value) * (v_ipitch_level.value * 0.5);
404 viewent->angles[YAW] -= v_idlescale * sin(pparams->time*v_iyaw_cycle.value) * v_iyaw_level.value;
405
406 VectorCopy( viewent->angles, viewent->curstate.angles );
407 VectorCopy( viewent->angles, viewent->latched.prevangles );
408}

813 {
814 VectorSubtract( neworg, pparams->simorg, delta );
815
816 VectorAdd( pparams->simorg, delta, pparams->simorg );
817 VectorAdd( pparams->vieworg, delta, pparams->vieworg );
818 VectorAdd( view->origin, delta, view->origin );

821 }

882void V_SmoothInterpolateAngles( float * startAngle, float * endAngle, float * finalAngle, float degreesPerSec )
883{
884 float absd,frac,d,threshhold;
885
886 NormalizeAngles( startAngle );
887 NormalizeAngles( endAngle );
888
889 for ( int i = 0 ; i < 3 ; i++ )
890 {
891 d = endAngle[i] - startAngle[i];
892
893 if ( d > 180.0f )
894 {
895 d -= 360.0f;
896 }
897 else if ( d < -180.0f )
898 {
899 d += 360.0f;
900 }
901
902 absd = fabs(d);
903
904 if ( absd > 0.01f )
905 {
906 frac = degreesPerSec * v_frametime;
907
908 threshhold= degreesPerSec / 4;
909
910 if ( absd < threshhold )
911 {
912 float h = absd / threshhold;
913 h *= h;
914 frac*= h; // slow down last degrees
915 }
916
917 if ( frac > absd )
918 {
919 finalAngle[i] = endAngle[i];
920 }
921 else
922 {
923 if ( d>0)
924 finalAngle[i] = startAngle[i] + frac;
925 else
926 finalAngle[i] = startAngle[i] - frac;
927 }
928 }
929 else
930 {
931 finalAngle[i] = endAngle[i];
932 }
933
934 }
935
936 NormalizeAngles( finalAngle );
937}
