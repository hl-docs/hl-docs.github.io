13// For every module and executable we store a debugging level and flags
14// for the types of output that are desired. Constants for the types are
15// defined in WXDEBUG.H and more can be added.
16// The keys are stored in the registry under the
17// HKEY_LOCAL_MACHINE\SOFTWARE\Debug<Module Name>\Type and
18// HKEY_LOCAL_MACHINE\SOFTWARE\Debug<Module Name>\Level key values
19//
20// There are also global values under SOFTWARE\Debug\Global which are loaded
21// after the module-specific values. The Types specified there are OR'ed with
22// the module specific types and m_dwLevel is set to the greater of the global
23// and the module specific settings.

114void WINAPI DbgInitModuleName()
115{
116 TCHAR FullName[iDEBUGINFO]; // Buffer to load the full path and module name
117 TCHAR *pName; // Pointer to search for the last backslash
118
119 GetModuleFileName(m_hInst, FullName, iDEBUGINFO); // Load the module file name
120 pName = _tcsrchr(FullName, '\\'); // Find the last backslash
121 if (pName == NULL) // If no backslash is found
122 {
123 pName = FullName; // Use the full name
124 }
125 else
126 {
127 pName++; // Move past the backslash
128 }
129 lstrcpy(m_ModuleName, pName); // Copy the module name

138void WINAPI DbgInitModuleSettings()
139{
140 LONG lReturn; // Return value for registry operations
141 TCHAR szInfo[iDEBUGINFO]; // Buffer for constructing key names
142 HKEY hModuleKey; // Handle for the module key
143
144 // Construct the base key name
145 wsprintf(szInfo, TEXT("%s\\%s"), m_pBaseKey, m_ModuleName);
146
147 // Create or open the key for this module
148 lReturn = RegCreateKeyEx(HKEY_LOCAL_MACHINE, // Handle of an open key
149 szInfo, // Address of subkey name
150 (DWORD)0, // Reserved value
151 NULL, // Address of class name
152 (DWORD)0, // Special options flags
153 KEY_ALL_ACCESS, // Desired security access
154 NULL, // Key security descriptor
155 &hModuleKey, // Opened handle buffer
156 NULL); // What really happened
157
158 if (lReturn != ERROR_SUCCESS) // Check if the key was successfully created or opened
159 {
160 DbgLogInfo(LOG_ERROR, 0, TEXT("Could not access module key")); // Log error
161 return; // Exit the function
162 }
163
164 DbgInitLogTo(hModuleKey); // Initialize logging settings
165 DbgInitKeyLevels(hModuleKey, &m_dwTypes, &m_dwLevel); // Initialize key levels
166 RegCloseKey(hModuleKey); // Close the module key
167}

175void WINAPI DbgInitGlobalSettings()
176{
177 LONG lReturn; // Return value for registry operations
178 TCHAR szInfo[iDEBUGINFO]; // Buffer for constructing key names
179 HKEY hGlobalKey; // Handle for the global key
180 DWORD dwTypes = 0; // Variable to store types
181 DWORD dwLevel = 0; // Variable to store level
182
183 // Construct the global base key name
184 wsprintf(szInfo, TEXT("%s\\%s"), m_pBaseKey, m_pGlobalKey);
185
186 // Create or open the key for this module
187 lReturn = RegCreateKeyEx(HKEY_LOCAL_MACHINE, // Handle of an open key
188 szInfo, // Address of subkey name
189 (DWORD)0, // Reserved value
190 NULL, // Address of class name
191 (DWORD)0, // Special options flags
192 KEY_ALL_ACCESS, // Desired security access
193 NULL, // Key security descriptor
194 &hGlobalKey, // Opened handle buffer
195 NULL); // What really happened
196
197 if (lReturn != ERROR_SUCCESS) // Check if the key was successfully created or opened
198 {
199 DbgLogInfo(LOG_ERROR, 0, TEXT("Could not access GLOBAL module key")); // Log error
200 return; // Exit the function
201 }
202
203 DbgInitKeyLevels(hGlobalKey, &dwTypes, &dwLevel); // Initialize key levels
204 RegCloseKey(hGlobalKey); // Close the global key
205
206 m_dwTypes |= dwTypes; // Combine global types with module types
207 if (dwLevel > m_dwLevel) // Update level if global level is greater
208 m_dwLevel = dwLevel; // Set the new level
209}

219void WINAPI DbgInitLogTo(HKEY hKey)
220{
221 LONG lReturn; // Return value for registry operations
222 DWORD dwKeyType; // Type of the key
223 DWORD dwKeySize; // Size of the key value
224 TCHAR szFile[MAX_PATH] = {0}; // Buffer for the log file name
225 static const TCHAR cszKey[] = TEXT("LogToFile"); // Key name for logging
226
227 dwKeySize = MAX_PATH; // Set the size of the buffer
228 lReturn = RegQueryValueEx(
229 hKey, // Handle to an open key
230 cszKey, // Subkey name
231 NULL, // Reserved field
232 &dwKeyType, // Returns the field type
233 (LPBYTE)szFile, // Returns the field's value
234 &dwKeySize); // Number of bytes transferred
235
236 // Create an empty key if it does not already exist
237 if (lReturn != ERROR_SUCCESS || dwKeyType != REG_SZ)
238 {
239 dwKeySize = 1; // Set size to 1 for empty key
240 lReturn = RegSetValueEx(
241 hKey, // Handle of an open key
242 cszKey, // Address of subkey name
243 (DWORD)0, // Reserved field
244 REG_SZ, // Type of the key field
245 (PBYTE)szFile, // Value for the field
246 dwKeySize); // Size of the field buffer
247 }
248
249 // If an output-to was specified, try to open it.
250 if (m_hOutput != INVALID_HANDLE_VALUE) // Check if the output handle is valid
251 {
252 DBGASSERTEXECUTE(CloseHandle(m_hOutput)); // Close the previous output handle
253 m_hOutput = INVALID_HANDLE_VALUE; // Reset the output handle
254 }
255 if (szFile[0] != 0) // Check if the file name is not empty
256 {
257 if (!lstrcmpi(szFile, TEXT("Console"))) // Check if output is to console
258 {
259 m_hOutput = GetStdHandle(STD_OUTPUT_HANDLE); // Get the standard output handle
260 if (m_hOutput == INVALID_HANDLE_VALUE) // Check if the handle is valid
261 {
262 AllocConsole(); // Allocate a console if needed
263 m_hOutput = GetStdHandle(STD_OUTPUT_HANDLE); // Get the new standard output handle
264 }
265 SetConsoleTitle(TEXT("Valve Debug Output")); // Set the console title
266 }
267 else if (szFile[0] && lstrcmpi(szFile, TEXT("Debug")) && lstrcmpi(szFile, TEXT("Debugger")) && lstrcmpi(szFile, TEXT("Deb")))
268 {
269 m_hOutput = CreateFile(szFile, GENERIC_WRITE, FILE_SHARE_READ, NULL, OPEN_ALWAYS, FILE_ATTRIBUTE_NORMAL, NULL); // Create or open the log file
270 if (INVALID_HANDLE_VALUE != m_hOutput) // Check if the file handle is valid
271 {
272 static const TCHAR cszBar[] = TEXT("\r\n\r\n=====DbgInitialize()=====\r\n\r\n"); // Log header
273 SetFilePointer(m_hOutput, 0, NULL, FILE_END); // Move to the end of the file
274 DbgOutString(cszBar); // Output the header to the log
275 }
276 }
277 }

290void WINAPI DbgInitKeyLevels(HKEY hKey, DWORD *pdwTypes, DWORD *pdwLevel)
291{
292 LONG lReturn; // Return value for registry operations
293 DWORD dwKeySize; // Size of the key value
294 DWORD dwKeyType; // Receives the type of the key
295
296 // Get the Types value
297 dwKeySize = sizeof(DWORD); // Set the size for DWORD
298 lReturn = RegQueryValueEx(
299 hKey, // Handle to an open key
300 pKeyNames[0], // Subkey name for types
301 NULL, // Reserved field
302 &dwKeyType, // Returns the field type
303 (LPBYTE)pdwTypes, // Returns the field's value
304 &dwKeySize); // Number of bytes transferred
305
306 // If the key was not available or it was not a DWORD value
307 if (lReturn != ERROR_SUCCESS || dwKeyType != REG_DWORD)
308 {
309 *pdwTypes = 0; // Set types to 0
310 lReturn = RegSetValueEx(
311 hKey, // Handle of an open key
312 pKeyNames[0], // Address of subkey name
313 (DWORD)0, // Reserved field
314 REG_DWORD, // Type of the key field
315 (PBYTE)pdwTypes, // Value for the field
316 sizeof(DWORD)); // Size of the field buffer
317
318 if (lReturn != ERROR_SUCCESS) // Check if the value was set successfully
319 {
320 DbgLogInfo(LOG_ERROR, 0, TEXT("Could not create subkey %s"), pKeyNames[0]); // Log error
321 *pdwTypes = 0; // Ensure types is set to 0
322 }
323 }
324
325 // Get the Level value
326 dwKeySize = sizeof(DWORD); // Set the size for DWORD
327 lReturn = RegQueryValueEx(
328 hKey, // Handle to an open key
329 pKeyNames[1], // Subkey name for level
330 NULL, // Reserved field
331 &dwKeyType, // Returns the field type
332 (LPBYTE)pdwLevel, // Returns the field's value
333 &dwKeySize); // Number of bytes transferred
334
335 // If the key was not available or it was not a DWORD value
336 if (lReturn != ERROR_SUCCESS || dwKeyType != REG_DWORD)
337 {
338 *pdwLevel = 0; // Set level to 0
339 lReturn = RegSetValueEx(
340 hKey, // Handle of an open key
341 pKeyNames[1], // Address of subkey name
342 (DWORD)0, // Reserved field
343 REG_DWORD, // Type of the key field
344 (PBYTE)pdwLevel, // Value for the field
345 sizeof(DWORD)); // Size of the field buffer
346
347 if (lReturn != ERROR_SUCCESS) // Check if the value was set successfully
348 {
349 DbgLogInfo(LOG_ERROR, 0, TEXT("Could not create subkey %s"), pKeyNames[1]); // Log error
350 *pdwLevel = 0; // Ensure level is set to 0
351 }
352 }

389void WINAPI DbgLogInfo(DWORD Type, DWORD Level, const TCHAR *pFormat,...)
390{
391 if (!m_bInit) // Check if the library is initialized
392 return; // Exit if not initialized
393 // Check the current level for this type combination
394 if (((Type & m_dwTypes) == 0) || (m_dwLevel < Level))
395 return; // Exit if the type is not enabled or level is too low
396
397 TCHAR szInfo[2000]; // Buffer for the log message
398
399 // Format the variable length parameter list
400 va_list va; // Variable argument list
401 va_start(va, pFormat); // Initialize the argument list
402
403 // lstrcpy(szInfo, m_ModuleName); // Uncomment if module name is needed
404 // lstrcat(szInfo, TEXT(": ")); // Uncomment if module name is needed
405 wvsprintf(szInfo /* + lstrlen(szInfo) */, pFormat, va); // Format the message
406 // lstrcat(szInfo, TEXT("\r\n")); // Uncomment if newline is needed
407 DbgOutString(szInfo); // Output the formatted string
408
409 va_end(va); // Clean up the argument list
410}
