277void CTentacle :: Spawn( )
278{
279 Precache( );
280
281 pev->solid = SOLID_BBOX;
282 pev->movetype = MOVETYPE_FLY;
283 pev->effects = 0;
284 pev->health = 75;
285 pev->sequence = 0;
286
287 SET_MODEL(ENT(pev), "models/tentacle2.mdl");
288 UTIL_SetSize( pev, Vector( -32, -32, 0 ), Vector( 32, 32, 64 ) );
289
290 pev->takedamage = DAMAGE_AIM;
291 pev->flags |= FL_MONSTER;
292
293 m_bloodColor = BLOOD_COLOR_GREEN;
294
295 SetThink( &CTentacle::Start );
296 SetTouch( &CTentacle::HitTouch );
297 SetUse( &CTentacle::CommandUse );
298
299 pev->nextthink = gpGlobals->time + 0.2;
300
301 ResetSequenceInfo( );
302 m_iDir = 1;
303
304 pev->yaw_speed = 18;
305 m_flInitialYaw = pev->angles.y;
306 pev->ideal_yaw = m_flInitialYaw;
307
308 g_fFlySound = FALSE;
309 g_fSquirmSound = FALSE;
310
311 m_iHitDmg = 20;
312
313 if (m_flMaxYaw <= 0)
314 m_flMaxYaw = 65;
315
316 m_MonsterState = MONSTERSTATE_IDLE;
317
318 UTIL_SetOrigin( pev, pev->origin );
319}

527void CTentacle::Cycle(void)
528{
529 pev->nextthink = gpGlobals->time + 0.1;
530
531 if (m_MonsterState == MONSTERSTATE_SCRIPT || m_IdealMonsterState == MONSTERSTATE_SCRIPT)
532 {
533 pev->angles.y = m_flInitialYaw;
534 pev->ideal_yaw = m_flInitialYaw;
535 ClearConditions(IgnoreConditions());
536 MonsterThink();
537 m_iGoalAnim = TENTACLE_ANIM_Pit_Idle;
538 return;
539 }
540
541 DispatchAnimEvents();
542 StudioFrameAdvance();
543
544 ChangeYaw(pev->yaw_speed);
545
546 CSound *pSound;
547
548 Listen();
549
550 if (HasConditions(bits_COND_HEAR_SOUND))
551 pSound = PBestSound();
552 else
553 pSound = NULL;
554
555 if (pSound)
556 {
557 Vector vecDir;
558 if (gpGlobals->time - m_flPrevSoundTime < 0.5)
559 {
560 float dt = gpGlobals->time - m_flPrevSoundTime;
561 vecDir = pSound->m_vecOrigin + (pSound->m_vecOrigin - m_vecPrevSound) / dt - pev->origin;
562 }
563 else
564 {
565 vecDir = pSound->m_vecOrigin - pev->origin;
566 }
567 m_flPrevSoundTime = gpGlobals->time;
568 m_vecPrevSound = pSound->m_vecOrigin;
569
570 m_flSoundYaw = UTIL_VecToYaw(vecDir) - m_flInitialYaw;
571 m_iSoundLevel = Level(vecDir.z);
572
573 if (m_flSoundYaw < -180)
574 m_flSoundYaw += 360;
575 if (m_flSoundYaw > 180)
576 m_flSoundYaw -= 360;
577
578 if (m_flSoundTime < gpGlobals->time)
579 {
580 char *sound;
581
582 switch(RANDOM_LONG(0, 1))
583 {
584 case 0: sound = "tentacle/te_alert1.wav"; break;
585 case 1: sound = "tentacle/te_alert2.wav"; break;
586 }
587 }
588 m_flSoundTime = gpGlobals->time + RANDOM_FLOAT(5.0, 10.0);
589 }
590
591 float dy = m_flSoundYaw;
592 switch (pev->sequence)
593 {
594 case TENTACLE_ANIM_Floor_Rear:
595 case TENTACLE_ANIM_Floor_Rear_Idle:
596 case TENTACLE_ANIM_Lev1_Rear:
597 case TENTACLE_ANIM_Lev1_Rear_Idle:
598 case TENTACLE_ANIM_Lev2_Rear:
599 case TENTACLE_ANIM_Lev2_Rear_Idle:
600 case TENTACLE_ANIM_Lev3_Rear:
601 case TENTACLE_ANIM_Lev3_Rear_Idle:
602 if (dy < 0 && dy > -m_flMaxYaw)
603 dy = -m_flMaxYaw;
604 if (dy > 0 && dy < m_flMaxYaw)
605 dy = m_flMaxYaw;
606 break;
607 default:
608 if (dy < -m_flMaxYaw)
609 dy = -m_flMaxYaw;
610 if (dy > m_flMaxYaw)
611 dy = m_flMaxYaw;
612 }
613 pev->ideal_yaw = m_flInitialYaw + dy;
614
615 if (m_fSequenceFinished)
616 {
617 if (pev->health <= 1)
618 {
619 m_iGoalAnim = TENTACLE_ANIM_Pit_Idle;
620 if (pev->sequence == TENTACLE_ANIM_Pit_Idle)
621 {
622 pev->health = 75;
623 }
624 }
625 else if (m_flSoundTime > gpGlobals->time)
626 {
627 if (m_flSoundYaw >= -(m_flMaxYaw + 30) && m_flSoundYaw <= (m_flMaxYaw + 30))
628 {
629 m_iGoalAnim = LookupActivity(ACT_T_STRIKE + m_iSoundLevel);
630 }
631 else if (m_flSoundYaw >= -m_flMaxYaw * 2 && m_flSoundYaw <= m_flMaxYaw * 2)
632 {
633 m_iGoalAnim = LookupActivity(ACT_T_TAP + m_iSoundLevel);
634 }
635 else
636 {
637 m_iGoalAnim = LookupActivity(ACT_T_REARIDLE + m_iSoundLevel);
638 }
639 }
640 else if (pev->sequence == TENTACLE_ANIM_Pit_Idle)
641 {
642 m_iGoalAnim = TENTACLE_ANIM_Pit_Idle;
643 }
644 else if (pev->sequence == m_iGoalAnim)
645 {
646 if (MyLevel() >= 0 && gpGlobals->time < m_flSoundTime)
647 {
648 if (RANDOM_LONG(0, 9) < m_flSoundTime - gpGlobals->time)
649 {
650 m_iGoalAnim = LookupActivity(ACT_T_STRIKE + m_iSoundLevel);
651 }
652 else
653 {
654 m_iGoalAnim = LookupActivity(ACT_T_TAP + m_iSoundLevel);
655 }
656 }
657 else if (MyLevel() < 0)
658 {
659 m_iGoalAnim = LookupActivity(ACT_T_IDLE + 0);
660 }
661 else
662 {
663 if (m_flNextSong < gpGlobals->time)
664 {
665 char *sound;
666
667 switch(RANDOM_LONG(0, 1))
668 {
669 case 0: sound = "tentacle/te_sing1.wav"; break;
670 case 1: sound = "tentacle/te_sing2.wav"; break;
671 }
672
673 EMIT_SOUND(ENT(pev), CHAN_VOICE, sound, 1.0, ATTN_NORM);
674
675 m_flNextSong = gpGlobals->time + RANDOM_FLOAT(10, 20);
676 }
677
678 if (RANDOM_LONG(0, 15) == 0)
679 {
680 m_iGoalAnim = LookupActivity(ACT_T_IDLE + RANDOM_LONG(0, 3));
681 }
682 else if (RANDOM_LONG(0, 3) == 0)
683 {
684 m_iGoalAnim = LookupActivity(ACT_T_TAP + MyLevel());
685 }
686 else
687 {
688 m_iGoalAnim = LookupActivity(ACT_T_IDLE + MyLevel());
689 }
690 }
691 if (m_flSoundYaw < 0)
692 m_flSoundYaw += RANDOM_FLOAT(2, 8);
693 else
694 m_flSoundYaw -= RANDOM_FLOAT(2, 8);
695 }
696
697 pev->sequence = FindTransition(pev->sequence, m_iGoalAnim, &m_iDir);
698
699 if (m_iDir > 0)
700 {
701 pev->frame = 0;
702 }
703 else
704 {
705 m_iDir = -1;
706 pev->frame = 255;
707 }
708 ResetSequenceInfo();
709
710 m_flFramerateAdj = RANDOM_FLOAT(-0.2, 0.2);
711 pev->framerate = m_iDir * 1.0 + m_flFramerateAdj;
712
713 switch (pev->sequence)
714 {
715 case TENTACLE_ANIM_Floor_Tap:
716 case TENTACLE_ANIM_Lev1_Tap:
717 case TENTACLE_ANIM_Lev2_Tap:
718 case TENTACLE_ANIM_Lev3_Tap:
719 {
720 Vector vecSrc;
721 UTIL_MakeVectors(pev->angles);
722
723 TraceResult tr1, tr2;
724
725 vecSrc = pev->origin + Vector(0, 0, MyHeight() - 4);
726 UTIL_TraceLine(vecSrc, vecSrc + gpGlobals->v_forward * 512, ignore_monsters, ENT(pev), &tr1);
727
728 vecSrc = pev->origin + Vector(0, 0, MyHeight() + 8);
729 UTIL_TraceLine(vecSrc, vecSrc + gpGlobals->v_forward * 512, ignore_monsters, ENT(pev), &tr2);
730
731 m_flTapRadius = SetBlending(0, RANDOM_FLOAT(tr1.flFraction * 512, tr2.flFraction * 512));
732 }
733 break;
734 default:
735 m_flTapRadius = 336;
736 break;
737 }
738 pev->view_ofs.z = MyHeight();
739 }
740
741 if (m_flPrevSoundTime + 2.0 > gpGlobals->time)
742 {
743 pev->framerate = m_iDir * 1.5 + m_flFramerateAdj;
744 }
745 else if (m_flPrevSoundTime + 5.0 > gpGlobals->time)
746 {
747 pev->framerate = m_iDir + m_iDir * (5 - (gpGlobals->time - m_flPrevSoundTime)) / 2 + m_flFramerateAdj;
748 }
749}

762void CTentacle::CommandUse(CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
763{
764 switch (useType)
765 {
766 case USE_OFF:
767 pev->takedamage = DAMAGE_NO;
768 SetThink(&CTentacle::DieThink);
769 m_iGoalAnim = TENTACLE_ANIM_Engine_Death1;
770 break;
771 case USE_ON:
