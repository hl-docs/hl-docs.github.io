271void CClassMenuPanel::Update()
272{
273 // Don't allow the player to join a team if they're not in a team
274 if (!g_iTeamNumber)
275 return;
276
277 int iYPos = CLASSMENU_TOPLEFT_BUTTON_Y;
278
279 // Cycle through the rest of the buttons
280#ifdef _TFC
281 for (int i = 0; i <= PC_RANDOM; i++)
282 {
283 bool bCivilian = (gViewPort->GetValidClasses(g_iTeamNumber) == -1);
284
285 if ( bCivilian )
286 {
287 // If this team can only be civilians, only the civilian button's visible
288 if (i == 0)
289 {
290 m_pButtons[0]->setVisible( true );
291 SetActiveInfo( 0 );
292 iYPos += CLASSMENU_BUTTON_SIZE_Y + CLASSMENU_BUTTON_SPACER_Y;
293 }
294 else
295 {
296 m_pButtons[i]->setVisible( false );
297 }
298 }
299 else
300 {
301 if ( m_pButtons[i]->IsNotValid() || i == 0 )
302 {
303 m_pButtons[i]->setVisible( false );
304 }
305 else
306 {
307 m_pButtons[i]->setVisible( true );
308 m_pButtons[i]->setPos( CLASSMENU_TOPLEFT_BUTTON_X, iYPos );
309 iYPos += CLASSMENU_BUTTON_SIZE_Y + CLASSMENU_BUTTON_SPACER_Y;
310
311 // Start with the first option up
312 if (!m_iCurrentInfo)
313 SetActiveInfo( i );
314 }
315 }
316
317 // Now count the number of teammembers of this class
318 int iTotal = 0;
319 for ( int j = 1; j < MAX_PLAYERS; j++ )
320 {
321 if ( g_PlayerInfoList[j].name == NULL )
322 continue; // empty player slot, skip
323 if ( g_PlayerExtraInfo[j].teamname[0] == 0 )
324 continue; // skip over players who are not in a team
325 if ( g_PlayerInfoList[j].thisplayer )
326 continue; // skip this player
327 if ( g_PlayerExtraInfo[j].teamnumber != g_iTeamNumber )
328 continue; // skip over players in other teams
329
330 // If this team is forced to be civilians, just count the number of teammates
331 if ( g_PlayerExtraInfo[j].playerclass != i && !bCivilian )
332 continue;
333
334 iTotal++;
335 }
336
337 char sz[256];
338 sprintf(sz, m_sPlayersOnTeamString, iTotal);
339 m_pPlayers[i]->setText( "%s", sz );
340
341 // Set the text color to the teamcolor
342 m_pPlayers[i]->setFgColor( iTeamColors[g_iTeamNumber % iNumberOfTeamColors][0],
343 iTeamColors[g_iTeamNumber % iNumberOfTeamColors][1],
344 iTeamColors[g_iTeamNumber % iNumberOfTeamColors][2],
345 0 );
346
347 // set the graphic to be the team pick
348 for ( int team = 0; team < MAX_TEAMS; team++ )
349 {
350 // unset all the other images
351 if ( m_pClassImages[team][i] )
352 {
353 m_pClassImages[team][i]->setVisible( false );
354 }
355
356 // set the current team image
357 if ( m_pClassImages[g_iTeamNumber-1][i] != NULL )
358 {
359 m_pClassImages[g_iTeamNumber-1][i]->setVisible( true );
360 }
361 else if ( m_pClassImages[0][i] )
362 {
363 m_pClassImages[0][i]->setVisible( true );
364 }
365 }
366 }
367#endif
368
369 // If the player already has a class, make the cancel button visible
370 if ( g_iPlayerClass )
371 {
372 m_pCancelButton->setPos( CLASSMENU_TOPLEFT_BUTTON_X, iYPos );
373 m_pCancelButton->setVisible( true );
374 }
375 else
376 {
377 m_pCancelButton->setVisible( false );
378 }
379}

390bool CClassMenuPanel::SlotInput( int iSlot )
391{
392 if ( (iSlot < 0) || (iSlot > 9) )
393 return false;
394 if ( !m_pButtons[ iSlot ] )
395 return false;
396
397 // Is the button pushable? (0 is special case)
398 if (iSlot == 0)
399 {
400 // Selects Civilian and RandomPC
401 if ( gViewPort->GetValidClasses(g_iTeamNumber) == -1 )
402 {
403 m_pButtons[ 0 ]->fireActionSignal();
404 return true;
405 }
406
407 // Select RandomPC
408 iSlot = 10;
409 }
410
411 if ( !(m_pButtons[ iSlot ]->IsNotValid()) )
412 {
413 m_pButtons[ iSlot ]->fireActionSignal();
414 return true;
415 }
416
417 return false;
418}

453void CClassMenuPanel::SetActiveInfo( int iInput )
454{
455 // Remove all the Info panels and bring up the specified one
456#ifdef _TFC
457 for (int i = 0; i <= PC_RANDOM; i++)
458 {
459 m_pButtons[i]->setArmed( false );
460 m_pClassInfoPanel[i]->setVisible( false );
461 }
462
463 if ( iInput > PC_RANDOM || iInput < 0 )
464#endif
465 iInput = 0;
466
467 m_pButtons[iInput]->setArmed( true );
468 m_pClassInfoPanel[iInput]->setVisible( true );
469 m_iCurrentInfo = iInput;
470
471 m_pScrollPanel->setScrollValue(0,0);
472 m_pScrollPanel->validate();
473}
