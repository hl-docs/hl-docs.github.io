179void CMP5::PrimaryAttack()
180{
181 // Don't fire underwater
182 if (m_pPlayer->pev->waterlevel == 3)
183 {
184 PlayEmptySound(); // Play empty sound
185 m_flNextPrimaryAttack = 0.15; // Set next attack delay
186 return; // Exit
187 }
188
189 if (m_iClip <= 0) // Check if the clip is empty
190 {
191 PlayEmptySound(); // Play empty sound
192 m_flNextPrimaryAttack = 0.15; // Set next attack delay
193 return; // Exit
194 }
195
196 m_pPlayer->m_iWeaponVolume = NORMAL_GUN_VOLUME; // Set weapon volume
197 m_pPlayer->m_iWeaponFlash = NORMAL_GUN_FLASH; // Set weapon flash
198
199 m_iClip--; // Decrement the clip count
200
201 m_pPlayer->pev->effects = (int)(m_pPlayer->pev->effects) | EF_MUZZLEFLASH; // Set muzzle flash effect
202
203 // Player "shoot" animation
204 m_pPlayer->SetAnimation(PLAYER_ATTACK1);
205
206 Vector vecSrc = m_pPlayer->GetGunPosition(); // Get the gun position
207 Vector vecAiming = m_pPlayer->GetAutoaimVector(AUTOAIM_5DEGREES); // Get the aiming vector
208 Vector vecDir; // Direction vector
209
210#ifdef CLIENT_DLL
211 if (!bIsMultiplayer()) // Check if not in multiplayer
212#else
213 if (!g_pGameRules->IsMultiplayer()) // Check if not in multiplayer
214#endif
215 {
216 // Optimized multiplayer. Widened to make it easier to hit a moving player
217 vecDir = m_pPlayer->FireBulletsPlayer(1, vecSrc, vecAiming, VECTOR_CONE_6DEGREES, 8192, BULLET_PLAYER_MP5, 2, 0, m_pPlayer->pev, m_pPlayer->random_seed);
218 }
219 else
220 {
221 // Single player spread
222 vecDir = m_pPlayer->FireBulletsPlayer(1, vecSrc, vecAiming, VECTOR_CONE_3DEGREES, 8192, BULLET_PLAYER_MP5, 2, 0, m_pPlayer->pev, m_pPlayer->random_seed);
223 }
224
225 int flags;
226#if defined(CLIENT_WEAPONS)
227 flags = FEV_NOTHOST; // Set flags for client weapons
228#else
229 flags = 0; // No flags for server
230#endif
231
232 PLAYBACK_EVENT_FULL(flags, m_pPlayer->edict(), m_usMP5, 0.0, (float *)&g_vecZero, (float *)&g_vecZero, vecDir.x, vecDir.y, 0, 0, 0, 0); // Play the firing event
233
234 if (!m_iClip && m_pPlayer->m_rgAmmo[m_iPrimaryAmmoType] <= 0)
235 // HEV suit - indicate out of ammo condition
236 m_pPlayer->SetSuitUpdate("!HEV_AMO0", FALSE, 0); // Update suit with out of ammo message
237
238 m_flNextPrimaryAttack = GetNextAttackDelay(0.1); // Set next attack delay
239
240 if (m_flNextPrimaryAttack < UTIL_WeaponTimeBase())
241 m_flNextPrimaryAttack = UTIL_WeaponTimeBase() + 0.1; // Ensure next attack delay is valid
242
243 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + UTIL_SharedRandomFloat(m_pPlayer->random_seed, 10, 15); // Set idle time
244}

252void CMP5::SecondaryAttack(void)
253{
254 // Don't fire underwater
255 if (m_pPlayer->pev->waterlevel == 3)
256 {
257 PlayEmptySound(); // Play empty sound
258 m_flNextPrimaryAttack = 0.15; // Set next attack delay
259 return; // Exit
260 }
261
262 if (m_pPlayer->m_rgAmmo[m_iSecondaryAmmoType] == 0) // Check if secondary ammo is available
263 {
264 PlayEmptySound(); // Play empty sound
265 return; // Exit
266 }
267
268 m_pPlayer->m_iWeaponVolume = NORMAL_GUN_VOLUME; // Set weapon volume
269 m_pPlayer->m_iWeaponFlash = BRIGHT_GUN_FLASH; // Set weapon flash
270
271 m_pPlayer->m_iExtraSoundTypes = bits_SOUND_DANGER; // Set extra sound types
272 m_pPlayer->m_flStopExtraSoundTime = UTIL_WeaponTimeBase() + 0.2; // Set time to stop extra sound
273
274 m_pPlayer->m_rgAmmo[m_iSecondaryAmmoType]--; // Decrement secondary ammo
275
276 // Player "shoot" animation
277 m_pPlayer->SetAnimation(PLAYER_ATTACK1);
278
279 UTIL_MakeVectors(m_pPlayer->pev->v_angle + m_pPlayer->pev->punchangle); // Update vectors for shooting
280
281 // Create and shoot the grenade
282 CGrenade::ShootContact(m_pPlayer->pev,
283 m_pPlayer->pev->origin + m_pPlayer->pev->view_ofs + gpGlobals->v_forward * 16,
284 gpGlobals->v_forward * 800);
285
286 int flags;
287#if defined(CLIENT_WEAPONS)
288 flags = FEV_NOTHOST; // Set flags for client weapons
289#else
290 flags = 0; // No flags for server
291#endif
292
293 PLAYBACK_EVENT(flags, m_pPlayer->edict(), m_usMP52); // Play the grenade launch event
294
295 m_flNextPrimaryAttack = GetNextAttackDelay(1); // Set next primary attack delay
296 m_flNextSecondaryAttack = UTIL_WeaponTimeBase() + 1; // Set next secondary attack delay
297 m_flTimeWeaponIdle = UTIL_WeaponTimeBase() + 5; // Set idle time after shooting
298
299 if (!m_pPlayer->m_rgAmmo[m_iSecondaryAmmoType])
300 // HEV suit - indicate out of ammo condition
301 m_pPlayer->SetSuitUpdate("!HEV_AMO0", FALSE, 0); // Update suit with out of ammo message
302}

324void CMP5::WeaponIdle(void)
325{
326 ResetEmptySound(); // Reset the empty sound
327
328 m_pPlayer->GetAutoaimVector(AUTOAIM_5DEGREES); // Update autoaim vector
329
330 if (m_flTimeWeaponIdle > UTIL_WeaponTimeBase())
331 return; // Exit if still in idle time
332
333 int iAnim; // Variable to hold the animation type
334 switch (RANDOM_LONG(0, 1)) // Randomly select an idle animation
335 {
336 case 0:
337 iAnim = MP5_LONGIDLE; // Set long idle animation
338 break;
339
340 default:
341 case 1:
342 iAnim = MP5_IDLE1; // Set short idle animation
343 break;
344 }
345
346 SendWeaponAnim(iAnim); // Play the selected idle animation
347
348 m_flTimeWeaponIdle = UTIL_SharedRandomFloat(m_pPlayer->random_seed, 10, 15); // Set time until next idle action
349}
