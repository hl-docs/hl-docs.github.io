44int CFlyingMonster :: CheckLocalMove ( const Vector &vecStart, const Vector &vecEnd, CBaseEntity *pTarget, float *pflDist )
45{
46 // UNDONE: need to check more than the endpoint
47 if (FBitSet(pev->flags, FL_SWIM) && (UTIL_PointContents(vecEnd) != CONTENTS_WATER))
48 {
49 // ALERT(at_aiconsole, "can't swim out of water\n");
50 return FALSE;
51 }
52
53 TraceResult tr;
54
55 UTIL_TraceHull( vecStart + Vector( 0, 0, 32 ), vecEnd + Vector( 0, 0, 32 ), dont_ignore_monsters, large_hull, edict(), &tr );
56
57 // ALERT( at_console, "%.0f %.0f %.0f : ", vecStart.x, vecStart.y, vecStart.z );
58 // ALERT( at_console, "%.0f %.0f %.0f\n", vecEnd.x, vecEnd.y, vecEnd.z );
59
60 if (pflDist)
61 {
62 *pflDist = ( (tr.vecEndPos - Vector( 0, 0, 32 )) - vecStart ).Length();// get the distance.
63 }
64
65 // ALERT( at_console, "check %d %d %f\n", tr.fStartSolid, tr.fAllSolid, tr.flFraction );
66 if (tr.fStartSolid || tr.flFraction < 1.0)
67 {
68 if ( pTarget && pTarget->edict() == gpGlobals->trace_ent )
69 return LOCALMOVE_VALID;
70 return LOCALMOVE_INVALID;
71 }
72
73 return LOCALMOVE_VALID;
74}

227BOOL CFlyingMonster:: ShouldAdvanceRoute( float flWaypointDist )
228{
229 // Get true 3D distance to the goal so we actually reach the correct height
230 if ( m_Route[ m_iRouteIndex ].iType & bits_MF_IS_GOAL )
231 flWaypointDist = ( m_Route[ m_iRouteIndex ].vecLocation - pev->origin ).Length();
232
233 if ( flWaypointDist <= 64 + (m_flGroundSpeed * gpGlobals->frametime) )
234 return TRUE;

249void CFlyingMonster::MoveExecute( CBaseEntity *pTargetEnt, const Vector &vecDir, float flInterval )
250{
251 if ( pev->movetype == MOVETYPE_FLY )
252 {
253 if ( gpGlobals->time - m_stopTime > 1.0 )
254 {
255 if ( m_IdealActivity != m_movementActivity )
256 {
257 m_IdealActivity = m_movementActivity;
258 m_flGroundSpeed = m_flightSpeed = 200;
259 }
260 }
261 Vector vecMove = pev->origin + (( vecDir + (m_vecTravel * m_momentum) ).Normalize() * (m_flGroundSpeed * flInterval));
262
263 if ( m_IdealActivity != m_movementActivity )
264 {
265 m_flightSpeed = UTIL_Approach( 100, m_flightSpeed, 75 * gpGlobals->frametime );
266 if ( m_flightSpeed < 100 )
267 m_stopTime = gpGlobals->time;
268 }
269 else
270 m_flightSpeed = UTIL_Approach( 20, m_flightSpeed, 300 * gpGlobals->frametime );
271
272 if ( CheckLocalMove ( pev->origin, vecMove, pTargetEnt, NULL ) )
273 {
274 m_vecTravel = (vecMove - pev->origin);
275 m_vecTravel = m_vecTravel.Normalize();
276 UTIL_MoveToOrigin(ENT(pev), vecMove, (m_flGroundSpeed * flInterval), MOVE_STRAFE);
277 }
278 else
279 {
280 m_IdealActivity = GetStoppedActivity();
281 m_stopTime = gpGlobals->time;
282 m_vecTravel = g_vecZero;
283 }
284 }
285 else
286 CBaseMonster::MoveExecute( pTargetEnt, vecDir, flInterval );
287}

299float CFlyingMonster::CeilingZ( const Vector &position )
300{

329BOOL CFlyingMonster::ProbeZ( const Vector &position, const Vector &probe, float *pFraction)
330{
331 int conPosition = UTIL_PointContents(position);
332 if ( (((pev->flags) & FL_SWIM) == FL_SWIM) ^ (conPosition == CONTENTS_WATER))
333 {
334 // SWIMING & !WATER
335 // or FLYING & WATER
336 //
337 *pFraction = 0.0;
338 return TRUE; // We hit a water boundary because we are where we don't belong.
339 }
340 int conProbe = UTIL_PointContents(probe);
341 if (conProbe == conPosition)
342 {
343 // The probe is either entirely inside the water (for fish) or entirely
344 // outside the water (for birds).
345 //
346 *pFraction = 1.0;
347 return FALSE;
348 }
349
350 Vector ProbeUnit = (probe-position).Normalize();
351 float ProbeLength = (probe-position).Length();
352 float maxProbeLength = ProbeLength;
353 float minProbeLength = 0;
354
355 float diff = maxProbeLength - minProbeLength;
356 while (diff > 1.0)
357 {
358 float midProbeLength = minProbeLength + diff/2.0;
359 Vector midProbeVec = midProbeLength * ProbeUnit;
360 if (UTIL_PointContents(position+midProbeVec) == conPosition)
361 {
362 minProbeLength = midProbeLength;
363 }
364 else
365 {
366 maxProbeLength = midProbeLength;
367 }
368 diff = maxProbeLength - minProbeLength;
369 }
370 *pFraction = minProbeLength/ProbeLength;
371
372 return TRUE;
373}
