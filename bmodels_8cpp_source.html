247void CFuncConveyor::UpdateSpeed(float speed)
248{
249 // Encode it as an integer with 4 fractional bits
250 int speedCode = (int)(fabs(speed) * 16.0); // Calculate speed code
251
252 if (speed < 0) // Check if speed is negative
253 pev->rendercolor.x = 1; // Set rendercolor to indicate negative speed
254 else
255 pev->rendercolor.x = 0; // Set rendercolor to indicate positive speed
256
257 pev->rendercolor.y = (speedCode >> 8); // Store high byte of speed code
258 pev->rendercolor.z = (speedCode & 0xFF); // Store low byte of speed code
259}

478void CFuncRotating::Spawn()
479{
480 m_pitch = PITCH_NORM - 1; // Set final pitch
481
482 // Maintain compatibility with previous maps
483 if (m_flVolume == 0.0)
484 m_flVolume = 1.0; // Default volume
485
486 // Default sound attenuation
487 m_flAttenuation = ATTN_NORM;
488
489 // Set attenuation based on spawn flags
490 if (FBitSet(pev->spawnflags, SF_BRUSH_ROTATE_SMALLRADIUS))
491 m_flAttenuation = ATTN_IDLE;
492 else if (FBitSet(pev->spawnflags, SF_BRUSH_ROTATE_MEDIUMRADIUS))
493 m_flAttenuation = ATTN_STATIC;
494 else if (FBitSet(pev->spawnflags, SF_BRUSH_ROTATE_LARGERADIUS))
495 m_flAttenuation = ATTN_NORM;
496
497 // Prevent divide by zero if friction is not set
498 if (m_flFanFriction == 0)
499 m_flFanFriction = 1;
500
501 // Set movement direction based on spawn flags
502 if (FBitSet(pev->spawnflags, SF_BRUSH_ROTATE_Z_AXIS))
503 pev->movedir = Vector(0, 0, 1);
504 else if (FBitSet(pev->spawnflags, SF_BRUSH_ROTATE_X_AXIS))
505 pev->movedir = Vector(1, 0, 0);
506 else
507 pev->movedir = Vector(0, 1, 0); // Default to Y-axis
508
509 // Check for reverse rotation
510 if (FBitSet(pev->spawnflags, SF_BRUSH_ROTATE_BACKWARDS))
511 pev->movedir = pev->movedir * -1;
512
513 // Set solid type and movement type based on spawn flags
514 if (FBitSet(pev->spawnflags, SF_ROTATING_NOT_SOLID))
515 {
516 pev->solid = SOLID_NOT; // Set to not solid
517 pev->skin = CONTENTS_EMPTY; // Set skin to empty
518 pev->movetype = MOVETYPE_PUSH; // Set movement type to push
519 }
520 else
521 {
522 pev->solid = SOLID_BSP; // Set solid type to BSP
523 pev->movetype = MOVETYPE_PUSH; // Set movement type to push
524 }
525
526 UTIL_SetOrigin(pev, pev->origin); // Set the entity's origin
527 SET_MODEL(ENT(pev), STRING(pev->model)); // Set the model for the entity
528
529 SetUse(&CFuncRotating::RotatingUse); // Set the use function
530
531 // Ensure speed is set
532 if (pev->speed <= 0)
533 pev->speed = 0;
534
535 // Check for instant-use brush
536 if (FBitSet(pev->spawnflags, SF_BRUSH_ROTATE_INSTANT))
537 {
538 SetThink(&CFuncRotating::SUB_CallUseToggle); // Set think function for instant use
539 pev->nextthink = pev->ltime + 1.5; // Delay for client startup
540 }
541
542 // Check if this brush can inflict damage
543 if (FBitSet(pev->spawnflags, SF_BRUSH_HURT))
544 {
545 SetTouch(&CFuncRotating::HurtTouch); // Set touch function for damage
546 }
547
548 Precache(); // Precache sounds and resources
549}

557void CFuncRotating::Precache(void)
558{
559 char* szSoundFile = (char*)STRING(pev->message); // Get the sound file path
560
561 // Set up fan sounds
562 if (!FStringNull(pev->message) && strlen(szSoundFile) > 0)
563 {
564 PRECACHE_SOUND(szSoundFile); // Pre-cache the specified sound file
565 pev->noiseRunning = ALLOC_STRING(szSoundFile); // Allocate string for running sound
566 }
567 else
568 {
569 // Use preset sounds based on the number of sounds
570 switch (m_sounds)
571 {
572 case 1:
573 PRECACHE_SOUND("fans/fan1.wav");
574 pev->noiseRunning = ALLOC_STRING("fans/fan1.wav");
575 break;
576 case 2:
577 PRECACHE_SOUND("fans/fan2.wav");
578 pev->noiseRunning = ALLOC_STRING("fans/fan2.wav");
579 break;
580 case 3:
581 PRECACHE_SOUND("fans/fan3.wav");
582 pev->noiseRunning = ALLOC_STRING("fans/fan3.wav");
583 break;
584 case 4:
585 PRECACHE_SOUND("fans/fan4.wav");
586 pev->noiseRunning = ALLOC_STRING("fans/fan4.wav");
587 break;
588 case 5:
589 PRECACHE_SOUND("fans/fan5.wav");
590 pev->noiseRunning = ALLOC_STRING("fans/fan5.wav");
591 break;
592 case 0:
593 default:
594 if (!FStringNull(pev->message) && strlen(szSoundFile) > 0)
595 {
596 PRECACHE_SOUND(szSoundFile); // Pre-cache the specified sound file
597 pev->noiseRunning = ALLOC_STRING(szSoundFile); // Allocate string for running sound
598 break;
599 }
600 else
601 {
602 pev->noiseRunning = ALLOC_STRING("common/null.wav"); // Default to null sound
603 break;
604 }
605 }
606 }
607
608 if (pev->avelocity != g_vecZero)
609 {
610 // Restart sound if the fan was spinning
611 SetThink(&CFuncRotating::SpinUp);
612 pev->nextthink = pev->ltime + 1.5; // Delay for sound restart
613 }

624void CFuncRotating::HurtTouch(CBaseEntity *pOther)
625{
626 entvars_t *pevOther = pOther->pev; // Get the other entity's variables
627
628 // Ignore entities that cannot take damage
629 if (!pevOther->takedamage)
630 return;
631
632 // Calculate damage based on rotation speed
633 pev->dmg = pev->avelocity.Length() / 10; // Damage based on angular velocity
634 pOther->TakeDamage(pev, pev, pev->dmg, DMG_CRUSH); // Inflict damage
635
636 // Apply force to the other entity based on the rotation
637 pevOther->velocity = (pevOther->origin - VecBModelOrigin(pev)).Normalize() * pev->dmg;
638}

648void CFuncRotating::RampPitchVol(int fUp)
649{
650 Vector vecAVel = pev->avelocity; // Get current angular velocity
651 vec_t vecCur; // Current angular velocity magnitude
652 vec_t vecFinal; // Target angular velocity magnitude
653 float fpct; // Percentage of final speed
654 float fvol; // Volume
655 float fpitch; // Pitch
656 int pitch; // Final pitch value
657
658 // Get current angular velocity
659 vecCur = abs(vecAVel.x != 0 ? vecAVel.x : (vecAVel.y != 0 ? vecAVel.y : vecAVel.z));
660
661 // Get target angular velocity
662 vecFinal = (pev->movedir.x != 0 ? pev->movedir.x : (pev->movedir.y != 0 ? pev->movedir.y : pev->movedir.z));
663 vecFinal *= pev->speed; // Scale by speed
664 vecFinal = abs(vecFinal); // Get absolute value
665
666 // Calculate volume and pitch as a percentage of final values
667 fpct = vecCur / vecFinal;
668 fvol = m_flVolume * fpct; // Volume ramps down to 0
669
670 fpitch = FANPITCHMIN + (FANPITCHMAX - FANPITCHMIN) * fpct; // Calculate pitch based on speed
671
672 pitch = (int)fpitch; // Convert to integer pitch
673 if (pitch == PITCH_NORM)
674 pitch = PITCH_NORM - 1; // Adjust pitch if necessary
675
676 // Change the fan's volume and pitch
677 EMIT_SOUND_DYN(ENT(pev), CHAN_STATIC, (char *)STRING(pev->noiseRunning),
678 fvol, m_flAttenuation, SND_CHANGE_PITCH | SND_CHANGE_VOL, pitch);
679}

687void CFuncRotating::SpinUp(void)
688{
689 Vector vecAVel; // Rotational velocity
690
691 pev->nextthink = pev->ltime + 0.1; // Set next think time
692 pev->avelocity = pev->avelocity + (pev->movedir * (pev->speed * m_flFanFriction)); // Increase angular velocity
693
694 vecAVel = pev->avelocity; // Cache entity's rotational velocity
695
696 // Check if target speed is reached
697 if (abs(vecAVel.x) >= abs(pev->movedir.x * pev->speed) &&
698 abs(vecAVel.y) >= abs(pev->movedir.y * pev->speed) &&
699 abs(vecAVel.z) >= abs(pev->movedir.z * pev->speed))
700 {
701 pev->avelocity = pev->movedir * pev->speed; // Set to target speed
702 EMIT_SOUND_DYN(ENT(pev), CHAN_STATIC, (char *)STRING(pev->noiseRunning),
703 m_flVolume, m_flAttenuation, SND_CHANGE_PITCH | SND_CHANGE_VOL, FANPITCHMAX);
704
705 SetThink(&CFuncRotating::Rotate); // Set think function to rotate
706 Rotate(); // Call rotate function
707 }
708 else
709 {
710 RampPitchVol(TRUE); // Ramp up pitch and volume
711 }
712}

720void CFuncRotating::SpinDown(void)
721{
722 Vector vecAVel; // Rotational velocity
723 vec_t vecdir; // Direction of rotation
724
725 pev->nextthink = pev->ltime + 0.1; // Set next think time
726
727 pev->avelocity = pev->avelocity - (pev->movedir * (pev->speed * m_flFanFriction)); // Decrease angular velocity
728
729 vecAVel = pev->avelocity; // Cache entity's rotational velocity
730
731 if (pev->movedir.x != 0)
732 vecdir = pev->movedir.x; // Get direction based on movedir
733 else if (pev->movedir.y != 0)
734 vecdir = pev->movedir.y;
735 else
736 vecdir = pev->movedir.z;
737
738 // Check if target speed is reached
739 if (((vecdir > 0) && (vecAVel.x <= 0 && vecAVel.y <= 0 && vecAVel.z <= 0)) ||
740 ((vecdir < 0) && (vecAVel.x >= 0 && vecAVel.y >= 0 && vecAVel.z >= 0)))
741 {
742 pev->avelocity = g_vecZero; // Set speed to zero
743
744 // Stop sound, we're done
745 EMIT_SOUND_DYN(ENT(pev), CHAN_STATIC, (char *)STRING(pev->noiseRunning),
746 0, 0, SND_STOP, m_pitch);
747
748 SetThink(&CFuncRotating::Rotate); // Set think function to rotate
749 Rotate(); // Call rotate function
750 }
751 else
752 {
753 RampPitchVol(FALSE); // Ramp down pitch and volume
754 }
755}

778void CFuncRotating::RotatingUse(CBaseEntity *pActivator, CBaseEntity *pCaller, USE_TYPE useType, float value)
779{
780 // Check if this is a brush that should accelerate and decelerate
781 if (FBitSet(pev->spawnflags, SF_BRUSH_ACCDCC))
782 {
783 // Fan is spinning, so stop it.
784 if (pev->avelocity != g_vecZero)
785 {
786 SetThink(&CFuncRotating::SpinDown); // Set think function to spin down
787 pev->nextthink = pev->ltime + 0.1; // Set next think time
788 }
789 else // Fan is not moving, so start it
790 {
791 SetThink(&CFuncRotating::SpinUp); // Set think function to spin up
792 EMIT_SOUND_DYN(ENT(pev), CHAN_STATIC, (char *)STRING(pev->noiseRunning),
793 0.01, m_flAttenuation, 0, FANPITCHMIN); // Start sound
794
795 pev->nextthink = pev->ltime + 0.1; // Set next think time
796 }
797 }
798 else if (!FBitSet(pev->spawnflags, SF_BRUSH_ACCDCC)) // Normal start/stop brush
799 {
800 if (pev->avelocity != g_vecZero) // If currently spinning
801 {
802 SetThink(&CFuncRotating::SpinDown); // Set think function to spin down
803 pev->nextthink = pev->ltime + 0.1; // Set next think time
804 }
805 else // If not spinning, start it
806 {
807 EMIT_SOUND_DYN(ENT(pev), CHAN_STATIC, (char *)STRING(pev->noiseRunning),
808 m_flVolume, m_flAttenuation, 0, FANPITCHMAX); // Start sound
809 pev->avelocity = pev->movedir * pev->speed; // Set angular velocity
810
811 SetThink(&CFuncRotating::Rotate); // Set think function to rotate
812 Rotate(); // Call rotate function
813 }
814 }

1026 dt = gpGlobals->time - m_time; // Calculate elapsed time
1027 m_time = gpGlobals->time; // Update last time called
1028

1030 pev->speed -= m_accel * dt; // Integrate velocity
1031 else
1032 pev->speed += m_accel * dt; // Update speed
1033
1034 // Clamp speed to maximum limits

1036 pev->speed = m_maxSpeed;
1037 else if (pev->speed < -m_maxSpeed)
1038 pev->speed = -m_maxSpeed;

1052 pev->speed = 0; // Stop the pendulum

1055 }
