632 // no way to get there =(
633 ALERT ( at_aiconsole, "GetPathToEnemy failed!!\n" );
634 TaskFail();
635 }
636 break;
637 }
638 default:
639 CSquadMonster :: StartTask ( pTask );
640 break;
641 }

658Vector Intersect( Vector vecSrc, Vector vecDst, Vector vecMove, float flSpeed )
659{
660 Vector vecTo = vecDst - vecSrc;
661
662 float a = DotProduct( vecMove, vecMove ) - flSpeed * flSpeed;
663 float b = 0 * DotProduct(vecTo, vecMove); // why does this work?
664 float c = DotProduct( vecTo, vecTo );
665
666 float t;
667 if (a == 0)
668 {
669 t = c / (flSpeed * flSpeed);
670 }
671 else
672 {
673 t = b * b - 4 * a * c;
674 t = sqrt( t ) / (2.0 * a);
675 float t1 = -b +t;
676 float t2 = -b -t;
677
678 if (t1 < 0 || t2 < t1)
679 t = t2;
680 else
681 t = t1;
682 }
683
684 // ALERT( at_console, "Intersect %f\n", t );
685
686 if (t < 0.1)
687 t = 0.1;
688 if (t > 10.0)
689 t = 10.0;
690
691 Vector vecHit = vecTo + vecMove * t;
692 return vecHit.Normalize( ) * flSpeed;
693}

704int CController::LookupFloat( )
705{
706 if (m_velocity.Length( ) < 32.0)
707 {
708 return LookupSequence( "up" );
709 }
710
711 UTIL_MakeAimVectors( pev->angles );
712 float x = DotProduct( gpGlobals->v_forward, m_velocity );
713 float y = DotProduct( gpGlobals->v_right, m_velocity );
714 float z = DotProduct( gpGlobals->v_up, m_velocity );
715
716 if (fabs(x) > fabs(y) && fabs(x) > fabs(z))
717 {
718 if (x > 0)
719 return LookupSequence( "forward");
720 else
721 return LookupSequence( "backward");
722 }
723 else if (fabs(y) > fabs(z))
724 {
725 if (y > 0)
726 return LookupSequence( "right");
727 else
728 return LookupSequence( "left");
729 }
730 else
731 {
732 if (z > 0)
733 return LookupSequence( "up");
734 else
735 return LookupSequence( "down");
736 }
737}

740//=========================================================
741// RunTask
742//=========================================================
751void CController :: RunTask ( Task_t *pTask )
752{
753
754 if (m_flShootEnd > gpGlobals->time)
755 {
756 Vector vecHand, vecAngle;
757
758 GetAttachment( 2, vecHand, vecAngle );
759
760 while (m_flShootTime < m_flShootEnd && m_flShootTime < gpGlobals->time)
761 {
762 Vector vecSrc = vecHand + pev->velocity * (m_flShootTime - gpGlobals->time);
763 Vector vecDir;
764
765 if (m_hEnemy != NULL)
766 {
767 if (HasConditions( bits_COND_SEE_ENEMY ))
768 {
769 m_vecEstVelocity = m_vecEstVelocity * 0.5 + m_hEnemy->pev->velocity * 0.5;
770 }
771 else
772 {
773 m_vecEstVelocity = m_vecEstVelocity * 0.8;
774 }
775 vecDir = Intersect( vecSrc, m_hEnemy->BodyTarget( pev->origin ), m_vecEstVelocity, gSkillData.controllerSpeedBall );
776 float delta = 0.03490; // +-2 degree
777 vecDir = vecDir + Vector( RANDOM_FLOAT( -delta, delta ), RANDOM_FLOAT( -delta, delta ), RANDOM_FLOAT( -delta, delta ) ) * gSkillData.controllerSpeedBall;
778
779 vecSrc = vecSrc + vecDir * (gpGlobals->time - m_flShootTime);
780 CBaseMonster *pBall = (CBaseMonster*)Create( "controller_energy_ball", vecSrc, pev->angles, edict() );
781 pBall->pev->velocity = vecDir;
782 }
783 m_flShootTime += 0.2;
784 }
785
786 if (m_flShootTime > m_flShootEnd)
787 {
788 m_iBall[0] = 64;
789 m_iBallTime[0] = m_flShootEnd;
790 m_iBall[1] = 64;
791 m_iBallTime[1] = m_flShootEnd;
792 m_fInCombat = FALSE;
793 }
794 }
795
796 switch ( pTask->iTask )
797 {
798 case TASK_WAIT_FOR_MOVEMENT:
799 case TASK_WAIT:
800 case TASK_WAIT_FACE_ENEMY:
801 case TASK_WAIT_PVS:
802 MakeIdealYaw( m_vecEnemyLKP );
803 ChangeYaw( pev->yaw_speed );
804
805 if (m_fSequenceFinished)
806 {
807 m_fInCombat = FALSE;
808 }
809
810 CSquadMonster :: RunTask ( pTask );
811
812 if (!m_fInCombat)
813 {
814 if (HasConditions ( bits_COND_CAN_RANGE_ATTACK1 ))
815 {
816 pev->sequence = LookupActivity( ACT_RANGE_ATTACK1 );
817 pev->frame = 0;
818 ResetSequenceInfo( );
819 m_fInCombat = TRUE;
820 }
821 else if (HasConditions ( bits_COND_CAN_RANGE_ATTACK2 ))
822 {
823 pev->sequence = LookupActivity( ACT_RANGE_ATTACK2 );
824 pev->frame = 0;
825 ResetSequenceInfo( );
826 m_fInCombat = TRUE;
827 }
828 else
829 {
830 int iFloat = LookupFloat( );
831 if (m_fSequenceFinished || iFloat != pev->sequence)
832 {
833 pev->sequence = iFloat;
834 pev->frame = 0;
835 ResetSequenceInfo( );
836 }
837 }
838 }
839 break;
840 default:
841 CSquadMonster :: RunTask ( pTask );
842 break;
843 }
844}

1258 RouteSimplify( m_hEnemy );
1259 else
1260 RouteSimplify( m_hTargetEnt );
1261 FRefreshRoute();
