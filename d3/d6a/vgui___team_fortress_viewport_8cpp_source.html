343bool CCommandMenu::RecalculateVisibles( int iYOffset, bool bHideAll )
344{
345 int i, iCurrentY = 0;
346 int iVisibleButtons = 0;
347
348 // Cycle through all the buttons in this menu, and see which will be visible
349 for (i = 0; i < m_iButtons; i++)
350 {
351 int iClass = m_aButtons[i]->GetPlayerClass();
352
353 if ( (iClass && iClass != g_iPlayerClass ) || ( m_aButtons[i]->IsNotValid() ) || bHideAll )
354 {
355 m_aButtons[i]->setVisible( false );
356 if ( m_aButtons[i]->GetSubMenu() != NULL )
357 {
358 (m_aButtons[i]->GetSubMenu())->RecalculateVisibles( 0, true );
359 }
360 }
361 else
362 {
363 // If it's got a submenu, force it to check visibilities
364 if ( m_aButtons[i]->GetSubMenu() != NULL )
365 {
366 if ( !(m_aButtons[i]->GetSubMenu())->RecalculateVisibles( 0 , false ) )
367 {
368 // The submenu had no visible buttons, so don't display this button
369 m_aButtons[i]->setVisible( false );
370 continue;
371 }
372 }
373
374 m_aButtons[i]->setVisible( true );
375 iVisibleButtons++;
376 }
377 }
378
379 // Set Size
380 setSize( _size[0], (iVisibleButtons * (m_flButtonSizeY-1)) + 1 );
381
382 if ( iYOffset )
383 {
384 m_iYOffset = iYOffset;
385 }
386
387 for (i = 0; i < m_iButtons; i++)
388 {
389 if ( m_aButtons[i]->isVisible() )
390 {
391 if ( m_aButtons[i]->GetSubMenu() != NULL )
392 (m_aButtons[i]->GetSubMenu())->RecalculateVisibles( iCurrentY + m_iYOffset, false );
393
394
395 // Make sure it's at the right Y position
396 // m_aButtons[i]->getPos( iXPos, iYPos );
397
398 if ( m_iDirection )
399 {
400 m_aButtons[i]->setPos( 0, (iVisibleButtons-1) * (m_flButtonSizeY-1) - iCurrentY );
401 }
402 else
403 {
404 m_aButtons[i]->setPos( 0, iCurrentY );
405 }
406
407 iCurrentY += (m_flButtonSizeY-1);
408 }
409 }
410
411 return iVisibleButtons?true:false;
412}

420void CCommandMenu::RecalculatePositions( int iYOffset )
421{
422 int iTop;
423 int iAdjust = 0;
424
425 m_iYOffset+= iYOffset;
426
427 if ( m_iDirection )
428 iTop = ScreenHeight - (m_iYOffset + _size[1] );
429 else
430 iTop = m_iYOffset;
431
432 if ( iTop < 0 )
433 iTop = 0;
434
435 // Calculate if this is going to fit onscreen, and shuffle it up if it won't
436 int iBottom = iTop + _size[1];
437
438 if ( iBottom > ScreenHeight )
439 {
440 // Move in increments of button sizes
441 while (iAdjust < (iBottom - ScreenHeight))
442 {
443 iAdjust += m_flButtonSizeY - 1;
444 }
445
446 iTop -= iAdjust;
447
448 // Make sure it doesn't move off the top of the screen (the menu's too big to fit it all)
449 if ( iTop < 0 )
450 {
451 iAdjust -= (0 - iTop);
452 iTop = 0;
453 }
454 }
455
456 setPos( _pos[0], iTop );
457
458 // We need to force all menus below this one to update their positions now, because they
459 // might have submenus riding off buttons in this menu that have just shifted.
460 for (int i = 0; i < m_iButtons; i++)
461 m_aButtons[i]->UpdateSubMenus( iAdjust );
462}

524 pMenu->setParent(this);
525 pButton->AddSubMenu( pMenu );
526 pButton->setFont( Scheme::sf_primary3 );
527 pMenu->m_flButtonSizeY = m_pCurrentCommandMenu->m_flButtonSizeY;
528
529 // Create the Submenu-open signal
530 InputSignal *pISignal = new CMenuHandler_PopupSubMenuInput(pButton, pMenu);
531 pButton->addInputSignal(pISignal);
532
533 // Put a > to show it's a submenu
534 CImageLabel *pLabel = new CImageLabel( "arrowright", XRES(CMENU_SIZE_X - SUBMENU_SIZE_X), YRES(SUBMENU_SIZE_Y) );
535 pLabel->setParent(pButton);
536 pLabel->addInputSignal(pISignal);
537
538 // Reposition
539 pLabel->getPos( iXPos, iYPos );
540 pLabel->setPos( pButton->getWide() - pLabel->getImageWide()-4, -4 );
541
542 // Create the mouse off signal for the Label too
543 if (!pButton->m_bNoHighlight)
544 pLabel->addInputSignal( new CHandler_CommandButtonHighlight(pButton) );
545
546 return pMenu;
547}

785int TeamFortressViewport::CreateCommandMenu( char * menuFile, int direction, int yOffset, bool flatDesign, float flButtonSizeX, float flButtonSizeY, int xOffset )
786{
787 // COMMAND MENU
788 // Create the root of this new Command Menu
789
790 int newIndex = m_iNumMenus;
791
792 m_pCommandMenus[newIndex] = new CCommandMenu(NULL, direction, xOffset, yOffset, flButtonSizeX, 300); // This will be resized once we know how many items are in it
793 m_pCommandMenus[newIndex]->setParent(this);
794 m_pCommandMenus[newIndex]->setVisible(false);
795 m_pCommandMenus[newIndex]->m_flButtonSizeY = flButtonSizeY;
796 m_pCommandMenus[newIndex]->m_iSpectCmdMenu = direction;
797
798 m_iNumMenus++;
799
800 // Read Command Menu from the txt file
801 char token[1024];
802 char *pfile = (char*)gEngfuncs.COM_LoadFile( menuFile, 5, NULL);
803 if (!pfile)
804 {
805 gEngfuncs.Con_DPrintf( "Unable to open %s\n", menuFile);
806 SetCurrentCommandMenu( NULL );
807 return newIndex;
808 }
809
810#ifdef _WIN32
811try
812{
813#endif
814 // First, read in the localisation strings
815
816 // Detpack strings
817 gHUD.m_TextMessage.LocaliseTextString( "#DetpackSet_For5Seconds", m_sDetpackStrings[0], MAX_BUTTON_SIZE );
818 gHUD.m_TextMessage.LocaliseTextString( "#DetpackSet_For20Seconds", m_sDetpackStrings[1], MAX_BUTTON_SIZE );
819 gHUD.m_TextMessage.LocaliseTextString( "#DetpackSet_For50Seconds", m_sDetpackStrings[2], MAX_BUTTON_SIZE );
820
821 // Now start parsing the menu structure
822 m_pCurrentCommandMenu = m_pCommandMenus[newIndex];
823 char szLastButtonText[32] = "file start";
824 pfile = gEngfuncs.COM_ParseFile(pfile, token);
825 while ( ( strlen ( token ) > 0 ) && ( m_iNumMenus < MAX_MENUS ) )
826 {
827 // Keep looping until we hit the end of this menu
828 while ( token[0] != '}' && ( strlen( token ) > 0 ) )
829 {
830 char cText[32] = "";
831 char cBoundKey[32] = "";
832 char cCustom[32] = "";
833 static const int cCommandLength = 128;
834 char cCommand[cCommandLength] = "";
835 char szMap[MAX_MAPNAME] = "";
836 int iPlayerClass = 0;
837 int iCustom = false;
838 int iTeamOnly = -1;
839 int iToggle = 0;
840 int iButtonY;
841 bool bGetExtraToken = true;
842 CommandButton *pButton = NULL;
843
844 // We should never be here without a Command Menu
845 if (!m_pCurrentCommandMenu)
846 {
847 gEngfuncs.Con_Printf("Error in %s file after '%s'.\n",menuFile, szLastButtonText );
848 m_iInitialized = false;
849 return newIndex;
850 }
851
852 // token should already be the bound key, or the custom name
853 strncpy( cCustom, token, 32 );
854 cCustom[31] = '\0';
855
856 // See if it's a custom button
857 if (!strcmp(cCustom, "CUSTOM") )
858 {
859 iCustom = true;
860
861 // Get the next token
862 pfile = gEngfuncs.COM_ParseFile(pfile, token);
863 }
864 // See if it's a map
865 else if (!strcmp(cCustom, "MAP") )
866 {
867 // Get the mapname
868 pfile = gEngfuncs.COM_ParseFile(pfile, token);
869 strncpy( szMap, token, MAX_MAPNAME );
870 szMap[MAX_MAPNAME-1] = '\0';
871
872 // Get the next token
873 pfile = gEngfuncs.COM_ParseFile(pfile, token);
874 }

904 }

906 }

908 // Get the button bound key
909 strncpy( cBoundKey, token, 32 );
910 cText[31] = '\0';
911
912 // Get the button text
913 pfile = gEngfuncs.COM_ParseFile(pfile, token);
914 strncpy( cText, token, 32 );
915 cText[31] = '\0';
916
917 // save off the last button text we've come across (for error reporting)
918 strcpy( szLastButtonText, cText );
919
920 // Get the button command
921 pfile = gEngfuncs.COM_ParseFile(pfile, token);
922 strncpy( cCommand, token, cCommandLength );

924

926
927 // Custom button handling
928 if ( iCustom )
929 {
930 pButton = CreateCustomButton( cText, cCommand, iButtonY );
931
932 // Get the next token to see if we're a menu
933 pfile = gEngfuncs.COM_ParseFile(pfile, token);
934
935 if ( token[0] == '{' )
936 {
937 strcpy( cCommand, token );
938 }
939 else
940 {
941 bGetExtraToken = false;
942 }
943 }
944 else if ( szMap[0] != '\0' )
945 {
946 // create a map button
947 pButton = new MapButton(szMap, cText, xOffset, iButtonY, flButtonSizeX, flButtonSizeY );

952 pButton = new TeamOnlyCommandButton( iTeamOnly, cText, xOffset, iButtonY, flButtonSizeX, flButtonSizeY, flatDesign );

956 pButton = new ToggleCommandButton( cCommand, cText, xOffset, iButtonY, flButtonSizeX, flButtonSizeY, flatDesign );
957 }
958 else if ( direction == 1 )
959 {
960 if ( iToggle )
961 pButton = new SpectToggleButton( cCommand, cText, xOffset, iButtonY, flButtonSizeX, flButtonSizeY, flatDesign );

963 pButton = new SpectButton( iPlayerClass, cText, xOffset, iButtonY, flButtonSizeX, flButtonSizeY );
964 }

966 {
967 // normal button
968 pButton = new CommandButton( iPlayerClass, cText, xOffset, iButtonY, flButtonSizeX, flButtonSizeY, flatDesign );
969 }
970

972 if ( pButton )
973 {
974 m_pCurrentCommandMenu->AddButton( pButton );
975 pButton->setBoundKey( cBoundKey[0] );
976 pButton->setParentMenu( m_pCurrentCommandMenu );
977
978 // Override font in CommandMenu
979 pButton->setFont( Scheme::sf_primary3 );
980 }
981

983 if ( cCommand[0] == '{' )
984 {
985 if ( m_iNumMenus >= MAX_MENUS )
986 {
987 gEngfuncs.Con_Printf( "Too many menus in %s past '%s'\n",menuFile, szLastButtonText );
988 }
989 else
990 {
991 // Create the menu
992 m_pCommandMenus[m_iNumMenus] = CreateSubMenu(pButton, m_pCurrentCommandMenu, iButtonY );
993 m_pCurrentCommandMenu = m_pCommandMenus[m_iNumMenus];
994 m_iNumMenus++;
995 }
996 }
997 else if ( !iCustom )
998 {
999 // Create the button and attach it to the current menu
1000 if ( iToggle )
1001 pButton->addActionSignal(new CMenuHandler_ToggleCvar(cCommand));
1002 else
1003 pButton->addActionSignal(new CMenuHandler_StringCommand(cCommand));
1004 // Create an input signal that'll popup the current menu
1005 pButton->addInputSignal( new CMenuHandler_PopupSubMenuInput(pButton, m_pCurrentCommandMenu) );
1006 }
1007
1008 // Get the next token
1009 if ( bGetExtraToken )
1010 {
1011 pfile = gEngfuncs.COM_ParseFile(pfile, token);
1012 }
1013 }
1014
1015 // Move back up a menu
1016 m_pCurrentCommandMenu = m_pCurrentCommandMenu->GetParentMenu();
1017
1018 pfile = gEngfuncs.COM_ParseFile(pfile, token);
1019 }
1020#ifdef _WIN32
1021}
1022catch( CException *e )
1023{
1024 e;
1025 //e->Delete();
1026 e = NULL;
1027 m_iInitialized = false;
1028 return newIndex;
1029}
1030#endif
1031
1032 SetCurrentMenu( NULL );
1033 SetCurrentCommandMenu( NULL );
1034 gEngfuncs.COM_FreeFile( pfile );
1035
1036 m_iInitialized = true;
1037 return newIndex;
1038}
