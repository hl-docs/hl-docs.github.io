392void CTalkMonster::StartTask(Task_t* pTask)
393{
394 switch (pTask->iTask)
395 {
396 case TASK_TLK_SPEAK:
397 // ask question or make statement
398 FIdleSpeak();
399 TaskComplete();
400 break;
401
402 case TASK_TLK_RESPOND:
403 // respond to question
404 IdleRespond();
405 TaskComplete();
406 break;
407
408 case TASK_TLK_HELLO:
409 // greet player
410 FIdleHello();
411 TaskComplete();
412 break;
413
414
415 case TASK_TLK_STARE:
416 // let the player know I know he's staring at me.
417 FIdleStare();
418 TaskComplete();
419 break;
420
421 case TASK_FACE_PLAYER:
422 case TASK_TLK_LOOK_AT_CLIENT:
423 case TASK_TLK_CLIENT_STARE:
424 // track head to the client for a while.
425 m_flWaitFinished = gpGlobals->time + pTask->flData;
426 break;
427
428 case TASK_TLK_EYECONTACT:
429 break;
430
431 case TASK_TLK_IDEALYAW:
432 if (m_hTalkTarget != NULL)
433 {
434 pev->yaw_speed = 60;
435 float yaw = VecToYaw(m_hTalkTarget->pev->origin - pev->origin) - pev->angles.y;
436
437 if (yaw > 180) yaw -= 360;
438 if (yaw < -180) yaw += 360;
439
440 if (yaw < 0)
441 {
442 pev->ideal_yaw = min(yaw + 45, 0) + pev->angles.y;
443 }
444 else
445 {
446 pev->ideal_yaw = max(yaw - 45, 0) + pev->angles.y;
447 }
448 }
449 TaskComplete();
450 break;
451
452 case TASK_TLK_HEADRESET:
453 // reset head position after looking at something
454 m_hTalkTarget = NULL;
455 TaskComplete();
456 break;
457
458 case TASK_TLK_STOPSHOOTING:
459 // tell player to stop shooting
460 PlaySentence(m_szGrp[TLK_NOSHOOT], RANDOM_FLOAT(2.8, 3.2), VOL_NORM, ATTN_NORM);
461 TaskComplete();
462 break;
463
464 case TASK_CANT_FOLLOW:
465 StopFollowing(FALSE);
466 PlaySentence(m_szGrp[TLK_STOP], RANDOM_FLOAT(2, 2.5), VOL_NORM, ATTN_NORM);
467 TaskComplete();
468 break;
469
470 case TASK_WALK_PATH_FOR_UNITS:
471 m_movementActivity = ACT_WALK;
472 break;
473
474 case TASK_MOVE_AWAY_PATH:
475 {
476 Vector dir = pev->angles;
477 dir.y = pev->ideal_yaw + 180;
478 Vector move;
479
480 UTIL_MakeVectorsPrivate(dir, move, NULL, NULL);
481 dir = pev->origin + move * pTask->flData;
482 if (MoveToLocation(ACT_WALK, 2, dir))
483 {
484 TaskComplete();
485 }
486 else if (FindCover(pev->origin, pev->view_ofs, 0, CoverRadius()))
487 {
488 // then try for plain ole cover
489 m_flMoveWaitFinished = gpGlobals->time + 2;
490 TaskComplete();
491 }
492 else
493 {
494 // nowhere to go?
495 TaskFail();
496 }
497 }
498 break;
499
500 case TASK_PLAY_SCRIPT:
501 m_hTalkTarget = NULL;
502 CBaseMonster::StartTask(pTask);
503 break;
504
505 default:
506 CBaseMonster::StartTask(pTask);
507 }
508}

519void CTalkMonster::RunTask(Task_t* pTask)
520{
521 switch (pTask->iTask)
522 {
523 case TASK_TLK_CLIENT_STARE:
524 case TASK_TLK_LOOK_AT_CLIENT:
525
526 edict_t* pPlayer;
527
528 // track head to the client for a while.
529 if (m_MonsterState == MONSTERSTATE_IDLE &&
530 !IsMoving() &&
531 !IsTalking())
532 {
533 // Get edict for one player
534 pPlayer = g_engfuncs.pfnPEntityOfEntIndex(1);
535
536 if (pPlayer)
537 {
538 IdleHeadTurn(pPlayer->v.origin);
539 }
540 }
541 else
542 {
543 // started moving or talking
544 TaskFail();
545 return;
546 }
547
548 if (pTask->iTask == TASK_TLK_CLIENT_STARE)
549 {
550 // fail out if the player looks away or moves away.
551 if ((pPlayer->v.origin - pev->origin).Length2D() > TLK_STARE_DIST)
552 {
553 // player moved away.
554 TaskFail();
555 }
556
557 UTIL_MakeVectors(pPlayer->v.angles);
558 if (UTIL_DotPoints(pPlayer->v.origin, pev->origin, gpGlobals->v_forward) < m_flFieldOfView)
559 {
560 // player looked away
561 TaskFail();
562 }
563 }
564
565 if (gpGlobals->time > m_flWaitFinished)
566 {
567 TaskComplete();
568 }
569 break;
570
571 case TASK_FACE_PLAYER:
572 {
573 // Get edict for one player
574 edict_t* pPlayer = g_engfuncs.pfnPEntityOfEntIndex(1);
575
576 if (pPlayer)
577 {
578 MakeIdealYaw(pPlayer->v.origin);
579 ChangeYaw(pev->yaw_speed);
580 IdleHeadTurn(pPlayer->v.origin);
581 if (gpGlobals->time > m_flWaitFinished && FlYawDiff() < 10)
582 {
583 TaskComplete();
584 }
585 }
586 else
587 {
588 TaskFail();
589 }
590 }
591 break;
592
593 case TASK_TLK_EYECONTACT:
594 if (!IsMoving() && IsTalking() && m_hTalkTarget != NULL)
595 {
596 // ALERT( at_console, "waiting %f\n", m_flStopTalkTime - gpGlobals->time );
597 IdleHeadTurn(m_hTalkTarget->pev->origin);
598 }
599 else
600 {
601 TaskComplete();
602 }
603 break;
604
605 case TASK_WALK_PATH_FOR_UNITS:
606 {
607 float distance;
608
609 distance = (m_vecLastPosition - pev->origin).Length2D();
610
611 // Walk path until far enough away
612 if (distance > pTask->flData || MovementIsComplete())
613 {
614 TaskComplete();
615 RouteClear(); // Stop moving
616 }
617 }
618 break;
619 case TASK_WAIT_FOR_MOVEMENT:
620 if (IsTalking() && m_hTalkTarget != NULL)
621 {
622 // ALERT(at_console, "walking, talking\n");
623 IdleHeadTurn(m_hTalkTarget->pev->origin);
624 }
625 else
626 {
627 IdleHeadTurn(pev->origin);
628 // override so that during walk, a scientist may talk and greet player
629 FIdleHello();
630 if (RANDOM_LONG(0, m_nSpeak * 20) == 0)
631 {
632 FIdleSpeak();
633 }
634 }
635
636 CBaseMonster::RunTask(pTask);
637 if (TaskIsComplete())
638 IdleHeadTurn(pev->origin);
639 break;
640
641 default:
642 if (IsTalking() && m_hTalkTarget != NULL)
643 {
644 IdleHeadTurn(m_hTalkTarget->pev->origin);
645 }
646 else
647 {
648 SetBoneController(0, 0);
649 }
650 CBaseMonster::RunTask(pTask);
651 }
652}

693CBaseEntity* CTalkMonster::EnumFriends(CBaseEntity* pPrevious, int listNumber, BOOL bTrace)
694{
695 CBaseEntity* pFriend = pPrevious;
696 char* pszFriend;
697 TraceResult tr;
698 Vector vecCheck;
699
700 pszFriend = m_szFriends[FriendNumber(listNumber)];
701 while (pFriend = UTIL_FindEntityByClassname(pFriend, pszFriend))
702 {
703 if (pFriend == this || !pFriend->IsAlive())
704 // don't talk to self or dead people
705 continue;
706 if (bTrace)
707 {
708 vecCheck = pFriend->pev->origin;
709 vecCheck.z = pFriend->pev->absmax.z;
710
711 UTIL_TraceLine(pev->origin, vecCheck, ignore_monsters, ENT(pev), &tr);
712 }
713 else
714 tr.flFraction = 1.0;
715
716 if (tr.flFraction == 1.0)
717 {
718 return pFriend;
719 }
720 }
721
722 return NULL;
723}

878//=========================================================
879// FindNearestFriend
880// Scan for nearest, visible friend. If fPlayer is true, look for
881// nearest player
882//=========================================================
891CBaseEntity* CTalkMonster::FindNearestFriend(BOOL fPlayer)
892{
893 CBaseEntity* pFriend = NULL;
894 CBaseEntity* pNearest = NULL;
895 float range = 10000000.0;
896 TraceResult tr;
897 Vector vecStart = pev->origin;
898 Vector vecCheck;
899 int i;
900 char* pszFriend;
901 int cfriends;
902
903 vecStart.z = pev->absmax.z;
904
905 if (fPlayer)
906 cfriends = 1;
907 else
908 cfriends = TLK_CFRIENDS;
909
910 // for each type of friend...
911
912 for (i = cfriends - 1; i > -1; i--)
913 {
914 if (fPlayer)
915 pszFriend = "player";
916 else
917 pszFriend = m_szFriends[FriendNumber(i)];
918
919 if (!pszFriend)
920 continue;
921
922 // for each friend in this bsp...
923 while (pFriend = UTIL_FindEntityByClassname(pFriend, pszFriend))
924 {
925 if (pFriend == this || !pFriend->IsAlive())
926 // don't talk to self or dead people
927 continue;
928
929 CBaseMonster* pMonster = pFriend->MyMonsterPointer();
930
931 // If not a monster for some reason, or in a script, or prone
932 if (!pMonster || pMonster->m_MonsterState == MONSTERSTATE_SCRIPT || pMonster->m_MonsterState == MONSTERSTATE_PRONE)
933 continue;
934
935 vecCheck = pFriend->pev->origin;
936 vecCheck.z = pFriend->pev->absmax.z;
937
938 // if closer than previous friend, and in range, see if he's visible
939
940 if (range > (vecStart - vecCheck).Length())
941 {
942 UTIL_TraceLine(vecStart, vecCheck, ignore_monsters, ENT(pev), &tr);
943
944 if (tr.flFraction == 1.0)
945 {
946 // visible and in range, this is the new nearest scientist
947 if ((vecStart - vecCheck).Length() < TALKRANGE_MIN)
948 {
949 pNearest = pFriend;
950 range = (vecStart - vecCheck).Length();
951 }
952 }
953 }
954 }
955 }
956 return pNearest;
957}
