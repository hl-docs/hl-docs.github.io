156int CZombie::TakeDamage(entvars_t *pevInflictor, entvars_t *pevAttacker, float flDamage, int bitsDamageType)
157{
158 // Take 30% damage from bullets
159 if (bitsDamageType == DMG_BULLET)
160 {
161 Vector vecDir = pev->origin - (pevInflictor->absmin + pevInflictor->absmax) * 0.5; // Calculate direction of the hit
162 vecDir = vecDir.Normalize(); // Normalize the direction vector
163 float flForce = DamageForce(flDamage); // Calculate the force based on damage
164 pev->velocity = pev->velocity + vecDir * flForce; // Apply force to the zombie's velocity
165 flDamage *= 0.3; // Reduce damage taken to 30%
166 }
167
168 // HACK HACK -- until we fix this.
169 if (IsAlive())
170 PainSound(); // Play pain sound if the zombie is alive
171 return CBaseMonster::TakeDamage(pevInflictor, pevAttacker, flDamage, bitsDamageType); // Call the base class method
172}

237void CZombie::HandleAnimEvent(MonsterEvent_t *pEvent)
238{
239 switch (pEvent->event)
240 {
241 case ZOMBIE_AE_ATTACK_RIGHT:
242 {
243 // Handle right attack event
244 CBaseEntity *pHurt = CheckTraceHullAttack(70, gSkillData.zombieDmgOneSlash, DMG_SLASH); // Check for attack hit
245 if (pHurt)
246 {
247 if (pHurt->pev->flags & (FL_MONSTER | FL_CLIENT))
248 {
249 pHurt->pev->punchangle.z = -18; // Apply punch angle
250 pHurt->pev->punchangle.x = 5; // Apply punch angle
251 pHurt->pev->velocity = pHurt->pev->velocity - gpGlobals->v_right * 100; // Apply force to the hurt entity
252 }
253 // Play a random attack hit sound
254 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, pAttackHitSounds[RANDOM_LONG(0, ARRAYSIZE(pAttackHitSounds) - 1)], 1.0, ATTN_NORM, 0, 100 + RANDOM_LONG(-5, 5));
255 }
256 else // Play a random attack miss sound
257 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, pAttackMissSounds[RANDOM_LONG(0, ARRAYSIZE(pAttackMissSounds) - 1)], 1.0, ATTN_NORM, 0, 100 + RANDOM_LONG(-5, 5));
258
259 if (RANDOM_LONG(0, 1))
260 AttackSound(); // Play attack sound randomly
261 }
262 break;
263
264 case ZOMBIE_AE_ATTACK_LEFT:
265 {
266 // Handle left attack event
267 CBaseEntity *pHurt = CheckTraceHullAttack(70, gSkillData.zombieDmgOneSlash, DMG_SLASH); // Check for attack hit
268 if (pHurt)
269 {
270 if (pHurt->pev->flags & (FL_MONSTER | FL_CLIENT))
271 {
272 pHurt->pev->punchangle.z = 18; // Apply punch angle
273 pHurt->pev->punchangle.x = 5; // Apply punch angle
274 pHurt->pev->velocity = pHurt->pev->velocity + gpGlobals->v_right * 100; // Apply force to the hurt entity
275 }
276 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, pAttackHitSounds[RANDOM_LONG(0, ARRAYSIZE(pAttackHitSounds) - 1)], 1.0, ATTN_NORM, 0, 100 + RANDOM_LONG(-5, 5)); // Play attack hit sound
277 }
278 else
279 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, pAttackMissSounds[RANDOM_LONG(0, ARRAYSIZE(pAttackMissSounds) - 1)], 1.0, ATTN_NORM, 0, 100 + RANDOM_LONG(-5, 5)); // Play attack miss sound
280
281 if (RANDOM_LONG(0, 1))
282 AttackSound(); // Play attack sound randomly
283 }
284 break;
285
286 case ZOMBIE_AE_ATTACK_BOTH:
287 {
288 // Handle both attack event
289 CBaseEntity *pHurt = CheckTraceHullAttack(70, gSkillData.zombieDmgBothSlash, DMG_SLASH); // Check for attack hit
290 if (pHurt)
291 {
292 if (pHurt->pev->flags & (FL_MONSTER | FL_CLIENT))
293 {
294 pHurt->pev->punchangle.x = 5; // Apply punch angle
295 pHurt->pev->velocity = pHurt->pev->velocity + gpGlobals->v_forward * -100; // Apply force to the hurt entity
296 }
297 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, pAttackHitSounds[RANDOM_LONG(0, ARRAYSIZE(pAttackHitSounds) - 1)], 1.0, ATTN_NORM, 0, 100 + RANDOM_LONG(-5, 5)); // Play attack hit sound
298 }
299 else
300 EMIT_SOUND_DYN(ENT(pev), CHAN_WEAPON, pAttackMissSounds[RANDOM_LONG(0, ARRAYSIZE(pAttackMissSounds) - 1)], 1.0, ATTN_NORM, 0, 100 + RANDOM_LONG(-5, 5)); // Play attack miss sound
301
302 if (RANDOM_LONG(0, 1))
303 AttackSound(); // Play attack sound randomly
304 }
305 break;
306
307 default:
308 CBaseMonster::HandleAnimEvent(pEvent); // Call the base class method for unhandled events
309 break;
310 }
