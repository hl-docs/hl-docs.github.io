227void CPathTrack::Link(void)
228{
229 edict_t *pentTarget;
230
231 if (!FStringNull(pev->target))
232 {
233 pentTarget = FIND_ENTITY_BY_TARGETNAME(NULL, STRING(pev->target)); // Find the next path track
234 if (!FNullEnt(pentTarget))
235 {
236 m_pnext = CPathTrack::Instance(pentTarget); // Set the next path track

238 if (m_pnext) // If a next pointer exists, set the previous link
239 {
240 m_pnext->SetPrevious(this);

242 }
243 else
244 ALERT(at_console, "Dead end link %s\n", STRING(pev->target)); // Warn if no next path found
245 }
246
247 // Find "alternate" path
248 if (m_altName)
249 {
250 pentTarget = FIND_ENTITY_BY_TARGETNAME(NULL, STRING(m_altName)); // Find the alternate path track
251 if (!FNullEnt(pentTarget))

253 m_paltpath = CPathTrack::Instance(pentTarget); // Set the alternate path track
254
255 if (m_paltpath) // If a next pointer exists, set the previous link

396CPathTrack *CPathTrack::LookAhead(Vector *origin, float dist, int move)
397{
398 CPathTrack *pcurrent;
399 float originalDist = dist;
400
401 pcurrent = this; // Start from the current path track
402 Vector currentPos = *origin; // Store the current position
403
404 if (dist < 0) // Travelling backwards through path
405 {
406 dist = -dist; // Make distance positive
407 while (dist > 0)
408 {
409 Vector dir = pcurrent->pev->origin - currentPos; // Calculate direction to the current path track
410 float length = dir.Length(); // Get the length of the direction vector
411 if (!length) // If at the same position
412 {
413 if (!ValidPath(pcurrent->GetPrevious(), move)) // Check for a valid previous path
414 {
415 if (!move)
416 Project(pcurrent->GetNext(), pcurrent, origin, dist); // Project to the next path
417 return NULL; // Return NULL if no valid path
418 }
419 pcurrent = pcurrent->GetPrevious(); // Move to the previous path track
420 }
421 else if (length > dist) // Enough left in this path to move
422 {
423 *origin = currentPos + (dir * (dist / length)); // Calculate the new position
424 return pcurrent; // Return the current path track
425 }
426 else
427 {
428 dist -= length; // Decrease distance by the length of the path
429 currentPos = pcurrent->pev->origin; // Update current position
430 *origin = currentPos; // Set the origin to the current position
431 if (!ValidPath(pcurrent->GetPrevious(), move)) // Check for a valid previous path
432 return NULL; // Return NULL if no valid path
433
434 pcurrent = pcurrent->GetPrevious(); // Move to the previous path track
435 }
436 }
437 *origin = currentPos; // Set the final origin
438 return pcurrent; // Return the current path track
439 }
440 else
441 {
442 while (dist > 0)
443 {
444 if (!ValidPath(pcurrent->GetNext(), move)) // Check for a valid next path
445 {
446 if (!move)
447 Project(pcurrent->GetPrevious(), pcurrent, origin, dist); // Project to the previous path
448 return NULL; // Return NULL if no valid path
449 }
450 Vector dir = pcurrent->GetNext()->pev->origin - currentPos; // Calculate direction to the next path track
451 float length = dir.Length(); // Get the length of the direction vector
452 if (!length && !ValidPath(pcurrent->GetNext()->GetNext(), move)) // If at the same position and no valid next path
453 {
454 if (dist == originalDist) // HACK -- up against a dead end
455 return NULL; // Return NULL if at a dead end
456 return pcurrent; // Return the current path track
457 }
458 if (length > dist) // Enough left in this path to move
459 {
460 *origin = currentPos + (dir * (dist / length)); // Calculate the new position
461 return pcurrent; // Return the current path track
462 }
463 else
464 {
465 dist -= length; // Decrease distance by the length of the path
466 currentPos = pcurrent->GetNext()->pev->origin; // Update current position
467 pcurrent = pcurrent->GetNext(); // Move to the next path track
468 *origin = currentPos; // Set the origin to the current position
469 }
470 }
471 *origin = currentPos; // Set the final origin
472 }
473
474 return pcurrent; // Return the current path track
475}

486CPathTrack *CPathTrack::Nearest(Vector origin)
487{
488 int deadCount;
489 float minDist, dist;
490 Vector delta;
491 CPathTrack *ppath, *pnearest;
492
493 delta = origin - pev->origin; // Calculate the difference from the origin
494 delta.z = 0; // Ignore the z component
495 minDist = delta.Length(); // Get the initial minimum distance
496 pnearest = this; // Start with the current path track
497 ppath = GetNext(); // Get the next path track
498
499 // Iterate through the path tracks to find the nearest
500 deadCount = 0; // Initialize dead count
501 while (ppath && ppath != this)
502 {
503 deadCount++;
504 if (deadCount > 9999) // Prevent infinite loops
505 {
506 ALERT(at_error, "Bad sequence of path_tracks from %s", STRING(pev->targetname)); // Log an error
507 return NULL; // Return NULL if an error occurs
508 }
509 delta = origin - ppath->pev->origin; // Calculate the difference from the current path track
510 delta.z = 0; // Ignore the z component
511 dist = delta.Length(); // Get the distance to the current path track
512 if (dist < minDist) // If this path track is closer
513 {
514 minDist = dist; // Update the minimum distance
515 pnearest = ppath; // Update the nearest path track
516 }
517 ppath = ppath->GetNext(); // Move to the next path track
518 }
519 return pnearest; // Return the nearest path track
520}
