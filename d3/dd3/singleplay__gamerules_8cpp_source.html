110BOOL CHalfLifeRules::FShouldSwitchWeapon( CBasePlayer *pPlayer, CBasePlayerItem *pWeapon )
111{
112 if ( !pPlayer->m_pActiveItem )
113 {
114 // player doesn't have an active item!
115 return TRUE;
116 }
117
118 if ( !pPlayer->m_pActiveItem->CanHolster() )
119 {
120 return FALSE;
121 }
122
123 return TRUE;
124}

201float CHalfLifeRules::FlPlayerFallDamage( CBasePlayer *pPlayer )
202{
203 // subtract off the speed at which a player is allowed to fall without being hurt,
204 // so damage will be based on speed beyond that, not the entire fall
205 pPlayer->m_flFallVelocity -= PLAYER_MAX_SAFE_FALL_SPEED;
206 return pPlayer->m_flFallVelocity * DAMAGE_FOR_FALL_SPEED;
207}
